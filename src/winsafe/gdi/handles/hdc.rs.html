270	/// [`GetCurrentObject`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-getcurrentobject)
271	/// function.
272	///
273	/// # Examples
274	///
275	/// ```no_run
276	/// use winsafe::{self as w, prelude::*, co};
277	///
278	/// let hdc: w::HDC; // initialized somewhere
279	/// # let hdc = w::HDC::NULL;
280	///
281	/// let obj = hdc.GetCurrentObject(co::CUR_OBJ::BRUSH)?;
282	/// let w::CurObj::Brush(hbrush) = obj else { unreachable!() };
283	///
284	/// println!("HBRUSH: {}", hbrush);
285	/// w::SysResult::Ok(())
286	/// ```
287	#[must_use]
288	fn GetCurrentObject(&self, kind: co::CUR_OBJ) -> SysResult<CurObj> {
289		unsafe {
290			ptr_to_sysresult(ffi::GetCurrentObject(self.ptr(), kind.raw())).map(|h| match kind {
291				co::CUR_OBJ::BITMAP => CurObj::Bitmap(HBITMAP::from_ptr(h)),
292				co::CUR_OBJ::BRUSH => CurObj::Brush(HBRUSH::from_ptr(h)),
293				co::CUR_OBJ::FONT => CurObj::Font(HFONT::from_ptr(h)),
294				co::CUR_OBJ::PAL => CurObj::Pal(HPALETTE::from_ptr(h)),
295				co::CUR_OBJ::PEN => CurObj::Pen(HPEN::from_ptr(h)),
296				_ => panic!("co::OBJ_CUR not implemented: {}", kind),
297			})
298		}
299	}

311	/// [`GetDIBits`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-getdibits)
312	/// function.
313	///
314	/// # Safety
315	///
316	/// If `bmpDataBuf` is smaller than needed, you'll have a buffer overflow.
317	///
318	/// # Examples
319	///
320	/// Taking a screenshot and saving to file:
321	///
322	/// ```no_run
323	/// use winsafe::{self as w, prelude::*, co};
324	///
325	/// let cx_screen = w::GetSystemMetrics(co::SM::CXSCREEN);
326	/// let cy_screen = w::GetSystemMetrics(co::SM::CYSCREEN);
327	///
328	/// let hdc_screen = w::HWND::DESKTOP.GetDC()?;
329	/// let hbmp = hdc_screen.CreateCompatibleBitmap(cx_screen, cy_screen)?;
330	/// let hdc_mem = hdc_screen.CreateCompatibleDC()?;
331	/// let _hbmp_guard = hdc_mem.SelectObject(&*hbmp)?;
332	///
333	/// hdc_mem.BitBlt(w::POINT::new(0, 0), w::SIZE::new(cx_screen, cy_screen),
334	///     &hdc_screen, w::POINT::new(0, 0), co::ROP::SRCCOPY)?;
335	///
336	/// let mut bmp_obj = w::BITMAP::default();
337	/// hbmp.GetObject(&mut bmp_obj)?;
338	///
339	/// let mut bi = w::BITMAPINFO::default();
340	/// bi.bmiHeader.biWidth = cx_screen;
341	/// bi.bmiHeader.biHeight = cy_screen;
342	/// bi.bmiHeader.biPlanes = 1;
343	/// bi.bmiHeader.biBitCount = 32;
344	/// bi.bmiHeader.biCompression = co::BI::RGB;
345	///
346	/// let bmp_size = (bmp_obj.bmWidth * (bi.bmiHeader.biBitCount as i32) + 31)
347	///     / 32 * 4 * bmp_obj.bmHeight;
348	/// let mut data_buf = vec![0u8; bmp_size as _];
349	///
350	/// unsafe {
351	///     hdc_screen.GetDIBits(&hbmp, 0, cy_screen as _,
352	///         Some(&mut data_buf), &mut bi, co::DIB::RGB_COLORS)?;
353	/// }
354	///
355	/// let mut bfh = w::BITMAPFILEHEADER::default();
356	/// bfh.bfOffBits = (std::mem::size_of::<w::BITMAPFILEHEADER>()
357	///     + std::mem::size_of::<w::BITMAPINFOHEADER>()) as _;
358	/// bfh.bfSize = bfh.bfOffBits + (bmp_size as u32);
359	///
360	/// let fo = w::File::open("C:\\Temp\\foo.bmp", w::FileAccess::OpenOrCreateRW)?;
361	/// fo.write(bfh.serialize())?;
362	/// fo.write(bi.bmiHeader.serialize())?;
363	/// fo.write(&data_buf)?;
364	/// # w::SysResult::Ok(())
365	/// ```
366	unsafe fn GetDIBits(
367		&self,
368		hbm: &HBITMAP,
369		first_scan_line: u32,
370		num_scan_lines: u32,
371		bmp_data_buf: Option<&mut [u8]>,
372		bmi: &mut BITMAPINFO,
373		usage: co::DIB,
374	) -> SysResult<i32> {
375		let ret = ffi::GetDIBits(
376			self.ptr(),
377			hbm.ptr(),
378			first_scan_line,
379			num_scan_lines,
380			bmp_data_buf.map_or(std::ptr::null_mut(), |buf| buf.as_mut_ptr() as _),
381			bmi as *const _ as _,
382			usage.raw(),
383		);
384
385		if unsafe { co::ERROR::from_raw(ret as _) } == co::ERROR::INVALID_PARAMETER {
386			Err(co::ERROR::INVALID_PARAMETER)
387		} else if ret == 0 {
388			Err(GetLastError())
389		} else {
390			Ok(ret)
391		}
392	}

671	/// [`SelectObject`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-selectobject)
672	/// function.
673	///
674	/// In the original C implementation, `SelectObject` returns a handle to the
675	/// object being replaced. You must perform a cleanup operation, calling
676	/// `SelectObject` again, passing the handle to the replaced object.
677	///
678	/// Here, the cleanup is performed automatically, because `SelectObject`
679	/// returns a [`SelectObjectGuard`](crate::guard::SelectObjectGuard), which
680	/// stores the replaced handle and calls `SelectObject` automatically when
681	/// the guard goes out of scope. You must, however, keep the guard alive,
682	/// otherwise the cleanup will be performed right away.
683	///
684	/// # Examples
685	///
686	/// ```no_run
687	/// use winsafe::{self as w, prelude::*, co};
688	///
689	/// let hdc: w::HDC; // initialized somewhere
690	/// # let hdc = w::HDC::NULL;
691	///
692	/// let hpen = w::HPEN::CreatePen(
693	///     co::PS::SOLID,
694	///     1,
695	///     w::COLORREF::new(0xff, 0x00, 0x88),
696	/// )?;
697	///
698	/// let _pen_guard = hdc.SelectObject(&*hpen); // keep guard alive
699	/// # w::SysResult::Ok(())
700	/// ```
701	#[must_use]
702	fn SelectObject<G>(&self, hgdiobj: &G) -> SysResult<SelectObjectGuard<'_, Self, G>>
703	where
704		G: GdiObject,
705	{
706		unsafe {
707			ptr_to_sysresult(ffi::SelectObject(self.ptr(), hgdiobj.ptr())).map(|ptr| {
708				if hgdiobj.type_id() == TypeId::of::<HRGN>() {
709					SelectObjectGuard::new(
710						self,
711						G::NULL, // regions don't need cleanup
712						Some(co::REGION::from_raw(ptr as *mut _ as _)),
713					)
714				} else {
715					SelectObjectGuard::new(
716						self,
717						G::from_ptr(ptr), // GDI object to be passed to SelectObject at the end of scope
718						None,
719					)
720				}
721			})
722		}
723	}
