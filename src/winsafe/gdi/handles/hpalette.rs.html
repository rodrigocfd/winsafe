11/// This trait is enabled with the `gdi` feature, and provides methods for
12/// [`HFONT`](crate::HFONT).
13///
14/// Prefer importing this trait through the prelude:
15///
16/// ```no_run
17/// use winsafe::prelude::*;
18/// ```
19pub trait gdi_Hpalette: Handle {
20	/// [`CreatePalette`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-createpalette)
21	/// function.
22	#[must_use]
23	fn CreatePalette(pal: &LOGPALETTE) -> SysResult<DeleteObjectPaletteGuard> {
24		unsafe {
25			ptr_to_sysresult_handle(ffi::CreatePalette(pal as *const _ as _))
26				.map(|h| DeleteObjectPaletteGuard::new(h))
27		}
28	}
