18	/// [`CreateFont`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-createfontw)
19	/// function.
20	#[must_use]
21	pub fn CreateFont(
22		sz: SIZE,
23		escapement: i32,
24		orientation: i32,
25		weight: co::FW,
26		italic: bool,
27		underline: bool,
28		strike_out: bool,
29		char_set: co::CHARSET,
30		out_precision: co::OUT_PRECIS,
31		clip_precision: co::CLIP,
32		quality: co::QUALITY,
33		pitch_and_family: co::PITCH,
34		face_name: &str,
35	) -> SysResult<DeleteObjectGuard<HFONT>> {
36		unsafe {
37			PtrRet(ffi::CreateFontW(
38				sz.cy,
39				sz.cx,
40				escapement,
41				orientation,
42				weight.raw() as _,
43				italic as _,
44				underline as _,
45				strike_out as _,
46				char_set.raw() as _,
47				out_precision.raw() as _,
48				clip_precision.raw() as _,
49				quality.raw() as _,
50				pitch_and_family.raw() as _,
51				WString::from_str(face_name).as_ptr(),
52			))
53			.to_invalidparm_handle()
54			.map(|h| DeleteObjectGuard::new(h))
55		}
56	}
57
58	/// [`CreateFontIndirect`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-createfontindirectw)
59	/// function.
60	#[must_use]
61	pub fn CreateFontIndirect(lf: &LOGFONT) -> SysResult<DeleteObjectGuard<HFONT>> {
62		unsafe {
63			PtrRet(ffi::CreateFontIndirectW(pcvoid(lf)))
64				.to_invalidparm_handle()
65				.map(|h| DeleteObjectGuard::new(h))
66		}
67	}
68
69	/// [`GetObject`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-getobjectw)
70	/// function.
71	pub fn GetObject(&self) -> SysResult<LOGFONT> {
72		let mut lf = LOGFONT::default();
73		BoolRet(unsafe {
74			ffi::GetObjectW(self.ptr(), std::mem::size_of::<LOGFONT>() as _, pvoid(&mut lf))
75		})
76		.to_invalidparm()
77		.map(|_| lf)
78	}

80	/// [`GetStockObject`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-getstockobject)
81	/// function.
82	#[must_use]
83	pub fn GetStockObject(sf: co::STOCK_FONT) -> SysResult<HFONT> {
84		PtrRet(unsafe { ffi::GetStockObject(sf.raw()) }).to_invalidparm_handle()
85	}
