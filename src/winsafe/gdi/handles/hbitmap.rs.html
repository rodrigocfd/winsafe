28	/// [`GetObject`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-getobjectw)

31		let mut bm = BITMAP::default();
32		BoolRet(unsafe {
33			ffi::GetObjectW(self.ptr(), std::mem::size_of::<BITMAP>() as _, pvoid(&mut bm))
34		})
35		.to_invalidparm()
36		.map(|_| bm)
37	}
