12/// This trait is enabled with the `gdi` feature, and provides methods for
13/// [`HBITMAP`](crate::HBITMAP).
14///
15/// Prefer importing this trait through the prelude:
16///
17/// ```no_run
18/// use winsafe::prelude::*;
19/// ```
20pub trait gdi_Hbitmap: Handle {
21	/// [`CreateBitmap`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-createbitmap)
22	/// function.
23	#[must_use]
24	fn CreateBitmap(
25		sz: SIZE,
26		num_planes: u32,
27		bit_count: u32,
28		bits: *mut u8,
29	) -> SysResult<DeleteObjectGuard<HBITMAP>> {
30		unsafe {
31			ptr_to_invalidparm_handle(ffi::CreateBitmap(
32				sz.cx, sz.cy, num_planes, bit_count, bits as _,
33			))
34			.map(|h| DeleteObjectGuard::new(h))
35		}
36	}
37
38	/// [`GetObject`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-getobjectw)
39	/// function.
40	///
41	/// # Examples
42	///
43	/// ```no_run
44	/// use winsafe::{self as w, prelude::*};
45	///
46	/// let hbmp: w::HBITMAP; // initialized somewhere
47	/// # let hbmp = w::HBITMAP::NULL;
48	///
49	/// let mut bitmap = w::BITMAP::default();
50	/// hbmp.GetObject(&mut bitmap)?;
51	/// # w::SysResult::Ok(())
52	/// ```
53	fn GetObject(&self, pv: &mut BITMAP) -> SysResult<()> {
54		bool_to_invalidparm(unsafe {
55			ffi::GetObjectW(self.ptr(), std::mem::size_of::<BITMAP>() as _, pvoid(pv))
56		})
57	}
58}
