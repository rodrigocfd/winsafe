19/// This trait is enabled with the `gdi` feature, and provides methods for
20/// [`HPEN`](crate::HPEN).
21///
22/// Prefer importing this trait through the prelude:
23///
24/// ```no_run
25/// use winsafe::prelude::*;
26/// ```
27pub trait gdi_Hpen: Handle {
28	/// [`CreatePen`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-createpen)
29	/// function.
30	#[must_use]
31	fn CreatePen(style: co::PS, width: i32, color: COLORREF) -> SysResult<DeleteObjectGuard<HPEN>> {
32		unsafe {
33			ptr_to_invalidparm_handle(ffi::CreatePen(style.raw(), width, color.into()))
34				.map(|h| DeleteObjectGuard::new(h))
35		}
36	}
37
38	/// [`CreatePenIndirect`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-createpenindirect)
39	/// function.
40	#[must_use]
41	fn CreatePenIndirect(lp: &mut LOGPEN) -> SysResult<DeleteObjectGuard<HPEN>> {
42		unsafe {
43			ptr_to_invalidparm_handle(ffi::CreatePenIndirect(pcvoid(lp)))
44				.map(|h| DeleteObjectGuard::new(h))
45		}
46	}
47
48	/// [`GetStockObject`](https://learn.microsoft.com/en-us/windows/win32/api/wingdi/nf-wingdi-getstockobject)
49	/// function.
50	#[must_use]
51	fn GetStockObject(sp: co::STOCK_PEN) -> SysResult<HPEN> {
52		ptr_to_invalidparm_handle(unsafe { ffi::GetStockObject(sp.raw()) })
53	}
54}
