11/// [`ShellExecuteEx`](https://learn.microsoft.com/en-us/windows/win32/api/shellapi/nf-shellapi-shellexecuteexw)
12/// function.
13///
14/// Fill the [`SHELLEXECUTEINFO`](crate::SHELLEXECUTEINFO) fields you need, and
15/// leave the others as default. The needed mask flags will be automatically
16/// set.
17///
18/// # Examples
19///
20/// ```no_run
21/// use winsafe::{self as w, prelude::*, co};
22///
23/// w::ShellExecuteEx(&w::SHELLEXECUTEINFO {
24///     file: "C:\\Temp\\foo.exe",
25///     show: co::SW::SHOW,
26///     ..Default::default()
27/// })?;
28/// # w::SysResult::Ok(())
29/// ```
30pub fn ShellExecuteEx(exec_info: &SHELLEXECUTEINFO) -> SysResult<()> {
31	let mut buf = exec_info.to_raw();
32	bool_to_sysresult(unsafe { ffi::ShellExecuteExW(&mut buf.raw as *mut _ as _) })
33}

35/// [`SHGetKnownFolderPath`](https://learn.microsoft.com/en-us/windows/win32/api/shlobj_core/nf-shlobj_core-shgetknownfolderpath)
36/// function.
37///
38/// # Examples
39///
40/// Retrieving documents folder:
41///
42/// ```no_run
43/// use winsafe::{self as w, prelude::*, co};
44///
45/// let docs_folder = w::SHGetKnownFolderPath(
46///     &co::KNOWNFOLDERID::Documents,
47///     co::KF::DEFAULT,
48///     None,
49/// )?;
50///
51/// println!("Docs folder: {}", docs_folder);
52/// # w::HrResult::Ok(())
53/// ```
54#[must_use]
55pub fn SHGetKnownFolderPath(
56	folder_id: &co::KNOWNFOLDERID,
57	flags: co::KF,
58	token: Option<&HACCESSTOKEN>,
59) -> HrResult<String> {
60	let mut pstr = std::ptr::null_mut::<u16>();
61	ok_to_hrresult(unsafe {
62		ffi::SHGetKnownFolderPath(
63			folder_id as *const _ as _,
64			flags.raw(),
65			token.map_or(std::ptr::null_mut(), |t| t.ptr()),
66			&mut pstr,
67		)
68	})
69	.map(|_| {
70		let path = unsafe { WString::from_wchars_nullt(pstr) };
71		let _ = unsafe { CoTaskMemFreeGuard::new(pstr as _, 0) };
72		path.to_string()
73	})
74}
