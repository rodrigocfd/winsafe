9/// [`ShellExecuteEx`](https://learn.microsoft.com/en-us/windows/win32/api/shellapi/nf-shellapi-shellexecuteexw)
10/// function.
11///
12/// Fill the [`SHELLEXECUTEINFO`](crate::SHELLEXECUTEINFO) fields you need, and
13/// leave the others as default. The needed mask flags will be automatically
14/// set.
15///
16/// # Examples
17///
18/// ```no_run
19/// use winsafe::{self as w, prelude::*, co};
20///
21/// w::ShellExecuteEx(&w::SHELLEXECUTEINFO {
22///     file: "C:\\Temp\\foo.exe",
23///     show: co::SW::SHOW,
24///     ..Default::default()
25/// })?;
26/// # w::SysResult::Ok(())
27/// ```
28pub fn ShellExecuteEx(exec_info: &SHELLEXECUTEINFO) -> SysResult<()> {
29	let mut buf = exec_info.to_raw();
30	BoolRet(unsafe { ffi::ShellExecuteExW(pvoid(&mut buf.raw)) }).to_sysresult()
31}

33/// [`SHGetKnownFolderPath`](https://learn.microsoft.com/en-us/windows/win32/api/shlobj_core/nf-shlobj_core-shgetknownfolderpath)
34/// function.
35///
36/// # Examples
37///
38/// Retrieving documents folder:
39///
40/// ```no_run
41/// use winsafe::{self as w, prelude::*, co};
42///
43/// let docs_folder = w::SHGetKnownFolderPath(
44///     &co::KNOWNFOLDERID::Documents,
45///     co::KF::DEFAULT,
46///     None,
47/// )?;
48///
49/// println!("Docs folder: {}", docs_folder);
50/// # w::HrResult::Ok(())
51/// ```
52#[must_use]
53pub fn SHGetKnownFolderPath(
54	folder_id: &co::KNOWNFOLDERID,
55	flags: co::KF,
56	token: Option<&HACCESSTOKEN>,
57) -> HrResult<String> {
58	let mut pstr = std::ptr::null_mut::<u16>();
59	HrRet(unsafe {
60		ffi::SHGetKnownFolderPath(
61			pcvoid(folder_id),
62			flags.raw(),
63			token.map_or(std::ptr::null_mut(), |t| t.ptr()),
64			&mut pstr,
65		)
66	})
67	.to_hrresult()
68	.map(|_| htaskmem_ptr_to_str(pstr))
69}
