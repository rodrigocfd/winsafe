11/// This trait is enabled with `comctl` and `gdi` features, and provides methods
12/// for [`HIMAGELIST`](crate::HIMAGELIST).
13///
14/// Prefer importing this trait through the prelude:
15///
16/// ```no_run
17/// use winsafe::prelude::*;
18/// ```
19pub trait comctl_gdi_Himagelist: Handle {
20	/// [`ImageList_DrawIndirect`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_drawindirect)
21	/// function.
22	fn DrawIndirect(&self, imldp: &IMAGELISTDRAWPARAMS) -> HrResult<()> {
23		match unsafe { ffi::ImageList_DrawIndirect(self.ptr(), pcvoid(imldp)) } {
24			0 => Err(co::HRESULT::E_FAIL),
25			_ => Ok(()),
26		}
27	}
28}
