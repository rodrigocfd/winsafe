6impl HIMAGELIST {
7	/// Calls [`SHGetFileInfo`](crate::SHGetFileInfo) to retrieve one or more
8	/// shell file icons, then passes them to
9	/// [`AddIcon`](crate::HIMAGELIST::AddIcon).
10	///
11	/// # Examples
12	///
13	/// ```no_run
14	/// use winsafe::{self as w, prelude::*, co};
15	///
16	/// let mut himgl = w::HIMAGELIST::Create(
17	///     w::SIZE::with(16, 16),
18	///     co::ILC::COLOR32,
19	///     1,
20	///     1,
21	/// )?;
22	///
23	/// himgl.add_icons_from_shell(&["mp3", "wav"])?;
24	/// # w::HrResult::Ok(())
25	/// ```
26	pub fn add_icons_from_shell(&self, file_extensions: &[impl AsRef<str>]) -> HrResult<()> {
27		let sz = self.GetIconSize()?;
28		if sz != SIZE::with(16, 16) && sz != SIZE::with(32, 32) {
29			return Err(co::HRESULT::E_NOTIMPL); // only 16x16 or 32x32 icons can be loaded
30		}
31
32		for file_extension in file_extensions.iter() {
33			let (_, shfi) = SHGetFileInfo(
34				&format!("*.{}", file_extension.as_ref()),
35				co::FILE_ATTRIBUTE::NORMAL,
36				co::SHGFI::USEFILEATTRIBUTES
37					| co::SHGFI::ICON
38					| if sz == SIZE::with(16, 16) {
39						co::SHGFI::SMALLICON
40					} else {
41						co::SHGFI::LARGEICON
42					},
43			)?;
44			self.AddIcon(&shfi.hIcon)?;
45		}
46		Ok(())
47	}
48}
