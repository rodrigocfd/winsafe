3/// Parameters of a message which can be sent. Implemented by
4/// [all defined messages](crate::msg).
5///
6/// Allows the conversion to the generic [`WndMsg`](crate::msg::WndMsg)
7/// parameters, and also defines the return type of the message.
8///
9/// Used in functions like
10/// [`SendMessage`](crate::HWND::SendMessage) and
11/// [`DefWindowProc`](crate::HWND::DefWindowProc).
12pub trait MsgSend {
13	/// The specific type of the value returned by the message.
14	type RetType;
15
16	/// Unmarshaling method which converts the generic `isize` return value to
17	/// the specific type returned by the message.
18	///
19	/// # Safety
20	///
21	/// Return values often involve pointers and require casts, make sure the
22	/// conversions are correct.
23	#[must_use]
24	unsafe fn isize_to_ret(&self, v: isize) -> Self::RetType;
25
26	/// Marshaling method which converts the specific message parameters struct
27	/// into the generic [`WndMsg`](crate::msg::WndMsg) message struct.
28	#[must_use]
29	fn as_generic_wm(&mut self) -> WndMsg;
30}
31
32/// Parameters of a message which can be sent and handled (received).
33/// Implemented by [`WndMsg`](crate::msg::WndMsg) and all
34/// [msg::wm](`crate::msg::wm`) messages.
35///
36/// Allows the conversion from and to the generic [`WndMsg`](crate::msg::WndMsg)
37/// parameters, and also defines the return type of the message.
38pub trait MsgSendRecv: MsgSend {
39	/// Unmarshaling method which converts the generic
40	/// [`WndMsg`](crate::msg::WndMsg) parameters struct into the specific
41	/// message struct.
42	///
43	/// # Safety
44	///
45	/// Message parameters often involve pointers and require casts, make sure
46	/// the conversions are correct.
47	#[must_use]
48	unsafe fn from_generic_wm(parm: WndMsg) -> Self;
49}
