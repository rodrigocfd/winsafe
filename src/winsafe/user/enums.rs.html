8/// Variant parameter for:
9///
10/// * [`wm::Command`](crate::msg::wm::Command).
11pub enum AccelMenuCtrl {
12	/// Accelerator event. Contains the accelerator command ID.
13	Accel(u16),
14	/// Menu item click event. Contains the menu item command ID.
15	Menu(u16),
16	/// Some child control event. Contains
17	/// [`AccelMenuCtrlData`](crate::AccelMenuCtrlData) data.
18	Ctrl(AccelMenuCtrlData),
19}
20
21impl AccelMenuCtrl {
22	/// Returns the control ID.
23	#[must_use]
24	pub const fn ctrl_id(&self) -> u16 {
25		match self {
26			AccelMenuCtrl::Accel(id) => *id,
27			AccelMenuCtrl::Menu(id) => *id,
28			AccelMenuCtrl::Ctrl(data) => data.ctrl_id,
29		}
30	}
31
32	/// Returns the notification code.
33	#[must_use]
34	pub const fn code(&self) -> co::CMD {
35		match self {
36			AccelMenuCtrl::Accel(_) => co::CMD::Accelerator,
37			AccelMenuCtrl::Menu(_) => co::CMD::Menu,
38			AccelMenuCtrl::Ctrl(data) => data.notif_code,
39		}
40	}
41}

383	Id(u16),
384	/// Zero-based position.

386}
387
388impl IdPos {
389	/// Returns whether value is `Pos`.
390	#[must_use]
391	pub const fn is_by_pos(self) -> bool {
392		match self {
393			IdPos::Id(_) => false,
394			IdPos::Pos(_) => true,
395		}
396	}
397
398	/// Returns the ID or the position as a plain `u32`.
399	#[must_use]
400	pub const fn id_or_pos_u32(self) -> u32 {
401		match self {
402			IdPos::Id(id) => id as _,
403			IdPos::Pos(pos) => pos,
404		}
405	}
406
407	/// Returns [`MF::BYCOMMAND`](crate::co::MF::BYCOMMAND) if value is `Id`, or
408	/// [`MF::BYPOSITION`](crate::co::MF::BYPOSITION) if value is `Pos`.
409	#[must_use]
410	pub const fn mf_flag(self) -> co::MF {
411		match self {
412			IdPos::Id(_) => co::MF::BYCOMMAND,
413			IdPos::Pos(_) => co::MF::BYPOSITION,
414		}
415	}
