1use crate::co;

4/// Generic
5/// [window message](https://learn.microsoft.com/en-us/windows/win32/winmsg/about-messages-and-message-queues)
6/// parameters: `WPARAM` and `LPARAM`.
7///
8/// All message types can be converted to `WndMsg` via the
9/// [`as_generic_wm`](crate::prelude::MsgSend::as_generic_wm) method.
10///
11/// Return type: `isize`.
12#[derive(Clone, Copy, Debug)]
13pub struct WndMsg {
14	/// The [`co::WM`](crate::co::WM) constant that identifies the window
15	/// message.
16	pub msg_id: co::WM,
17	/// First message parameter.
18	pub wparam: usize,
19	/// Second message parameter.
20	pub lparam: isize,
21}
22
23impl MsgSend for WndMsg {
24	type RetType = isize;
25
26	unsafe fn isize_to_ret(&self, v: isize) -> Self::RetType {
27		v
28	}
29
30	fn as_generic_wm(&mut self) -> Self {
31		*self
32	}
33}
34
35impl MsgSendRecv for WndMsg {
36	unsafe fn from_generic_wm(p: Self) -> Self {
37		p
38	}
39}
40
41impl WndMsg {
42	/// Creates a new `WndMsg` from the given message parameters.
43	#[must_use]
44	pub const fn new(msg_id: co::WM, wparam: usize, lparam: isize) -> WndMsg {
45		Self { msg_id, wparam, lparam }
46	}
47}
