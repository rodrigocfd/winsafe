16/// This trait is enabled with the `user` feature, and provides methods for
17/// [`HMONITOR`](crate::HMONITOR).
18///
19/// Prefer importing this trait through the prelude:
20///
21/// ```no_run
22/// use winsafe::prelude::*;
23/// ```
24pub trait user_Hmonitor: Handle {
25	/// [`GetMonitorInfo`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-getmonitorinfow)
26	/// function.
27	fn GetMonitorInfo(&self) -> SysResult<MONITORINFOEX> {
28		let mut mi = MONITORINFOEX::default();
29		bool_to_sysresult(
30			unsafe { ffi::GetMonitorInfoW(self.ptr(), &mut mi as *mut _ as _) },
31		).map(|_| mi)
32	}
33
34	/// [`MonitorFromPoint`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-monitorfrompoint)
35	/// function.
36	#[must_use]
37	fn MonitorFromPoint(pt: POINT, flags: co::MONITOR) -> HMONITOR {
38		unsafe {
39			HMONITOR::from_ptr(ffi::MonitorFromPoint(pt.x, pt.y, flags.raw()))
40		}
41	}
42
43	/// [`MonitorFromRect`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-monitorfromrect)
44	/// function.
45	#[must_use]
46	fn MonitorFromRect(rc: RECT, flags: co::MONITOR) -> HMONITOR {
47		unsafe {
48			HMONITOR::from_ptr(
49				ffi::MonitorFromRect(&rc as *const _ as _, flags.raw()),
50			)
51		}
52	}
53}
