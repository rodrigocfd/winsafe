26	/// A more convenient
27	/// [`HMENU::AppendMenu`](crate::prelude::user_Hmenu::AppendMenu).
28	///
29	/// # Examples
30	///
31	/// Adding multiple entries at once, with their command IDs:
32	///
33	/// ```no_run
34	/// use winsafe::{self as w, prelude::*, seq_ids};
35	///
36	/// seq_ids! {
37	///     ID_FILE_OPEN = 2001;
38	///     ID_FILE_SAVE
39	///     ID_FILE_EXIT
40	/// }
41	///
42	/// let hmenu: w::HMENU; // initialized somewhere
43	/// # let hmenu = w::HMENU::NULL;
44	///
45	/// hmenu.append_item(&[
46	///     w::MenuItem::Entry(ID_FILE_OPEN, "&Open"),
47	///     w::MenuItem::Entry(ID_FILE_OPEN, "&Save"),
48	///     w::MenuItem::Separator,
49	///     w::MenuItem::Entry(ID_FILE_EXIT, "E&xit"),
50	/// ])?;
51	/// # w::SysResult::Ok(())
52	/// ```
53	fn append_item(&self, items: &[MenuItem]) -> SysResult<()> {
54		items
55			.iter()
56			.map(|item| match item {
57				MenuItem::Entry(cmd_id, text) => {
58					self.AppendMenu(co::MF::STRING, IdMenu::Id(*cmd_id), BmpPtrStr::from_str(*text))
59				},
60				MenuItem::Separator => {
61					self.AppendMenu(co::MF::SEPARATOR, IdMenu::None, BmpPtrStr::None)
62				},
63				MenuItem::Submenu(hmenu, text) => {
64					self.AppendMenu(co::MF::POPUP, IdMenu::Menu(hmenu), BmpPtrStr::from_str(*text))
65				},
66			})
67			.collect::<Result<Vec<_>, _>>()?;
68
69		Ok(())
70	}

72	/// Simpler version of
73	/// [`HMENU::GetMenuItemInfo`](crate::prelude::user_Hmenu::GetMenuItemInfo),
74	/// which returns a [`MenuItemInfo`](crate::MenuItemInfo) instead of the
75	/// tricky [`MENUITEMINFO`](crate::MENUITEMINFO).
76	///
77	/// # Examples
78	///
79	/// ```rust,no_run
80	/// use winsafe::{self as w, prelude::*};
81	///
82	/// let hmenu: w::HMENU; // initialized somewhere
83	/// # let hmenu = w::HMENU::NULL;
84	///
85	/// let item_info = hmenu.item_info(w::IdPos::Id(0))?;
86	/// match item_info {
87	///     w::MenuItemInfo::Entry(id, txt) => println!("item {} {}", id, txt),
88	///     w::MenuItemInfo::Separator => println!("separator"),
89	///     w::MenuItemInfo::Submenu(hsub, txt) => println!("submenu {} {}", hsub, txt),
90	/// }
91	/// # w::SysResult::Ok(())
92	/// ```
93	#[must_use]
94	fn item_info(&self, id_or_pos: IdPos) -> SysResult<MenuItemInfo> {
95		let mut mii = MENUITEMINFO::default();
96		mii.fMask = co::MIIM::FTYPE | co::MIIM::ID | co::MIIM::STATE | co::MIIM::SUBMENU;
97		self.GetMenuItemInfo(id_or_pos, &mut mii)?;
98
99		let nfo = if mii.fType == co::MFT::SEPARATOR {
100			MenuItemInfo::Separator
101		} else {
102			let text = self.GetMenuString(id_or_pos)?;
103			if mii.hSubMenu != HMENU::NULL {
104				MenuItemInfo::Submenu(mii.hSubMenu, text)
105			} else {
106				MenuItemInfo::Entry(mii.wID as _, text)
107			}
108		};
109
110		Ok(nfo)
111	}

139	/// Shows the popup menu anchored at the given coordinates using
140	/// [`TrackPopupMenu`](crate::prelude::user_Hmenu::TrackPopupMenu), and
141	/// performs other needed operations.
142	///
143	/// This method will block until the menu disappears.
144	fn track_popup_menu_at_point(
145		&self,
146		pos: POINT,
147		hwnd_parent: &HWND,
148		hwnd_coords_relative_to: &HWND,
149	) -> SysResult<()> {
150		let pos = hwnd_coords_relative_to.ClientToScreen(pos)?; // now relative to screen
151		hwnd_parent.SetForegroundWindow();
152		self.TrackPopupMenu(co::TPM::LEFTBUTTON, pos, hwnd_parent)?;
153		unsafe {
154			hwnd_parent.PostMessage(wm::Null {})?; // necessary according to TrackPopupMenu docs
155		}
156		Ok(())
157	}

378	/// [`GetMenuString`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-getmenustringw)
379	/// function.
380	#[must_use]
381	fn GetMenuString(&self, id_or_pos: IdPos) -> SysResult<String> {
382		let mut buf_sz = WString::SSO_LEN; // start with no string heap allocation
383		loop {
384			let mut buf = WString::new_alloc_buf(buf_sz);
385
386			let returned_chars = match unsafe {
387				// char count without terminating null
388				ffi::GetMenuStringW(
389					self.ptr(),
390					id_or_pos.id_or_pos_u32(),
391					buf.as_mut_ptr(),
392					buf.buf_len() as _,
393					id_or_pos.mf_flag().raw(),
394				)
395			} {
396				0 => return Err(GetLastError()),
397				n => n + 1, // plus terminating null count
398			};
399
400			if (returned_chars as usize) < buf_sz {
401				return Ok(buf.to_string()); // to break, must have at least 1 char gap
402			}
403
404			buf_sz *= 2; // double the buffer size to try again
405		}
406	}

493	/// [`TrackPopupMenu`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-trackpopupmenu)
494	/// function.
495	///
496	/// **Note:** If you just want to display a popup menu, consider the simpler
497	/// [`HMENU::track_popup_menu_at_point`](crate::prelude::user_Hmenu::track_popup_menu_at_point).
498	fn TrackPopupMenu(
499		&self,
500		flags: co::TPM,
501		location: POINT,
502		hwnd: &HWND,
503	) -> SysResult<Option<i32>> {
504		let ret = unsafe {
505			ffi::TrackPopupMenu(
506				self.ptr(),
507				flags.raw(),
508				location.x,
509				location.y,
510				0,
511				hwnd.ptr(),
512				std::ptr::null(),
513			)
514		};
515
516		if flags.has(co::TPM::RETURNCMD) {
517			match ret {
518				0 => match GetLastError() {
519					co::ERROR::SUCCESS => Ok(None), // success, user cancelled the menu
520					error => Err(error),
521				},
522				id => Ok(Some(id)), // success, menu item identifier
523			}
524		} else {
525			match ret {
526				0 => Err(GetLastError()),
527				_ => Ok(None),
528			}
529		}
530	}
