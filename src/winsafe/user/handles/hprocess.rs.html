8impl HPROCESS {
9	/// [`SetUserObjectInformation`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-setuserobjectinformationw)
10	/// function.
11	///
12	/// # Safety
13	///
14	/// The `pv_info` type varies according to `index`. If you set it wrong,
15	/// you're likely to cause a buffer overrun.
16	pub unsafe fn SetUserObjectInformation<T>(
17		&self,
18		index: co::UOI,
19		pv_info: &mut T,
20	) -> SysResult<()> {
21		BoolRet(unsafe {
22			ffi::SetUserObjectInformationW(
23				self.ptr(),
24				index.raw(),
25				pvoid(pv_info),
26				std::mem::size_of::<T>() as _,
27			)
28		})
29		.to_sysresult()
30	}
31
32	/// [`WaitForInputIdle`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-waitforinputidle)
33	/// function.
34	pub fn WaitForInputIdle(&self, milliseconds: u32) -> SysResult<SuccessTimeout> {
35		match unsafe { ffi::WaitForInputIdle(self.ptr(), milliseconds) } {
36			0 => Ok(SuccessTimeout::Success),
37			0x0000_0102 => Ok(SuccessTimeout::Timeout),
38			_ => Err(GetLastError()),
39		}
40	}
