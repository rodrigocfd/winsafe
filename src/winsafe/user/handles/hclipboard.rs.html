8handle! { HCLIPBOARD;
9	/// Handle to the
10	/// [clipboard](https://learn.microsoft.com/en-us/windows/win32/dataxchg/about-the-clipboard).
11	///
12	/// This handle doesn't exist natively, it's just an abstraction to safely
13	/// group the related clipboard operations.
14}
15
16impl HCLIPBOARD {
17	/// [`EmptyClipboard`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-emptyclipboard)
18	/// function.
19	pub fn EmptyClipboard(&self) -> SysResult<()> {
20		BoolRet(unsafe { ffi::EmptyClipboard() }).to_sysresult()
21	}
22
23	/// [`GetClipboardData`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-getclipboarddata)
24	/// function.
25	///
26	/// Calls [`HGLOBAL::GlobalSize`](crate::HGLOBAL::GlobalSize) and
27	/// [`HGLOBAL::GlobalLock`](crate::HGLOBAL::GlobalLock) internally to
28	/// retrieve a copy of the raw clipboard data.
29	///
30	/// Note that you should not trust the clipboard format â€“ the binary data
31	/// can be anything, despite what the format says. Be careful when parsing
32	/// the binary into your desired format.
33	///
34	/// # Examples
35	///
36	/// ```no_run
37	/// use winsafe::{self as w, prelude::*, co};
38	///
39	/// let hclip = w::HWND::NULL.OpenClipboard()?;
40	/// let data = hclip.GetClipboardData(co::CF::TEXT)?;
41	/// # w::SysResult::Ok(())
42	/// ```
43	#[must_use]
44	pub fn GetClipboardData(&self, format: co::CF) -> SysResult<Vec<u8>> {
45		let hglobal = PtrRet(unsafe { ffi::GetClipboardData(format.raw() as _) })
46			.to_sysresult_handle::<HGLOBAL>()?;
47		let copied = {
48			let block = hglobal.GlobalLock()?;
49			block.as_slice().to_vec()
50		};
51		Ok(copied)
52	}
53
54	/// [`GetClipboardSequenceNumber`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-getclipboardsequencenumber)
55	/// function.
56	#[must_use]
57	pub fn GetClipboardSequenceNumber(&self) -> u32 {
58		unsafe { ffi::GetClipboardSequenceNumber() }
59	}
60
61	/// [`SetClipboardData`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-setclipboarddata)
62	/// function.
63	///
64	/// Calls [`HGLOBAL::GlobalAlloc`](crate::HGLOBAL::GlobalAlloc) and
65	/// [`HGLOBAL::GlobalLock`](crate::HGLOBAL::GlobalLock) internally before
66	/// copying the data into the clipboard.
67	///
68	/// # Examples
69	///
70	/// ```no_run
71	/// use winsafe::{self as w, prelude::*, co};
72	///
73	/// let hclip = w::HWND::NULL.OpenClipboard()?;
74	///
75	/// let str_nullt = "foo"
76	///     .as_bytes()
77	///     .iter()
78	///     .map(|ch| *ch)
79	///     .chain(std::iter::once(0)) // null-terminated
80	///     .collect::<Vec<_>>();
81	///
82	/// hclip.SetClipboardData(co::CF::TEXT, &str_nullt)?;
83	/// # w::SysResult::Ok(())
84	/// ```
85	pub fn SetClipboardData(&self, format: co::CF, data: &[u8]) -> SysResult<()> {
86		let mut hglobal = HGLOBAL::GlobalAlloc(co::GMEM::MOVEABLE, data.len())?;
87		{
88			let mut block = hglobal.GlobalLock()?;
89			block.as_mut_slice().copy_from_slice(data); // copy the contents into HGLOBAL
90		}
91		PtrRet(unsafe { ffi::SetClipboardData(format.raw() as _, hglobal.leak().ptr() as _) })
92			.to_sysresult()?;
93		Ok(())
94	}
