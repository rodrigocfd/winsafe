16/// This trait is enabled with the `user` feature, and provides methods for
17/// [`HHOOK`](crate::HHOOK).
18///
19/// Prefer importing this trait through the prelude:
20///
21/// ```no_run
22/// use winsafe::prelude::*;
23/// ```
24pub trait user_Hhook: Handle {
25	/// [`CallNextHookEx`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-callnexthookex)
26	/// function.
27	fn CallNextHookEx(&self, code: co::WH, wparam: usize, lparam: isize) -> isize {
28		unsafe { ffi::CallNextHookEx(self.ptr(), code.raw(), wparam, lparam) }
29	}
30
31	/// [`SetWindowsHookEx`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-setwindowshookexw)
32	/// function.
33	fn SetWindowsHookEx(
34		hook_id: co::WH,
35		proc: HOOKPROC,
36		module: Option<&HINSTANCE>,
37		thread_id: Option<u32>,
38	) -> SysResult<HHOOK> {
39		ptr_to_sysresult_handle(unsafe {
40			ffi::SetWindowsHookExW(
41				hook_id.raw(),
42				proc as _,
43				module.map_or(std::ptr::null_mut(), |h| h.ptr()),
44				thread_id.unwrap_or_default(),
45			)
46		})
47	}
48
49	/// [`UnhookWindowsHookEx`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-unhookwindowshookex)
50	/// function.
51	///
52	/// After calling this method, the handle will be invalidated and further
53	/// operations will fail with
54	/// [`ERROR::INVALID_HANDLE`](crate::co::ERROR::INVALID_HANDLE) error code.
55	fn UnhookWindowsHookEx(&mut self) -> SysResult<()> {
56		let ret = bool_to_sysresult(unsafe { ffi::UnhookWindowsHookEx(self.ptr()) });
57		*self = Self::INVALID;
58		ret
59	}
60}
