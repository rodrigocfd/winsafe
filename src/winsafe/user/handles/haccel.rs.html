16/// This trait is enabled with the `user` feature, and provides methods for
17/// [`HACCEL`](crate::HACCEL).
18///
19/// Prefer importing this trait through the prelude:
20///
21/// ```no_run
22/// use winsafe::prelude::*;
23/// ```
24pub trait user_Haccel: Handle {
25	/// [`CreateAcceleratorTable`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createacceleratortablew)
26	/// function.
27	#[must_use]
28	fn CreateAcceleratorTable(accel: &[ACCEL]) -> SysResult<DestroyAcceleratorTableGuard> {
29		unsafe {
30			ptr_to_sysresult_handle(ffi::CreateAcceleratorTableW(
31				accel.as_ptr() as _,
32				accel.len() as _,
33			))
34			.map(|h| DestroyAcceleratorTableGuard::new(h))
35		}
36	}
37}
