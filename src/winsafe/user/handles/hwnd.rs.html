84	/// [`BeginPaint`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-beginpaint)
85	/// function.
86	///
87	/// In the original C implementation, `BeginPaint` returns a handle which
88	/// must be passed to
89	/// [`EndPaint`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-endpaint),
90	/// as a cleanup operation. Also, you must allocate and pass a
91	/// [`PAINTSTRUCT`](crate::PAINTSTRUCT) object.
92	///
93	/// Here, the cleanup is performed automatically, because `BeginPaint`
94	/// returns an [`EndPaintGuard`](crate::guard::EndPaintGuard), which stores
95	/// the `PAINTSTRUCT` and automatically calls `EndPaint` when the guard goes
96	/// out of scope. You must, however, keep the guard alive, otherwise the
97	/// cleanup will be performed right away.
98	///
99	/// # Examples
100	///
101	/// ```no_run
102	/// use winsafe::{self as w, prelude::*};
103	///
104	/// let hwnd: w::HWND; // initialized somewhere
105	/// # let hwnd = w::HWND::NULL;
106	///
107	/// let hdc = hwnd.BeginPaint()?;
108	///
109	/// // do your hdc painting...
110	///
111	/// // EndPaint() called automatically
112	/// # w::SysResult::Ok(())
113	/// ```
114	///
115	/// If you don't use the returned device context handle, you must still keep
116	/// the guard alive:
117	///
118	/// ```no_run
119	/// use winsafe::{self as w, prelude::*};
120	///
121	/// let hwnd: w::HWND; // initialized somewhere
122	/// # let hwnd = w::HWND::NULL;
123	///
124	/// let _hdc = hwnd.BeginPaint()?; // keep guard alive
125	///
126	/// // do your hdc painting...
127	///
128	/// // EndPaint() called automatically
129	/// # w::SysResult::Ok(())
130	/// ```
131	#[must_use]
132	pub fn BeginPaint(&self) -> SysResult<EndPaintGuard<'_>> {
133		let mut ps = PAINTSTRUCT::default();
134		unsafe {
135			PtrRet(ffi::BeginPaint(self.ptr(), pvoid(&mut ps)))
136				.to_sysresult_handle()
137				.map(|h| EndPaintGuard::new(self, h, ps))
138		}
139	}

1541	/// [`HWND::KillTimer`](crate::HWND::KillTimer).
1542	///
1543	/// The timer calls â€“ either
1544	/// [`wm_timer`](crate::prelude::GuiEventsWindow::wm_timer) message or
1545	/// callback function â€“ will continuously be executed until you call
1546	/// `KillTimer`. If you don't call `KillTimer`, the timer calls will
1547	/// continue until the window is destroyed â€“ at this point, any remaining
1548	/// timers will be automatically cleared.
1549	///
1550	/// # Why not closures?
1551	///
1552	/// A common C++ technique to use closures with `SetTimer` is allocating a
1553	/// closure on the heap and use its pointer as the timer ID. When the
1554	/// callback function is called, the pointer is dereferenced and the closure
1555	/// is then executed.
1556	///
1557	/// The problem with this approach is that the closure must be freed after
1558	/// `KillTimer`, which can be called from anywhere, including from the
1559	/// closure itself â€“ that means you must keep the pointer outside the
1560	/// closure and free it somehow after the closure finishes.
1561	///
1562	/// Such approach is, obviously, incredibly unsafe, and only possible within
1563	/// Rust's rigid ownership rules if we use some sort of garbage-collection,
1564	/// which will free the allocated closure some time after `KillTimer` is
1565	/// called and the closure itself finishes. Since that would incur in a
1566	/// performance penalty, the current implementation of `SetTimer` will only
1567	/// accept ordinary function pointers, not closures.
1568	///
1569	/// Handling the `wm_timer` message is simply more practical and efficient,
1570	/// so the use of a callback is discouraged here.
1571	pub fn SetTimer(

1576	) -> SysResult<usize> {
1577		match unsafe {
1578			ffi::SetTimer(
1579				self.ptr(),
1580				event_id,
1581				elapse_ms,
1582				timer_func.map_or(std::ptr::null(), |lp| lp as _),
1583			)
1584		} {
1585			0 => Err(GetLastError()),
1586			tid => Ok(tid),
1587		}
1588	}
