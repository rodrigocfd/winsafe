84	/// [`BeginPaint`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-beginpaint)
85	/// function.
86	///
87	/// In the original C implementation, `BeginPaint` returns a handle which
88	/// must be passed to
89	/// [`EndPaint`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-endpaint),
90	/// as a cleanup operation. Also, you must allocate and pass a
91	/// [`PAINTSTRUCT`](crate::PAINTSTRUCT) object.
92	///
93	/// Here, the cleanup is performed automatically, because `BeginPaint`
94	/// returns an [`EndPaintGuard`](crate::guard::EndPaintGuard), which stores
95	/// the `PAINTSTRUCT` and automatically calls `EndPaint` when the guard goes
96	/// out of scope. You must, however, keep the guard alive, otherwise the
97	/// cleanup will be performed right away.
98	///
99	/// # Examples
100	///
101	/// ```no_run
102	/// use winsafe::{self as w, prelude::*};
103	///
104	/// let hwnd: w::HWND; // initialized somewhere
105	/// # let hwnd = w::HWND::NULL;
106	///
107	/// let hdc = hwnd.BeginPaint()?;
108	///
109	/// // do your hdc painting...
110	///
111	/// // EndPaint() called automatically
112	/// # w::SysResult::Ok(())
113	/// ```
114	///
115	/// If you don't use the returned device context handle, you must still keep
116	/// the guard alive:
117	///
118	/// ```no_run
119	/// use winsafe::{self as w, prelude::*};
120	///
121	/// let hwnd: w::HWND; // initialized somewhere
122	/// # let hwnd = w::HWND::NULL;
123	///
124	/// let _hdc = hwnd.BeginPaint()?; // keep guard alive
125	///
126	/// // do your hdc painting...
127	///
128	/// // EndPaint() called automatically
129	/// # w::SysResult::Ok(())
130	/// ```
131	#[must_use]
132	pub fn BeginPaint(&self) -> SysResult<EndPaintGuard<'_>> {
133		let mut ps = PAINTSTRUCT::default();
134		unsafe {
135			PtrRet(ffi::BeginPaint(self.ptr(), pvoid(&mut ps)))
136				.to_sysresult_handle()
137				.map(|h| EndPaintGuard::new(self, h, ps))
138		}
139	}

820	/// ```

830			0 => match GetLastError() {

836	}

1546	/// This method returns the timer ID, to be passed to
1547	/// [`HWND::KillTimer`](crate::HWND::KillTimer).
1548	///
1549	/// The timer calls – either
1550	/// [`wm_timer`](crate::gui::events::WindowEvents::wm_timer) message or
1551	/// callback function – will continuously be executed until you call
1552	/// `KillTimer`. If you don't call `KillTimer`, the timer calls will
1553	/// continue until the window is destroyed – at this point, any remaining
1554	/// timers will be automatically cleared.
1555	///
1556	/// # Why not closures?
1557	///
1558	/// A common C++ technique to use closures with `SetTimer` is allocating a
1559	/// closure on the heap and use its pointer as the timer ID. When the
1560	/// callback function is called, the pointer is dereferenced and the closure
1561	/// is then executed.
1562	///
1563	/// The problem with this approach is that the closure must be freed after
1564	/// `KillTimer`, which can be called from anywhere, including from the
1565	/// closure itself – that means you must keep the pointer outside the
1566	/// closure and free it somehow after the closure finishes.
1567	///
1568	/// Such approach is, obviously, incredibly unsafe, and only possible within
1569	/// Rust's rigid ownership rules if we use some sort of garbage-collection,
1570	/// which will free the allocated closure some time after `KillTimer` is
1571	/// called and the closure itself finishes. Since that would incur in a
1572	/// performance penalty, the current implementation of `SetTimer` will only
1573	/// accept ordinary function pointers, not closures.
1574	///
1575	/// Handling the `wm_timer` message is simply more practical and efficient,
1576	/// so the use of a callback is discouraged here.
1577	pub fn SetTimer(
1578		&self,
1579		event_id: usize,
1580		elapse_ms: u32,
1581		timer_func: Option<TIMERPROC>,
1582	) -> SysResult<usize> {
1583		match unsafe {
1584			ffi::SetTimer(
1585				self.ptr(),
1586				event_id,
1587				elapse_ms,
1588				timer_func.map_or(std::ptr::null(), |lp| lp as _),
1589			)
1590		} {
1591			0 => Err(GetLastError()),
1592			tid => Ok(tid),
1593		}
1594	}
