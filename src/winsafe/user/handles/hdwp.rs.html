11	/// Handle to a
12	/// [deferred window position](https://learn.microsoft.com/en-us/windows/win32/winprog/windows-data-types#hdwp).
13}
14
15impl HDWP {
16	/// [`BeginDeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-begindeferwindowpos)
17	/// function.
18	///
19	/// In the original C implementation, `BeginDeferWindowPos` returns a handle
20	/// which must be passed to
21	/// [`EndDeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-enddeferwindowpos)
22	/// as a cleanup operation.
23	///
24	/// Here, the cleanup is performed automatically, because
25	/// `BeginDeferWindowPos` returns an
26	/// [`EndDeferWindowPosGuard`](crate::guard::EndDeferWindowPosGuard), which
27	/// automatically calls `EndDeferWindowPos` when the guard goes out of
28	/// scope.
29	#[must_use]
30	pub fn BeginDeferWindowPos(num_windows: u32) -> SysResult<EndDeferWindowPosGuard> {
31		unsafe {
32			PtrRet(ffi::BeginDeferWindowPos(num_windows as _))
33				.to_sysresult_handle()
34				.map(|h| EndDeferWindowPosGuard::new(h))
35		}
36	}
37
38	/// [`DeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-deferwindowpos)
39	/// function.
40	///
41	/// Originally this method returns the handle to the reallocated memory
42	/// object; here the original handle is automatically updated.
43	pub fn DeferWindowPos(
44		&mut self,
45		hwnd: &HWND,
46		hwnd_insert_after: HwndPlace,
47		top_left: POINT,
48		sz: SIZE,
49		flags: co::SWP,
50	) -> SysResult<()> {
51		match unsafe {
52			ffi::DeferWindowPos(
53				self.ptr(),
54				hwnd.ptr(),
55				hwnd_insert_after.as_ptr(),
56				top_left.x,
57				top_left.y,
58				sz.cx,
59				sz.cy,
60				flags.raw(),
61			)
62			.as_mut()
63		} {
64			Some(ptr) => {
65				*self = unsafe { Self::from_ptr(ptr) };
66				Ok(())
67			},
68			None => {
69				*self = Self::INVALID; // prevent EndDeferWindowPos()
70				Err(GetLastError())
71			},
72		}
73	}
