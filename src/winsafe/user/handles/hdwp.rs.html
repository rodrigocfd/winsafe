11	/// Handle to a
12	/// [deferred window position](https://learn.microsoft.com/en-us/windows/win32/winprog/windows-data-types#hdwp).
13}
14
15impl HDWP {
16	/// [`BeginDeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-begindeferwindowpos)
17	/// function.
18	///
19	/// In the original C implementation, `BeginDeferWindowPos` returns a handle
20	/// which must be passed to
21	/// [`EndDeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-enddeferwindowpos)
22	/// as a cleanup operation.
23	///
24	/// Here, the cleanup is performed automatically, because
25	/// `BeginDeferWindowPos` returns an
26	/// [`EndDeferWindowPosGuard`](crate::guard::EndDeferWindowPosGuard), which
27	/// automatically calls `EndDeferWindowPos` when the guard goes out of
28	/// scope.
29	#[must_use]
30	pub fn BeginDeferWindowPos(num_windows: u32) -> SysResult<EndDeferWindowPosGuard> {
31		unsafe {
32			ptr_to_sysresult_handle(ffi::BeginDeferWindowPos(num_windows as _))
33				.map(|h| EndDeferWindowPosGuard::new(h))
34		}
35	}
36
37	/// [`DeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-deferwindowpos)
38	/// function.
39	///
40	/// Originally this method returns the handle to the reallocated memory
41	/// object; here the original handle is automatically updated.
42	pub fn DeferWindowPos(
43		&mut self,
44		hwnd: &HWND,
45		hwnd_insert_after: HwndPlace,
46		top_left: POINT,
47		sz: SIZE,
48		flags: co::SWP,
49	) -> SysResult<()> {
50		match unsafe {
51			ffi::DeferWindowPos(
52				self.ptr(),
53				hwnd.ptr(),
54				hwnd_insert_after.as_ptr(),
55				top_left.x,
56				top_left.y,
57				sz.cx,
58				sz.cy,
59				flags.raw(),
60			)
61			.as_mut()
62		} {
63			Some(ptr) => {
64				*self = unsafe { Self::from_ptr(ptr) };
65				Ok(())
66			},
67			None => {
68				*self = Self::INVALID; // prevent EndDeferWindowPos()
69				Err(GetLastError())
70			},
71		}
72	}
73}
