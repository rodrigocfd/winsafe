17/// This trait is enabled with the `user` feature, and provides methods for
18/// [`HDWP`](crate::HDWP).
19///
20/// Prefer importing this trait through the prelude:
21///
22/// ```no_run
23/// use winsafe::prelude::*;
24/// ```
25pub trait user_Hdwp: Handle {
26	/// [`BeginDeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-begindeferwindowpos)
27	/// function.
28	///
29	/// In the original C implementation, `BeginDeferWindowPos` returns a handle
30	/// which must be passed to
31	/// [`EndDeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-enddeferwindowpos)
32	/// as a cleanup operation.
33	///
34	/// Here, the cleanup is performed automatically, because
35	/// `BeginDeferWindowPos` returns an
36	/// [`EndDeferWindowPosGuard`](crate::guard::EndDeferWindowPosGuard), which
37	/// automatically calls `EndDeferWindowPos` when the guard goes out of
38	/// scope.
39	#[must_use]
40	fn BeginDeferWindowPos(
41		num_windows: u32,
42	) -> SysResult<EndDeferWindowPosGuard>
43	{
44		unsafe {
45			ptr_to_sysresult_handle(ffi::BeginDeferWindowPos(num_windows as _))
46				.map(|h| EndDeferWindowPosGuard::new(h))
47		}
48	}
49
50	/// [`DeferWindowPos`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-deferwindowpos)
51	/// function.
52	///
53	/// Originally this method returns the handle to the reallocated memory
54	/// object; here the original handle is automatically updated.
55	fn DeferWindowPos(&mut self,
56		hwnd: &HWND,
57		hwnd_insert_after: HwndPlace,
58		top_left: POINT,
59		sz: SIZE,
60		flags: co::SWP,
61	) -> SysResult<()>
62	{
63		match unsafe {
64			ffi::DeferWindowPos(
65				self.ptr(),
66				hwnd.ptr(),
67				hwnd_insert_after.as_ptr(),
68				top_left.x, top_left.y, sz.cx, sz.cy,
69				flags.raw(),
70			).as_mut()
71		} {
72			Some(ptr) => {
73				*self = unsafe { Self::from_ptr(ptr) };
74				Ok(())
75			},
76			None => {
77				*self = Self::INVALID; // prevent EndDeferWindowPos()
78				Err(GetLastError())
79			},
80		}
81	}
82}
