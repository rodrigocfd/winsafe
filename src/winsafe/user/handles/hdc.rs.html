16/// This trait is enabled with the `user` feature, and provides methods for
17/// [`HDC`](crate::HDC).
18///
19/// Prefer importing this trait through the prelude:
20///
21/// ```no_run
22/// use winsafe::prelude::*;
23/// ```
24pub trait user_Hdc: Handle {
25	/// [`DrawFocusRect`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-drawfocusrect)
26	/// function.
27	fn DrawFocusRect(&self, rect: RECT) -> SysResult<()> {
28		bool_to_sysresult(unsafe { ffi::DrawFocusRect(self.ptr(), &rect as *const _ as _) })
29	}
30
31	/// [`DrawText`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-drawtextw)
32	/// function.
33	fn DrawText(&self, text: &str, bounds: RECT, format: co::DT) -> SysResult<i32> {
34		let wtext = WString::from_str(text);
35		match unsafe {
36			ffi::DrawTextW(
37				self.ptr(),
38				wtext.as_ptr(),
39				wtext.str_len() as _,
40				&bounds as *const _ as _,
41				format.raw(),
42			)
43		} {
44			0 => Err(GetLastError()),
45			i => Ok(i),
46		}
47	}
48
49	/// [`DrawTextExW`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-drawtextexw)
50	/// function.
51	fn DrawTextEx(
52		&self,
53		text: &str,
54		bounds: RECT,
55		format: co::DT,
56		dtp: Option<&DRAWTEXTPARAMS>,
57	) -> SysResult<i32> {
58		let wtext = WString::from_str(text);
59		match unsafe {
60			ffi::DrawTextExW(
61				self.ptr(),
62				wtext.as_ptr(),
63				wtext.str_len() as _,
64				&bounds as *const _ as _,
65				format.raw(),
66				dtp.map_or(std::ptr::null(), |p| p as *const _ as _),
67			)
68		} {
69			0 => Err(GetLastError()),
70			i => Ok(i),
71		}
72	}
73
74	/// [`EnumDisplayMonitors`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-enumdisplaymonitors)
75	/// function.
76	///
77	/// # Examples
78	///
79	/// ```no_run
80	/// use winsafe::{self as w, prelude::*};
81	///
82	/// let hdc: w::HDC; // initialized somewhere
83	/// # let hdc = w::HDC::NULL;
84	///
85	/// hdc.EnumDisplayMonitors(
86	///     None,
87	///     |hmon: w::HMONITOR, hdc: w::HDC, rc: &w::RECT| -> bool {
88	///         println!("HMONITOR: {}, ", hmon);
89	///         true
90	///     },
91	/// )?;
92	/// # w::SysResult::Ok(())
93	/// ```
94	fn EnumDisplayMonitors<F>(&self, rc_clip: Option<RECT>, func: F) -> SysResult<()>
95	where
96		F: FnMut(HMONITOR, HDC, &RECT) -> bool,
97	{
98		bool_to_sysresult(unsafe {
99			ffi::EnumDisplayMonitors(
100				self.ptr(),
101				rc_clip.map_or(std::ptr::null_mut(), |rc| &rc as *const _ as _),
102				proc::hdc_enum_display_monitors::<F> as _,
103				&func as *const _ as _,
104			)
105		})
106	}
107
108	/// [`FrameRect`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-framerect)
109	/// function.
110	fn FrameRect(&self, rc: RECT, hbr: &HBRUSH) -> SysResult<()> {
111		bool_to_sysresult(unsafe { ffi::FrameRect(self.ptr(), &rc as *const _ as _, hbr.ptr()) })
112	}
113
114	/// [`InvertRect`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-invertrect)
115	/// function.
116	fn InvertRect(&self, rc: RECT) -> SysResult<()> {
117		bool_to_sysresult(unsafe { ffi::InvertRect(self.ptr(), &rc as *const _ as _) })
118	}
119
120	/// [`PaintDesktop`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-paintdesktop)
121	/// function.
122	fn PaintDesktop(&self) -> SysResult<()> {
123		bool_to_sysresult(unsafe { ffi::PaintDesktop(self.ptr()) })
124	}
125
126	/// [`WindowFromDC`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-windowfromdc)
127	/// function.
128	#[must_use]
129	fn WindowFromDC(&self) -> Option<HWND> {
130		ptr_to_option_handle(unsafe { ffi::WindowFromDC(self.ptr()) })
131	}
132}
