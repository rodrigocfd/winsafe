3/// This trait is enabled with the `gui` feature, and is implemented by the
4/// parameters of messages which can be sent. Implemented by
5/// [all defined messages](crate::msg).
6///
7/// Allows the conversion to the generic [`Wm`](crate::msg::Wm) parameters, and
8/// also defines the return type of the message.
9///
10/// Used in functions like [`SendMessage`](crate::HWND::SendMessage) and
11/// [`DefWindowProc`](crate::HWND::DefWindowProc).
12///
13/// Prefer importing this trait through the prelude:
14///
15/// ```no_run
16/// use winsafe::prelude::*;
17/// ```
18pub trait MsgSend {
19	/// The specific type of the value returned by the message.
20	type RetType;
21
22	/// Unmarshaling method which converts the generic `isize` return value to
23	/// the specific type returned by the message.
24	///
25	/// # Safety
26	///
27	/// Return values often involve pointers and require casts, make sure the
28	/// conversions are correct.
29	#[must_use]
30	unsafe fn isize_to_ret(&self, v: isize) -> Self::RetType;
31
32	/// Marshaling method which converts the specific message parameters struct
33	/// into the generic [`Wm`](crate::msg::Wm) message struct.
34	#[must_use]
35	fn as_generic_wm(&mut self) -> Wm;
36}

38/// This trait is enabled with the `gui` feature, and is implemented by the
39/// parameters of messages which can be sent and handled (received).
40///
41/// Allows the conversion from and to the generic [`Wm`](crate::msg::Wm)
42/// parameters, and also defines the return type of the message.
43///
44/// Prefer importing this trait through the prelude:
45///
46/// ```no_run
47/// use winsafe::prelude::*;
48/// ```
49pub trait MsgSendRecv: MsgSend {
50	/// Unmarshaling method which converts the generic [`Wm`](crate::msg::Wm)
51	/// parameters struct into the specific message struct.
52	///
53	/// # Safety
54	///
55	/// Message parameters often involve pointers and require casts, make sure
56	/// the conversions are correct.
57	#[must_use]
58	unsafe fn from_generic_wm(parm: Wm) -> Self;
59}
