3/// This trait is enabled with the `gui` feature, and is implemented by the
4/// parameters of messages which can be sent. Implemented by
5/// [all defined messages](crate::msg).
6///
7/// Allows the conversion to the generic [`WndMsg`](crate::msg::WndMsg)
8/// parameters, and also defines the return type of the message.
9///
10/// Used in functions like
11/// [`SendMessage`](crate::HWND::SendMessage) and
12/// [`DefWindowProc`](crate::HWND::DefWindowProc).
13///
14/// Prefer importing this trait through the prelude:
15///
16/// ```no_run
17/// use winsafe::prelude::*;
18/// ```
19pub trait MsgSend {
20	/// The specific type of the value returned by the message.
21	type RetType;
22
23	/// Unmarshaling method which converts the generic `isize` return value to
24	/// the specific type returned by the message.
25	///
26	/// # Safety
27	///
28	/// Return values often involve pointers and require casts, make sure the
29	/// conversions are correct.
30	#[must_use]
31	unsafe fn isize_to_ret(&self, v: isize) -> Self::RetType;
32
33	/// Marshaling method which converts the specific message parameters struct
34	/// into the generic [`WndMsg`](crate::msg::WndMsg) message struct.
35	#[must_use]
36	fn as_generic_wm(&mut self) -> WndMsg;
37}
38
39/// This trait is enabled with the `gui` feature, and is implemented by the
40/// parameters of messages which can be sent and handled (received). Implemented
41/// by [`WndMsg`](crate::msg::WndMsg) and all [msg::wm](`crate::msg::wm`)
42/// messages.
43///
44/// Allows the conversion from and to the generic [`WndMsg`](crate::msg::WndMsg)
45/// parameters, and also defines the return type of the message.
46///
47/// Prefer importing this trait through the prelude:
48///
49/// ```no_run
50/// use winsafe::prelude::*;
51/// ```
52pub trait MsgSendRecv: MsgSend {
53	/// Unmarshaling method which converts the generic
54	/// [`WndMsg`](crate::msg::WndMsg) parameters struct into the specific
55	/// message struct.
56	///
57	/// # Safety
58	///
59	/// Message parameters often involve pointers and require casts, make sure
60	/// the conversions are correct.
61	#[must_use]
62	unsafe fn from_generic_wm(parm: WndMsg) -> Self;
63}
