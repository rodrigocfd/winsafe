8/// [`InternetCanonicalizeUrl`](https://learn.microsoft.com/en-us/windows/win32/api/wininet/nf-wininet-internetcanonicalizeurlw)
9/// function.
10#[must_use]
11pub fn InternetCanonicalizeUrl(url: &str, flags: co::ICU) -> SysResult<String> {
12	let wurl = WString::from_str(url);
13	let mut buf_sz = WString::SSO_LEN as u32; // start with no string heap allocation
14
15	loop {
16		let mut buf = WString::new_alloc_buf(buf_sz as _);
17		match bool_to_sysresult(unsafe {
18			ffi::InternetCanonicalizeUrlW(wurl.as_ptr(), buf.as_mut_ptr(), &mut buf_sz, flags.raw())
19		}) {
20			Ok(_) => return Ok(buf.to_string()),
21			Err(err) => match err {
22				co::ERROR::INSUFFICIENT_BUFFER => continue,
23				err => return Err(err),
24			},
25		}
26	}
27}

57/// [`InternetCrackUrl`](https://learn.microsoft.com/en-us/windows/win32/api/wininet/nf-wininet-internetcrackurlw)
58/// function.
59#[must_use]
60pub fn InternetCrackUrl(url: &str, flags: co::ICU) -> SysResult<URL_COMPONENTS> {
61	let w_url = WString::from_str(url);
62	let mut buf = URL_COMPONENTS_buf::new();
63	buf.set_initial_ptrs();
64
65	loop {
66		match bool_to_sysresult(unsafe {
67			ffi::InternetCrackUrlW(
68				w_url.as_ptr(),
69				url.chars().count() as _,
70				flags.raw(),
71				pvoid(&mut buf.raw),
72			)
73		}) {
74			Ok(_) => return Ok(buf.to_final()),
75			Err(err) => match err {
76				co::ERROR::INSUFFICIENT_BUFFER => {
77					buf.alloc_more_strs();
78					continue;
79				},
80				err => return Err(err),
81			},
82		}
83	}
84}
