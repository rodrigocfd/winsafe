8/// [`InternetCanonicalizeUrl`](https://learn.microsoft.com/en-us/windows/win32/api/wininet/nf-wininet-internetcanonicalizeurlw)
9/// function.
10#[must_use]
11pub fn InternetCanonicalizeUrl(url: &str, flags: co::ICU) -> SysResult<String> {
12	let wurl = WString::from_str(url);
13	let mut buf_sz = WString::SSO_LEN as u32; // start with no string heap allocation
14
15	loop {
16		let mut buf = WString::new_alloc_buf(buf_sz as _);
17		match BoolRet(unsafe {
18			ffi::InternetCanonicalizeUrlW(wurl.as_ptr(), buf.as_mut_ptr(), &mut buf_sz, flags.raw())
19		})
20		.to_sysresult()
21		{
22			Ok(_) => return Ok(buf.to_string()),
23			Err(err) => match err {
24				co::ERROR::INSUFFICIENT_BUFFER => continue,
25				err => return Err(err),
26			},
27		}
28	}
29}

98#[must_use]
99pub fn InternetCreateUrl(components: &URL_COMPONENTS, flags: co::ICU) -> SysResult<String> {
100	let mut scheme_w = WString::from_str(&components.scheme);
101	let mut host_name_w = WString::from_str(&components.host_name);
102	let mut user_name_w = WString::from_str(&components.user_name);
103	let mut password_w = WString::from_str(&components.password);
104	let mut url_path_w = WString::from_str(&components.url_path);
105	let mut extra_info_w = WString::from_str(&components.extra_info);
106
107	let mut raw = URL_COMPONENTS_raw::new_create(&components);
108	if scheme_w.is_allocated() {
109		raw.lpszScheme = unsafe { scheme_w.as_mut_ptr() };
110		raw.dwSchemeLength = scheme_w.str_len() as _;
111	}
112	if host_name_w.is_allocated() {
113		raw.lpszHostName = unsafe { host_name_w.as_mut_ptr() };
114		raw.dwHostNameLength = host_name_w.str_len() as _;
115	}
116	if user_name_w.is_allocated() {
117		raw.lpszUserName = unsafe { user_name_w.as_mut_ptr() };
118		raw.dwUserNameLength = user_name_w.str_len() as _;
119	}
120	if password_w.is_allocated() {
121		raw.lpszPassword = unsafe { password_w.as_mut_ptr() };
122		raw.dwPasswordLength = password_w.str_len() as _;
123	}
124	if url_path_w.is_allocated() {
125		raw.lpszUrlPath = unsafe { url_path_w.as_mut_ptr() };
126		raw.dwUrlPathLength = url_path_w.str_len() as _;
127	}
128	if extra_info_w.is_allocated() {
129		raw.lpszExtraInfo = unsafe { extra_info_w.as_mut_ptr() };
130		raw.dwExtraInfoLength = extra_info_w.str_len() as _;
131	}
132
133	let mut url_len = 0u32;
134	match BoolRet(unsafe {
135		ffi::InternetCreateUrlW(pcvoid(&raw), flags.raw(), std::ptr::null_mut(), &mut url_len) // first call to retrieve len
136	})
137	.to_sysresult()
138	{
139		Ok(_) => {}, // should never happen
140		Err(err) => match err {
141			co::ERROR::INSUFFICIENT_BUFFER => {}, // expected
142			err => return Err(err),
143		},
144	}
145
146	let mut buf = WString::new_alloc_buf(url_len as _);
147	BoolRet(unsafe {
148		ffi::InternetCreateUrlW(pcvoid(&raw), flags.raw(), buf.as_mut_ptr(), &mut url_len)
149	})
150	.to_sysresult()
151	.map(|_| buf.to_string())
152}
