33/// RAII implementation for [`SID`](crate::SID), returned by
34/// [`AllocateAndInitializeSid`](crate::AllocateAndInitializeSid), which
35/// automatically calls
36/// [`FreeSid`](https://learn.microsoft.com/en-us/windows/win32/api/securitybaseapi/nf-securitybaseapi-freesid)
37/// when the object goes out of scope.
38pub struct FreeSidGuard {
39	psid: *mut SID,
40}
41
42impl Drop for FreeSidGuard {
43	fn drop(&mut self) {
44		if !self.psid.is_null() {
45			unsafe {
46				ffi::FreeSid(self.psid as *mut _ as _); // ignore errors
47			}
48		}
49	}
50}
51
52impl Deref for FreeSidGuard {
53	type Target = SID;
54
55	fn deref(&self) -> &Self::Target {
56		unsafe { &*self.psid }
57	}
58}
59
60impl std::fmt::Display for FreeSidGuard {
61	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
62		std::fmt::Display::fmt(self.deref(), f) // delegate to the underlying SID
63	}
64}

66impl FreeSidGuard {
67	/// Constructs the guard by taking ownership of the pointer.
68	///
69	/// # Safety
70	///
71	/// Be sure the pointer must be freed with
72	/// [`FreeSid`](https://learn.microsoft.com/en-us/windows/win32/api/securitybaseapi/nf-securitybaseapi-freesid).
73	#[must_use]
74	pub const unsafe fn new(psid: *mut SID) -> Self {
75		Self { psid }
76	}
77
78	/// Ejects the underlying pointer, leaving a null pointer in its place.
79	///
80	/// Since the internal pointer will be invalidated, the destructor will not
81	/// run. It's your responsability to run it, otherwise you'll cause a
82	/// resource leak.
83	#[must_use]
84	pub fn leak(&mut self) -> *mut SID {
85		std::mem::replace(&mut self.psid, std::ptr::null_mut())
86	}
87}

89/// RAII implementation for [`HKEY`](crate::HKEY) which automatically calls
90/// [`RegCloseKey`](https://learn.microsoft.com/en-us/windows/win32/api/winreg/nf-winreg-regclosekey)
91/// when the object goes out of scope.
92pub struct RegCloseKeyGuard {
93	hkey: HKEY,
94}
95
96impl Drop for RegCloseKeyGuard {
97	fn drop(&mut self) {
98		if let Some(h) = self.hkey.as_opt() {
99			// Don't call on predefined keys, these belong to the system.
100			if !self.is_predef_key() {
101				unsafe {
102					ffi::RegCloseKey(h.ptr()); // ignore errors
103				}
104			}
105		}
106	}
107}
108

110	type Target = HKEY;
111
112	fn deref(&self) -> &Self::Target {
113		&self.hkey
114	}
115}
116
117impl DerefMut for RegCloseKeyGuard {
118	fn deref_mut(&mut self) -> &mut Self::Target {
119		&mut self.hkey
120	}
121}

222/// RAII implementation for [`TOKEN_GROUPS`](crate::TOKEN_GROUPS) which manages
223/// the allocated memory.
224pub struct TokenGroupsGuard<'a> {
225	ptr: GlobalFreeGuard,
226	_groups: PhantomData<&'a ()>,
227}
228
229impl<'a> Deref for TokenGroupsGuard<'a> {
230	type Target = TOKEN_GROUPS<'a>;
231
232	fn deref(&self) -> &Self::Target {
233		unsafe { &*(self.ptr.ptr() as *const _) }
234	}
235}
236
237impl<'a> DerefMut for TokenGroupsGuard<'a> {
238	fn deref_mut(&mut self) -> &mut Self::Target {
239		unsafe { &mut *(self.ptr.ptr() as *mut _) }
240	}
241}
242
243impl<'a> TokenGroupsGuard<'a> {
244	#[must_use]
245	pub(in crate::advapi) fn new(groups: &'a [SID_AND_ATTRIBUTES<'a>]) -> SysResult<Self> {
246		let sz = std::mem::size_of::<TOKEN_GROUPS>() // size in bytes of the allocated struct
247			- std::mem::size_of::<SID_AND_ATTRIBUTES>()
248			+ (groups.len() * std::mem::size_of::<SID_AND_ATTRIBUTES>());
249		let mut new_self = Self {
250			ptr: HGLOBAL::GlobalAlloc(co::GMEM::FIXED | co::GMEM::ZEROINIT, sz)?,
251			_groups: PhantomData,
252		};
253		new_self.GroupCount = groups.len() as _;
254		groups
255			.iter()
256			.zip(new_self.Groups_mut())
257			.for_each(|(src, dest)| *dest = src.clone()); // copy all SID_AND_ATTRIBUTES into struct room
258		Ok(new_self)
259	}
260}
