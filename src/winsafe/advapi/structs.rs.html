166/// [`SID`](https://learn.microsoft.com/en-us/windows/win32/api/winnt/ns-winnt-sid)
167/// struct.
168///
169/// Note that you cannot directly instantiate this struct, because the
170/// `SubAuthority` field is dynamically allocated.
171///
172/// Possible ways:
173///
174/// * [`AllocateAndInitializeSid`](crate::AllocateAndInitializeSid) as [`FreeSidGuard`](crate::guard::FreeSidGuard);
175/// * [`ConvertStringSidToSid`](crate::ConvertStringSidToSid) as [`LocalFreeSidGuard`](crate::guard::LocalFreeSidGuard);
176/// * [`CopySid`](crate::CopySid) as [`SidGuard`](crate::guard::SidGuard);
177/// * [`CreateWellKnownSid`](crate::CreateWellKnownSid) as [`SidGuard`](crate::guard::SidGuard);
178/// * [`GetWindowsAccountDomainSid`](crate::GetWindowsAccountDomainSid) as [`SidGuard`](crate::guard::SidGuard);
179/// * [`LookupAccountName`](crate::LookupAccountName) as [`SidGuard`](crate::guard::SidGuard).
180#[repr(C)]
181pub struct SID {
182	pub Revision: u8,
183	pub(in crate::advapi) SubAuthorityCount: u8,
184	pub IdentifierAuthority: SID_IDENTIFIER_AUTHORITY,
185	SubAuthority: [co::RID; 1],
186}
187
188impl std::fmt::Display for SID {
189	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
190		match ConvertSidToStringSid(self) {
191			Ok(name) => write!(f, "{}", name),
192			Err(err) => write!(f, "{}", err),
193		}
194	}
195}
196
197impl PartialEq<SID> for SID {
198	fn eq(&self, other: &SID) -> bool {
199		EqualSid(self, other).unwrap() // assumes valid references
200	}
201}
202
203impl Eq for SID {}
204
205impl SID {
206	/// Returns the `SubAuthorityCount` field.
207	#[must_use]

210	}
