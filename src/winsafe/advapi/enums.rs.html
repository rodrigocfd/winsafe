15/// Registry value types.
16///
17/// This is a high-level abstraction over the [`co::REG`](crate::co::REG)
18/// constants, plus the value they carry.
19#[derive(Clone, Debug)]
20pub enum RegistryValue {
21	/// Binary value, defined as [`REG::BINARY`](crate::co::REG::BINARY).
22	Binary(Vec<u8>),
23	/// An `u32` integer value, defined as [`REG::DWORD`](crate::co::REG::DWORD).
24	Dword(u32),
25	/// An `u64` integer value, defined as [`REG::QWORD`](crate::co::REG::QWORD).
26	Qword(u64),
27	/// String value, defined as [`REG::SZ`](crate::co::REG::SZ).
28	Sz(String),
29	/// String value that contains unexpanded references to environment
30	/// variables, for example, `%PATH%`. To expand the environment variable
31	/// references, use
32	/// [`ExpandEnvironmentStrings`](crate::ExpandEnvironmentStrings).
33	ExpandSz(String),
34	/// Multiple strings, defined as [`REG::MULTI_SZ`](crate::co::REG::MULTI_SZ).
35	MultiSz(Vec<String>),
36	/// No value, defined as [`REG::NONE`](crate::co::REG::NONE). Also used for
37	/// non-implemented value types.
38	None,
39}
40
41impl std::fmt::Display for RegistryValue {
42	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
43		use RegistryValue::*;
44		match self {
45			Binary(b) => write!(
46				f,
47				"[REG_BINARY] {}",
48				b.iter()
49					.map(|n| format!("{:02x}", *n))
50					.collect::<Vec<_>>()
51					.join(" "),
52			),
53			Dword(n) => write!(f, "[REG_DWORD] {}", *n),
54			Qword(n) => write!(f, "[REG_QWORD] {}", *n),
55			Sz(s) => write!(f, "[REG_SZ] \"{}\"", s),
56			ExpandSz(s) => write!(f, "[REG_EXPAND_SZ] \"{}\"", s),
57			MultiSz(v) => write!(
58				f,
59				"[REG_MULTI_SZ] {}",
60				v.iter()
61					.map(|s| format!("\"{}\"", s))
62					.collect::<Vec<_>>()
63					.join(", "),
64			),
65			None => write!(f, "[REG_NONE]"),
66		}
67	}
68}

70impl RegistryValue {
71	/// Parses a binary data block as a `RegistryValue`.
72	///
73	/// This method can be used as an escape hatch to interoperate with other
74	/// libraries.
75	///
76	/// # Safety
77	///
78	/// Whilst a few validations are made, assumes the binary data block has the
79	/// correct content, according to the informed [`co::REG`](crate::co::REG).
80	#[must_use]
81	pub unsafe fn from_raw(buf: Vec<u8>, reg_type: co::REG) -> SysResult<RegistryValue> {
82		match reg_type {
83			co::REG::BINARY => Ok(RegistryValue::Binary(buf)),
84			co::REG::DWORD => {
85				if buf.len() != std::mem::size_of::<u32>() {
86					Err(co::ERROR::INVALID_DATA) // validate buf size
87				} else {
88					Ok(RegistryValue::Dword(u32::from_ne_bytes(unsafe {
89						*std::mem::transmute::<_, *const [u8; 4]>(buf.as_ptr())
90					})))
91				}
92			},
93			co::REG::QWORD => {
94				if buf.len() != std::mem::size_of::<u64>() {
95					Err(co::ERROR::INVALID_DATA) // validate buf size
96				} else {
97					Ok(RegistryValue::Qword(u64::from_ne_bytes(unsafe {
98						*std::mem::transmute::<_, *const [u8; 8]>(buf.as_ptr())
99					})))
100				}
101			},
102			co::REG::SZ => {
103				let (_, vec16, _) = unsafe { buf.align_to::<u16>() };
104				Ok(RegistryValue::Sz(WString::from_wchars_slice(&vec16).to_string()))
105			},
106			co::REG::EXPAND_SZ => {
107				let (_, vec16, _) = unsafe { buf.align_to::<u16>() };
108				Ok(RegistryValue::ExpandSz(WString::from_wchars_slice(&vec16).to_string()))
109			},
110			co::REG::MULTI_SZ => {
111				let (_, vec16, _) = unsafe { buf.align_to::<u16>() };
112				Ok(RegistryValue::MultiSz(unsafe {
113					parse_multi_z_str(vec16.as_ptr(), Some(vec16.len()))
114				}))
115			},
116			co::REG::NONE => Ok(RegistryValue::None),
117			_ => Err(co::ERROR::CALL_NOT_IMPLEMENTED), // other types not implemented yet
118		}
119	}
120
121	/// Returns a pointer to the raw data, along with the raw data length in bytes.
122	#[must_use]
123	pub fn as_ptr_with_len(&self, str_buf: &mut WString) -> (*const std::ffi::c_void, u32) {
124		use RegistryValue::*;
125		match self {
126			Binary(b) => (b.as_ptr() as _, b.len() as _),
127			Dword(n) => (n as *const _ as _, std::mem::size_of::<u32>() as _),
128			Qword(n) => (n as *const _ as _, std::mem::size_of::<u64>() as _),
129			Sz(s) => {
130				*str_buf = WString::from_str(s); // serialize the string into the external buffer
131				let buf_sz = (str_buf.str_len() + 1) * std::mem::size_of::<u16>(); // count terminating null
132				(str_buf.as_ptr() as _, buf_sz as _)
133			},
134			ExpandSz(s) => {
135				*str_buf = WString::from_str(s); // serialize the string into the external buffer
136				let buf_sz = (str_buf.str_len() + 1) * std::mem::size_of::<u16>(); // count terminating null
137				(str_buf.as_ptr() as _, buf_sz as _)
138			},
139			MultiSz(v) => {
140				*str_buf = WString::from_str_vec(v); // serialize the string into the external buffer
141				let tot_chars = v.iter() // number of chars of all strings, including terminating nulls
142					.fold(0, |tot, s| tot + s.chars().count() + 1) // include terminating null
143					+ 1; // double terminating null
144				let buf_sz = tot_chars * std::mem::size_of::<u16>();
145				(str_buf.as_ptr() as _, buf_sz as _)
146			},
147			None => (std::ptr::null(), 0),
148		}
149	}
150
151	/// Returns the correspondent [`co::REG`](crate::co::REG) constant.
152	#[must_use]
153	pub const fn reg_type(&self) -> co::REG {
154		use RegistryValue::*;
155		match self {
156			Binary(_) => co::REG::BINARY,
157			Dword(_) => co::REG::DWORD,
158			Qword(_) => co::REG::QWORD,
159			Sz(_) => co::REG::SZ,
160			ExpandSz(_) => co::REG::EXPAND_SZ,
161			MultiSz(_) => co::REG::MULTI_SZ,
162			None => co::REG::NONE,
163		}
164	}

194impl<'a> SvcCtl<'a> {
195	/// Constructs the enum according to the raw data.
196	///
197	/// # Safety
198	///
199	/// This enum is constructed when building the output of
200	/// [`HSERVICESTATUS::RegisterServiceCtrlHandlerEx`](crate::HSERVICESTATUS::RegisterServiceCtrlHandlerEx)
201	/// callback, make sure all parameters are correct.
202	#[must_use]
203	pub unsafe fn from_raw(
204		control: u32,
205		event_type: u32,
206		event_data: *mut std::ffi::c_void,
207	) -> Self {
208		match unsafe { co::SERVICE_CONTROL::from_raw(control) } {
209			co::SERVICE_CONTROL::CONTINUE => Self::Continue,
210			co::SERVICE_CONTROL::INTERROGATE => Self::Interrogate,
211			co::SERVICE_CONTROL::NETBINDADD => Self::NetBindAdd,
212			co::SERVICE_CONTROL::NETBINDDISABLE => Self::NetBindDisable,
213			co::SERVICE_CONTROL::NETBINDENABLE => Self::NetBindEnable,
214			co::SERVICE_CONTROL::NETBINDREMOVE => Self::NetBindRemove,
215			co::SERVICE_CONTROL::PARAMCHANGE => Self::ParamChange,
216			co::SERVICE_CONTROL::PAUSE => Self::Pause,
217			co::SERVICE_CONTROL::PRESHUTDOWN => Self::PreShutdown,
218			co::SERVICE_CONTROL::SHUTDOWN => Self::Shutdown,
219			co::SERVICE_CONTROL::STOP => Self::Stop,
220
221			co::SERVICE_CONTROL::DEVICEEVENT => unsafe {
222				Self::DeviceEvent(
223					co::DBT::from_raw(event_type as _),
224					SvcCtlDeviceEvent::from_raw(&*(event_data as *const _)),
225				)
226			},
227			co::SERVICE_CONTROL::HARDWAREPROFILECHANGE => {
228				Self::HardwareProfileChange(unsafe { co::DBT::from_raw(event_type as _) })
229			},
230			co::SERVICE_CONTROL::POWEREVENT => Self::PowerEvent(unsafe {
231				SvcCtlPowerEvent::from_raw(co::PBT::from_raw(event_type), event_data)
232			}),
233			co::SERVICE_CONTROL::SESSIONCHANGE => unsafe {
234				Self::SessionChange(co::WTS::from_raw(event_type as _), &*(event_data as *const _))
235			},
236			co::SERVICE_CONTROL::TIMECHANGE => {
237				Self::TimeChange(unsafe { &*(event_data as *const _) })
238			},
239			co::SERVICE_CONTROL::TRIGGEREVENT => Self::TriggerEvent,
240			co::SERVICE_CONTROL::USERMODEREBOOT => Self::UserModeReboot,
241
242			_ => Self::UserDefined(control as _, event_type, event_data as _),
243		}
244	}
245}

250	Handle(&'a DEV_BROADCAST_HANDLE),

256impl<'a> SvcCtlDeviceEvent<'a> {
257	/// Constructs the enum according to the raw data.
258	///
259	/// # Panics
260	///
261	/// Panics if `dbch_devicetype` field is invalid.
262	///
263	/// # Safety
264	///
265	/// This enum is constructed when building the output of
266	/// [`HSERVICESTATUS::RegisterServiceCtrlHandlerEx`](crate::HSERVICESTATUS::RegisterServiceCtrlHandlerEx)
267	/// callback, make sure all parameters are correct.
268	#[must_use]
269	pub const unsafe fn from_raw(event_data: &DEV_BROADCAST_HDR) -> Self {
270		let ptr = event_data as *const DEV_BROADCAST_HDR;
271		unsafe {
272			match event_data.dbch_devicetype {
273				co::DBT_DEVTYP::DEVICEINTERFACE => Self::Interface(&*(ptr as *const _)),
274				co::DBT_DEVTYP::HANDLE => Self::Handle(&*(ptr as *const _)),
275				co::DBT_DEVTYP::OEM => Self::Oem(&*(ptr as *const _)),
276				co::DBT_DEVTYP::PORT => Self::Port(&*(ptr as *const _)),
277				co::DBT_DEVTYP::VOLUME => Self::Volume(&*(ptr as *const _)),
278				_ => panic!("Invalid co::DBT_DEVTYP."),
279			}
280		}
281	}
282}
