15/// Registry value types.
16///
17/// This is a high-level abstraction over the [`co::REG`](crate::co::REG)
18/// constants, plus the value they carry.
19#[derive(Clone, Debug)]
20pub enum RegistryValue {
21	/// Binary value, defined as [`REG::BINARY`](crate::co::REG::BINARY).
22	Binary(Vec<u8>),
23	/// An `u32` integer value, defined as [`REG::DWORD`](crate::co::REG::DWORD).
24	Dword(u32),
25	/// An `u64` integer value, defined as [`REG::QWORD`](crate::co::REG::QWORD).
26	Qword(u64),
27	/// String value, defined as [`REG::SZ`](crate::co::REG::SZ).
28	Sz(String),
29	/// String value that contains unexpanded references to environment
30	/// variables, for example, `%PATH%`. To expand the environment variable
31	/// references, use
32	/// [`ExpandEnvironmentStrings`](crate::ExpandEnvironmentStrings).
33	ExpandSz(String),
34	/// Multiple strings, defined as [`REG::MULTI_SZ`](crate::co::REG::MULTI_SZ).
35	MultiSz(Vec<String>),
36	/// No value, defined as [`REG::NONE`](crate::co::REG::NONE). Also used for
37	/// non-implemented value types.
38	None,
39}
40
41impl std::fmt::Display for RegistryValue {
42	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
43		match self {
44			Self::Binary(b) => write!(
45				f,
46				"[REG_BINARY] {}",
47				b.iter()
48					.map(|n| format!("{:02x}", *n))
49					.collect::<Vec<_>>()
50					.join(" "),
51			),
52			Self::Dword(n) => write!(f, "[REG_DWORD] {}", *n),
53			Self::Qword(n) => write!(f, "[REG_QWORD] {}", *n),
54			Self::Sz(s) => write!(f, "[REG_SZ] \"{}\"", s),
55			Self::ExpandSz(s) => write!(f, "[REG_EXPAND_SZ] \"{}\"", s),
56			Self::MultiSz(v) => write!(
57				f,
58				"[REG_MULTI_SZ] {}",
59				v.iter()
60					.map(|s| format!("\"{}\"", s))
61					.collect::<Vec<_>>()
62					.join(", "),
63			),
64			Self::None => write!(f, "[REG_NONE]"),
65		}
66	}
67}

69impl RegistryValue {
70	/// Parses a binary data block as a `RegistryValue`.
71	///
72	/// This method can be used as an escape hatch to interoperate with other
73	/// libraries.
74	///
75	/// # Safety
76	///
77	/// Whilst a few validations are made, assumes the binary data block has the
78	/// correct content, according to the informed [`co::REG`](crate::co::REG).
79	#[must_use]
80	pub unsafe fn from_raw(buf: Vec<u8>, reg_type: co::REG) -> SysResult<RegistryValue> {
81		match reg_type {
82			co::REG::BINARY => Ok(RegistryValue::Binary(buf)),
83			co::REG::DWORD => {
84				if buf.len() != std::mem::size_of::<u32>() {
85					Err(co::ERROR::INVALID_DATA) // validate buf size
86				} else {
87					Ok(RegistryValue::Dword(u32::from_ne_bytes(unsafe {
88						*std::mem::transmute::<_, *const [u8; 4]>(buf.as_ptr())
89					})))
90				}
91			},
92			co::REG::QWORD => {
93				if buf.len() != std::mem::size_of::<u64>() {
94					Err(co::ERROR::INVALID_DATA) // validate buf size
95				} else {
96					Ok(RegistryValue::Qword(u64::from_ne_bytes(unsafe {
97						*std::mem::transmute::<_, *const [u8; 8]>(buf.as_ptr())
98					})))
99				}
100			},
101			co::REG::SZ => {
102				let (_, vec16, _) = unsafe { buf.align_to::<u16>() };
103				Ok(RegistryValue::Sz(WString::from_wchars_slice(&vec16).to_string()))
104			},
105			co::REG::EXPAND_SZ => {
106				let (_, vec16, _) = unsafe { buf.align_to::<u16>() };
107				Ok(RegistryValue::ExpandSz(WString::from_wchars_slice(&vec16).to_string()))
108			},
109			co::REG::MULTI_SZ => {
110				let (_, vec16, _) = unsafe { buf.align_to::<u16>() };
111				Ok(RegistryValue::MultiSz(unsafe {
112					parse_multi_z_str(vec16.as_ptr(), Some(vec16.len()))
113				}))
114			},
115			co::REG::NONE => Ok(RegistryValue::None),
116			_ => Err(co::ERROR::CALL_NOT_IMPLEMENTED), // other types not implemented yet
117		}
118	}
119
120	/// Returns a pointer to the raw data, along with the raw data length.
121	#[must_use]
122	pub fn as_ptr_with_len(&self, str_buf: &mut WString) -> (*const std::ffi::c_void, u32) {
123		match self {
124			Self::Binary(b) => (b.as_ptr() as _, b.len() as _),
125			Self::Dword(n) => (n as *const _ as _, std::mem::size_of::<u32>() as _),
126			Self::Qword(n) => (n as *const _ as _, std::mem::size_of::<u64>() as _),
127			Self::Sz(s) => {
128				*str_buf = WString::from_str(s);
129				Self::as_ptr_with_len_str(&str_buf)
130			},
131			Self::ExpandSz(s) => {
132				*str_buf = WString::from_str(s);
133				Self::as_ptr_with_len_str(&str_buf)
134			},
135			Self::MultiSz(v) => {
136				*str_buf = WString::from_str_vec(v);
137				Self::as_ptr_with_len_str(&str_buf)
138			},
139			Self::None => (std::ptr::null(), 0),
140		}
141	}
142
143	fn as_ptr_with_len_str(str_buf: &WString) -> (*const std::ffi::c_void, u32) {
144		(
145			str_buf.as_ptr() as _,
146			(str_buf.buf_len() * std::mem::size_of::<u16>()) as _, // will include terminating null
147		)
148	}

150	/// Returns the correspondent [`co::REG`](crate::co::REG) constant.
151	#[must_use]
152	pub const fn reg_type(&self) -> co::REG {
153		match self {
154			Self::Binary(_) => co::REG::BINARY,
155			Self::Dword(_) => co::REG::DWORD,
156			Self::Qword(_) => co::REG::QWORD,
157			Self::Sz(_) => co::REG::SZ,
158			Self::ExpandSz(_) => co::REG::EXPAND_SZ,
159			Self::MultiSz(_) => co::REG::MULTI_SZ,
160			Self::None => co::REG::NONE,
161		}
162	}

192impl<'a> SvcCtl<'a> {
193	/// Constructs the enum according to the raw data.
194	///
195	/// # Safety
196	///
197	/// This enum is constructed when building the output of
198	/// [`HSERVICESTATUS::RegisterServiceCtrlHandlerEx`](crate::HSERVICESTATUS::RegisterServiceCtrlHandlerEx)
199	/// callback, make sure all parameters are correct.
200	#[must_use]
201	pub unsafe fn from_raw(
202		control: u32,
203		event_type: u32,
204		event_data: *mut std::ffi::c_void,
205	) -> Self {
206		match unsafe { co::SERVICE_CONTROL::from_raw(control) } {
207			co::SERVICE_CONTROL::CONTINUE => Self::Continue,
208			co::SERVICE_CONTROL::INTERROGATE => Self::Interrogate,
209			co::SERVICE_CONTROL::NETBINDADD => Self::NetBindAdd,
210			co::SERVICE_CONTROL::NETBINDDISABLE => Self::NetBindDisable,
211			co::SERVICE_CONTROL::NETBINDENABLE => Self::NetBindEnable,
212			co::SERVICE_CONTROL::NETBINDREMOVE => Self::NetBindRemove,
213			co::SERVICE_CONTROL::PARAMCHANGE => Self::ParamChange,
214			co::SERVICE_CONTROL::PAUSE => Self::Pause,
215			co::SERVICE_CONTROL::PRESHUTDOWN => Self::PreShutdown,
216			co::SERVICE_CONTROL::SHUTDOWN => Self::Shutdown,
217			co::SERVICE_CONTROL::STOP => Self::Stop,
218
219			co::SERVICE_CONTROL::DEVICEEVENT => unsafe {
220				Self::DeviceEvent(
221					co::DBT::from_raw(event_type as _),
222					SvcCtlDeviceEvent::from_raw(&*(event_data as *const _)),
223				)
224			},
225			co::SERVICE_CONTROL::HARDWAREPROFILECHANGE => {
226				Self::HardwareProfileChange(unsafe { co::DBT::from_raw(event_type as _) })
227			},
228			co::SERVICE_CONTROL::POWEREVENT => Self::PowerEvent(unsafe {
229				SvcCtlPowerEvent::from_raw(co::PBT::from_raw(event_type), event_data)
230			}),
231			co::SERVICE_CONTROL::SESSIONCHANGE => unsafe {
232				Self::SessionChange(co::WTS::from_raw(event_type as _), &*(event_data as *const _))
233			},
234			co::SERVICE_CONTROL::TIMECHANGE => {
235				Self::TimeChange(unsafe { &*(event_data as *const _) })
236			},
237			co::SERVICE_CONTROL::TRIGGEREVENT => Self::TriggerEvent,
238			co::SERVICE_CONTROL::USERMODEREBOOT => Self::UserModeReboot,
239
240			_ => Self::UserDefined(control as _, event_type, event_data as _),
241		}
242	}
243}

254impl<'a> SvcCtlDeviceEvent<'a> {
255	/// Constructs the enum according to the raw data.
256	///
257	/// # Panics
258	///
259	/// Panics if `dbch_devicetype` field is invalid.
260	///
261	/// # Safety
262	///
263	/// This enum is constructed when building the output of
264	/// [`HSERVICESTATUS::RegisterServiceCtrlHandlerEx`](crate::HSERVICESTATUS::RegisterServiceCtrlHandlerEx)
265	/// callback, make sure all parameters are correct.
266	#[must_use]
267	pub const unsafe fn from_raw(event_data: &DEV_BROADCAST_HDR) -> Self {
268		let ptr = event_data as *const DEV_BROADCAST_HDR;
269		unsafe {
270			match event_data.dbch_devicetype {
271				co::DBT_DEVTYP::DEVICEINTERFACE => Self::Interface(&*(ptr as *const _)),
272				co::DBT_DEVTYP::HANDLE => Self::Handle(&*(ptr as *const _)),
273				co::DBT_DEVTYP::OEM => Self::Oem(&*(ptr as *const _)),
274				co::DBT_DEVTYP::PORT => Self::Port(&*(ptr as *const _)),
275				co::DBT_DEVTYP::VOLUME => Self::Volume(&*(ptr as *const _)),
276				_ => panic!("Invalid co::DBT_DEVTYP."),
277			}
278		}
279	}
280}
