18/// This trait is enabled with the `advapi` feature, and provides methods for
19/// [`HTRANSACTION`](crate::HTRANSACTION).
20///
21/// Prefer importing this trait through the prelude:
22///
23/// ```no_run
24/// use winsafe::prelude::*;
25/// ```
26pub trait advapi_Htransaction: Handle {
27	/// [`CommitTransaction`](https://learn.microsoft.com/en-us/windows/win32/api/ktmw32/nf-ktmw32-committransaction)
28	/// function.
29	fn CommitTransaction(&self) -> SysResult<()> {
30		bool_to_sysresult(unsafe { ffi::CommitTransaction(self.ptr()) })
31	}
32
33	/// [`CreateTransaction`](https://learn.microsoft.com/en-us/windows/win32/api/ktmw32/nf-ktmw32-createtransaction)
34	/// function.
35	#[must_use]
36	fn CreateTransaction(
37		transaction_attributes: Option<&SECURITY_ATTRIBUTES>,
38		options: Option<co::TRANSACTION_OPT>,
39		timeout: Option<u32>,
40		description: &str,
41	) -> SysResult<CloseHandleGuard<HTRANSACTION>> {
42		unsafe {
43			match HTRANSACTION(ffi::CreateTransaction(
44				pcvoid_or_null(transaction_attributes),
45				std::ptr::null_mut(),
46				options.unwrap_or_default().raw(),
47				0,
48				0,
49				timeout.unwrap_or_default(),
50				WString::from_str(description).as_ptr() as _,
51			)) {
52				HTRANSACTION::INVALID => Err(GetLastError()),
53				handle => Ok(CloseHandleGuard::new(handle)),
54			}
55		}
56	}
57
58	/// [`GetTransactionId`](https://learn.microsoft.com/en-us/windows/win32/api/ktmw32/nf-ktmw32-gettransactionid)
59	/// function.
60	#[must_use]
61	fn GetTransactionId(&self) -> SysResult<GUID> {
62		let mut guid = GUID::default();
63		bool_to_sysresult(unsafe { ffi::GetTransactionId(self.ptr(), pvoid(&mut guid)) })
64			.map(|_| guid)
65	}
66
67	/// [`OpenTransaction`](https://learn.microsoft.com/en-us/windows/win32/api/ktmw32/nf-ktmw32-opentransaction)
68	/// function.
69	#[must_use]
70	fn OpenTransaction(
71		desired_access: co::TRANSACTION,
72		transaction_id: &GUID,
73	) -> SysResult<CloseHandleGuard<HTRANSACTION>> {
74		unsafe {
75			match HTRANSACTION(ffi::OpenTransaction(desired_access.raw(), pcvoid(transaction_id))) {
76				HTRANSACTION::INVALID => Err(GetLastError()),
77				handle => Ok(CloseHandleGuard::new(handle)),
78			}
79		}
80	}
81
82	/// [`RollbackTransaction`](https://learn.microsoft.com/en-us/windows/win32/api/ktmw32/nf-ktmw32-rollbacktransaction)
83	/// function.
84	fn RollbackTransaction(&self) -> SysResult<()> {
85		bool_to_sysresult(unsafe { ffi::RollbackTransaction(self.ptr()) })
86	}
87}
