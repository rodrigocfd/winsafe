15impl HSC {
16	/// [`CreateService`](https://learn.microsoft.com/en-us/windows/win32/api/winsvc/nf-winsvc-createservicew)
17	/// function.
18	#[must_use]
19	pub fn CreateService(
20		&self,
21		service_name: &str,
22		display_name: Option<&str>,
23		desired_access: co::SERVICE,
24		service_type: co::SERVICE_TYPE,
25		start_type: co::SERVICE_START,
26		error_control: co::SERVICE_ERROR,
27		binary_path_name: Option<&str>,
28		load_order_group: Option<&str>,
29		tag_id: Option<&mut u32>,
30		dependencies: &[impl AsRef<str>],
31		service_start_name: Option<&str>,
32		password: Option<&str>,
33	) -> SysResult<CloseServiceHandleSvcGuard> {
34		let binary_path_name_quoted = binary_path_name
35			.map(|s| if s.starts_with('"') { s.to_owned() } else { format!("\"{}\"", s) });
36
37		unsafe {
38			PtrRet(ffi::CreateServiceW(
39				self.ptr(),
40				WString::from_str(service_name).as_ptr(),
41				WString::from_opt_str(display_name).as_ptr(),
42				desired_access.raw(),
43				service_type.raw(),
44				start_type.raw(),
45				error_control.raw(),
46				WString::from_opt_str(binary_path_name_quoted).as_ptr(),
47				WString::from_opt_str(load_order_group).as_ptr(),
48				tag_id.map_or(std::ptr::null_mut(), |n| n),
49				WString::from_str_vec(dependencies).as_ptr(),
50				WString::from_opt_str(service_start_name).as_ptr(),
51				WString::from_opt_str(password).as_ptr(),
52			))
53			.to_sysresult_handle()
54			.map(|h| CloseServiceHandleSvcGuard::new(h))
55		}
56	}
57
