18/// This trait is enabled with the `advapi` feature, and provides methods for
19/// [`HEVENTLOG`](crate::HEVENTLOG).
20///
21/// Prefer importing this trait through the prelude:
22///
23/// ```no_run
24/// use winsafe::prelude::*;
25/// ```
26pub trait advapi_Heventlog: Handle {
27	/// [`RegisterEventSource`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-registereventsourcew)
28	/// function.
29	#[must_use]
30	fn RegisterEventSource(
31		unc_server_name: Option<&str>,
32		source_name: &str,
33	) -> SysResult<DeregisterEventSourceGuard> {
34		unsafe {
35			ptr_to_sysresult_handle(ffi::RegisterEventSourceW(
36				WString::from_opt_str(unc_server_name).as_ptr(),
37				WString::from_str(source_name).as_ptr(),
38			))
39			.map(|h| DeregisterEventSourceGuard::new(h))
40		}
41	}
42
43	/// [`ReportEvent`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-reporteventw)
44	/// function.
45	fn ReportEvent(
46		&self,
47		event_type: co::EVENTLOG,
48		category: u16,
49		event_id: u32,
50		user_sid: Option<&SID>,
51		strings: Option<&[impl AsRef<str>]>,
52		raw_data: Option<&[u8]>,
53	) -> SysResult<()> {
54		let (_wstrs, pwstrs) = create_wstr_ptr_vecs(strings);
55		bool_to_sysresult(unsafe {
56			ffi::ReportEventW(
57				self.ptr(),
58				event_type.raw(),
59				category,
60				event_id,
61				pcvoid_or_null(user_sid),
62				strings.map_or(0, |ss| ss.len() as _),
63				raw_data.map_or(0, |d| d.len() as _),
64				vec_ptr(&pwstrs),
65				raw_data.map_or(std::ptr::null(), |d| vec_ptr(d) as _),
66			)
67		})
68	}
69}
