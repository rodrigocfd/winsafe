27	/// [`AdjustTokenPrivileges`](https://learn.microsoft.com/en-us/windows/win32/api/securitybaseapi/nf-securitybaseapi-adjusttokenprivileges)
28	/// function.
29	///
30	/// # Examples
31	///
32	/// ```no_run
33	/// use winsafe::{self as w, prelude::*, co};
34	///
35	/// let htoken = w::HPROCESS::GetCurrentProcess()
36	///     .OpenProcessToken(co::TOKEN::ADJUST_PRIVILEGES | co::TOKEN::QUERY)?;
37	///
38	/// let luid = w::LookupPrivilegeValue(None, co::SE_PRIV::SHUTDOWN_NAME)?;
39	///
40	/// let privs = w::TOKEN_PRIVILEGES::new(&[
41	///     w::LUID_AND_ATTRIBUTES::new(luid, co::SE_PRIV_ATTR::ENABLED),
42	/// ])?;
43	///
44	/// htoken.AdjustTokenPrivileges(w::DisabPriv::Privs(&privs))?;
45	/// # w::SysResult::Ok(())
46	/// ```
47	fn AdjustTokenPrivileges(&self, new_state: DisabPriv) -> SysResult<()> {
48		bool_to_sysresult(
49			unsafe {
50				ffi::AdjustTokenPrivileges(
51					self.ptr(),
52					match new_state {
53						DisabPriv::Disab => 1,
54						_ => 0,
55					},
56					match new_state {
57						DisabPriv::Privs(privs) => privs as *const _ as _,
58						_ => std::ptr::null(),
59					},
60					0,
61					std::ptr::null_mut(),
62					std::ptr::null_mut(),
63				)
64			},
65		)
66	}

142	/// [`GetTokenInformation`](https://learn.microsoft.com/en-us/windows/win32/api/securitybaseapi/nf-securitybaseapi-gettokeninformation)
143	/// function.
144	///
145	/// The returned enum variant will correspond to the passed
146	/// `information_class`.
147	///
148	/// # Examples
149	///
150	/// Retrieving the `Groups` information:
151	///
152	/// ```no_run
153	/// use winsafe::{self as w, prelude::*, co};
154	///
155	/// let htoken = w::HPROCESS::GetCurrentProcess()
156	///     .OpenProcessToken(co::TOKEN::QUERY)?;
157	///
158	/// let nfo = htoken.GetTokenInformation(co::TOKEN_INFORMATION_CLASS::Groups)?;
159	/// let w::TokenInfo::Groups(groups) = nfo else { panic!("never") };
160	///
161	/// for (idx, g) in groups.Groups().iter().enumerate() {
162	///     println!("{}: {}", idx, g.Sid().unwrap());
163	/// }
164	/// # w::SysResult::Ok(())
165	/// ```
166	#[must_use]
167	fn GetTokenInformation(&self,
168		information_class: co::TOKEN_INFORMATION_CLASS,
169	) -> SysResult<TokenInfo>
170	{
171		let mut num_bytes = u32::default();
172		match bool_to_sysresult(
173			unsafe {
174				ffi::GetTokenInformation(
175					self.ptr(),
176					information_class.raw(),
177					std::ptr::null_mut(),
178					0,
179					&mut num_bytes,
180				)
181			},
182		) {
183			Err(err) => match err {
184				co::ERROR::INSUFFICIENT_BUFFER => {}, // all good
185				err => return Err(err),
186			},
187			Ok(_) => return Err(co::ERROR::INVALID_PARAMETER), // should never happen
188		};
189
190		let mut buf = vec![0u8; num_bytes as usize].into_boxed_slice();
191
192		unsafe {
193			bool_to_sysresult(
194				ffi::GetTokenInformation(
195					self.ptr(),
196					information_class.raw(),
197					buf.as_mut_ptr() as _,
198					num_bytes,
199					&mut num_bytes,
200				),
201			).map(|_| {
202				match information_class {
203					co::TOKEN_INFORMATION_CLASS::User => TokenInfo::User(
204						Box::from_raw(Box::into_raw(buf) as *mut TOKEN_USER),
205					),
206					co::TOKEN_INFORMATION_CLASS::Groups => TokenInfo::Groups(
207						Box::from_raw(Box::into_raw(buf) as *mut TOKEN_GROUPS),
208					),

210						Box::from_raw(Box::into_raw(buf) as *mut TOKEN_PRIVILEGES),
211					),
212					co::TOKEN_INFORMATION_CLASS::Owner => TokenInfo::Owner(
213						Box::from_raw(Box::into_raw(buf) as *mut TOKEN_OWNER),
214					),
215					co::TOKEN_INFORMATION_CLASS::PrimaryGroup => TokenInfo::PrimaryGroup(

217					),

313	#[must_use]
314	fn IsTokenRestricted(&self) -> SysResult<bool> {
315		match unsafe { ffi::IsTokenRestricted(self.ptr()) } {
316			0 => match GetLastError() {
