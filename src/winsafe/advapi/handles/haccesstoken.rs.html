17	/// [`AdjustTokenPrivileges`](https://learn.microsoft.com/en-us/windows/win32/api/securitybaseapi/nf-securitybaseapi-adjusttokenprivileges)
18	/// function.
19	///
20	/// # Examples
21	///
22	/// ```no_run
23	/// use winsafe::{self as w, prelude::*, co};
24	///
25	/// let htoken = w::HPROCESS::GetCurrentProcess()
26	///     .OpenProcessToken(co::TOKEN::ADJUST_PRIVILEGES | co::TOKEN::QUERY)?;
27	///
28	/// let luid = w::LookupPrivilegeValue(None, co::SE_PRIV::SHUTDOWN_NAME)?;
29	///
30	/// let privs = w::TOKEN_PRIVILEGES::new(&[
31	///     w::LUID_AND_ATTRIBUTES::new(luid, co::SE_PRIV_ATTR::ENABLED),
32	/// ])?;
33	///
34	/// htoken.AdjustTokenPrivileges(w::DisabPriv::Privs(&privs))?;
35	/// # w::SysResult::Ok(())
36	/// ```
37	pub fn AdjustTokenPrivileges(&self, new_state: DisabPriv) -> SysResult<()> {
38		bool_to_sysresult(unsafe {
39			ffi::AdjustTokenPrivileges(
40				self.ptr(),
41				match new_state {
42					DisabPriv::Disab => 1,
43					_ => 0,
44				},
45				match new_state {
46					DisabPriv::Privs(privs) => pcvoid(privs),
47					_ => std::ptr::null(),
48				},
49				0,
50				std::ptr::null_mut(),
51				std::ptr::null_mut(),
52			)
53		})
54	}

116	/// [`GetTokenInformation`](https://learn.microsoft.com/en-us/windows/win32/api/securitybaseapi/nf-securitybaseapi-gettokeninformation)
117	/// function.
118	///
119	/// The returned enum variant will correspond to the passed
120	/// `information_class`.
121	///
122	/// # Examples
123	///
124	/// Retrieving the `Groups` information:
125	///
126	/// ```no_run
127	/// use winsafe::{self as w, prelude::*, co};
128	///
129	/// let htoken = w::HPROCESS::GetCurrentProcess()
130	///     .OpenProcessToken(co::TOKEN::QUERY)?;
131	///
132	/// let nfo = htoken.GetTokenInformation(co::TOKEN_INFORMATION_CLASS::Groups)?;
133	/// let w::TokenInfo::Groups(groups) = nfo else { unreachable!() };
134	///
135	/// for (idx, g) in groups.Groups().iter().enumerate() {
136	///     println!("{}: {}", idx, g.Sid().unwrap());
137	/// }
138	/// # w::SysResult::Ok(())
139	/// ```
140	#[must_use]
141	pub fn GetTokenInformation(
142		&self,
143		information_class: co::TOKEN_INFORMATION_CLASS,
144	) -> SysResult<TokenInfo> {
145		let mut num_bytes = 0u32;
146		match bool_to_sysresult(unsafe {
147			ffi::GetTokenInformation(
148				self.ptr(),
149				information_class.raw(),
150				std::ptr::null_mut(),
151				0,
152				&mut num_bytes,
153			)
154		}) {
155			Err(err) => match err {
156				co::ERROR::INSUFFICIENT_BUFFER => {}, // all good
157				err => return Err(err),
158			},
159			Ok(_) => return Err(co::ERROR::INVALID_PARAMETER), // should never happen
160		};
161
162		let mut buf = vec![0u8; num_bytes as usize].into_boxed_slice();
163
164		unsafe {
165			bool_to_sysresult(ffi::GetTokenInformation(
166				self.ptr(),
167				information_class.raw(),
168				buf.as_mut_ptr() as _,
169				num_bytes,
170				&mut num_bytes,
171			))
172			.map(|_| match information_class {
173				co::TOKEN_INFORMATION_CLASS::User => {
174					TokenInfo::User(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_USER))
175				},
176				co::TOKEN_INFORMATION_CLASS::Groups => {
177					TokenInfo::Groups(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_GROUPS))
178				},
179				co::TOKEN_INFORMATION_CLASS::Privileges => TokenInfo::Privileges(Box::from_raw(
180					Box::into_raw(buf) as *mut TOKEN_PRIVILEGES,
181				)),
182				co::TOKEN_INFORMATION_CLASS::Owner => {
183					TokenInfo::Owner(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_OWNER))
184				},
185				co::TOKEN_INFORMATION_CLASS::PrimaryGroup => TokenInfo::PrimaryGroup(
186					Box::from_raw(Box::into_raw(buf) as *mut TOKEN_PRIMARY_GROUP),
187				),
188				co::TOKEN_INFORMATION_CLASS::DefaultDacl => TokenInfo::DefaultDacl(Box::from_raw(
189					Box::into_raw(buf) as *mut TOKEN_DEFAULT_DACL,
190				)),
191				co::TOKEN_INFORMATION_CLASS::Source => {
192					TokenInfo::Source(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_SOURCE))
193				},
194				co::TOKEN_INFORMATION_CLASS::Type => {
195					TokenInfo::Type(Box::from_raw(Box::into_raw(buf) as *mut co::TOKEN_TYPE))
196				},
197				co::TOKEN_INFORMATION_CLASS::ImpersonationLevel => TokenInfo::ImpersonationLevel(
198					Box::from_raw(Box::into_raw(buf) as *mut co::SECURITY_IMPERSONATION),
199				),
200				co::TOKEN_INFORMATION_CLASS::Statistics => TokenInfo::Statistics(Box::from_raw(
201					Box::into_raw(buf) as *mut TOKEN_STATISTICS,
202				)),
203				co::TOKEN_INFORMATION_CLASS::RestrictedSids => TokenInfo::RestrictedSids(
204					Box::from_raw(Box::into_raw(buf) as *mut TOKEN_GROUPS),
205				),
206				co::TOKEN_INFORMATION_CLASS::SessionId => {
207					TokenInfo::SessionId(Box::from_raw(Box::into_raw(buf) as *mut u32))
208				},
209				co::TOKEN_INFORMATION_CLASS::GroupsAndPrivileges => TokenInfo::GroupsAndPrivileges(
210					Box::from_raw(Box::into_raw(buf) as *mut TOKEN_GROUPS_AND_PRIVILEGES),
211				),
212				co::TOKEN_INFORMATION_CLASS::SandBoxInert => {
213					TokenInfo::SandBoxInert(Box::from_raw(Box::into_raw(buf) as *mut u32))
214				},
215				co::TOKEN_INFORMATION_CLASS::Origin => {
216					TokenInfo::Origin(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_ORIGIN))
217				},
218				co::TOKEN_INFORMATION_CLASS::ElevationType => TokenInfo::ElevationType(
219					Box::from_raw(Box::into_raw(buf) as *mut co::TOKEN_ELEVATION_TYPE),
220				),
221				co::TOKEN_INFORMATION_CLASS::LinkedToken => TokenInfo::LinkedToken(Box::from_raw(
222					Box::into_raw(buf) as *mut TOKEN_LINKED_TOKEN,
223				)),
224				co::TOKEN_INFORMATION_CLASS::Elevation => {
225					TokenInfo::Elevation(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_ELEVATION))
226				},
227				co::TOKEN_INFORMATION_CLASS::HasRestrictions => {
228					TokenInfo::HasRestrictions(Box::from_raw(Box::into_raw(buf) as *mut u32))
229				},
230				co::TOKEN_INFORMATION_CLASS::AccessInformation => TokenInfo::AccessInformation(
231					Box::from_raw(Box::into_raw(buf) as *mut TOKEN_ACCESS_INFORMATION),
232				),
233				co::TOKEN_INFORMATION_CLASS::VirtualizationAllowed => {
234					TokenInfo::VirtualizationAllowed(Box::from_raw(Box::into_raw(buf) as *mut u32))
235				},
236				co::TOKEN_INFORMATION_CLASS::VirtualizationEnabled => {
237					TokenInfo::VirtualizationEnabled(Box::from_raw(Box::into_raw(buf) as *mut u32))
238				},
239				co::TOKEN_INFORMATION_CLASS::IntegrityLevel => TokenInfo::IntegrityLevel(
240					Box::from_raw(Box::into_raw(buf) as *mut TOKEN_MANDATORY_LABEL),
241				),
242				co::TOKEN_INFORMATION_CLASS::UIAccess => {
243					TokenInfo::UIAccess(Box::from_raw(Box::into_raw(buf) as *mut u32))
244				},
245				co::TOKEN_INFORMATION_CLASS::MandatoryPolicy => TokenInfo::MandatoryPolicy(
246					Box::from_raw(Box::into_raw(buf) as *mut TOKEN_MANDATORY_POLICY),
247				),
248				co::TOKEN_INFORMATION_CLASS::LogonSid => {
249					TokenInfo::LogonSid(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_GROUPS))
250				},
251				co::TOKEN_INFORMATION_CLASS::IsAppContainer => {
252					TokenInfo::IsAppContainer(Box::from_raw(Box::into_raw(buf) as *mut u32))
253				},
254				co::TOKEN_INFORMATION_CLASS::Capabilities => {
255					TokenInfo::Capabilities(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_GROUPS))

257				co::TOKEN_INFORMATION_CLASS::AppContainerNumber => {
258					TokenInfo::AppContainerNumber(Box::from_raw(Box::into_raw(buf) as *mut u32))
259				},
260				co::TOKEN_INFORMATION_CLASS::DeviceClaimAttributes => {
261					TokenInfo::DeviceClaimAttributes(Box::from_raw(
262						Box::into_raw(buf) as *mut CLAIM_SECURITY_ATTRIBUTES_INFORMATION
263					))
264				},
265				co::TOKEN_INFORMATION_CLASS::DeviceGroups => {
266					TokenInfo::DeviceGroups(Box::from_raw(Box::into_raw(buf) as *mut TOKEN_GROUPS))
267				},
268				co::TOKEN_INFORMATION_CLASS::RestrictedDeviceGroups => {
269					TokenInfo::RestrictedDeviceGroups(Box::from_raw(
270						Box::into_raw(buf) as *mut TOKEN_GROUPS
271					))
272				},
273				_ => {
274					panic!("co::TOKEN_INFORMATION_CLASS not implemented yet: {}", information_class)
275				},
276			})
277		}
278	}
