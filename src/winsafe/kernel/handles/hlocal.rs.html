3use crate::co;

8handle! { HLOCAL;
9	/// Handle to a
10	/// [local memory block](https://learn.microsoft.com/en-us/windows/win32/winprog/windows-data-types#hlocal).
11	///
12	/// The allocated memory block is accessible through the
13	/// [`LocalLock`](crate::HLOCAL::LocalLock) method.
14}
15
16impl HLOCAL {
17	/// [`LocalAlloc`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-localalloc)
18	/// function.
19	#[must_use]
20	pub fn LocalAlloc(flags: Option<co::LMEM>, num_bytes: usize) -> SysResult<LocalFreeGuard> {
21		unsafe {
22			PtrRet(ffi::LocalAlloc(flags.unwrap_or_default().raw(), num_bytes))
23				.to_sysresult_handle()
24				.map(|h| LocalFreeGuard::new(h))
25		}
26	}
27
28	/// [`LocalFlags`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-localflags)
29	/// function.
30	#[must_use]
31	pub fn LocalFlags(&self) -> SysResult<co::LMEM> {
32		match unsafe { ffi::LocalFlags(self.ptr()) } {
33			LMEM_INVALID_HANDLE => Err(GetLastError()),
34			flags => Ok(unsafe { co::LMEM::from_raw(flags) }),
35		}
36	}
37
38	/// [`LocalLock`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-locallock)
39	/// function.
40	///
41	/// Calls [`LocalSize`](crate::HLOCAL::LocalSize) to retrieve the size of
42	/// the memory block.
43	///
44	/// # Examples
45	///
46	/// ```no_run
47	/// use winsafe::{self as w, prelude::*, co};
48	///
49	/// let hlocal = w::HLOCAL::LocalAlloc(
50	///     Some(co::LMEM::FIXED | co::LMEM::ZEROINIT),
51	///     120,
52	/// )?;
53	///
54	/// let mut block = hlocal.LocalLock()?;
55	///
56	/// block.as_mut_slice()[0] = 40;
57	///
58	/// // LocalUnlock() called automatically
59	///
60	/// // LocalFree() called automatically
61	/// # w::SysResult::Ok(())
62	/// ```
63	#[must_use]
64	pub fn LocalLock(&self) -> SysResult<LocalUnlockGuard<'_>> {
65		let mem_sz = self.LocalSize()?;
66		unsafe {
67			PtrRet(ffi::LocalLock(self.ptr()))
68				.to_sysresult()
69				.map(|ptr| LocalUnlockGuard::new(self, ptr, mem_sz))
70		}
71	}
72
73	/// [`LocalReAlloc`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-localrealloc)
74	/// function.
75	///
76	/// Originally this method returns the handle to the reallocated memory
77	/// object; here the original handle is automatically updated.
78	pub fn LocalReAlloc(&mut self, num_bytes: usize, flags: Option<co::LMEM>) -> SysResult<()> {
79		PtrRet(unsafe { ffi::LocalReAlloc(self.ptr(), num_bytes, flags.unwrap_or_default().raw()) })
80			.to_sysresult_handle()
81			.map(|h| {
82				*self = h;
83			})
84	}
85
86	/// [`LocalSize`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-localsize)
87	/// function.
88	#[must_use]
89	pub fn LocalSize(&self) -> SysResult<usize> {
90		match unsafe { ffi::LocalSize(self.ptr()) } {
91			0 => Err(GetLastError()),
92			sz => Ok(sz),
93		}
94	}
