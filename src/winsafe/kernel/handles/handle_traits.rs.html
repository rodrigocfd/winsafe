3use std::{fmt, hash};
4
5/// A native
6/// [handle](https://learn.microsoft.com/en-us/windows/win32/sysinfo/handles-and-objects),
7/// implemented by all handle types.
8///
9/// Prefer importing this trait through the prelude:
10///
11/// ```no_run
12/// use winsafe::prelude::*;
13/// ```
14pub trait Handle: Sized
15	+ PartialEq + Eq + Send + hash::Hash
16	+ fmt::Debug + fmt::Display
17	+ fmt::LowerHex + fmt::UpperHex
18{
19	/// The null, uninitialized handle; equals to `0`.
20	const NULL: Self;
21
22	/// The invalid handle; equals to `-1`.
23	///
24	/// Operations upon this handle will fail with
25	/// [`ERROR::INVALID_HANDLE`](crate::co::ERROR::INVALID_HANDLE) error code.
26	const INVALID: Self;
27
28	/// Creates a new handle object by wrapping a pointer.
29	///
30	/// This method can be used as an escape hatch to interoperate with other
31	/// libraries.
32	///
33	/// # Safety
34	///
35	/// Be sure the pointer has the correct type and isn't owned by anyone else,
36	/// otherwise you may cause memory access violations.
37	#[must_use]
38	unsafe fn from_ptr(p: *mut std::ffi::c_void) -> Self;
39
40	/// Returns a raw copy of the underlying handle pointer.
41	///
42	/// # Safety
43	///
44	/// As the name implies, `raw_copy` returns a raw copy of the handle, so
45	/// closing one of the copies won't close the others. This means a handle
46	/// can be used after it has been closed, what can lead to errors and
47	/// undefined behavior. Even worse: sometimes Windows reuses handle values,
48	/// so you can call a method on a completely different handle type, what can
49	/// be catastrophic.
50	///
51	/// However, in some cases the Windows API *demands* a copy of the handle â€“
52	/// `raw_copy` is an escape hatch to fill this gap.
53	#[must_use]
54	unsafe fn raw_copy(&self) -> Self {
55		Self::from_ptr(self.ptr())
56	}
57
58	/// Returns a mutable reference to the underlying raw pointer.
59	///
60	/// This method can be used as an escape hatch to interoperate with other
61	/// libraries.
62	///
63	/// # Safety
64	///
65	/// This method exposes the raw pointer used by raw Windows calls. It's an
66	/// opaque pointer to an internal Windows structure, and no dereferencings
67	/// should be attempted.
68	#[must_use]
69	unsafe fn as_mut(&mut self) -> &mut *mut std::ffi::c_void;
70
71	/// Returns the underlying raw pointer.
72	///
73	/// This method exposes the raw pointer used by raw Windows calls. It's an
74	/// opaque pointer to an internal Windows structure, and no dereferencings
75	/// should be attempted.
76	///
77	/// This method can be used as an escape hatch to interoperate with other
78	/// libraries.
79	#[must_use]
80	fn ptr(&self) -> *mut std::ffi::c_void;
81
82	/// Returns `None` if the handle is null or invalid, otherwise returns
83	/// `Some(&self)`.
84	///
85	/// # Examples
86	///
87	/// ```no_run
88	/// use winsafe::{self as w, prelude::*};
89	///
90	/// let hfile = w::HFILE::NULL;
91	///
92	/// match hfile.as_opt() {
93	///     Some(hfile) => println!("Never prints"),
94	///     None => println!("The handle is null"),
95	/// }
96	/// ```
97	#[must_use]
98	fn as_opt(&self) -> Option<&Self> {
99		if *self == Self::NULL || *self == Self::INVALID {
100			None
101		} else {
102			Some(self)
103		}
104	}
105}
