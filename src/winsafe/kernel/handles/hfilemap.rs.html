20/// This trait is enabled with the `kernel` feature, and provides methods for
21/// [`HFILEMAP`](crate::HFILEMAP).
22///
23/// Prefer importing this trait through the prelude:
24///
25/// ```no_run
26/// use winsafe::prelude::*;
27/// ```
28pub trait kernel_Hfilemap: Handle {
29	/// [`MapViewOfFile`](https://learn.microsoft.com/en-us/windows/win32/api/memoryapi/nf-memoryapi-mapviewoffile)
30	/// function.
31	#[must_use]
32	fn MapViewOfFile(
33		&self,
34		desired_access: co::FILE_MAP,
35		offset: u64,
36		number_of_bytes_to_map: Option<usize>,
37	) -> SysResult<UnmapViewOfFileGuard> {
38		unsafe {
39			ptr_to_sysresult_handle(ffi::MapViewOfFileFromApp(
40				self.ptr(),
41				desired_access.raw(),
42				offset,
43				number_of_bytes_to_map.unwrap_or_default(),
44			))
45			.map(|h| UnmapViewOfFileGuard::new(h))
46		}
47	}
48}
