18/// This trait is enabled with the `kernel` feature, and provides methods for
19/// [`HFILEMAPVIEW`](crate::HFILEMAPVIEW).
20///
21/// Prefer importing this trait through the prelude:
22///
23/// ```no_run
24/// use winsafe::prelude::*;
25/// ```
26pub trait kernel_Hfilemapview: Handle {
27	/// Returns a slice representing the mapped memory. You can modify the
28	/// contents. You should call this method only if the file has write access.
29	///
30	/// **Note**: If the file is resized to a smaller size, the slice will still
31	/// map the bytes beyond the file. This may cause serious errors. So, if the
32	/// file is resized, re-generate the slice by calling `as_slice` again.
33	#[must_use]
34	fn as_mut_slice(&self, len: usize) -> &mut [u8] {
35		unsafe { std::slice::from_raw_parts_mut(self.ptr() as _, len) }
36	}
37
38	/// Returns a slice representing the mapped memory.
39	///
40	/// **Note**: If the file is resized to a smaller size, the slice will still
41	/// map the bytes beyond the file. This may cause serious errors. So, if the
42	/// file is resized, re-generate the slice by calling `as_slice` again.
43	///
44	/// # Examples
45	///
46	/// Reading the contents of a file into a string:
47	///
48	/// ```no_run
49	/// use winsafe::{self as w, prelude::*, co};
50	///
51	/// let (hfile, _) = w::HFILE::CreateFile(
52	///     "C:\\Temp\\test.txt",
53	///     co::GENERIC::READ,
54	///     Some(co::FILE_SHARE::READ),
55	///     None,
56	///     co::DISPOSITION::OPEN_EXISTING,
57	///     co::FILE_ATTRIBUTE::NORMAL,
58	///     None,
59	///     None,
60	///     None,
61	/// )?;
62	///
63	/// let hmap = hfile.CreateFileMapping(
64	///     None,
65	///     co::PAGE::READONLY,
66	///     None,
67	///     None,
68	/// )?;
69	///
70	/// let view = hmap.MapViewOfFile(co::FILE_MAP::READ, 0, None)?;
71	///
72	/// let slice = view.as_slice(hfile.GetFileSizeEx()? as _);
73	/// let text = std::str::from_utf8(slice)?;
74	///
75	/// println!("{}", text);
76	/// # w::AnyResult::Ok(())
77	/// ```
78	#[must_use]
79	fn as_slice(&self, len: usize) -> &[u8] {
80		unsafe { std::slice::from_raw_parts(self.ptr() as _, len) }
81	}
82
83	/// [`FlushViewOfFile`](https://learn.microsoft.com/en-us/windows/win32/api/memoryapi/nf-memoryapi-flushviewoffile)
84	/// function.
85	fn FlushViewOfFile(&self, start_at_byte: usize, num_bytes: usize) -> SysResult<()> {
86		bool_to_sysresult(unsafe {
87			ffi::FlushViewOfFile(self.ptr().offset(start_at_byte as _), num_bytes)
88		})
89	}
