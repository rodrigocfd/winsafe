16	/// Returns a slice representing the mapped memory. You can modify the
17	/// contents. You should call this method only if the file has write access.
18	///
19	/// **Note**: If the file is resized to a smaller size, the slice will still
20	/// map the bytes beyond the file. This may cause serious errors. So, if the
21	/// file is resized, re-generate the slice by calling `as_slice` again.
22	#[must_use]
23	pub fn as_mut_slice(&self, len: usize) -> &mut [u8] {
24		unsafe { std::slice::from_raw_parts_mut(self.ptr() as _, len) }
25	}

27	/// Returns a slice representing the mapped memory.
28	///
29	/// **Note**: If the file is resized to a smaller size, the slice will still
30	/// map the bytes beyond the file. This may cause serious errors. So, if the
31	/// file is resized, re-generate the slice by calling `as_slice` again.
32	///
33	/// # Examples
34	///
35	/// Reading the contents of a file into a string:
36	///
37	/// ```no_run
38	/// use winsafe::{self as w, prelude::*, co};
39	///
40	/// let (hfile, _) = w::HFILE::CreateFile(
41	///     "C:\\Temp\\test.txt",
42	///     co::GENERIC::READ,
43	///     Some(co::FILE_SHARE::READ),
44	///     None,
45	///     co::DISPOSITION::OPEN_EXISTING,
46	///     co::FILE_ATTRIBUTE::NORMAL,
47	///     None,
48	///     None,
49	///     None,
50	/// )?;
51	///
52	/// let hmap = hfile.CreateFileMapping(
53	///     None,
54	///     co::PAGE::READONLY,
55	///     None,
56	///     None,
57	///     None,
58	/// )?;
59	///
60	/// let view = hmap.MapViewOfFile(co::FILE_MAP::READ, 0, None)?;
61	///
62	/// let slice = view.as_slice(hfile.GetFileSizeEx()? as _);
63	/// let text = std::str::from_utf8(slice)?;
64	///
65	/// println!("{}", text);
66	/// # w::AnyResult::Ok(())
67	/// ```
68	#[must_use]
69	pub fn as_slice(&self, len: usize) -> &[u8] {
70		unsafe { std::slice::from_raw_parts(self.ptr() as _, len) }
71	}
72
73	/// [`FlushViewOfFile`](https://learn.microsoft.com/en-us/windows/win32/api/memoryapi/nf-memoryapi-flushviewoffile)
74	/// function.
75	pub fn FlushViewOfFile(&self, start_at_byte: usize, num_bytes: usize) -> SysResult<()> {
76		bool_to_sysresult(unsafe {
77			ffi::FlushViewOfFile(self.ptr().offset(start_at_byte as _), num_bytes)
78		})
79	}
