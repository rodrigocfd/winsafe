9handle! { HFINDFILE;
10	/// Handle to a
11	/// [file search](https://learn.microsoft.com/en-us/windows/win32/api/fileapi/nf-fileapi-findfirstfilew).
12	/// Originally just a `HANDLE`.
13}
14
15impl HFINDFILE {
16	/// [`FindFirstFile`](https://learn.microsoft.com/en-us/windows/win32/api/fileapi/nf-fileapi-findfirstfilew)
17	/// function.
18	///
19	/// This method is rather tricky, consider using
20	/// [`path::dir_list_flat`](crate::path::dir_list_flat) or
21	/// [`path::dir_list_recursive`](crate::path::dir_list_recursive)
22	#[must_use]
23	pub fn FindFirstFile(
24		file_name: &str,
25		wfd: &mut WIN32_FIND_DATA,
26	) -> SysResult<(FindCloseGuard, bool)> {
27		unsafe {
28			match ffi::FindFirstFileW(WString::from_str(file_name).as_ptr(), pvoid(wfd)).as_mut() {
29				Some(ptr) => {
30					let h = HFINDFILE::from_ptr(ptr);
31					// When using a filter, if no files are found, the function
32					// is successful but it returns an invalid handle.
33					let has_something = h != HFINDFILE::INVALID;
34					Ok((FindCloseGuard::new(h), has_something))
35				},
36				None => match GetLastError() {
37					co::ERROR::FILE_NOT_FOUND => Ok((
38						FindCloseGuard::new(HFINDFILE::NULL), // not an error, first file not found
39						false,
40					)),
41					err => Err(err),
42				},
43			}
44		}
45	}
46
47	/// [`FindNextFile`](https://learn.microsoft.com/en-us/windows/win32/api/fileapi/nf-fileapi-findnextfilew)
48	/// function.
49	///
50	/// This method is rather tricky, consider using
51	/// [`path::dir_list_flat`](crate::path::dir_list_flat) or
52	/// [`path::dir_list_recursive`](crate::path::dir_list_recursive)
53	#[must_use]
54	pub fn FindNextFile(&self, wfd: &mut WIN32_FIND_DATA) -> SysResult<bool> {
55		match unsafe { ffi::FindNextFileW(self.ptr(), pvoid(wfd)) } {
56			0 => match GetLastError() {
57				co::ERROR::NO_MORE_FILES => Ok(false), // not an error, no further files found
58				err => Err(err),
59			},
60			_ => Ok(true),
61		}
62	}
