17/// This trait is enabled with the `kernel` feature, and provides methods for
18/// [`HFINDFILE`](crate::HFINDFILE).
19///
20/// Prefer importing this trait through the prelude:
21///
22/// ```no_run
23/// use winsafe::prelude::*;
24/// ```
25pub trait kernel_Hfindfile: Handle {
26	/// [`FindFirstFile`](https://learn.microsoft.com/en-us/windows/win32/api/fileapi/nf-fileapi-findfirstfilew)
27	/// function.
28	///
29	/// This method is rather tricky, consider using
30	/// [`path::dir_list`](crate::path::dir_list).
31	#[must_use]
32	fn FindFirstFile(
33		file_name: &str,
34		wfd: &mut WIN32_FIND_DATA,
35	) -> SysResult<(FindCloseGuard, bool)> {
36		unsafe {
37			match ffi::FindFirstFileW(WString::from_str(file_name).as_ptr(), wfd as *mut _ as _)
38				.as_mut()
39			{
40				Some(ptr) => {
41					let h = HFINDFILE::from_ptr(ptr);
42					// When using a filter, if no files are found, the function
43					// is successful but it returns an invalid handle.
44					let has_something = h != HFINDFILE::INVALID;
45					Ok((FindCloseGuard::new(h), has_something))
46				},
47				None => match GetLastError() {
48					co::ERROR::FILE_NOT_FOUND => Ok((
49						FindCloseGuard::new(HFINDFILE::NULL), // not an error, first file not found
50						false,
51					)),
52					err => Err(err),
53				},
54			}
55		}
56	}
57
58	/// [`FindNextFile`](https://learn.microsoft.com/en-us/windows/win32/api/fileapi/nf-fileapi-findnextfilew)
59	/// function.
60	///
61	/// This method is rather tricky, consider using
62	/// [`path::dir_list`](crate::path::dir_list).
63	#[must_use]
64	fn FindNextFile(&self, wfd: &mut WIN32_FIND_DATA) -> SysResult<bool> {
65		match unsafe { ffi::FindNextFileW(self.ptr(), wfd as *mut _ as _) } {
66			0 => match GetLastError() {
67				co::ERROR::NO_MORE_FILES => Ok(false), // not an error, no further files found
68				err => Err(err),
69			},
70			_ => Ok(true),
71		}
72	}
73}
