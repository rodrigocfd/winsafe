17/// This trait is enabled with the `kernel` feature, and provides methods for
18/// [`HEVENT`](crate::HEVENT).
19///
20/// Prefer importing this trait through the prelude:
21///
22/// ```no_run
23/// use winsafe::prelude::*;
24/// ```
25pub trait kernel_Hevent: Handle {
26	/// [`CreateEvent`](https://learn.microsoft.com/en-us/windows/win32/api/synchapi/nf-synchapi-createeventw)
27	/// function.
28	#[must_use]
29	fn CreateEvent(
30		security_attributes: Option<&mut SECURITY_ATTRIBUTES>,
31		manual_reset: bool,
32		initial_state: bool,
33		name: Option<&str>,
34	) -> SysResult<CloseHandleGuard<HEVENT>>
35	{
36		unsafe {
37			ptr_to_sysresult_handle(
38				ffi::CreateEventW(
39					security_attributes.map_or(std::ptr::null_mut(), |sa| sa as *const _ as _),
40					manual_reset as _,
41					initial_state as _,
42					WString::from_opt_str(name).as_ptr(),
43				)
44			).map(|h| CloseHandleGuard::new(h))
45		}
46	}
47
48	/// [`CreateEventEx`](https://learn.microsoft.com/en-us/windows/win32/api/synchapi/nf-synchapi-createeventexw)
49	/// method.
50	#[must_use]
51	fn CreateEventEx(
52		security_attributes: Option<&mut SECURITY_ATTRIBUTES>,
53		name: Option<&str>,
54		flags: co::CREATE_EVENT,
55		desired_access: co::EVENT_RIGHTS,
56	) -> SysResult<CloseHandleGuard<HEVENT>>
57	{
58		unsafe {
59			ptr_to_sysresult_handle(
60				ffi::CreateEventExW(
61					security_attributes.map_or(std::ptr::null_mut(), |sa| sa as *const _ as _),
62					WString::from_opt_str(name).as_ptr(),
63					flags.raw(),
64					desired_access.raw(),
65				)
66			).map(|h| CloseHandleGuard::new(h))
67		}
68	}
69
70	/// [`OpenEvent`](https://learn.microsoft.com/en-us/windows/win32/api/synchapi/nf-synchapi-openeventw)
71	/// function.
72	#[must_use]
73	fn OpenEvent(&self,
74		desired_access: co::EVENT_RIGHTS,
75		inherit_handle: bool,
76		name: &str,
77	) -> SysResult<CloseHandleGuard<HEVENT>>
78	{
79		unsafe {
80			ptr_to_sysresult_handle(
81				ffi::OpenEventW(
82					desired_access.raw(),
83					inherit_handle as _,
84					WString::from_str(name).as_ptr(),
85				)
86			).map(|h| CloseHandleGuard::new(h))
87		}
88	}
89
90	/// [`PulseEvent`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-pulseevent)
91	/// function.
92	fn PulseEvent(&self) -> SysResult<()> {
93		bool_to_sysresult(unsafe { ffi::PulseEvent(self.ptr()) })
94	}
95
96	/// [`ResetEvent`](https://learn.microsoft.com/en-us/windows/win32/api/synchapi/nf-synchapi-resetevent)
97	/// function.
98	fn ResetEvent(&self) -> SysResult<()> {
99		bool_to_sysresult(unsafe { ffi::ResetEvent(self.ptr()) })
100	}
101
102	/// [`SetEvent`](https://learn.microsoft.com/en-us/windows/win32/api/synchapi/nf-synchapi-setevent)
103	/// function.
104	fn SetEvent(&self) -> SysResult<()> {
105		bool_to_sysresult(unsafe { ffi::SetEvent(self.ptr()) })
106	}
107
108	/// [`WaitForSingleObject`](https://learn.microsoft.com/en-us/windows/win32/api/synchapi/nf-synchapi-waitforsingleobject)
109	/// function.
110	fn WaitForSingleObject(&self,
111		milliseconds: Option<u32>,
112	) -> SysResult<co::WAIT>
113	{
114		match unsafe {
115			co::WAIT::from_raw(
116				ffi::WaitForSingleObject(
117					self.ptr(),
118					milliseconds.unwrap_or(INFINITE),
119				),
120			)
121		} {
122			co::WAIT::FAILED => Err(GetLastError()),
123			wait => Ok(wait),
124		}
125	}
126}
