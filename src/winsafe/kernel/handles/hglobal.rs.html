8handle! { HGLOBAL;
9	/// Handle to a
10	/// [global memory block](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-globalalloc).
11	/// Originally just a `HANDLE`.
12	///
13	/// The allocated memory block is accessible through the
14	/// [`GlobalLock`](crate::HGLOBAL::GlobalLock) method.
15}
16
17impl HGLOBAL {
18	/// [`GlobalAlloc`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-globalalloc)
19	/// function.
20	#[must_use]
21	pub fn GlobalAlloc(flags: co::GMEM, num_bytes: usize) -> SysResult<GlobalFreeGuard> {
22		unsafe {
23			PtrRet(ffi::GlobalAlloc(flags.raw(), num_bytes))
24				.to_sysresult_handle()
25				.map(|h| GlobalFreeGuard::new(h))
26		}
27	}
28
29	/// [`GlobalFlags`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-globalflags)
30	/// function.
31	#[must_use]
32	pub fn GlobalFlags(&self) -> SysResult<co::GMEM> {
33		match unsafe { ffi::GlobalFlags(self.ptr()) } {
34			GMEM_INVALID_HANDLE => Err(GetLastError()),
35			flags => Ok(unsafe { co::GMEM::from_raw(flags) }),
36		}
37	}
38
39	/// [`GlobalLock`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-globallock)
40	/// function.
41	///
42	/// Calls [`GlobalSize`](crate::HGLOBAL::GlobalSize) to retrieve the size of
43	/// the memory block.
44	///
45	/// # Examples
46	///
47	/// ```no_run
48	/// use winsafe::{self as w, prelude::*, co};
49	///
50	/// let hglobal = w::HGLOBAL::GlobalAlloc(
51	///     co::GMEM::FIXED | co::GMEM::ZEROINIT,
52	///     120,
53	/// )?;
54	///
55	/// let mut block = hglobal.GlobalLock()?;
56	///
57	/// block.as_mut_slice()[0] = 40;
58	///
59	/// // GlobalUnlock() called automatically
60	///
61	/// // GlobalFree() called automatically
62	/// # w::SysResult::Ok(())
63	/// ```
64	#[must_use]
65	pub fn GlobalLock(&self) -> SysResult<GlobalUnlockGuard<'_>> {
66		let mem_sz = self.GlobalSize()?;
67		unsafe {
68			PtrRet(ffi::GlobalLock(self.ptr()))
69				.to_sysresult()
70				.map(|ptr| GlobalUnlockGuard::new(self, ptr, mem_sz))
71		}
72	}
73
74	/// [`GlobalReAlloc`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-globalrealloc)
75	/// function.
76	///
77	/// Originally this method returns the handle to the reallocated memory
78	/// object; here the original handle is automatically updated.
79	pub fn GlobalReAlloc(&mut self, num_bytes: usize, flags: co::GMEM) -> SysResult<()> {
80		PtrRet(unsafe { ffi::GlobalReAlloc(self.ptr(), num_bytes, flags.raw()) })
81			.to_sysresult_handle()
82			.map(|h| {
83				*self = h;
84			})
85	}
86
87	/// [`GlobalSize`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-globalsize)
88	/// function.
89	#[must_use]
90	pub fn GlobalSize(&self) -> SysResult<usize> {
91		match unsafe { ffi::GlobalSize(self.ptr()) } {
92			0 => Err(GetLastError()),
93			sz => Ok(sz),
94		}
95	}
