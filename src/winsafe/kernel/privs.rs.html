143/// If the vector is empty, returns null, otherwise calls `as_ptr`.

153/// Converts a string to an ISO-8859-1 null-terminated byte array.
154#[must_use]
155pub(crate) fn str_to_iso88591(s: &str) -> Vec<u8> {
156	s.chars()
157		.map(|ch| ch as u8)
158		.chain(std::iter::once(0)) // append a terminating null
159		.collect()
160}

162/// Parses a null-delimited multi-string, ending with two terminating nulls.
163///
164/// # Safety
165///
166/// If `len` is not informed, make sure the string has two terminating nulls.
167#[must_use]
168pub(crate) unsafe fn parse_multi_z_str(src: *const u16, len: Option<usize>) -> Vec<String> {
169	let given_len = len.unwrap_or(usize::MAX);
170	let mut src = src;
171	let mut strings = Vec::<String>::new();
172	let mut ch = 0; // relative index of char in current string
173	let mut tot_ch = 0; // absolute index of char in original src
174
175	loop {
176		if unsafe { *src.add(ch) } == 0 || tot_ch == given_len {
177			let slice = unsafe { std::slice::from_raw_parts(src, ch) };
178			if slice.is_empty() {
179				break; // empty string means two consecutive nulls
180			}
181			strings.push(WString::from_wchars_slice(slice).to_string());
182			src = unsafe { src.add(ch + 1) };
183			ch = 0;
184		} else {
185			ch += 1;
186		}
187
188		if len.is_some() && tot_ch == given_len {
189			break;
190		}
191		tot_ch += 1;
192	}
193	strings
194}

196/// Creates two vectors:
197/// * the first with each string converted to `WString`;
198/// * the second with the pointers to each `WString` in the first vector.
199#[must_use]
200pub(crate) fn create_wstr_ptr_vecs(
201	strings: Option<&[impl AsRef<str>]>,
202) -> (Vec<WString>, Vec<*const u16>) {
203	match strings {
204		Some(ss) => {
205			let wstrs = ss.iter().map(|s| WString::from_str(s)).collect::<Vec<_>>();
206			let pwstrs = wstrs.iter().map(|w| w.as_ptr()).collect::<Vec<_>>();
207			(wstrs, pwstrs)
208		},
209		None => (Vec::default(), Vec::default()),
210	}
