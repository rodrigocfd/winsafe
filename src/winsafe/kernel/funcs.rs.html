67/// [`ExpandEnvironmentStrings`](https://learn.microsoft.com/en-us/windows/win32/api/processenv/nf-processenv-expandenvironmentstringsw)
68/// function.
69///
70/// # Examples
71///
72/// ```no_run
73/// use winsafe::{self as w, prelude::*};
74///
75/// let expanded = w::ExpandEnvironmentStrings(
76///     "Os %OS%, home %HOMEPATH% and temp %TEMP%",
77/// )?;
78///
79/// println!("{}", expanded);
80/// # w::SysResult::Ok(())
81/// ```
82#[must_use]
83pub fn ExpandEnvironmentStrings(src: &str) -> SysResult<String> {
84	let wsrc = WString::from_str(src);
85	let mut buf_sz =
86		match unsafe { ffi::ExpandEnvironmentStringsW(wsrc.as_ptr(), std::ptr::null_mut(), 0) } {
87			0 => return Err(GetLastError()),
88			n => n,
89		}; // includes terminating null count
90
91	loop {
92		let mut buf = WString::new_alloc_buf(buf_sz as _);
93		let required_sz = match unsafe {
94			ffi::ExpandEnvironmentStringsW(wsrc.as_ptr(), buf.as_mut_ptr(), buf_sz)
95		} {
96			0 => return Err(GetLastError()),
97			n => n,
98		}; // plus terminating null count
99
100		if required_sz <= buf_sz {
101			return Ok(buf.to_string());
102		}
103
104		buf_sz = required_sz; // includes terminating null count; set the new buffer size to try again
105	}
106}

136/// [`FormatMessage`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-formatmessagew)
137/// function.
138///
139/// You don't need to call this function: all error types implement the
140/// [`SystemError`](crate::prelude::SystemError) trait which will automatically
141/// call `FormatMessage`.
142#[must_use]
143pub unsafe fn FormatMessage(
144	flags: co::FORMAT_MESSAGE,
145	source: Option<*mut std::ffi::c_void>,
146	message_id: u32,
147	lang_id: LANGID,
148	args: Option<&[*mut std::ffi::c_void]>,
149) -> SysResult<String> {
150	let mut ptr_buf = std::ptr::null_mut() as *mut u16;
151
152	let nchars = match ffi::FormatMessageW(
153		flags.raw(),
154		source.unwrap_or(std::ptr::null_mut()),
155		message_id,
156		u16::from(lang_id) as _,
157		&mut ptr_buf as *mut *mut _ as _, // pass pointer to pointer
158		0,
159		args.map_or(std::ptr::null_mut(), |arr| arr.as_ptr() as _),
160	) as _
161	{
162		0 => Err(GetLastError()),
163		nchars => Ok(nchars),
164	}?;
165
166	let final_wstr = WString::from_wchars_count(ptr_buf, nchars as _);
167	let _ = LocalFreeGuard::new(HLOCAL::from_ptr(ptr_buf as _)); // free returned pointer
168	let final_str = final_wstr.to_string();
169	Ok(final_str)
170}

284/// [`GetEnvironmentStrings`](https://learn.microsoft.com/en-us/windows/win32/api/processenv/nf-processenv-getenvironmentstringsw)
285/// function.
286///
287/// Returns the parsed strings, and automatically frees the retrieved
288/// environment block with
289/// [`FreeEnvironmentStrings`](https://learn.microsoft.com/en-us/windows/win32/api/processenv/nf-processenv-freeenvironmentstringsw).
290///
291/// # Examples
292///
293/// Retrieving and printing the key/value pairs of all environment strings:
294///
295/// ```no_run
296/// use winsafe::{self as w, prelude::*};
297///
298/// let env_vars = w::GetEnvironmentStrings()?;
299/// for (k, v) in env_vars.iter() {
300///     println!("{} = {}", k, v);
301/// }
302/// # w::SysResult::Ok(())
303/// ```
304#[must_use]
305pub fn GetEnvironmentStrings() -> SysResult<Vec<(String, String)>> {
306	ptr_to_sysresult(unsafe { ffi::GetEnvironmentStringsW() } as _).map(|ptr| {
307		let vec_entries = unsafe { parse_multi_z_str(ptr as *mut _ as _, None) };
308		unsafe {
309			ffi::FreeEnvironmentStringsW(ptr);
310		}
311		vec_entries
312			.iter()
313			.map(|env_str| {
314				let mut pair = env_str.split("="); // assumes correctly formatted pairs
315				let key = pair.next().unwrap();
316				let val = pair.next().unwrap();
317				(key.to_owned(), val.to_owned())
318			})
319			.collect()
320	})
321}

474/// function.

476/// # Examples
477///
478/// Reading all key/value pairs of a section from an INI file:
479///
480/// ```no_run
481/// use winsafe::{self as w, prelude::*};

495///
496/// * [`GetPrivateProfileSectionNames`](crate::GetPrivateProfileSectionNames)
497/// * [`GetPrivateProfileString`](crate::GetPrivateProfileString)
498/// * [`WritePrivateProfileString`](crate::WritePrivateProfileString)
499#[must_use]
500pub fn GetPrivateProfileSection(
501	section_name: &str,
502	file_name: &str,
503) -> SysResult<Vec<(String, String)>> {
504	let mut buf_sz = WString::SSO_LEN; // start with no string heap allocation
505	loop {
506		let mut buf = WString::new_alloc_buf(buf_sz);
507		let returned_chars = unsafe {
508			// char count without terminating null
509			ffi::GetPrivateProfileSectionW(
510				WString::from_str(section_name).as_ptr(),
511				buf.as_mut_ptr(),
512				buf.buf_len() as _,
513				WString::from_str(file_name).as_ptr(),
514			)
515		} + 1 + 1; // plus terminating null count, plus weird extra count
516
517		if GetLastError() == co::ERROR::FILE_NOT_FOUND {
518			return Err(co::ERROR::FILE_NOT_FOUND);
519		} else if (returned_chars as usize) < buf_sz {
520			// to break, must have at least 1 char gap
521			return Ok(unsafe { parse_multi_z_str(buf.as_ptr(), Some(buf.buf_len())) }
522				.iter()
523				.map(|line| match line.split_once('=') {
524					Some((key, val)) => (key.to_owned(), val.to_owned()),
525					None => (String::new(), String::new()),
526				})
527				.collect());
528		}
529
530		buf_sz *= 2; // double the buffer size to try again
531	}

859			match name {
860				Some(_) => name_buf.0.as_mut_ptr(),
861				None => std::ptr::null_mut(),
862			},
863			name_buf.1 as u32,
864			serial_number.map_or(std::ptr::null_mut(), |n| n),
865			max_component_len.map_or(std::ptr::null_mut(), |m| m),
