49/// [`CreateProcess`](https://learn.microsoft.com/en-us/windows/win32/api/processthreadsapi/nf-processthreadsapi-createprocessw)
50/// function.
51#[must_use]
52pub fn CreateProcess(
53	application_name: Option<&str>,
54	command_line: Option<&str>,
55	process_attrs: Option<&SECURITY_ATTRIBUTES>,
56	thread_attrs: Option<&SECURITY_ATTRIBUTES>,
57	inherit_handles: bool,
58	creation_flags: co::CREATE,
59	environment_vars: &[(&str, &str)],
60	current_dir: Option<&str>,
61	si: &mut STARTUPINFO,
62) -> SysResult<CloseHandlePiGuard> {
63	let mut buf_cmd_line = WString::from_opt_str(command_line);
64	let mut pi = PROCESS_INFORMATION::default();
65
66	let mut _env_buf = WString::new();
67	let env_ptr = if environment_vars.is_empty() {
68		std::ptr::null_mut()
69	} else {
70		let env_buf = WString::from_str_vec(
71			&environment_vars
72				.iter()
73				.map(|(name, val)| format!("{}={}", name, val))
74				.collect::<Vec<_>>(),
75		);
76		env_buf.as_ptr() as _
77	};
78
79	unsafe {
80		BoolRet(ffi::CreateProcessW(
81			WString::from_opt_str(application_name).as_ptr(),
82			buf_cmd_line.as_mut_ptr(),
83			pcvoid_or_null(process_attrs),
84			pcvoid_or_null(thread_attrs),
85			inherit_handles as _,
86			(creation_flags | co::CREATE::UNICODE_ENVIRONMENT).raw(), // environment is always UTF-16
87			env_ptr,
88			WString::from_opt_str(current_dir).as_ptr(),
89			pvoid(si),
90			pvoid(&mut pi),
91		))
92		.to_sysresult()
93		.map(|_| CloseHandlePiGuard::new(pi))
94	}
95}

122/// [`ExpandEnvironmentStrings`](https://learn.microsoft.com/en-us/windows/win32/api/processenv/nf-processenv-expandenvironmentstringsw)
123/// function.
124///
125/// # Examples
126///
127/// ```no_run
128/// use winsafe::{self as w, prelude::*};
129///
130/// let expanded = w::ExpandEnvironmentStrings(
131///     "Os %OS%, home %HOMEPATH% and temp %TEMP%",
132/// )?;
133///
134/// println!("{}", expanded);
135/// # w::SysResult::Ok(())
136/// ```
137#[must_use]
138pub fn ExpandEnvironmentStrings(src: &str) -> SysResult<String> {
139	let wsrc = WString::from_str(src);
140	let mut buf_sz =
141		match unsafe { ffi::ExpandEnvironmentStringsW(wsrc.as_ptr(), std::ptr::null_mut(), 0) } {
142			0 => return Err(GetLastError()),
143			n => n,
144		}; // includes terminating null count
145
146	loop {
147		let mut buf = WString::new_alloc_buf(buf_sz as _);
148		let required_sz = match unsafe {
149			ffi::ExpandEnvironmentStringsW(wsrc.as_ptr(), buf.as_mut_ptr(), buf_sz)
150		} {
151			0 => return Err(GetLastError()),
152			n => n,
153		}; // plus terminating null count
154
155		if required_sz <= buf_sz {
156			return Ok(buf.to_string());
157		}
158
159		buf_sz = required_sz; // includes terminating null count; set the new buffer size to try again
160	}
161}

205/// [`FormatMessage`](https://learn.microsoft.com/en-us/windows/win32/api/winbase/nf-winbase-formatmessagew)
206/// function.
207///
208/// You don't need to call this function: all error types implement the
209/// [`SystemError`](crate::prelude::SystemError) trait which will automatically
210/// call `FormatMessage`.
211///
212/// # Safety
213///
214/// Incorrect usage of the flags and formatting string may lead to memory
215/// corruption.
216#[must_use]
217pub unsafe fn FormatMessage(
218	flags: co::FORMAT_MESSAGE,
219	source: Option<*mut std::ffi::c_void>,
220	message_id: u32,
221	lang_id: LANGID,
222	args: &[*mut std::ffi::c_void],
223) -> SysResult<String> {
224	let mut ptr_buf = std::ptr::null_mut::<u16>();
225
226	let nchars = match unsafe {
227		ffi::FormatMessageW(
228			flags.raw(),
229			source.unwrap_or(std::ptr::null_mut()),
230			message_id,
231			u16::from(lang_id) as _,
232			&mut ptr_buf as *mut *mut _ as _, // pass pointer to pointer
233			0,
234			vec_ptr(args) as _,
235		)
236	} as _
237	{
238		0 => Err(GetLastError()),
239		nchars => Ok(nchars),
240	}?;
241
242	let final_wstr = WString::from_wchars_count(ptr_buf, nchars as _);
243	let _ = unsafe { LocalFreeGuard::new(HLOCAL::from_ptr(ptr_buf as _)) }; // free returned pointer
244	Ok(final_wstr.to_string())
245}

362/// [`GetEnvironmentStrings`](https://learn.microsoft.com/en-us/windows/win32/api/processenv/nf-processenv-getenvironmentstringsw)
363/// function.
364///
365/// Returns the parsed strings, and automatically frees the retrieved
366/// environment block with
367/// [`FreeEnvironmentStrings`](https://learn.microsoft.com/en-us/windows/win32/api/processenv/nf-processenv-freeenvironmentstringsw).
368///
369/// # Examples
370///
371/// Retrieving and printing the key/value pairs of all environment strings:
372///
373/// ```no_run
374/// use winsafe::{self as w, prelude::*};
375///
376/// let env_vars = w::GetEnvironmentStrings()?;
377/// for (k, v) in env_vars.iter() {
378///     println!("{} = {}", k, v);
379/// }
380/// # w::SysResult::Ok(())
381/// ```
382#[must_use]
383pub fn GetEnvironmentStrings() -> SysResult<Vec<(String, String)>> {
384	PtrRet(unsafe { ffi::GetEnvironmentStringsW() } as _)
385		.to_sysresult()
386		.map(|ptr| {
387			let vec_entries = unsafe { parse_multi_z_str(ptr as _, None) };
388			unsafe {
389				ffi::FreeEnvironmentStringsW(ptr);
390			}
391			vec_entries
392				.iter()
393				.map(|env_str| {
394					let mut pair = env_str.split("="); // assumes correctly formatted pairs
395					let key = pair.next().unwrap();
396					let val = pair.next().unwrap();
397					(key.to_owned(), val.to_owned())
398				})
399				.collect()
400		})
401}

593#[must_use]
594pub fn GetPrivateProfileSection(
595	section_name: &str,
596	file_name: &str,
597) -> SysResult<Vec<(String, String)>> {
598	let mut buf_sz = WString::SSO_LEN; // start with no string heap allocation
599	loop {
600		let mut buf = WString::new_alloc_buf(buf_sz);
601		let returned_chars = unsafe {
602			// Char count without terminating null.
603			ffi::GetPrivateProfileSectionW(
604				WString::from_str(section_name).as_ptr(),
605				buf.as_mut_ptr(),
606				buf.buf_len() as _,
607				WString::from_str(file_name).as_ptr(),
608			)
609		} + 1 + 1; // plus terminating null count, plus weird extra count
610
611		if GetLastError() == co::ERROR::FILE_NOT_FOUND {
612			return Err(co::ERROR::FILE_NOT_FOUND);
613		} else if (returned_chars as usize) < buf_sz {
614			// to break, must have at least 1 char gap
615			return Ok(unsafe { parse_multi_z_str(buf.as_ptr(), Some(buf.buf_len())) }
616				.iter()
617				.map(|line| match line.split_once('=') {
618					Some((key, val)) => (key.to_owned(), val.to_owned()),
619					None => (String::new(), String::new()),
620				})
621				.collect());
622		}
623
624		buf_sz *= 2; // double the buffer size to try again
625	}
626}
