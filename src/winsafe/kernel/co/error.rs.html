5const_basic_decl! { ERROR: u32;
6	/// A native
7	/// [system error code](https://learn.microsoft.com/en-us/windows/win32/debug/system-error-codes)
8	/// (`u32`), commonly retrieved by
9	/// [`GetLastError`](crate::GetLastError) function.
10	///
11	/// The [`Result` alias](crate#errors-and-result-aliases) for this type is
12	/// [`SysResult`](crate::SysResult).
13	///
14	/// Implements the standard [`Error`](std::error::Error) trait.
15	///
16	/// Implements the [`Debug`](std::fmt::Debug) and
17	/// [`Display`](std::fmt::Display) traits to show the error code along with
18	/// the error description, taken from
19	/// [`SystemError`](crate::prelude::SystemError) trait. For example, the
20	/// code below:
21	///
22	/// ```no_run
23	/// use winsafe::{self as w, prelude::*, co};
24	///
25	/// println!("{}", co::ERROR::OUT_OF_PAPER);
26	/// ```
27	///
28	/// Will print:
29	///
30	/// ```text
31	/// [0x001c 28] The printer is out of paper.
32	/// ```
33	///
34	/// The numbers inside the brackets are the system error code, in
35	/// hexadecimal and decimal formats for convenience.
36}
37
38impl std::error::Error for ERROR {
39	fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
40		None
41	}
42}
43
44impl std::fmt::Display for ERROR {
45	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
46		write!(f, "[{}] {}", self.0, self.FormatMessage().trim_end())
47	}
48}
49impl std::fmt::Debug for ERROR {
50	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
51		if self.0 > 0xffff {
52			write!(f, "ERROR({:#010x} {}) {}", self.0, self.0, self.FormatMessage().trim_end())
53		} else {
54			write!(f, "ERROR({:#06x} {}) {}", self.0, self.0, self.FormatMessage().trim_end())
55		}
56	}
57}
