6/// Access types for [`File::open`](crate::File::open) and
7/// [`FileMapped::open`](crate::FileMapped::open).
8#[derive(Clone, Copy, PartialEq, Eq, Hash)]
9pub enum FileAccess {
10	/// Opens the file as read-only. Fails if the file doesn't exist.
11	ExistingReadOnly,
12	/// Opens the file as read/write. Fails if the file doesn't exist.
13	ExistingRW,
14	/// Opens the file as read/write. If the file doesn't exist, it will be
15	/// created.
16	OpenOrCreateRW,
17	/// Creates a new file as read/write. Fails if the file already exists.
18	CreateRW,
19}
20
21impl std::fmt::Display for FileAccess {
22	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
23		write!(f, "{}", match self {
24			FileAccess::ExistingReadOnly => "Existing file, read-only",
25			FileAccess::ExistingRW => "Existing file, read and write",
26			FileAccess::OpenOrCreateRW => "Open existing file or create new file, read and write",
27			FileAccess::CreateRW => "Create new file, read and write",
28		})
29	}
30}

32/// Manages an [`HFILE`](crate::HFILE) handle, which provides file read/write
33/// and other operations. It is closed automatically when the object goes out of
34/// scope.
35///
36/// This is an alternative to the standard [`std::fs::File`], with a possibly
37/// faster implementation since it's Windows-only.
38///
39/// If you just want to read the file, consider memory-mapping it with
40/// [`FileMapped`](crate::FileMapped), which tends to be faster.
41///
42/// # Examples
43///
44/// Reading the contents as a string:
45///
46/// ```no_run
47/// use winsafe::{self as w, prelude::*};
48///
49/// let f = w::File::open(
50///     "C:\\Temp\\foo.txt",
51///     w::FileAccess::ExistingRW,
52/// )?;
53/// let raw_bytes = f.read_all()?;
54/// let text = w::WString::parse(&raw_bytes)?.to_string();
55/// # w::SysResult::Ok(())
56/// ```
57///
58/// Erasing the file and writing a string:
59///
60/// ```no_run
61/// use winsafe::{self as w, prelude::*};
62///
63/// let f = w::File::open(
64///     "C:\\Temp\\foo.txt",
65///     w::FileAccess::OpenOrCreateRW,
66/// )?;
67/// f.set_size(0)?; // truncate
68/// f.write("My text".as_bytes())?;
69/// # w::SysResult::Ok(())
70/// ```
71pub struct File {
72	hfile: CloseHandleGuard<HFILE>,
73}
74
75impl File {
76	/// Opens a file with the desired access.
77	#[must_use]
78	pub fn open(file_path: &str, access: FileAccess) -> SysResult<Self> {
79		let (acc, share, disp) = match access {
80			FileAccess::ExistingReadOnly => (
81				co::GENERIC::READ,
82				Some(co::FILE_SHARE::READ),
83				co::DISPOSITION::OPEN_EXISTING,
84			),
85			FileAccess::ExistingRW => (
86				co::GENERIC::READ | co::GENERIC::WRITE,
87				None,
88				co::DISPOSITION::OPEN_EXISTING,
89			),
90			FileAccess::OpenOrCreateRW => (
91				co::GENERIC::READ | co::GENERIC::WRITE,
92				None,
93				co::DISPOSITION::OPEN_ALWAYS,
94			),
95			FileAccess::CreateRW => (
96				co::GENERIC::READ | co::GENERIC::WRITE,
97				None,
98				co::DISPOSITION::CREATE_NEW,
99			),
100		};
101
102		let (hfile, _) = HFILE::CreateFile(
103			file_path, acc, share, None, disp,
104			co::FILE_ATTRIBUTE::NORMAL, None, None, None)?;
105		Ok(Self { hfile })
106	}

108	/// Truncates the file size to zero, then writes the bytes.
109	pub fn erase_and_write(&self, data: &[u8]) -> SysResult<()> {
110		self.set_size(0)?;
111		self.write(data)
112	}

115	#[must_use]
116	pub fn hfile(&self) -> &HFILE {
117		&*self.hfile
118	}

120	/// Returns the position of the file pointer by calling
121	/// [`HFILE::SetFilePointerEx`](crate::prelude::kernel_Hfile::SetFilePointerEx).
122	#[must_use]
123	pub fn pointer_offset(&self) -> SysResult<u64> {
124		self.hfile.SetFilePointerEx(0, co::FILE_STARTING_POINT::CURRENT) // https://stackoverflow.com/a/17707021/6923555
125			.map(|off| off as _)
126	}
127
128	/// Returns the size of the file by calling
129	/// [`HFILE::GetFileSizeEx`](crate::prelude::kernel_Hfile::GetFileSizeEx),
130	/// allocates the `Vec` buffer, then reads all the file bytes by calling
131	/// [`HFILE::ReadFile`](crate::prelude::kernel_Hfile::ReadFile).
132	///
133	/// Note that the API limits the reading up to 4 GB.
134	#[must_use]
135	pub fn read_all(&self) -> SysResult<Vec<u8>> {
136		self.set_pointer_offset(0)?;
137		let mut buf = vec![0x00; self.size()? as _];
138		self.read_buffer(&mut buf)?;
139		Ok(buf)
140	}
141
142	/// Calls [`HFILE::ReadFile`](crate::prelude::kernel_Hfile::ReadFile) to
143	/// read at most `buffer.len()` bytes from the file, starting at the current
144	/// file pointer offset. Returns how many bytes were actually read. The file
145	/// pointer is then incremented by the number of bytes read.
146	///
147	/// Note that the API limits the reading up to 4 GB.
148	pub fn read_buffer(&self, buffer: &mut [u8]) -> SysResult<u32> {
149		self.hfile.ReadFile(buffer)
150	}
151
152	/// Sets the position of the file pointer by calling
153	/// [`HFILE::SetFilePointerEx`](crate::prelude::kernel_Hfile::SetFilePointerEx).
154	pub fn set_pointer_offset(&self, offset: u64) -> SysResult<()> {
155		self.hfile.SetFilePointerEx(offset as _, co::FILE_STARTING_POINT::BEGIN)
156			.map(|_| ())
157	}
158
159	/// Truncates or expands the file by calling
160	/// [`HFILE::SetFilePointerEx`](crate::prelude::kernel_Hfile::SetFilePointerEx)
161	/// and
162	/// [`HFILE::SetEndOfFile`](crate::prelude::kernel_Hfile::SetEndOfFile),
163	/// then sets the file pointer to the beginning of the file.
164	///
165	/// If the size is increased, the contents in the new area are undefined.
166	///
167	/// Note that sometimes the system will make the file larger than you
168	/// specified, so don't bindly trust this method.
169	pub fn set_size(&self, num_bytes: u64) -> SysResult<()> {
170		self.set_pointer_offset(num_bytes)?;
171		self.hfile.SetEndOfFile()?;
172		self.set_pointer_offset(0)
173	}
174
175	/// Returns the size of the file by calling
176	/// [`HFILE::GetFileSizeEx`](crate::prelude::kernel_Hfile::GetFileSizeEx).
177	#[must_use]
178	pub fn size(&self) -> SysResult<u64> {
179		self.hfile.GetFileSizeEx()
180	}
181
182	/// Returns, in current time zone, 3 times of the file, respectively:
183	/// 1. creation time;
184	/// 2. last access time;
185	/// 3. last write time.
186	#[must_use]
187	pub fn times(&self) -> SysResult<(SYSTEMTIME, SYSTEMTIME, SYSTEMTIME)> {
188		let (ft_creation, ft_last_access, ft_last_write) = self.hfile.GetFileTime()?;
189
190		let st_creation_utc = FileTimeToSystemTime(&ft_creation)?;
191		let st_last_access_utc = FileTimeToSystemTime(&ft_last_access)?;
192		let st_last_write_utc = FileTimeToSystemTime(&ft_last_write)?;
193
194		let st_creation_local = SystemTimeToTzSpecificLocalTime(None, &st_creation_utc)?;
195		let st_last_access_local = SystemTimeToTzSpecificLocalTime(None, &st_last_access_utc)?;
196		let st_last_write_local = SystemTimeToTzSpecificLocalTime(None, &st_last_write_utc)?;
197
198		Ok((st_creation_local, st_last_access_local, st_last_write_local))
199	}
200
201	/// Writes the bytes at the current file pointer by calling
202	/// [`HFILE::WriteFile`](crate::prelude::kernel_Hfile::WriteFile).
203	///
204	/// This method will fail if the file was opened with
205	/// [`FileAccess::ExistingReadOnly`](crate::FileAccess::ExistingReadOnly).
206	pub fn write(&self, data: &[u8]) -> SysResult<()> {
207		self.hfile.WriteFile(data)
208			.map(|_| ())
209	}
