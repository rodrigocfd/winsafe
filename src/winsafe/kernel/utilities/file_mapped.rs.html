6/// Manages an [`HFILEMAP`](crate::HFILEMAP) handle, which provides
7/// memory-mapped file operations, including read/write through slices. It is
8/// closed automatically when the object goes out of scope.
9///
10/// # Examples
11///
12/// Usually the fastest way to read a file as string is by memory-mapping it,
13/// like:
14///
15/// ```no_run
16/// use winsafe::{self as w, prelude::*};
17///
18/// let f = w::FileMapped::open(
19///     "C:\\Temp\\foo.txt",
20///     w::FileAccess::ExistingReadOnly,
21/// )?;
22/// let raw_bytes = f.as_slice();
23/// let text = w::WString::parse(raw_bytes)?.to_string();
24/// # w::SysResult::Ok(())
25/// ```
26pub struct FileMapped {
27	hview: UnmapViewOfFileGuard, // drop order is important
28	_hmap: CloseHandleGuard<HFILEMAP>,
29	file: File,
30	size: u64,
31}
32
33impl FileMapped {
34	/// Opens a file with the desired access, then map its contents in memory.
35	#[must_use]
36	pub fn open(file_path: &str, access: FileAccess) -> SysResult<Self> {
37		let file = File::open(file_path, access)?;
38		let hmap = file.hfile().CreateFileMapping(
39			None,
40			match access {
41				FileAccess::ExistingReadOnly => co::PAGE::READONLY,
42				FileAccess::ExistingRW
43					| FileAccess::OpenOrCreateRW
44					| FileAccess::CreateRW => co::PAGE::READWRITE,
45			},
46			None,
47			None,
48		)?;
49		let hview = hmap.MapViewOfFile(
50			match access {
51				FileAccess::ExistingReadOnly => co::FILE_MAP::READ,
52				FileAccess::ExistingRW
53					| FileAccess::OpenOrCreateRW
54					| FileAccess::CreateRW => co::FILE_MAP::READ | co::FILE_MAP::WRITE,
55			},
56			0,
57			None,
58		)?;
59		let size = file.hfile().GetFileSizeEx()?; // cache
60		Ok(Self { file, _hmap: hmap, hview, size })
61	}
62
63	/// Returns a mutable slice to the mapped memory.
64	#[must_use]
65	pub fn as_mut_slice(&mut self) -> &mut [u8] {
66		self.hview.as_mut_slice(self.size as _)
67	}
68
69	/// Returns a slice to the mapped memory.
70	#[must_use]
71	pub fn as_slice(&self) -> &[u8] {
72		self.hview.as_slice(self.size as _)
73	}
74
75	/// Returns the underlying file handle.
76	#[must_use]
77	pub fn hfile(&self) -> &HFILE {
78		self.file.hfile()
79	}
80
81	/// Returns the size of the file.
82	///
83	/// This value is cached.
84	#[must_use]
85	pub const fn size(&self) -> u64 {
86		self.size
87	}
88
89	/// Returns, in current time zone, 3 times of the file, respectively:
90	/// 1. creation time;
91	/// 2. last access time;
92	/// 3. last write time.
93	#[must_use]
94	pub fn times(&self) -> SysResult<(SYSTEMTIME, SYSTEMTIME, SYSTEMTIME)> {
95		self.file.times()
96	}
97}
