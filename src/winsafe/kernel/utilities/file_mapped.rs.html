5/// Manages an [`HFILEMAP`](crate::HFILEMAP) handle, which provides
6/// memory-mapped file operations, including read/write through slices. It is
7/// closed automatically when the object goes out of scope.
8///
9/// # Examples
10///
11/// Usually the fastest way to read a file as string is by memory-mapping it,
12/// like:
13///
14/// ```no_run
15/// use winsafe::{self as w, prelude::*};
16///
17/// let f = w::FileMapped::open(
18///     "C:\\Temp\\foo.txt",
19///     w::FileAccess::ExistingReadOnly,
20/// )?;
21/// let raw_bytes = f.as_slice();
22/// let text = w::WString::parse(raw_bytes)?.to_string();
23/// # w::SysResult::Ok(())
24/// ```
25pub struct FileMapped {
26	hview: UnmapViewOfFileGuard, // drop order is important
27	_hmap: CloseHandleGuard<HFILEMAP>,
28	file: File,
29	size: u64,
30}
31
32impl FileMapped {
33	/// Opens a file with the desired access, then map its contents in memory.
34	#[must_use]
35	pub fn open(file_path: &str, access: FileAccess) -> SysResult<Self> {
36		let file = File::open(file_path, access)?;
37		let hmap = file.hfile().CreateFileMapping(
38			None,
39			match access {
40				FileAccess::ExistingReadOnly => co::PAGE::READONLY,
41				FileAccess::ExistingRW | FileAccess::OpenOrCreateRW | FileAccess::CreateRW => {
42					co::PAGE::READWRITE
43				},
44			},
45			None,
46			None,
47			None,
48		)?;
49		let hview = hmap.MapViewOfFile(
50			match access {
51				FileAccess::ExistingReadOnly => co::FILE_MAP::READ,
52				FileAccess::ExistingRW | FileAccess::OpenOrCreateRW | FileAccess::CreateRW => {
53					co::FILE_MAP::READ | co::FILE_MAP::WRITE
54				},
55			},
56			0,
57			None,
58		)?;
59		let size = file.hfile().GetFileSizeEx()?; // cache
60		Ok(Self { file, _hmap: hmap, hview, size })
61	}
62
63	/// Returns a mutable slice to the mapped memory.
64	#[must_use]
65	pub fn as_mut_slice(&mut self) -> &mut [u8] {
66		self.hview.as_mut_slice(self.size as _)
67	}
68
69	/// Returns a slice to the mapped memory.
70	#[must_use]
71	pub fn as_slice(&self) -> &[u8] {
72		self.hview.as_slice(self.size as _)
73	}
74
75	/// Returns the underlying file handle.
76	#[must_use]
77	pub fn hfile(&self) -> &HFILE {
78		self.file.hfile()
79	}
80
81	/// Returns the size of the file.
82	///
83	/// This value is cached.
84	#[must_use]
85	pub const fn size(&self) -> u64 {
86		self.size
87	}
88
89	/// Returns, in current time zone, 3 times of the file, respectively:
90	/// 1. creation time;
91	/// 2. last access time;
92	/// 3. last write time.
93	#[must_use]
94	pub fn times(&self) -> SysResult<(SYSTEMTIME, SYSTEMTIME, SYSTEMTIME)> {
95		self.file.times()
96	}
97}
