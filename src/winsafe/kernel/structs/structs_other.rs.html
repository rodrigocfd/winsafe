139#[repr(C)]
140union CLAIM_SECURITY_ATTRIBUTE_V1_union0<'a> {
141	pInt64: *mut i64, // pointers because these are all arrays with ValueCount items
142	pUint64: *mut u64,
143	ppString: *mut *mut u16,
144	pFqbn: *mut CLAIM_SECURITY_ATTRIBUTE_FQBN_VALUE<'a>,
145	pOctetString: *mut CLAIM_SECURITY_ATTRIBUTE_OCTET_STRING_VALUE<'a>,
146}

148impl_default!(CLAIM_SECURITY_ATTRIBUTE_V1, 'a, 'b);

150impl<'a, 'b> CLAIM_SECURITY_ATTRIBUTE_V1<'a, 'b> {
151	pub_fn_string_ptr_get_set!('a, Name, set_Name);
152
153	/// Returns the low-word part of `Flags`.
154	#[must_use]
155	pub const fn FlagsLo(&self) -> co::CLAIM_SECURITY_ATTRIBUTE {
156		unsafe { co::CLAIM_SECURITY_ATTRIBUTE::from_raw(LOWORD(self.Flags)) }
157	}
158
159	/// Sets the low-word part of `Flags`.
160	pub const fn set_FlagsLo(&mut self, claim: co::CLAIM_SECURITY_ATTRIBUTE) {
161		self.Flags = MAKEDWORD(claim.raw(), self.FlagsHi());
162	}
163
164	/// Returns the high-word part of `Flags`.
165	#[must_use]
166	pub const fn FlagsHi(&self) -> u16 {
167		HIWORD(self.Flags)
168	}
169
170	/// Sets the high-word part of `Flags`.
171	pub const fn set_FlagsHi(&mut self, flags: u16) {
172		self.Flags = MAKEDWORD(self.FlagsLo().raw(), flags);
173	}
174
175	/// Returns the `Values` field.
176	///
177	/// # Panics
178	///
179	/// Panics if `ValueType` field is invalid.
180	#[must_use]
181	pub fn Values(&self) -> ClaimSecurityAttr<'_> {
182		unsafe {
183			match self.ValueType {
184				co::CLAIM_SECURITY_ATTRIBUTE_TYPE::INT64 => ClaimSecurityAttr::Int64(
185					std::slice::from_raw_parts(self.Values.pInt64, self.ValueCount as _),
186				),
187				co::CLAIM_SECURITY_ATTRIBUTE_TYPE::UINT64 => ClaimSecurityAttr::Uint64(
188					std::slice::from_raw_parts(self.Values.pUint64, self.ValueCount as _),
189				),
190				co::CLAIM_SECURITY_ATTRIBUTE_TYPE::STRING => ClaimSecurityAttr::String(
191					std::slice::from_raw_parts(self.Values.ppString, self.ValueCount as _)
192						.iter()
193						.map(|str_ptr| WString::from_wchars_nullt(*str_ptr).to_string())
194						.collect(),
195				),
196				co::CLAIM_SECURITY_ATTRIBUTE_TYPE::FQBN => ClaimSecurityAttr::Fbqn(
197					std::slice::from_raw_parts(self.Values.pFqbn, self.ValueCount as _),
198				),
199				co::CLAIM_SECURITY_ATTRIBUTE_TYPE::OCTET_STRING => ClaimSecurityAttr::OctetString(
200					std::slice::from_raw_parts(self.Values.pOctetString, self.ValueCount as _),
201				),
202				_ => panic!("Invalid ValueType."),
203			}
204		}
205	}
206}

249/// [`FILETIME`](https://learn.microsoft.com/en-us/windows/win32/api/minwinbase/ns-minwinbase-filetime)
250/// struct.
251///
252/// Can be converted to [`SYSTEMTIME`](crate::SYSTEMTIME) with
253/// [`FileTimeToSystemTime`](crate::FileTimeToSystemTime) function.
254#[repr(C)]
255#[derive(Default, Debug, Clone, Copy, PartialEq, Eq, Hash)]
256pub struct FILETIME {
257	pub dwLowDateTime: u32,
258	pub dwHighDateTime: u32,
259}
260
261impl From<u64> for FILETIME {
262	fn from(v: u64) -> Self {
263		Self {
264			dwLowDateTime: LODWORD(v),
265			dwHighDateTime: HIDWORD(v),
266		}
267	}
268}
269
270impl From<FILETIME> for u64 {
271	fn from(v: FILETIME) -> Self {
272		MAKEQWORD(v.dwLowDateTime, v.dwHighDateTime)
273	}
274}
275
276impl FILETIME {
277	/// Returns a new `FILETIME` with the milliseconds difference.
278	#[must_use]
279	pub const fn add_ms(self, ms: i64) -> Self {
280		let self64 = MAKEQWORD(self.dwLowDateTime, self.dwHighDateTime) as i64;
281		let new_self64 = self64 + (ms * 10_000);
282		Self {
283			dwLowDateTime: LODWORD(new_self64 as _),
284			dwHighDateTime: HIDWORD(new_self64 as _),
285		}
286	}
287
288	/// Returns a new `FILETIME` with the seconds difference.
289	#[must_use]
290	pub const fn add_secs(self, secs: i64) -> Self {
291		self.add_ms(secs * 1000)
292	}
293
294	/// Returns a new `FILETIME` with the minutes difference.
295	#[must_use]
296	pub const fn add_mins(self, mins: i64) -> Self {
297		self.add_secs(mins * 60)
298	}
299
300	/// Returns a new `FILETIME` with the hours difference.
301	#[must_use]
302	pub const fn add_hours(self, hours: i64) -> Self {
303		self.add_mins(hours * 60)
304	}
305
306	/// Returns a new `FILETIME` with the days difference.
307	#[must_use]
308	pub const fn add_days(self, days: i64) -> Self {
309		self.add_hours(days * 24)
310	}
311}

578	/// # Panics
579	///
580	/// Panics if `PowerSetting` identifier is invalid.
581	///
582	/// # Safety
583	///
584	/// Make sure the struct contains the correct size and data described by the
585	/// `PowerSetting` identifier.
586	#[must_use]
587	pub const unsafe fn data(&self) -> PowerSetting {
588		unsafe {
589			match self.PowerSetting {
590				co::POWER_SETTING::ACDC_POWER_SOURCE => {
591					PowerSetting::AcDcPowerSource(co::SYSTEM_POWER_CONDITION::from_raw(
592						std::slice::from_raw_parts(self.Data.as_ptr() as *const _, 1)[0],
593					))
594				},
595				co::POWER_SETTING::BATTERY_PERCENTAGE_REMAINING => {
596					PowerSetting::BatteryPercentageRemaining(
597						std::slice::from_raw_parts(self.Data.as_ptr() as *const u32, 1)[0] as _,
598					)
599				},
600				co::POWER_SETTING::CONSOLE_DISPLAY_STATE => {
601					PowerSetting::ConsoleDisplayState(co::MONITOR_DISPLAY_STATE::from_raw(
602						std::slice::from_raw_parts(self.Data.as_ptr() as *const _, 1)[0],
603					))
604				},
605				co::POWER_SETTING::GLOBAL_USER_PRESENCE => {
606					PowerSetting::GlobalUserPresence(co::USER_ACTIVITY_PRESENCE::from_raw(
607						std::slice::from_raw_parts(self.Data.as_ptr() as *const _, 1)[0],
608					))
609				},
610				co::POWER_SETTING::IDLE_BACKGROUND_TASK => PowerSetting::IdleBackgroundTask,
611				co::POWER_SETTING::MONITOR_POWER_ON => PowerSetting::MonitorPowerOn(
612					match std::slice::from_raw_parts(self.Data.as_ptr() as *const u32, 1)[0] {
613						0 => false,
614						_ => true,
615					},
616				),
617				co::POWER_SETTING::POWER_SAVING_STATUS => PowerSetting::PowerSavingStatus(
618					match std::slice::from_raw_parts(self.Data.as_ptr() as *const u32, 1)[0] {
619						0 => false,
620						_ => true,
621					},
622				),
623				co::POWER_SETTING::POWERSCHEME_PERSONALITY => PowerSetting::PowerSchemePersonality(
624					std::slice::from_raw_parts(self.Data.as_ptr() as *const co::POWER_SAVINGS, 1)
625						[0],
626				),
627				co::POWER_SETTING::SESSION_DISPLAY_STATUS => {
628					PowerSetting::SessionDisplayStatus(co::MONITOR_DISPLAY_STATE::from_raw(
629						std::slice::from_raw_parts(self.Data.as_ptr() as *const _, 1)[0],
630					))
631				},
632				co::POWER_SETTING::SESSION_USER_PRESENCE => {
633					PowerSetting::SessionUserPresence(co::USER_ACTIVITY_PRESENCE::from_raw(
634						std::slice::from_raw_parts(self.Data.as_ptr() as *const _, 1)[0],
635					))
636				},
637				co::POWER_SETTING::LIDSWITCH_STATE_CHANGE => PowerSetting::LidSwitchStateChange(
638					match std::slice::from_raw_parts(self.Data.as_ptr() as *const u8, 1)[0] {
639						0 => PowerSettingLid::Closed,
640						_ => PowerSettingLid::Opened,
641					},

645						0 => PowerSettingAwayMode::Exiting,
646						_ => PowerSettingAwayMode::Entering,
647					},
648				),

650			}
651		}
