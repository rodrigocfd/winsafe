3/// Builds a block of FFI bindings.
4macro_rules! extern_sys {
5	(
6		$dll:expr;
7		$(
8			$func:ident( $( $parm:ty ),* ) $( -> $ret:ty )?
9		)*
10	) => {
11		#[cfg_attr(not(feature = "raw-dylib"), link(name = $dll))]
12		#[cfg_attr(feature = "raw-dylib", link(name = $dll, kind = "raw-dylib"))]
13		unsafe extern "system" {
14			$(
15				pub(crate) fn $func( $( _x: $parm, )* ) $( -> $ret )?;
16			)*
17		}
18	};
19}
20pub(crate) use extern_sys;
