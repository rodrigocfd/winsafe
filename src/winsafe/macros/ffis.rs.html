3/// Builds one single FFI binding function.
4macro_rules! one_func {
5	($func:ident( $( $parm:ty ),* ) -> $ret:ty) => {
6		pub(crate) fn $func( $( _: $parm, )* ) -> $ret;
7	};
8
9	($func:ident( $( $parm:ty ),* )) => {
10		pub(crate) fn $func( $( _: $parm, )* );
11	};
12}
13
14/// Builds a block of FFI bindings.
15macro_rules! extern_sys {
16	(
17		$dll:expr;
18		$(
19			$func:ident( $( $parm:ty ),* ) $( -> $ret:ty )?
20		)*
21	) => {
22		#[cfg_attr(not(feature = "raw-dylib"), link(name = $dll))]
23		#[cfg_attr(feature = "raw-dylib", link(name = $dll, kind = "raw-dylib"))]
24		unsafe extern "system" {
25			$(
26				one_func!( $func( $( $parm ),* ) $(-> $ret)? );
27			)*
28		}
29	};
30}
