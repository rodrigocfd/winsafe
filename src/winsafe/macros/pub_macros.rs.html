1/// Generates sequential `u16` constants starting from the given value.
2///
3/// This macro is useful to generate constants for loaded resources, like menus
4/// or dialog windows.
5///
6/// Each constant also supports individual documentation.
7///
8/// # Examples
9///
10/// ```no_run
11/// use winsafe::seq_ids;
12///
13/// seq_ids! {
14///     MNU_FILE = 3000;
15///     MNU_FILE_OPEN
16///     MNU_FILE_SAVE
17///     /// This menu closes the application.
18///     MNU_FILE_CLOSE
19/// }
20/// ```
21///
22/// The code above will generate the following:
23///
24/// ```no_run
25/// pub const MNU_FILE: u16 = 3000;
26/// pub const MNU_FILE_OPEN: u16 = 3001;
27/// pub const MNU_FILE_SAVE: u16 = 3002;
28/// /// This menu closes the application.
29/// pub const MNU_FILE_CLOSE: u16 = 3003;
30/// ```
31#[cfg(feature = "user")]
32#[macro_export]
33macro_rules! seq_ids {
34	() => {};
35
36	($val:expr,) => {};
37
38	(
39		$( #[$comment:meta] )*
40		$name:ident = $val:expr;
41		$( $others:tt )*
42	) => {
43		$( #[$comment] )*
44		pub const $name: u16 = $val;
45		seq_ids!($val + 1, $( $others )*);
46	};
47
48	(
49		$next_val:expr,
50		$( #[$comment:meta] )*
51		$name:ident = $val:expr;
52		$( $others:tt )*
53	) => {
54		$( #[$comment] )*
55		pub const $name: u16 = $val;
56		seq_ids!($val + 1, $( $others )*);
57	};
58
59	(
60		$next_val:expr,
61		$( #[$comment:meta] )*
62		$name:ident
63		$( $others:tt )*
64	) => {
65		$( #[$comment] )*
66		pub const $name: u16 = $next_val;
67		seq_ids!($next_val + 1, $( $others )*);
68	};
69}

71/// Binds a function to an event which has no parameters.
72///
73/// This is just syntactic sugar to use a member function as an event closure.
74///
75/// # Example
76///
77/// ```no_run
78/// use winsafe::{self as w, prelude::*, gui, bind};
79///
80/// #[derive(Clone)]
81/// struct MyWindow {
82///     wnd: gui::WindowMain,
83/// }
84///
85/// impl MyWindow {
86///     fn new() -> Self {
87///         let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
88///         let new_self = Self { wnd };
89///         new_self.events();
90///         new_self
91///     }
92///
93///     fn events(&self) {
94///         self.wnd.on().wm_destroy(bind!(self, Self::on_destroy));
95///     }
96///
97///     fn on_destroy(&self) -> w::AnyResult<()> {
98///         Ok(())
99///     }
100/// }
101/// ```
102///
103/// # Related macros
104///
105/// * [`bind_ig`](crate::bind_ig)
106/// * [`bind_p`](crate::bind_p)
107#[cfg(feature = "gui")]
108#[macro_export]
109macro_rules! bind {
110	($arg:ident, $fun:expr) => {{
111		let arg2 = $arg.clone();
112		move || $fun(&arg2)
113	}};
114}

116/// Binds, to an event closure, a function receiving the event parameter.
117///
118/// This is just syntactic sugar to use a member function as an event closure.
119///
120/// # Example
121///
122/// ```no_run
123/// use winsafe::{self as w, prelude::*, gui, msg, bind_p};
124///
125/// #[derive(Clone)]
126/// struct MyWindow {
127///     wnd: gui::WindowMain,
128/// }
129///
130/// impl MyWindow {
131///     fn new() -> Self {
132///         let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
133///         let new_self = Self { wnd };
134///         new_self.events();
135///         new_self
136///     }
137///
138///     fn events(&self) {
139///         self.wnd.on().wm_create(bind_p!(self, Self::on_create));
140///     }
141///
142///     fn on_create(&self, p: msg::wm::Create) -> w::AnyResult<i32> {
143///         Ok(0)
144///     }
145/// }
146/// ```
147///
148/// # Related macros
149///
150/// * [`bind`](crate::bind)
151/// * [`bind_ig`](crate::bind_ig)
152#[cfg(feature = "gui")]
153#[macro_export]
154macro_rules! bind_p {
155	($arg:ident, $fun:expr) => {{
156		let arg2 = $arg.clone();
157		move |p| $fun(&arg2, p)
158	}};
159}
