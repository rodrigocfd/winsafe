21/// This trait is enabled with the `oleaut` feature, and provides methods for
22/// [`ITypeInfo`](crate::ITypeInfo).
23///
24/// Prefer importing this trait through the prelude:
25///
26/// ```no_run
27/// use winsafe::prelude::*;
28/// ```
29pub trait oleaut_ITypeInfo: ole_IUnknown {
30	/// [`ITypeInfo::AddressOfMember`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-itypeinfo-addressofmember)
31	/// method.
32	#[must_use]
33	fn AddressOfMember(
34		&self,
35		member_id: i32,
36		inv_kind: co::INVOKEKIND,
37	) -> HrResult<*mut std::ffi::c_void> {
38		let mut addr: *mut std::ffi::c_void = std::ptr::null_mut();
39		HrRet(unsafe {
40			(vt::<ITypeInfoVT>(self).AddressOfMember)(
41				self.ptr(),
42				member_id,
43				inv_kind.raw(),
44				&mut addr,
45			)
46		})
47		.to_hrresult()
48		.map(|_| addr)
49	}
50
51	/// [`ITypeInfo::CreateInstance`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-itypeinfo-createinstance)
52	/// method.
53	#[must_use]
54	fn CreateInstance<T>(&self, iunk_outer: Option<&impl ole_IUnknown>) -> HrResult<T>
55	where
56		T: ole_IUnknown,
57	{
58		let mut queried = unsafe { T::null() };
59		HrRet(unsafe {
60			(vt::<ITypeInfoVT>(self).CreateInstance)(
61				self.ptr(),
62				iunk_outer.map_or(std::ptr::null_mut(), |uo| uo.ptr()),
63				pcvoid(&T::IID),
64				queried.as_mut(),
65			)
66		})
67		.to_hrresult()
68		.map(|_| queried)
69	}
70
71	/// [`ITypeInfo::GetDllEntry`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-itypeinfo-getdllentry)
72	/// method.
73	///
74	/// Returns:
75	/// * DLL name;
76	/// * entry point name;
77	/// * ordinal.
78	#[must_use]
79	fn GetDllEntry(
80		&self,
81		member_id: i32,
82		inv_kind: co::INVOKEKIND,
83	) -> HrResult<(String, String, u16)> {
84		let (mut dll_name, mut name) = (BSTR::default(), BSTR::default());
85		let mut ordinal = 0u16;
86
87		HrRet(unsafe {
88			(vt::<ITypeInfoVT>(self).GetDllEntry)(
89				self.ptr(),
90				member_id,
91				inv_kind.raw(),
92				dll_name.as_mut_ptr(),
93				name.as_mut_ptr(),
94				&mut ordinal,
95			)
96		})
97		.to_hrresult()
98		.map(|_| {
99			let dll_name_str = dll_name.to_string();
100			let name_str = if name.as_ptr().is_null() { String::new() } else { name.to_string() };
101			(dll_name_str, name_str, ordinal)
102		})
103	}

105	/// [`ITypeInfo::GetDocumentation`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-itypeinfo-getdocumentation)
106	/// method.
107	///
108	/// Returns:
109	/// * item name;
110	/// * documentation;
111	/// * help localization context;
112	/// * fully qualified name of the file containing the DLL used for help file.
113	#[must_use]
114	fn GetDocumentation(&self, member_id: i32) -> HrResult<(String, String, u32, String)> {
115		let mut name = BSTR::default();
116		let mut doc = BSTR::default();
117		let mut context = 0u32;
118		let mut help_file = BSTR::default();
119
120		HrRet(unsafe {
121			(vt::<ITypeInfoVT>(self).GetDocumentation)(
122				self.ptr(),
123				member_id,
124				name.as_mut_ptr(),
125				doc.as_mut_ptr(),
126				&mut context,
127				help_file.as_mut_ptr(),
128			)
129		})
130		.to_hrresult()
131		.map(|_| (name.to_string(), doc.to_string(), context, help_file.to_string()))
132	}
133
134	/// [`ITypeInfo::GetIDsOfNames`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-itypeinfo-getidsofnames)
135	/// method.
136	#[must_use]
137	fn GetIDsOfNames(&self, names: &[impl AsRef<str>]) -> HrResult<Vec<i32>> {
138		let (_wstrs, pwstrs) = create_wstr_ptr_vecs(names);
139		let mut ids = vec![0i32; names.len()];
140
141		HrRet(unsafe {
142			(vt::<ITypeInfoVT>(self).GetIDsOfNames)(
143				self.ptr(),
144				vec_ptr(&pwstrs),
145				names.len() as _,
146				ids.as_mut_ptr() as _,
147			)
148		})
149		.to_hrresult()
150		.map(|_| ids)
151	}
