22/// This trait is enabled with the `oleaut` feature, and provides methods for
23/// [`IDispatch`](crate::IDispatch).
24///
25/// Prefer importing this trait through the prelude:
26///
27/// ```no_run
28/// use winsafe::prelude::*;
29/// ```
30pub trait oleaut_IDispatch: ole_IUnknown {
31	/// [`IDispatch::GetIDsOfNames`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-idispatch-getidsofnames)
32	/// method.
33	#[must_use]
34	fn GetIDsOfNames(&self, names: &[impl AsRef<str>], lcid: LCID) -> HrResult<Vec<i32>> {
35		let (_wstrs, pwstrs) = create_wstr_ptr_vecs(names);
36		let mut ids = vec![0i32; names.len()];
37
38		HrRet(unsafe {
39			(vt::<IDispatchVT>(self).GetIDsOfNames)(
40				self.ptr(),
41				pcvoid(&co::IID::default()),
42				vec_ptr(&pwstrs),
43				names.len() as _,
44				lcid.into(),
45				ids.as_mut_ptr() as _,
46			)
47		})
48		.to_hrresult()
49		.map(|_| ids)
50	}

79	/// [`IDispatch::Invoke`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-idispatch-invoke)
80	/// method.
81	///
82	/// If the remote call raises an exception, the returned error will be an
83	/// [`EXCEPINFO`](crate::EXCEPINFO).
84	///
85	/// This is a low-level method, prefer using:
86	/// * [`IDispatch::invoke_get`](crate::prelude::oleaut_IDispatch::invoke_get);
87	/// * [`IDispatch::invoke_method`](crate::prelude::oleaut_IDispatch::invoke_method); or
88	/// * [`IDispatch::invoke_put`](crate::prelude::oleaut_IDispatch::invoke_put).
89	fn Invoke(
90		&self,
91		disp_id_member: i32,
92		lcid: LCID,
93		flags: co::DISPATCH,
94		disp_params: &mut DISPPARAMS,
95	) -> AnyResult<VARIANT> {
96		let mut remote_res = VARIANT::default();
97		let mut remote_err = EXCEPINFO::default();
98		let mut arg_err = 0u32;
99
100		match HrRet(unsafe {
101			(vt::<IDispatchVT>(self).Invoke)(
102				self.ptr(),
103				disp_id_member,
104				pcvoid(&GUID::NULL),
105				lcid.raw(),
106				flags.raw(),
107				pvoid(disp_params),
108				pvoid(&mut remote_res),
109				pvoid(&mut remote_err),
110				&mut arg_err,
111			)
112		})
113		.to_hrresult()
114		{
115			Ok(_) => Ok(remote_res),
116			Err(hr) => match hr {
117				co::HRESULT::DISP_E_EXCEPTION => Err(Box::new(remote_err)),
118				_ => Err(hr.into()),
119			},
120		}
121	}

123	/// Calls
124	/// [`IDispatch::GetIDsOfNames`](crate::prelude::oleaut_IDispatch::GetIDsOfNames)
125	/// and [`IDispatch::Invoke`](crate::prelude::oleaut_IDispatch::Invoke) with
126	/// [`co::DISPATCH::PROPERTYGET`](co::DISPATCH::PROPERTYGET).
127	///
128	/// If the remote call raises an exception, the returned error will be an
129	/// [`EXCEPINFO`](crate::EXCEPINFO).
130	///
131	/// # Examples
132	///
133	/// ```no_run
134	/// use winsafe::{self as w, prelude::*, co};
135	///
136	/// let _com_guard = w::CoInitializeEx(
137	///     co::COINIT::APARTMENTTHREADED | co::COINIT::DISABLE_OLE1DDE)?;
138	///
139	/// let excel = w::CoCreateInstance::<w::IDispatch>(
140	///     &w::CLSIDFromProgID("Excel.Application")?,
141	///     None::<&w::IUnknown>,
142	///     co::CLSCTX::LOCAL_SERVER,
143	/// )?;
144	///
145	/// let books = excel.invoke_get("Workbooks", &[])?.unwrap_dispatch();
146	///
147	/// let workbook = books
148	///     .invoke_method("Open", &[&w::Variant::from_str("C:\\Temp\\bar.xlsx")])?
149	///     .unwrap_dispatch();
150	///
151	/// let worksheets = workbook.invoke_get("Worksheets", &[])?.unwrap_dispatch();
152	///
153	/// let sheet1 = worksheets
154	///     .invoke_get("Item", &[&w::Variant::from_str("Sheet1")])?
155	///     .unwrap_dispatch();
156	/// # w::AnyResult::Ok(())
157	/// ```
158	fn invoke_get(&self, property_name: &str, params: &[&Variant]) -> AnyResult<Variant> {
159		let member_ids = self.GetIDsOfNames(&[property_name], LCID::USER_DEFAULT)?;
160
161		let mut vars = params
162			.iter()
163			.rev() // in reverse order
164			.map(|param| param.to_raw())
165			.collect::<HrResult<Vec<_>>>()?;
166
167		let mut dp = DISPPARAMS::default();
168		dp.set_rvarg(Some(&mut vars));
169
170		let vari =
171			self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::PROPERTYGET, &mut dp)?;
172		Variant::from_raw(&vari).map_err(|err| err.into())
173	}

188	/// let _com_guard = w::CoInitializeEx(
189	///     co::COINIT::APARTMENTTHREADED | co::COINIT::DISABLE_OLE1DDE)?;

192	///     &w::CLSIDFromProgID("Excel.Application")?,
193	///     None::<&w::IUnknown>,

195	/// )?;
196	///
197	/// let books = excel.invoke_get("Workbooks", &[])?
198	///     .unwrap_dispatch();
199	///
200	/// let file = books.invoke_method(
201	///     "Open",
202	///     &[&w::Variant::from_str("C:\\Temp\\foo.xlsx")],
203	/// )?
204	/// .unwrap_dispatch();
205	///
206	/// file.invoke_method(
207	///     "SaveAs",
208	///     &[&w::Variant::from_str("C:\\Temp\\bar.xlsx")],
209	/// )?;
210	///
211	/// file.invoke_method("Close", &[])?;
212	/// # w::AnyResult::Ok(())
213	/// ```
214	fn invoke_method(&self, method_name: &str, params: &[&Variant]) -> AnyResult<Variant> {
215		let member_ids = self.GetIDsOfNames(&[method_name], LCID::USER_DEFAULT)?;
216
217		let mut vars = params
218			.iter()
219			.rev() // in reverse order
220			.map(|param| param.to_raw())
221			.collect::<HrResult<Vec<_>>>()?;
222
223		let mut dp = DISPPARAMS::default();
224		dp.set_rvarg(Some(&mut vars));
225
226		let vari = self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::METHOD, &mut dp)?;
227		Variant::from_raw(&vari).map_err(|err| err.into())
228	}

230	/// Calls
231	/// [`IDispatch::GetIDsOfNames`](crate::prelude::oleaut_IDispatch::GetIDsOfNames)
232	/// and [`IDispatch::Invoke`](crate::prelude::oleaut_IDispatch::Invoke) with
233	/// [`co::DISPATCH::PROPERTYPUT`](co::DISPATCH::PROPERTYPUT).
234	///
235	/// If the remote call raises an exception, the returned error will be an
236	/// [`EXCEPINFO`](crate::EXCEPINFO).
237	fn invoke_put(&self, property_name: &str, param: &Variant) -> AnyResult<Variant> {
238		let member_ids = self.GetIDsOfNames(&[property_name], LCID::USER_DEFAULT)?;
239
240		let mut vars = vec![param.to_raw()?]; // single parameter
241		let mut named_args = vec![co::DISPID::PROPERTYPUT];
242
243		let mut dp = DISPPARAMS::default();
244		dp.set_rvarg(Some(&mut vars));
245		dp.set_rgdispidNamedArgs(Some(&mut named_args));
246
247		let vari =
248			self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::PROPERTYPUT, &mut dp)?;
249		Variant::from_raw(&vari).map_err(|err| err.into())
250	}
