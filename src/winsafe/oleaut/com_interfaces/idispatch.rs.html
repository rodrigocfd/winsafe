21/// This trait is enabled with the `oleaut` feature, and provides methods for
22/// [`IDispatch`](crate::IDispatch).
23///
24/// Prefer importing this trait through the prelude:
25///
26/// ```no_run
27/// use winsafe::prelude::*;
28/// ```
29pub trait oleaut_IDispatch: ole_IUnknown {
30	/// [`IDispatch::GetIDsOfNames`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-idispatch-getidsofnames)
31	/// method.
32	#[must_use]
33	fn GetIDsOfNames(&self, names: &[impl AsRef<str>], lcid: LCID) -> HrResult<Vec<i32>> {
34		let (_wstrs, pwstrs) = create_wstr_ptr_vecs(names);
35		let mut ids = vec![0i32; names.len()];
36
37		HrRet(unsafe {
38			(vt::<IDispatchVT>(self).GetIDsOfNames)(
39				self.ptr(),
40				pcvoid(&co::IID::default()),
41				vec_ptr(&pwstrs),
42				names.len() as _,
43				lcid.into(),
44				ids.as_mut_ptr() as _,
45			)
46		})
47		.to_hrresult()
48		.map(|_| ids)
49	}

78	/// [`IDispatch::Invoke`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-idispatch-invoke)
79	/// method.
80	///
81	/// If the remote call raises an exception, the returned error will be an
82	/// [`EXCEPINFO`](crate::EXCEPINFO).
83	///
84	/// This is a low-level method, prefer using:
85	/// * [`IDispatch::invoke_get`](crate::prelude::oleaut_IDispatch::invoke_get);
86	/// * [`IDispatch::invoke_method`](crate::prelude::oleaut_IDispatch::invoke_method); or
87	/// * [`IDispatch::invoke_put`](crate::prelude::oleaut_IDispatch::invoke_put).
88	fn Invoke(
89		&self,
90		disp_id_member: i32,
91		lcid: LCID,
92		flags: co::DISPATCH,
93		disp_params: &mut DISPPARAMS,
94	) -> AnyResult<VARIANT> {
95		let mut remote_res = VARIANT::default();
96		let mut remote_err = EXCEPINFO::default();
97		let mut arg_err = 0u32;
98
99		match HrRet(unsafe {
100			(vt::<IDispatchVT>(self).Invoke)(
101				self.ptr(),
102				disp_id_member,
103				pcvoid(&GUID::NULL),
104				lcid.raw(),
105				flags.raw(),
106				pvoid(disp_params),
107				pvoid(&mut remote_res),
108				pvoid(&mut remote_err),
109				&mut arg_err,
110			)
111		})
112		.to_hrresult()
113		{
114			Ok(_) => Ok(remote_res),
115			Err(hr) => match hr {
116				co::HRESULT::DISP_E_EXCEPTION => Err(Box::new(remote_err)),
117				_ => Err(hr.into()),
118			},
119		}
120	}

122	/// Calls
123	/// [`IDispatch::GetIDsOfNames`](crate::prelude::oleaut_IDispatch::GetIDsOfNames)
124	/// and [`IDispatch::Invoke`](crate::prelude::oleaut_IDispatch::Invoke) with
125	/// [`co::DISPATCH::PROPERTYGET`](co::DISPATCH::PROPERTYGET).
126	///
127	/// If the remote call raises an exception, the returned error will be an
128	/// [`EXCEPINFO`](crate::EXCEPINFO).
129	///
130	/// # Examples
131	///
132	/// ```no_run
133	/// use winsafe::{self as w, prelude::*, co};
134	///
135	/// let _com_guard = w::CoInitializeEx(
136	///     co::COINIT::APARTMENTTHREADED | co::COINIT::DISABLE_OLE1DDE)?;
137	///
138	/// let excel = w::CoCreateInstance::<w::IDispatch>(
139	///     &w::CLSIDFromProgID("Excel.Application")?,
140	///     None::<&w::IUnknown>,
141	///     co::CLSCTX::LOCAL_SERVER,
142	/// )?;
143	///
144	/// let books = excel.invoke_get("Workbooks", &[])?.unwrap_dispatch();
145	///
146	/// let workbook = books
147	///     .invoke_method("Open", &[&w::Variant::from_str("C:\\Temp\\bar.xlsx")])?
148	///     .unwrap_dispatch();
149	///
150	/// let worksheets = workbook.invoke_get("Worksheets", &[])?.unwrap_dispatch();
151	///
152	/// let sheet1 = worksheets
153	///     .invoke_get("Item", &[&w::Variant::from_str("Sheet1")])?
154	///     .unwrap_dispatch();
155	/// # w::AnyResult::Ok(())
156	/// ```
157	fn invoke_get(&self, property_name: &str, params: &[&Variant]) -> AnyResult<Variant> {
158		let member_ids = self.GetIDsOfNames(&[property_name], LCID::USER_DEFAULT)?;
159
160		let mut vars = params
161			.iter()
162			.rev() // in reverse order
163			.map(|param| param.to_raw())
164			.collect::<HrResult<Vec<_>>>()?;
165
166		let mut dp = DISPPARAMS::default();
167		dp.set_rvarg(Some(&mut vars));
168
169		let vari =
170			self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::PROPERTYGET, &mut dp)?;
171		Variant::from_raw(&vari).map_err(|err| err.into())
172	}

175	/// [`IDispatch::GetIDsOfNames`](crate::prelude::oleaut_IDispatch::GetIDsOfNames)

182	/// # Examples
183	///
184	/// ```no_run
185	/// use winsafe::{self as w, prelude::*, co};

187	/// let _com_guard = w::CoInitializeEx(
188	///     co::COINIT::APARTMENTTHREADED | co::COINIT::DISABLE_OLE1DDE)?;

190	/// let excel = w::CoCreateInstance::<w::IDispatch>(
191	///     &w::CLSIDFromProgID("Excel.Application")?,
192	///     None::<&w::IUnknown>,
193	///     co::CLSCTX::LOCAL_SERVER,
194	/// )?;
195	///
196	/// let books = excel.invoke_get("Workbooks", &[])?
197	///     .unwrap_dispatch();
198	///
199	/// let file = books.invoke_method(
200	///     "Open",
201	///     &[&w::Variant::from_str("C:\\Temp\\foo.xlsx")],
202	/// )?
203	/// .unwrap_dispatch();
204	///
205	/// file.invoke_method(
206	///     "SaveAs",
207	///     &[&w::Variant::from_str("C:\\Temp\\bar.xlsx")],
208	/// )?;
209	///
210	/// file.invoke_method("Close", &[])?;
211	/// # w::AnyResult::Ok(())
212	/// ```
213	fn invoke_method(&self, method_name: &str, params: &[&Variant]) -> AnyResult<Variant> {
214		let member_ids = self.GetIDsOfNames(&[method_name], LCID::USER_DEFAULT)?;
215
216		let mut vars = params
217			.iter()
218			.rev() // in reverse order
219			.map(|param| param.to_raw())
220			.collect::<HrResult<Vec<_>>>()?;
221
222		let mut dp = DISPPARAMS::default();
223		dp.set_rvarg(Some(&mut vars));
224
225		let vari = self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::METHOD, &mut dp)?;
226		Variant::from_raw(&vari).map_err(|err| err.into())
227	}

229	/// Calls
230	/// [`IDispatch::GetIDsOfNames`](crate::prelude::oleaut_IDispatch::GetIDsOfNames)
231	/// and [`IDispatch::Invoke`](crate::prelude::oleaut_IDispatch::Invoke) with
232	/// [`co::DISPATCH::PROPERTYPUT`](co::DISPATCH::PROPERTYPUT).
233	///
234	/// If the remote call raises an exception, the returned error will be an
235	/// [`EXCEPINFO`](crate::EXCEPINFO).
236	fn invoke_put(&self, property_name: &str, param: &Variant) -> AnyResult<Variant> {
237		let member_ids = self.GetIDsOfNames(&[property_name], LCID::USER_DEFAULT)?;
238
239		let mut vars = vec![param.to_raw()?]; // single parameter
240		let mut named_args = vec![co::DISPID::PROPERTYPUT];
241
242		let mut dp = DISPPARAMS::default();
243		dp.set_rvarg(Some(&mut vars));
244		dp.set_rgdispidNamedArgs(Some(&mut named_args));
245
246		let vari =
247			self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::PROPERTYPUT, &mut dp)?;
248		Variant::from_raw(&vari).map_err(|err| err.into())
249	}
