21/// This trait is enabled with the `oleaut` feature, and provides methods for
22/// [`IDispatch`](crate::IDispatch).
23///
24/// Prefer importing this trait through the prelude:
25///
26/// ```no_run
27/// use winsafe::prelude::*;
28/// ```
29pub trait oleaut_IDispatch: ole_IUnknown {
30	/// [`IDispatch::GetIDsOfNames`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-idispatch-getidsofnames)
31	/// method.
32	#[must_use]
33	fn GetIDsOfNames(&self, names: &[impl AsRef<str>], lcid: LCID) -> HrResult<Vec<i32>> {
34		let (_wstrs, pwstrs) = create_wstr_ptr_vecs(Some(names));
35		let mut ids = vec![i32::default(); names.len()];
36
37		ok_to_hrresult(unsafe {
38			(vt::<IDispatchVT>(self).GetIDsOfNames)(
39				self.ptr(),
40				pcvoid(&co::IID::default()),
41				vec_ptr(&pwstrs),
42				names.len() as _,
43				lcid.into(),
44				ids.as_mut_ptr() as _,
45			)
46		})
47		.map(|_| ids)
48	}

77	/// [`IDispatch::Invoke`](https://learn.microsoft.com/en-us/windows/win32/api/oaidl/nf-oaidl-idispatch-invoke)
78	/// method.
79	///
80	/// If the remote call raises an exception, the returned error will be an
81	/// [`EXCEPINFO`](crate::EXCEPINFO).
82	///
83	/// This is a low-level method, prefer using:
84	/// * [`IDispatch::invoke_get`](crate::prelude::oleaut_IDispatch::invoke_get);
85	/// * [`IDispatch::invoke_method`](crate::prelude::oleaut_IDispatch::invoke_method); or
86	/// * [`IDispatch::invoke_put`](crate::prelude::oleaut_IDispatch::invoke_put).
87	fn Invoke(
88		&self,
89		disp_id_member: i32,
90		lcid: LCID,
91		flags: co::DISPATCH,
92		disp_params: &mut DISPPARAMS,
93	) -> AnyResult<VARIANT> {
94		let mut remote_res = VARIANT::default();
95		let mut remote_err = EXCEPINFO::default();
96		let mut arg_err = u32::default();
97
98		match ok_to_hrresult(unsafe {
99			(vt::<IDispatchVT>(self).Invoke)(
100				self.ptr(),
101				disp_id_member,
102				pcvoid(&GUID::NULL),
103				lcid.raw(),
104				flags.raw(),
105				pvoid(disp_params),
106				pvoid(&mut remote_res),
107				pvoid(&mut remote_err),
108				&mut arg_err,
109			)
110		}) {
111			Ok(_) => Ok(remote_res),
112			Err(hr) => match hr {
113				co::HRESULT::DISP_E_EXCEPTION => Err(Box::new(remote_err)),
114				_ => Err(hr.into()),
115			},
116		}
117	}

119	/// Calls
120	/// [`IDispatch::GetIDsOfNames`](crate::prelude::oleaut_IDispatch::GetIDsOfNames)
121	/// and [`IDispatch::Invoke`](crate::prelude::oleaut_IDispatch::Invoke) with
122	/// [`co::DISPATCH::PROPERTYGET`](co::DISPATCH::PROPERTYGET).
123	///
124	/// If the remote call raises an exception, the returned error will be an
125	/// [`EXCEPINFO`](crate::EXCEPINFO).
126	///
127	/// # Examples
128	///
129	/// ```no_run
130	/// use winsafe::{self as w, prelude::*, co};
131	///
132	/// let _com_guard = w::CoInitializeEx(
133	///     co::COINIT::APARTMENTTHREADED | co::COINIT::DISABLE_OLE1DDE)?;
134	///
135	/// let excel = w::CoCreateInstance::<w::IDispatch>(
136	///     &w::CLSIDFromProgID("Excel.Application")?,
137	///     None,
138	///     co::CLSCTX::LOCAL_SERVER,
139	/// )?;
140	///
141	/// let books = excel.invoke_get("Workbooks", &[])?.unwrap_dispatch();
142	///
143	/// let workbook = books
144	///     .invoke_method("Open", &[&w::Variant::from_str("C:\\Temp\\bar.xlsx")])?
145	///     .unwrap_dispatch();
146	///
147	/// let worksheets = workbook.invoke_get("Worksheets", &[])?.unwrap_dispatch();
148	///
149	/// let sheet1 = worksheets
150	///     .invoke_get("Item", &[&w::Variant::from_str("Sheet1")])?
151	///     .unwrap_dispatch();
152	/// # w::AnyResult::Ok(())
153	/// ```
154	fn invoke_get(&self, property_name: &str, params: &[&Variant]) -> AnyResult<Variant> {
155		let member_ids = self.GetIDsOfNames(&[property_name], LCID::USER_DEFAULT)?;
156
157		let mut vars = params
158			.iter()
159			.rev() // in reverse order
160			.map(|param| param.to_raw())
161			.collect::<HrResult<Vec<_>>>()?;
162
163		let mut dp = DISPPARAMS::default();
164		dp.set_rvarg(Some(&mut vars));
165
166		let vari =
167			self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::PROPERTYGET, &mut dp)?;
168		Variant::from_raw(&vari).map_err(|err| err.into())
169	}

171	/// Calls
172	/// [`IDispatch::GetIDsOfNames`](crate::prelude::oleaut_IDispatch::GetIDsOfNames)
173	/// and [`IDispatch::Invoke`](crate::prelude::oleaut_IDispatch::Invoke) with
174	/// [`co::DISPATCH::METHOD`](co::DISPATCH::METHOD).
175	///
176	/// If the remote call raises an exception, the returned error will be an
177	/// [`EXCEPINFO`](crate::EXCEPINFO).
178	///
179	/// # Examples
180	///
181	/// ```no_run
182	/// use winsafe::{self as w, prelude::*, co};
183	///
184	/// let _com_guard = w::CoInitializeEx(
185	///     co::COINIT::APARTMENTTHREADED | co::COINIT::DISABLE_OLE1DDE)?;
186	///
187	/// let excel = w::CoCreateInstance::<w::IDispatch>(
188	///     &w::CLSIDFromProgID("Excel.Application")?,
189	///     None,
190	///     co::CLSCTX::LOCAL_SERVER,
191	/// )?;
192	///
193	/// let books = excel.invoke_get("Workbooks", &[])?
194	///     .unwrap_dispatch();
195	///
196	/// let file = books.invoke_method(
197	///     "Open",
198	///     &[&w::Variant::from_str("C:\\Temp\\foo.xlsx")],
199	/// )?
200	/// .unwrap_dispatch();
201	///
202	/// file.invoke_method(
203	///     "SaveAs",
204	///     &[&w::Variant::from_str("C:\\Temp\\bar.xlsx")],
205	/// )?;
206	///
207	/// file.invoke_method("Close", &[])?;
208	/// # w::AnyResult::Ok(())
209	/// ```
210	fn invoke_method(&self, method_name: &str, params: &[&Variant]) -> AnyResult<Variant> {
211		let member_ids = self.GetIDsOfNames(&[method_name], LCID::USER_DEFAULT)?;
212
213		let mut vars = params
214			.iter()
215			.rev() // in reverse order
216			.map(|param| param.to_raw())
217			.collect::<HrResult<Vec<_>>>()?;
218
219		let mut dp = DISPPARAMS::default();
220		dp.set_rvarg(Some(&mut vars));
221
222		let vari = self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::METHOD, &mut dp)?;
223		Variant::from_raw(&vari).map_err(|err| err.into())
224	}

226	/// Calls
227	/// [`IDispatch::GetIDsOfNames`](crate::prelude::oleaut_IDispatch::GetIDsOfNames)
228	/// and [`IDispatch::Invoke`](crate::prelude::oleaut_IDispatch::Invoke) with
229	/// [`co::DISPATCH::PROPERTYPUT`](co::DISPATCH::PROPERTYPUT).
230	///
231	/// If the remote call raises an exception, the returned error will be an
232	/// [`EXCEPINFO`](crate::EXCEPINFO).
233	fn invoke_put(&self, property_name: &str, param: &Variant) -> AnyResult<Variant> {
234		let member_ids = self.GetIDsOfNames(&[property_name], LCID::USER_DEFAULT)?;
235
236		let mut vars = vec![param.to_raw()?]; // single parameter
237		let mut named_args = vec![co::DISPID::PROPERTYPUT];
238
239		let mut dp = DISPPARAMS::default();
240		dp.set_rvarg(Some(&mut vars));
241		dp.set_rgdispidNamedArgs(Some(&mut named_args));
242
243		let vari =
244			self.Invoke(member_ids[0], LCID::USER_DEFAULT, co::DISPATCH::PROPERTYPUT, &mut dp)?;
245		Variant::from_raw(&vari).map_err(|err| err.into())
246	}
