7/// High-level representation of the [`PROPVARIANT`](crate::PROPVARIANT) struct,
8/// which is automatically converted into its low-level representation when
9/// needed.
10///
11/// Not to be confused with the similar [`Variant`].
12///
13/// The fields are named according to the
14/// [`VARENUM`](https://learn.microsoft.com/en-us/windows/win32/api/wtypes/ne-wtypes-varenum)
15/// enumeration.
16#[derive(Clone)]
17pub enum PropVariant {
18	/// Nothing.
19	Empty,
20	/// 2 byte signed int (`i16`).
21	I2(i16),
22	/// 4 byte signed int (`i32`).
23	I4(i32),
24	/// 4 byte real (`f32`).
25	R4(f32),
26	/// 8 byte real (`f64`).
27	R8(f64),
28	/// Date.
29	Date(SYSTEMTIME),
30	/// OLE Automation string.
31	Bstr(String),
32	/// [`IDispatch`](crate::IDispatch) pointer.
33	Dispatch(IDispatch),
34	/// Boolean value (`bool`).
35	Bool(bool),
36	/// [`IUnknown`](crate::IUnknown) pointer.
37	Unknown(IUnknown),
38	/// Signed char (`i8`).
39	I1(i8),
40	/// Unsigned char (`u8`).
41	UI1(u8),
42	/// Unsigned short (`u16`).
43	UI2(u16),
44	/// Unsigned long (`u32`).
45	UI4(u32),
46	/// Signed 64-bit int (`i64`).
47	I8(i64),
48	/// Unsigned 64-bit int (`u64`).
49	UI8(u64),
50}

77	#[must_use]
78	pub(crate) fn from_raw(v: &PROPVARIANT) -> HrResult<Self> {
79		Ok(match v.vt {
80			co::VT::EMPTY => Self::Empty,
81			co::VT::I2 => Self::I2(unsafe { v.data.iVal }),
82			co::VT::I4 => Self::I4(unsafe { v.data.lVal }),
83			co::VT::R4 => Self::R4(unsafe { v.data.fltVal }),
84			co::VT::R8 => Self::R8(unsafe { v.data.dblVal }),
85			co::VT::DATE => {
86				let st = VariantTimeToSystemTime(unsafe { v.data.dblVal })
87					.map_err(|err| err.to_hresult())?;
88				Self::Date(st)
89			},
90			co::VT::BSTR => {
91				let bstr = ManuallyDrop::new(unsafe { BSTR::from_ptr(v.data.ptr as _) }); // won't release the stored pointer
92				Self::Bstr(bstr.to_string())
93			},
94			co::VT::DISPATCH => {
95				let obj = ManuallyDrop::new(unsafe { IDispatch::from_ptr(v.data.ptr as _) }); // won't release the stored pointer
96				let cloned = IDispatch::clone(&obj); // call AddRef
97				Self::Dispatch(cloned)
98			},
99			co::VT::BOOL => Self::Bool(unsafe { v.data.iVal != 0 }),
100			co::VT::UNKNOWN => {
101				let obj = ManuallyDrop::new(unsafe { IUnknown::from_ptr(v.data.ptr as _) }); // won't release the stored pointer
102				let cloned = IUnknown::clone(&obj); // call AddRef
103				Self::Unknown(cloned)
104			},
105			co::VT::I1 => Self::I1(unsafe { v.data.cVal }),
106			co::VT::UI1 => Self::UI1(unsafe { v.data.bVal }),
107			co::VT::UI2 => Self::UI2(unsafe { v.data.uiVal }),
108			co::VT::UI4 => Self::UI4(unsafe { v.data.ulVal }),
109			co::VT::I8 => Self::I8(unsafe { v.data.hVal }),
110			co::VT::UI8 => Self::UI8(unsafe { v.data.uhVal }),
111			_ => panic!("PROPVARIANT type not implemented."),
112		})
113	}

116	pub(crate) fn to_raw(&self) -> HrResult<PROPVARIANT> {
117		let mut v = PROPVARIANT::default();
118		match self {
119			Self::Empty => {},
120			Self::I2(n) => {
121				v.vt = co::VT::I2;
122				v.data.iVal = *n;
123			},
124			Self::I4(n) => {
125				v.vt = co::VT::I4;
126				v.data.lVal = *n;
127			},
128			Self::R4(n) => {
129				v.vt = co::VT::R4;
130				v.data.fltVal = *n;
131			},
132			Self::R8(n) => {
133				v.vt = co::VT::R8;
134				v.data.dblVal = *n;
135			},
136			Self::Date(st) => {
137				v.vt = co::VT::DATE;
138				v.data.dblVal = SystemTimeToVariantTime(st).map_err(|err| err.to_hresult())?;
139			},
140			Self::Bstr(s) => {
141				v.vt = co::VT::BSTR;
142				let mut bstr = BSTR::SysAllocString(s)?;
143				v.data.ptr = bstr.leak() as _; // the VARIANT will own the pointer
144			},
145			Self::Dispatch(p) => {
146				v.vt = co::VT::DISPATCH;
147				let mut cloned = p.clone(); // call AddRef
148				v.data.ptr = cloned.leak(); // the VARIANT will own the pointer
149			},
150			Self::Bool(b) => {
151				v.vt = co::VT::BOOL;
152				v.data.iVal = if *b { -1 } else { 0 };
153			},
154			Self::Unknown(p) => {
155				v.vt = co::VT::UNKNOWN;
156				let mut cloned = p.clone(); // call AddRef
157				v.data.ptr = cloned.leak(); // the VARIANT will own the pointer
158			},
159			Self::I1(n) => {
160				v.vt = co::VT::I1;
161				v.data.cVal = *n;
162			},
163			Self::UI1(n) => {
164				v.vt = co::VT::UI1;
165				v.data.bVal = *n;
166			},
167			Self::UI2(n) => {
168				v.vt = co::VT::UI2;
169				v.data.uiVal = *n;
170			},
171			Self::UI4(n) => {
172				v.vt = co::VT::UI4;
173				v.data.ulVal = *n;
174			},
175			Self::I8(n) => {
176				v.vt = co::VT::I8;
177				v.data.hVal = *n;
178			},
179			Self::UI8(n) => {
180				v.vt = co::VT::UI8;
181				v.data.uhVal = *n;
182			},
183		}
184		Ok(v)
185	}

367	/// If the value is [`Variant::Bool`](crate::Variant::Bool), returns it;
368	/// otherwise panics.
369	///
370	/// This is a syntactic sugar method to be used when you are sure of the
371	/// `Variant` content, its general use is discouraged.
372	///
373	/// # Panics
374	///
375	/// Panics if the value is not `Variant::Bool`.
376	#[must_use]
377	pub const fn unwrap_bool(&self) -> bool {
378		match self {
379			Self::Bool(b) => *b,
380			_ => panic!("Variant does not contain Bool."),
381		}
382	}

401	/// If the value is [`Variant::Dispatch`](crate::Variant::Dispatch), returns
402	/// a clone of it; otherwise panics.
403	///
404	/// This is a syntactic sugar method to be used when you are sure of the
405	/// `Variant` content, its general use is discouraged.
406	///
407	/// # Panics
408	///
409	/// Panics if the value is not [`IDispatch`](crate::IDispatch).
410	///
411	/// # Examples
412	///
413	/// ```no_run
414	/// use winsafe::{self as w, prelude::*, co};
415	///
416	/// let _com_guard = w::CoInitializeEx(
417	///     co::COINIT::APARTMENTTHREADED | co::COINIT::DISABLE_OLE1DDE)?;
418	///
419	/// let excel = w::CoCreateInstance::<w::IDispatch>(
420	///     &w::CLSIDFromProgID("Excel.Application")?,
421	///     None::<&w::IUnknown>,
422	///     co::CLSCTX::LOCAL_SERVER,
423	/// )?;
424	///
425	/// let books = excel.invoke_get("Workbooks", &[])?.unwrap_dispatch();
426	/// # w::AnyResult::Ok(())
427	/// ```
428	#[must_use]
429	pub fn unwrap_dispatch(&self) -> IDispatch {
430		match self {
431			Self::Dispatch(disp) => disp.clone(),
432			_ => panic!("Variant does not contain Dispatch."),
433		}
434	}
