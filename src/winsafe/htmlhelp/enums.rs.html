4/// Variable parameter for:
5///
6/// * [`HWND::HtmlHelp`](crate::HWND::HtmlHelp)
7pub enum HhCmd<'a> {
8	/// [`HH_CLOSE_ALL`](https://learn.microsoft.com/en-us/previous-versions/windows/desktop/htmlhelp/hh-close-all-command)
9	/// command.
10	CloseAll,
11	/// [`HH_DISPLAY_INDEX`](https://learn.microsoft.com/en-us/previous-versions/windows/desktop/htmlhelp/hh-display-index-command)
12	/// command.
13	///
14	/// Receives the keyword to select in the index file.
15	DisplayIndex(&'a str),
16	/// [`HH_DISPLAY_TOC`](https://learn.microsoft.com/en-us/previous-versions/windows/desktop/htmlhelp/hh-display-toc-command)
17	/// command.
18	DisplayToc,
19	/// [`HH_TP_HELP_CONTEXTMENU`](https://learn.microsoft.com/en-us/previous-versions/windows/desktop/htmlhelp/hh-tp-help-contextmenu-command)
20	/// command.
21	///
22	/// Receives pairs composed of:
23	/// - dialog box control ID;
24	/// - help topic ID.
25	TpHelpContextMenu(&'a [(u16, u16)]),
26	/// [`HH_TP_HELP_WM_HELP`](https://learn.microsoft.com/en-us/previous-versions/windows/desktop/htmlhelp/hh-tp-help-wm-help-command)
27	/// command.
28	///
29	/// Receives pairs composed of:
30	/// - dialog box control ID;
31	/// - help topic ID.
32	TpHelpWmHelp(&'a [(u16, u16)]),
33}
34
35impl<'a> HhCmd<'a> {
36	/// Returns the [`co::HH`](crate::co::HH) constant and serialized data. So
37	/// far, only a limited subset has been implemented. The ones involving
38	/// pointers are still up to debate.
39	#[must_use]
40	pub(in crate::htmlhelp) fn as_data(&self) -> (co::HH, HhCmdData) {
41		use HhCmd::*;
42		match self {
43			CloseAll => (co::HH::CLOSE_ALL, HhCmdData::None),
44			DisplayIndex(s) => (co::HH::DISPLAY_INDEX, Self::gen_str(s)),
45			DisplayToc => (co::HH::DISPLAY_TOC, HhCmdData::None),
46			TpHelpContextMenu(arr) => (co::HH::TP_HELP_CONTEXTMENU, Self::gen_vec(arr)),
47			TpHelpWmHelp(arr) => (co::HH::TP_HELP_WM_HELP, Self::gen_vec(arr)),
48		}
49	}
50
51	#[must_use]
52	fn gen_str(s: &str) -> HhCmdData {
53		HhCmdData::Str(WString::from_str(s))
54	}
55
56	#[must_use]
57	fn gen_vec(array: &[(u16, u16)]) -> HhCmdData {
58		let mut buf = Vec::<u32>::with_capacity(array.len() * 2 + 1);
59		for (ctrl_id, topic_id) in array.iter() {
60			buf.push(*ctrl_id as _);
61			buf.push(*topic_id as _);
62		}
63		buf.push(0);
64		HhCmdData::Ids(buf)
65	}
66}

68/// Serialized data of [`HhCmd`].
69pub(in crate::htmlhelp) enum HhCmdData {
70	None,
71	Str(WString),
72	Ids(Vec<u32>),
73}
74
75impl HhCmdData {
76	#[must_use]
77	pub(in crate::htmlhelp) fn serialize(&self) -> usize {
78		use HhCmdData::*;
79		match self {
80			None => 0,
81			Str(s) => s.as_ptr() as _,
82			Ids(arr) => arr.as_ptr() as _,
83		}
84	}
