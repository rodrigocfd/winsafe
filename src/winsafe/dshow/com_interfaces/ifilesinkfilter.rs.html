29	/// [`IFileSinkFilter::GetCurFile`](https://learn.microsoft.com/en-us/windows/win32/api/strmif/nf-strmif-ifilesinkfilter-getcurfile)
30	/// method.
31	///
32	/// # Safety
33	///
34	/// If you pass an [`AM_MEDIA_TYPE`](crate::AM_MEDIA_TYPE) reference to
35	/// `pmt`, its `pbFormat` field may return a valid reference to a format
36	/// block. If so, you must free it with
37	/// [`CoTaskMemFree`](crate::guard::CoTaskMemFreeGuard), or you'll have a
38	/// memory leak.
39	///
40	/// # Examples
41	///
42	/// ```no_run
43	/// use winsafe::{self as w, prelude::*, guard};
44	///
45	/// let sinkf: w::IFileSinkFilter; // initialized somewhere
46	/// # let sinkf = unsafe { w::IFileSinkFilter::null() };
47	///
48	/// let mut ammt = w::AM_MEDIA_TYPE::default();
49	/// unsafe {
50	///     sinkf.GetCurFile(Some(&mut ammt))?;
51	///     if let Some(pb_format) = ammt.pbFormat::<w::DVINFO>() { // valid reference?
52	///         let _ = guard::CoTaskMemFreeGuard::new(pb_format as *mut _  as _, 0);
53	///     }
54	/// }
55	/// # w::HrResult::Ok(())
56	/// ```
57	#[must_use]
58	unsafe fn GetCurFile(&self, mt: Option<&mut AM_MEDIA_TYPE>) -> HrResult<String> {
59		let mut pstr = std::ptr::null_mut::<u16>();
60		HrRet(unsafe {
61			(vt::<IFileSinkFilterVT>(self).GetCurFile)(self.ptr(), &mut pstr, pvoid_or_null(mt))
62		})
63		.to_hrresult()
64		.map(|_| htaskmem_ptr_to_str(pstr))
65	}
