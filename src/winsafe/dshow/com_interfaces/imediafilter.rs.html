31	/// [`IMediaFilter::GetState`](https://learn.microsoft.com/en-us/windows/win32/api/strmif/nf-strmif-imediafilter-getstate)
32	/// method.
33	#[must_use]
34	fn GetState(&self, ms_timeout: Option<u32>) -> HrResult<co::FILTER_STATE> {
35		let mut fs = co::FILTER_STATE::default();
36		HrRet(unsafe {
37			(vt::<IMediaFilterVT>(self).GetState)(
38				self.ptr(),
39				ms_timeout.unwrap_or(INFINITE),
40				fs.as_mut(),
41			)
42		})
43		.to_hrresult()
44		.map(|_| fs)
45	}
