50	/// [`IBaseFilter::FindPin`](https://learn.microsoft.com/en-us/windows/win32/api/strmif/nf-strmif-ibasefilter-findpin)
51	/// method.
52	#[must_use]
53	fn FindPin(&self, id: &str) -> HrResult<IPin> {
54		let mut queried = unsafe { IPin::null() };
55		ok_to_hrresult(unsafe {
56			(vt::<IBaseFilterVT>(self).FindPin)(
57				self.ptr(),
58				WString::from_str(id).as_ptr(),
59				queried.as_mut(),
60			)
61		})
62		.map(|_| queried)
63	}

65	/// [`IBaseFilter::JoinFilterGraph`](https://learn.microsoft.com/en-us/windows/win32/api/strmif/nf-strmif-ibasefilter-joinfiltergraph)
66	/// method.
67	fn JoinFilterGraph(&self, graph: Option<&impl dshow_IFilterGraph>, name: &str) -> HrResult<()> {
68		ok_to_hrresult(unsafe {

71				graph.map_or(std::ptr::null_mut(), |g| g.ptr()),
72				WString::from_str(name).as_ptr(),

74		})
75	}

85	/// [`IBaseFilter::QueryVendorInfo`](https://learn.microsoft.com/en-us/windows/win32/api/strmif/nf-strmif-ibasefilter-queryvendorinfo)
86	/// method.
87	#[must_use]
88	fn QueryVendorInfo(&self) -> HrResult<String> {
89		let mut pstr = std::ptr::null_mut::<u16>();
90		ok_to_hrresult(unsafe {
91			(vt::<IBaseFilterVT>(self).QueryVendorInfo)(self.ptr(), &mut pstr)
92		})
93		.map(|_| {
