49	/// [`IBaseFilter::FindPin`](https://learn.microsoft.com/en-us/windows/win32/api/strmif/nf-strmif-ibasefilter-findpin)
50	/// method.
51	#[must_use]
52	fn FindPin(&self, id: &str) -> HrResult<IPin> {
53		let mut queried = unsafe { IPin::null() };
54		HrRet(unsafe {
55			(vt::<IBaseFilterVT>(self).FindPin)(
56				self.ptr(),
57				WString::from_str(id).as_ptr(),
58				queried.as_mut(),
59			)
60		})
61		.to_hrresult()
62		.map(|_| queried)
63	}
