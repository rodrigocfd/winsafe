10/// Identifies whether a window is dialog-based.

21		match self {
22			Self::Raw => co::WM::CREATE,

37/// Global horizontal and vertical system DPI factor, cached.
38pub(in crate::gui) static mut DPI: (i32, i32) = (0, 0);
39
40/// Stuff that should run before anything else.
41pub(in crate::gui) fn initial_gui_setup() {
42	if unsafe { DPI } == (0, 0) {
43		if IsWindowsVistaOrGreater().expect(DONTFAIL) {
44			SetProcessDPIAware().expect(DONTFAIL);
45		}
46
47		InitCommonControls();
48
49		if IsWindows8OrGreater().expect(DONTFAIL) {
50			// https://github.com/rodrigocfd/winsafe-examples/issues/6
51			let mut b_val = 0; // FALSE
52			match unsafe {
53				HPROCESS::GetCurrentProcess().SetUserObjectInformation(
54					co::UOI::TIMERPROC_EXCEPTION_SUPPRESSION, // SetTimer() safety
55					&mut b_val,
56				)
57			} {
58				Err(e) if e == co::ERROR::INVALID_PARAMETER => {
59					// Do nothing: Wine doesn't support SetUserObjectInformation for now.
60					// https://bugs.winehq.org/show_bug.cgi?id=54951
61				},
62				Err(e) => panic!("TIMERPROC_EXCEPTION_SUPPRESSION failed: {e:?}"), // should never happen
63				_ => {},
64			}
65		}
66
67		let hdc_screen = HWND::NULL.GetDC().expect(DONTFAIL);
68		let x = hdc_screen.GetDeviceCaps(co::GDC::LOGPIXELSX);
69		let y = hdc_screen.GetDeviceCaps(co::GDC::LOGPIXELSY);
70		unsafe {
71			DPI = (x, y);
72		}
73	}
74}

76pub(in crate::gui) mod quit_error {
77	use std::error::Error;
78	use std::sync::Mutex;
79
80	use crate::decl::*;
81	use crate::gui::*;
82	use crate::msg::*;
83
84	/// Global return error originated from an event handling closure; will be taken
85	/// in main loop.
86	pub(in crate::gui) static QUIT_ERROR: Mutex<Option<MsgError>> = Mutex::new(None);
87
88	/// Calls `PostQuitMessage` to terminate the program with the given error.
89	pub(in crate::gui) fn post_quit_error(src_msg: WndMsg, err: Box<dyn Error + Send + Sync>) {
90		{
91			let mut msg_error = QUIT_ERROR.lock().unwrap();
92			*msg_error = Some(MsgError::new(src_msg, err)); // store the error, so the main window loop can grab it
93		};
94		PostQuitMessage(-1); // this -1 will be discarded in the main loop
95	}
96}

98pub(in crate::gui) mod ui_font {
99	use crate::co;
100	use crate::decl::*;
101	use crate::guard::*;
102	use crate::gui::privs::*;
103	use crate::msg::*;
104
105	/// Global UI font object, cached.
106	static mut UI_HFONT: Option<DeleteObjectGuard<HFONT>> = None;
107
108	// Returns the global UI font, creating it of not yet.
109	pub(in crate::gui) fn get() -> HFONT {
110		unsafe {
111			match &*&raw const UI_HFONT {
112				None => {
113					let mut ncm = NONCLIENTMETRICS::default(); // create the font once
114					SystemParametersInfo(
115						co::SPI::GETNONCLIENTMETRICS,
116						std::mem::size_of::<NONCLIENTMETRICS>() as _,
117						&mut ncm,
118						co::SPIF::NoValue,
119					)
120					.expect(DONTFAIL);
121
122					let font = HFONT::CreateFontIndirect(&ncm.lfMenuFont).expect(DONTFAIL);
123					let ret_font = font.raw_copy();
124					UI_HFONT = Some(font);
125					ret_font
126				},
127				Some(font) => font.raw_copy(),
128			}
129		}
130	}
131
132	/// Sets the global UI font on the given window.
133	pub(in crate::gui) fn set(hwnd: &HWND) {
134		unsafe {
135			hwnd.SendMessage(wm::SetFont { hfont: get(), redraw: true });
136		}
137	}
138
139	/// Frees the global UI font object.
140	pub(in crate::gui) fn delete() {
141		unsafe {
142			UI_HFONT = None; // https://users.rust-lang.org/t/why-drop-trait-not-called-when-use-global-static
143		}
144	}
145}

147pub(in crate::gui) mod auto_id {
148	/// Next auto control ID to be assigned to controls without one.
149	static mut BASE_CTRL_ID: u16 = 0xdfff; // https://stackoverflow.com/a/18192766/6923555
150
151	/// Returns the next sequential control ID.
152	#[must_use]
153	pub(in crate::gui) const fn next() -> u16 {
154		unsafe {
155			let ret_id = BASE_CTRL_ID;
156			BASE_CTRL_ID -= 1; // go down
157			ret_id
158		}
159	}
160
161	/// If the given ID is zero, returns the next sequential control ID.
162	#[must_use]
163	pub(in crate::gui) const fn set_if_zero(id: u16) -> u16 {
164		if id == 0 { next() } else { id }
165	}
166}

169	use crate::co;

173	/// "&He && she" becomes "He & she".
174	#[must_use]
175	pub(in crate::gui) fn remove_accel_ampersands(text: &str) -> String {
176		let mut txt_no_ampersands = String::with_capacity(text.len());
177		let mut last_ch = 'a'; // initial value will be skipped
178
179		for (idx, ch) in text.char_indices() {
180			if idx == 0 {
181				// first char
182				if ch != '&' {
183					txt_no_ampersands.push(ch);
184				}
185			} else if ch != '&' || (ch == '&' && last_ch == '&') {
186				txt_no_ampersands.push(ch);
187			}
188			last_ch = ch;
189		}
190
191		txt_no_ampersands
192	}
193
194	/// Calculates the bound rectangle to fit the text with current system font.
195	#[must_use]
196	pub(in crate::gui) fn bound_box(text: &str) -> SIZE {
197		let desktop_hwnd = HWND::GetDesktopWindow();
198		let desktop_hdc = desktop_hwnd.GetDC().expect(DONTFAIL);
199		let clone_dc = desktop_hdc.CreateCompatibleDC().expect(DONTFAIL);
200		let _prev_font = clone_dc.SelectObject(&ui_font::get()).expect(DONTFAIL);
201
202		let mut bounds = clone_dc
203			.GetTextExtentPoint32(if text.trim().is_empty() { "Pj" } else { text }) // "Pj" is a placeholder to get the height
204			.expect(DONTFAIL);
205
206		if text.is_empty() {
207			bounds.cx = 0; // if no text was given, return just the height
208		}
209		bounds
210	}
211
212	/// Calculates the bound rectangle to fit the text with current system font,
213	/// along with the system check/radio box.
214	#[must_use]
215	pub(in crate::gui) fn bound_box_with_check(text: &str) -> SIZE {
216		let mut bounds = bound_box(text);
217		bounds.cx += GetSystemMetrics(co::SM::CXMENUCHECK) // https://stackoverflow.com/a/1165052/6923555
218			+ GetSystemMetrics(co::SM::CXEDGE);
219
220		let cy_check = GetSystemMetrics(co::SM::CYMENUCHECK);
221		if cy_check > bounds.cy {
222			bounds.cy = cy_check; // if the check is taller than the font, use its height
223		}
224		bounds
225	}
226}

228/// Paints the themed border of an user control, if it has the proper styles.
229pub(in crate::gui) fn paint_control_borders(hwnd: &HWND, wm_ncp: wm::NcPaint) {
230	unsafe {
231		hwnd.DefWindowProc(wm_ncp); // let the system draw the scrollbar for us
232	}
233
234	if !hwnd.style_ex().has(co::WS_EX::CLIENTEDGE) // no border
235		|| !IsThemeActive()
236		|| !IsAppThemed()
237	{
238		return;
239	}
240
241	let mut rc = hwnd.GetWindowRect().expect(DONTFAIL); // window outmost coordinates, including margins
242	rc = hwnd.ScreenToClientRc(rc).expect(DONTFAIL);
243	rc = OffsetRect(rc, 2, 2).expect(DONTFAIL); // because it comes up anchored at -2,-2
244
245	let hdc = hwnd.GetWindowDC().expect(DONTFAIL);
246
247	// The HRGN which comes in WM_NCPAINT seems to be invalid, so we carve our own.
248	let hrgn_hole =
249		HRGN::CreateRectRgnIndirect(InflateRect(rc, -2, -2).expect(DONTFAIL)).expect(DONTFAIL);
250	let hrgn_clip = HRGN::CreateRectRgnIndirect(rc).expect(DONTFAIL);
251	hrgn_clip
252		.CombineRgn(&hrgn_clip, &hrgn_hole, co::RGN::DIFF)
253		.expect(DONTFAIL);
254	hdc.SelectClipRgn(&hrgn_clip).expect(DONTFAIL);
255
256	if let Some(htheme) = hwnd.OpenThemeData("EDIT") {
257		htheme
258			.DrawThemeBackground(&hdc, co::VS::EDIT_EDITTEXT_NORMAL, rc, None)
259			.expect(DONTFAIL);
260	}
