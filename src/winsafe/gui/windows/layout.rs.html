12/// Specifies the horizontal behavior of the control when the parent window is
13/// resized.
14///
15/// The values are analog to [`gui::Vert`](crate::gui::Vert).
16#[derive(Clone, Copy, PartialEq, Eq, Hash)]
17pub enum Horz {
18	/// Nothing will be done when parent window is resized.
19	None,
20	/// When parent window resizes, the control will move anchored at right.
21	/// Size of the control will remain fixed.
22	Repos,
23	/// When parent window resizes, the control width will stretch/shrink
24	/// accordingly. Position will remain fixed.
25	Resize,
26}
27
28/// Specifies the vertical behavior of the control when the parent window is
29/// resized.
30///
31/// The values are analog to [`gui::Horz`](crate::gui::Horz).
32#[derive(Clone, Copy, PartialEq, Eq, Hash)]
33pub enum Vert {
34	/// Nothing will be done when parent window is resized.
35	None,
36	/// When parent window resizes, the control will move anchored at bottom.
37	/// Size of the control will remain fixed.
38	Repos,
39	/// When parent window resizes, the control height will stretch/shrink
40	/// accordingly. Position will remain fixed.
41	Resize,
42}
43
44struct ChildInfo {
45	hchild: HWND,
46	horz: Horz,
47	vert: Vert,
48	rc_orig: Option<RECT>, // original coordinates relative to parent, filled at 1st WM_SIZE
49}
50
51struct LayoutObj {
52	ctrls: UnsafeCell<Vec<ChildInfo>>,
53	sz_parent_orig: UnsafeCell<Option<SIZE>>, // original parent client area, filled at WM_CREATE/INITDIALOG
54	_pin: PhantomPinned,
55}
56
57/// Rearranges the stored controls according to their predefined rules.
58#[derive(Clone)]
59pub(in crate::gui) struct Layout(Pin<Arc<LayoutObj>>);
60
61impl Layout {
62	#[must_use]
63	pub(in crate::gui) fn new() -> Self {
64		Self(Arc::pin(LayoutObj {
65			ctrls: UnsafeCell::new(Vec::new()),
66			sz_parent_orig: UnsafeCell::new(None),
67			_pin: PhantomPinned,
68		}))
69	}

71	/// Adds a new child control to the internal list, so this control will have
72	/// its position and size rearranged when requested.
73	pub(in crate::gui) fn add_child(
74		&self,
75		hparent: &HWND,
76		hchild: &HWND,
77		resize_behavior: (Horz, Vert),
78	) {
79		if *hparent == HWND::NULL || *hchild == HWND::NULL {
80			panic!("Cannot add resizer entries before window/control creation.");
81		}
82
83		let (horz, vert) = resize_behavior;
84		if horz != Horz::None || vert != Vert::None {
85			// if nothing to do, don't even add it
86			let ctrls = unsafe { &mut *self.0.ctrls.get() };
87
88			if ctrls.is_empty() {
89				let rc_parent = hparent.GetClientRect().expect(DONTFAIL);
90				*unsafe { &mut *self.0.sz_parent_orig.get() } = Some(
91					SIZE::with(rc_parent.right, rc_parent.bottom), // save parent client area
92				);
93			}
94
95			ctrls.push(ChildInfo {
96				hchild: unsafe { hchild.raw_copy() },
97				horz,
98				vert,
99				rc_orig: None,
100			});
101		}
102	}
103
104	/// Rearranges all child controls to fit the new width/height of parent
105	/// window.
106	pub(in crate::gui) fn rearrange(&self, p: wm::Size) {
107		let ctrls = unsafe { &mut *self.0.ctrls.get() };
108		if ctrls.is_empty() // no controls
109			|| p.request == co::SIZE_R::MINIMIZED
110		{
111			return; // we're minimized
112		}
113
114		let sz_parent_orig = match unsafe { &mut *self.0.sz_parent_orig.get() } {
115			Some(sz) => *sz,
116			None => panic!("Original parent client area was not saved."),
117		};
118
119		let mut hdwp = HDWP::BeginDeferWindowPos(ctrls.len() as _).expect(DONTFAIL);
120
121		for ctrl in ctrls.iter_mut() {
122			let mut uflags = co::SWP::NOZORDER;
123			if ctrl.horz == Horz::Repos && ctrl.vert == Vert::Repos {
124				uflags |= co::SWP::NOSIZE; // reposition both vert & horz
125			} else if ctrl.horz == Horz::Resize && ctrl.vert == Vert::Resize {
126				uflags |= co::SWP::NOMOVE; // resize both vert & horz
127			}
128
129			let rc_orig = match &ctrl.rc_orig {
130				Some(rc) => *rc,
131				None => {
132					let rc = ctrl
133						.hchild
134						.GetParent()
135						.expect(DONTFAIL)
136						.ScreenToClientRc(ctrl.hchild.GetWindowRect().expect(DONTFAIL))
137						.expect(DONTFAIL);
138					ctrl.rc_orig = Some(rc); // save control client coordinates relative to parent
139					rc
140				},
141			};
142
143			hdwp.DeferWindowPos(
144				&ctrl.hchild,
145				HwndPlace::None,
146				POINT::with(
147					match ctrl.horz {
148						Horz::Repos => p.client_area.cx - sz_parent_orig.cx + rc_orig.left,
149						_ => rc_orig.left, // keep original x pos
150					},
151					match ctrl.vert {
152						Vert::Repos => p.client_area.cy - sz_parent_orig.cy + rc_orig.top,
153						_ => rc_orig.top, // keep original y pos
154					},
155				),
156				SIZE::with(
157					match ctrl.horz {
158						Horz::Resize => {
159							p.client_area.cx - sz_parent_orig.cx + rc_orig.right - rc_orig.left
160						},
161						_ => rc_orig.right - rc_orig.left, // keep original width
162					},
163					match ctrl.vert {
164						Vert::Resize => {
165							p.client_area.cy - sz_parent_orig.cy + rc_orig.bottom - rc_orig.top
166						},
167						_ => rc_orig.bottom - rc_orig.top, // keep original height
168					},
169				),
170				uflags,
171			)
172			.expect(DONTFAIL);
173		}
174	}
