7/// Base to all dialog windows.
8///
9/// Owns the window procedure for all dialog windows.
10pub(in crate::gui) struct DlgBase {
11	base: Base,
12	dialog_id: u16,
13}
14
15impl Drop for DlgBase {
16	fn drop(&mut self) {
17		if *self.base.hwnd() != HWND::NULL {
18			unsafe { self.base.hwnd().SetWindowLongPtr(co::GWLP::DWLP_USER, 0); } // clear passed pointer
19		}
20	}
21}
22
23impl DlgBase {
24	#[must_use]
25	pub(in crate::gui) fn new(
26		parent: Option<&impl AsRef<Base>>,
27		dialog_id: u16,
28	) -> Self
29	{
30		Self {
31			base: Base::new(true, parent),
32			dialog_id,
33		}
34	}
35
36	#[must_use]
37	pub(in crate::gui) const fn base(&self) -> &Base {
38		&self.base
39	}

41	pub(in crate::gui) fn create_dialog_param(&self) -> SysResult<()> {
42		if *self.base.hwnd() != HWND::NULL {
43			panic!("Cannot create dialog twice.");
44		}
45
46		// Our hwnd member is set during WM_INITDIALOG processing; already set
47		// when CreateDialogParam returns.
48		unsafe {
49			self.base.parent_hinstance()?.CreateDialogParam(
50				IdStr::Id(self.dialog_id),
51				self.base.parent().map(|parent| parent.hwnd()),
52				Self::dialog_proc,
53				// Pass pointer to Self.
54				// At this moment, the parent struct is already created and pinned.
55				Some(self as *const _ as _),
56			)?;
57		}
58
59		Ok(())
60	}

62	pub(in crate::gui) fn dialog_box_param(&self) -> SysResult<i32> {
63		if *self.base.hwnd() != HWND::NULL {
64			panic!("Cannot create dialog twice.");
65		}
66
67		// Our hwnd member is set during WM_INITDIALOG processing; already set
68		// when DialogBoxParam returns.
69		let ret = unsafe {
70			self.base.parent_hinstance()?.DialogBoxParam(
71				IdStr::Id(self.dialog_id),
72				self.base.parent().map(|parent| parent.hwnd()),
73				Self::dialog_proc,
74				// Pass pointer to Self.
75				// At this moment, the parent struct is already created and pinned.
76				Some(self as *const _ as _),
77			)?
78		};
79
80		Ok(ret as _)
81	}

83	extern "system" fn dialog_proc(
84		hwnd: HWND,
85		msg: co::WM,
86		wparam: usize,
87		lparam: isize,
88	) -> isize
89	{
90		let wm_any = WndMsg::new(msg, wparam, lparam);
91		Self::dialog_proc_proc(hwnd, wm_any)
92			.unwrap_or_else(|err| { post_quit_error(wm_any, err); true as _ })
93	}
94
95	fn dialog_proc_proc(hwnd: HWND, wm_any: WndMsg) -> AnyResult<isize> {
96		let ptr_self = match wm_any.msg_id {
97			co::WM::INITDIALOG => { // first message being handled
98				let wm_idlg = unsafe { wm::InitDialog::from_generic_wm(wm_any) };
99				let ptr_self = wm_idlg.additional_data as *mut Self;
100				unsafe { hwnd.SetWindowLongPtr(co::GWLP::DWLP_USER, ptr_self as _); } // store
101				let ref_self = unsafe { &mut *ptr_self };
102				ref_self.base.set_hwnd(unsafe { hwnd.raw_copy() }); // store HWND in struct field
103				ptr_self
104			},
105			_ => hwnd.GetWindowLongPtr(co::GWLP::DWLP_USER) as *mut Self, // retrieve
106		};
107
108		// If no pointer stored, then no processing is done.
109		// Prevents processing before WM_INITDIALOG and after WM_NCDESTROY.
110		if ptr_self.is_null() {
111			return Ok(0); // FALSE
112		}
113
114		// Execute before-user closures, keep track if at least one was executed.
115		let ref_self = unsafe { &mut *ptr_self };
116		let at_least_one_before_user = ref_self.base.process_before_user_messages(wm_any)?;
117
118		if wm_any.msg_id == co::WM::INITDIALOG {
119			// Child controls are created in before-user closures, so we set the
120			// system font only after all them.
121			unsafe {
122				ref_self.base.hwnd().SendMessage(wm::SetFont { // on the window itself
123					hfont: ui_font(),
124					redraw: false,
125				});
126			}
127			ref_self.base.hwnd().EnumChildWindows(|hchild| {
128				unsafe {
129					hchild.SendMessage(wm::SetFont { // on each child control
130						hfont: ui_font(),
131						redraw: false,
132					});
133				}
134				true
135			});
136		}
137
138		// Execute user closure, if any.
139		let process_result = ref_self.base.process_user_message(wm_any)?;
140
141		// Execute post-user closures, keep track if at least one was executed.
142		let at_least_one_after_user = ref_self.base.process_after_user_messages(wm_any)?;
143
144		if wm_any.msg_id == co::WM::NCDESTROY { // always check
145			unsafe { hwnd.SetWindowLongPtr(co::GWLP::DWLP_USER, 0); } // clear passed pointer
146			ref_self.base.set_hwnd(HWND::NULL); // clear stored HWND
147			ref_self.base.clear_events(); // prevents circular references
148		}
149
150		Ok(match process_result {
151			WmRet::HandledWithRet(res) => res,
152			WmRet::HandledOk => 1, // TRUE
153			WmRet::NotHandled => if at_least_one_before_user || at_least_one_after_user {
154				1 // TRUE
155			} else {
156				0 // FALSE
157			},
158		})
159	}
