12struct ChildInfo {
13	hchild: HWND,
14	horz: Horz,
15	vert: Vert,
16	rc_orig: Option<RECT>, // original coordinates relative to parent, filled at 1st WM_SIZE
17}
18
19struct Obj { // actual fields of LayoutArranger
20	ctrls: UnsafeCell<Vec<ChildInfo>>,
21	sz_parent_orig: UnsafeCell<Option<SIZE>>, // original parent client area, filled at WM_CREATE/INITDIALOG
22	_pin: PhantomPinned,
23}
24
25/// Rearranges the stored controls according to their predefined rules.
26#[derive(Clone)]
27pub(in crate::gui) struct LayoutArranger(Pin<Arc<Obj>>);
28
29impl LayoutArranger {
30	#[must_use]
31	pub(in crate::gui) fn new() -> Self {
32		Self(
33			Arc::pin(
34				Obj {
35					ctrls: UnsafeCell::new(Vec::new()),
36					sz_parent_orig: UnsafeCell::new(None),
37					_pin: PhantomPinned,
38				},
39			),
40		)
41	}
42
43	/// Adds a new child control to the internal list, so this control will have
44	/// its position and size rearranged when requested.
45	pub(in crate::gui) fn add_child(&self,
46		hparent: &HWND,
47		hchild: &HWND,
48		resize_behavior: (Horz, Vert),
49	) -> SysResult<()>
50	{
51		if *hparent == HWND::NULL || *hchild == HWND::NULL {
52			panic!("Cannot add resizer entries before window/control creation.");
53		}
54
55		let (horz, vert) = resize_behavior;
56		if horz != Horz::None || vert != Vert::None { // if nothing to do, don't even add it
57			unsafe { &mut *self.0.ctrls.get() }.push(
58				ChildInfo {
59					hchild: unsafe { hchild.raw_copy() },
60					horz,
61					vert,
62					rc_orig: None,
63				},
64			);
65		}
66
67		Ok(())
68	}
69
70	/// Saves the original client area of the parent window.
71	pub(in crate::gui) fn save_original_client_area(&self,
72		hparent: &HWND,
73	) -> SysResult<()>
74	{
75		let rc_parent = hparent.GetClientRect()?;
76		*unsafe { &mut *self.0.sz_parent_orig.get() } = Some(
77			SIZE::new(rc_parent.right, rc_parent.bottom),
78		);
79		Ok(())
80	}
81
82	/// Rearranges all child controls to fit the new width/height of parent
83	/// window.
84	pub(in crate::gui) fn rearrange(&self, p: wm::Size) -> SysResult<()> {
85		let ctrls = unsafe { &mut *self.0.ctrls.get() };
86		if ctrls.is_empty() // no controls
87			|| p.request == co::SIZE_R::MINIMIZED { // we're minimized
88			return Ok(());
89		}
90
91		let sz_parent_orig = match unsafe { &mut *self.0.sz_parent_orig.get() } {
92			Some(sz) => *sz,
93			None => panic!("Original parent client area was not saved."),
94		};
95
96		let mut hdwp = HDWP::BeginDeferWindowPos(ctrls.len() as _)?;
97
98		for ctrl in ctrls.iter_mut() {
99			let mut uflags = co::SWP::NOZORDER;
100			if ctrl.horz == Horz::Repos && ctrl.vert == Vert::Repos { // reposition both vert & horz
101				uflags |= co::SWP::NOSIZE;
102			} else if ctrl.horz == Horz::Resize && ctrl.vert == Vert::Resize { // resize both vert & horz
103				uflags |= co::SWP::NOMOVE;
104			}
105
106			let rc_orig = match &ctrl.rc_orig {
107				Some(rc) => *rc,
108				None => {
109					let rc = ctrl.hchild.GetParent()?
110						.ScreenToClientRc(ctrl.hchild.GetWindowRect()?)?;
111					ctrl.rc_orig = Some(rc); // save control client coordinates relative to parent
112					rc
113				},
114			};
115
116			hdwp.DeferWindowPos(
117				&ctrl.hchild,
118				HwndPlace::None,
119				POINT::new(
120					match ctrl.horz {
121						Horz::Repos => p.client_area.cx - sz_parent_orig.cx + rc_orig.left,
122						_ => rc_orig.left // keep original x pos
123					},
124					match ctrl.vert {
125						Vert::Repos => p.client_area.cy - sz_parent_orig.cy + rc_orig.top,
126						_ => rc_orig.top // keep original y pos
127					},
128				),
129				SIZE::new(
130					match ctrl.horz {
131						Horz::Resize => p.client_area.cx - sz_parent_orig.cx + rc_orig.right - rc_orig.left,
132						_ => rc_orig.right - rc_orig.left // keep original width
133					},
134					match ctrl.vert {
135						Vert::Resize => p.client_area.cy - sz_parent_orig.cy + rc_orig.bottom - rc_orig.top,
136						_ => rc_orig.bottom - rc_orig.top // keep original height
137					},
138				),
139				uflags,
140			)?;
141		}
142
143		Ok(())
144	}
