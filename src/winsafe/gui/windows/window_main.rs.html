15/// An user main window, which can handle events. Usually, this is the first
16/// window of your application, launched directly from the `main` function. Can
17/// be programmatically created or load a dialog resource from a `.res` file.
18#[derive(Clone)]
19pub struct WindowMain(RawDlg);
20
21unsafe impl Send for WindowMain {}
22
23impl AsRef<BaseWnd> for WindowMain {
24	fn as_ref(&self) -> &BaseWnd {
25		match &self.0 {
26			RawDlg::Raw(r) => r.raw_base().base(),
27			RawDlg::Dlg(d) => d.dlg_base().base(),
28		}
29	}
30}

32impl GuiWindow for WindowMain {

37	fn as_any(&self) -> &dyn Any {
38		self
39	}

44impl WindowMain {
45	/// Instantiates a new `WindowMain` object, to be created internally with
46	/// [`HWND::CreateWindowEx`](crate::HWND::CreateWindowEx).
47	#[must_use]
48	pub fn new(opts: WindowMainOpts) -> Self {
49		Self(RawDlg::Raw(RawMain::new(opts)))
50	}
51
52	/// Instantiates a new `WindowMain` object, to be loaded from a dialog
53	/// resource with
54	/// [`HINSTANCE::CreateDialogParam`](crate::HINSTANCE::CreateDialogParam).
55	#[must_use]
56	pub fn new_dlg(dlg_id: u16, icon_id: Option<u16>, accel_tbl_id: Option<u16>) -> Self {
57		Self(RawDlg::Dlg(DlgMain::new(dlg_id, icon_id, accel_tbl_id)))
58	}

60	/// Physically creates the window, then runs the main application loop. This
61	/// method will block until the window is closed.
62	///
63	/// The `cmd_show` parameter defaults to
64	/// [`co::SW::SHOW`](crate::co::SW::SHOW).
65	///
66	/// # Panics
67	///
68	/// Panics if the window is already created.
69	///
70	/// Panics if the creation process fails.
71	pub fn run_main(&self, cmd_show: Option<co::SW>) -> AnyResult<i32> {
72		if IsWindowsVistaOrGreater()? {
73			SetProcessDPIAware()?;
74		}
75
76		InitCommonControls();
77
78		if IsWindows8OrGreater()? {
79			// https://github.com/rodrigocfd/winsafe-examples/issues/6
80			let mut b_val = 0; // FALSE
81			match unsafe {
82				HPROCESS::GetCurrentProcess().SetUserObjectInformation(
83					co::UOI::TIMERPROC_EXCEPTION_SUPPRESSION, // SetTimer() safety
84					&mut b_val,
85				)
86			} {
87				Err(e) if e == co::ERROR::INVALID_PARAMETER => {
88					// Do nothing: Wine doesn't support SetUserObjectInformation for now.
89					// https://bugs.winehq.org/show_bug.cgi?id=54951
90				},
91				Err(e) => panic!("TIMERPROC_EXCEPTION_SUPPRESSION failed: {e:?}"), // should never happen
92				_ => {},
93			}
94		}
95
96		let hinst = HINSTANCE::GetModuleHandle(None)?;
97		let res = match &self.0 {
98			RawDlg::Raw(r) => r.run_main(&hinst, cmd_show),
99			RawDlg::Dlg(d) => d.run_main(&hinst, cmd_show),
100		};
101
102		ui_font::delete(); // cleanup
103		res
104	}
105
106	/// Closes the window by posting a [`WM_CLOSE`](crate::msg::wm::Close)
107	/// message. This is the safest way to close any popup window, because
108	/// you'll able to process the
109	/// [`wm_close`](crate::gui::events::WindowEvents::wm_close) event, just
110	/// like if the user clicked the window "X" button.
111	pub fn close(&self) {
112		unsafe {
113			self.hwnd().PostMessage(wm::Close {}).unwrap();
114		}
115	}
116}
