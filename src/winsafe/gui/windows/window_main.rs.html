9/// Switches between raw and dialog implementations.
10///
11/// Hierarchy: `BaseWnd` -> `(Raw|Dlg)Base` -> `(Raw|Dlg)Main` -> `WindowMain`.
12#[derive(Clone)]
13enum RawDlg {
14	Raw(RawMain),
15	Dlg(DlgMain),
16}
17
18/// An user main window, which can handle events. Usually, this is the first
19/// window of your application, launched directly from the `main` function. Can
20/// be programmatically created or load a dialog resource from a `.res` file.
21///
22/// # Examples
23///
24/// Basic structure of a program with a main window, created programmatically:
25///
26/// ```no_run
27/// use winsafe::{self as w, co, gui, prelude::*};
28///
29/// fn main() {
30///     if let Err(err) = Main::create_and_run() {
31///         w::HWND::NULL
32///             .MessageBox(&err.to_string(), "Uncaught error", co::MB::ICONERROR)
33///             .unwrap();
34///     }
35/// }
36///
37/// #[derive(Clone)]
38/// struct Main {
39///     wnd: gui::WindowMain,
40/// }
41///
42/// impl Main {
43///     #[must_use]
44///     fn create_and_run() -> w::AnyResult<i32> {
45///         let wnd = gui::WindowMain::new(gui::WindowMainOpts {
46///             title: "Main window",
47///             ..Default::default()
48///         });
49///
50///         let new_self = Self { wnd };
51///         new_self.events();
52///
53///         new_self.wnd.run_main(None)
54///     }
55///
56///     fn events(&self) {
57///         let self2 = self.clone();
58///         self.wnd.on().wm_create(move |_| {
59///             self2.wnd.hwnd().SetWindowText("Hello")?;
60///             Ok(0)
61///         });
62///     }
63/// }
64/// ```
65#[derive(Clone)]
66pub struct WindowMain(RawDlg);
67
68unsafe impl Send for WindowMain {}
69
70impl AsRef<BaseWnd> for WindowMain {
71	fn as_ref(&self) -> &BaseWnd {
72		match &self.0 {
73			RawDlg::Raw(r) => r.raw_base().base(),
74			RawDlg::Dlg(d) => d.dlg_base().base(),
75		}
76	}
77}

105	}

107	/// Physically creates the window, then runs the main application loop. This
108	/// method will block until the window is closed.
109	///
110	/// The `cmd_show` parameter defaults to
111	/// [`co::SW::SHOW`](crate::co::SW::SHOW).
112	///
113	/// # Panics
114	///
115	/// Panics if the window is already created.
116	///
117	/// Panics if the creation process fails.
118	pub fn run_main(&self, cmd_show: Option<co::SW>) -> AnyResult<i32> {
119		initial_gui_setup();
120
121		let hinst = HINSTANCE::GetModuleHandle(None).expect(DONTFAIL);
122		let res = match &self.0 {
123			RawDlg::Raw(r) => r.run_main(&hinst, cmd_show),
124			RawDlg::Dlg(d) => d.run_main(&hinst, cmd_show),
125		};
126
127		ui_font::delete(); // cleanup
128		res
129	}
130
131	/// Exposes methods to handle window messages.
132	///
133	/// # Panics
134	///
135	/// Panics if the window is already created. Events must be set before
136	/// window creation.
137	#[must_use]
138	pub fn on(&self) -> &impl GuiEventsParent {
139		self.as_ref().on()
140	}
141
142	/// Closes the window by posting a [`WM_CLOSE`](crate::msg::wm::Close)
143	/// message. This is the safest way to close any popup window, because
144	/// you'll able to process the
145	/// [`wm_close`](crate::prelude::GuiEventsWindow::wm_close) event, just like
146	/// if the user clicked the window "X" button.
147	pub fn close(&self) {
148		unsafe {
149			self.hwnd().PostMessage(wm::Close {}).unwrap();
150		}
151	}
