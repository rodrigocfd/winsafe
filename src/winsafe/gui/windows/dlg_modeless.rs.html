15/// A dialog-based modeless window.

18#[derive(Clone)]
19pub(in crate::gui) struct DlgModeless(Pin<Arc<DlgModelessObj>>);
20
21impl DlgModeless {
22	#[must_use]
23	pub(in crate::gui) fn new(
24		parent: &(impl GuiParent + 'static),
25		dlg_id: u16,
26		position: (i32, i32),
27	) -> Self {
28		let new_self = Self(Arc::pin(DlgModelessObj {
29			dlg_base: DlgBase::new(dlg_id),
30			_pin: PhantomPinned,
31		}));
32
33		let self2 = new_self.clone();
34		let parent2 = parent.clone();
35		parent
36			.as_ref()
37			.before_on()
38			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
39				let hinst = parent2.hwnd().hinstance();
40				self2
41					.0
42					.dlg_base
43					.create_dialog_param(&hinst, Some(parent2.hwnd()));
44
45				let hwnd = self2.0.dlg_base.base().hwnd();
46				hwnd.ShowWindow(co::SW::SHOW);
47
48				let rc_parent = parent2
49					.hwnd()
50					.ClientToScreenRc(parent2.hwnd().GetClientRect().expect(DONTFAIL))
51					.expect(DONTFAIL);
52
53				hwnd.SetWindowPos(
54					HwndPlace::None,
55					POINT::with(position.0 + rc_parent.left, position.1 + rc_parent.top),
56					SIZE::default(),
57					co::SWP::NOZORDER | co::SWP::NOSIZE,
58				)
59				.expect(DONTFAIL);
60
61				Ok(0) // ignored
62			});
63
64		new_self
65	}
66
67	#[must_use]
68	pub(in crate::gui) fn dlg_base(&self) -> &DlgBase {
69		&self.0.dlg_base
70	}
