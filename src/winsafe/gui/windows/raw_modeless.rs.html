15/// An ordinary modeless window.
16///
17/// Hierarchy: `BaseWnd` -> `RawBase` -> `RawModeless`.
18#[derive(Clone)]
19pub(in crate::gui) struct RawModeless(Pin<Arc<RawModelessObj>>);
20
21impl RawModeless {
22	#[must_use]
23	pub(in crate::gui) fn new(
24		parent: &(impl GuiParent + 'static),
25		opts: WindowModelessOpts,
26	) -> Self {
27		let new_self = Self(Arc::pin(RawModelessObj {
28			raw_base: RawBase::new(),
29			_pin: PhantomPinned,
30		}));
31
32		let self2 = new_self.clone();
33		let parent2 = parent.clone();
34		let opts2: WindowModelessOptsObj = opts.into();
35		parent
36			.as_ref()
37			.before_on()
38			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
39				let hinst = parent2.hwnd().hinstance();
40				let atom = self2.0.raw_base.register_class(
41					&hinst,
42					&opts2.class_name,
43					opts2.class_style,
44					&opts2.class_icon,
45					&opts2.class_bg_brush,
46					&opts2.class_cursor,
47				);
48
49				let rc_parent = parent2
50					.hwnd()
51					.ClientToScreenRc(parent2.hwnd().GetClientRect().expect(DONTFAIL))
52					.expect(DONTFAIL);
53
54				self2.0.raw_base.create_window(
55					opts2.ex_style,
56					atom,
57					Some(&opts2.title),
58					opts2.style,
59					POINT::with(
60						opts2.position.0 + rc_parent.left,
61						opts2.position.1 + rc_parent.top,
62					),
63					opts2.size.into(),
64					Some(parent2.hwnd()),
65					IdMenu::None,
66					&hinst,
67				);
68
69				Ok(0) // ignored
70			});
71
72		new_self
73	}
74
75	#[must_use]
76	pub(in crate::gui) fn raw_base(&self) -> &RawBase {
77		&self.0.raw_base
78	}
79}

81/// Options to create a [`WindowModeless`](crate::gui::WindowModeless)
82/// programmatically with
83/// [`WindowModeless::new`](crate::gui::WindowModeless::new).
84pub struct WindowModelessOpts<'a> {
85	/// Window class name to be
86	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
87	///
88	/// Defaults to an auto-generated string.
89	pub class_name: &'a str,
90	/// Window class styles to be
91	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
92	///
93	/// Defaults to `co::CS::DBLCLKS`.
94	pub class_style: co::CS,
95	/// Window main icon to be
96	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
97	///
98	/// Defaults to `gui::Icon::None`.
99	pub class_icon: Icon,
100	/// Window cursor to be
101	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
102	///
103	/// Defaults to `gui::Cursor::Idc(co::IDC::ARROW)`.
104	pub class_cursor: Cursor,
105	/// Window background brush to be
106	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
107	///
108	/// Defaults to `gui::Brush::Color(co::COLOR::BTNFACE)`.
109	pub class_bg_brush: Brush,
110
111	/// Window title to be
112	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
113	///
114	/// Defaults to empty string.
115	pub title: &'a str,
116	/// Left and top position coordinates of control within parent's client
117	/// area, to be
118	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
119	///
120	/// Defaults to `gui::dpi(0, 0)`.
121	pub position: (i32, i32),
122	/// Width and height of window client area, in pixels, to be
123	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
124	/// Does not include title bar or borders.
125	///
126	/// Defaults to `gui::dpi(220, 150)`.
127	pub size: (i32, i32),
128	/// Window styles to be
129	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
130	///
131	/// Defaults to `WS::CAPTION | WS::SYSMENU | WS::CLIPCHILDREN | WS::BORDER | WS::VISIBLE`.
132	///
133	/// Suggestions:
134	/// * `WS::SIZEBOX` to make the window resizable.
135	pub style: co::WS,
136	/// Extended window styles to be
137	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
138	///
139	/// Defaults to `WS_EX::LEFT | WS_EX::TOOLWINDOW`.
140	pub ex_style: co::WS_EX,
141}

143impl<'a> Default for WindowModelessOpts<'a> {
144	fn default() -> Self {
145		Self {
146			class_name: "",
147			class_style: co::CS::DBLCLKS,
148			class_icon: Icon::None,
149			class_cursor: Cursor::Idc(co::IDC::ARROW),
150			class_bg_brush: Brush::Color(co::COLOR::BTNFACE),
151			title: "",
152			position: dpi(0, 0),
153			size: dpi(220, 150),
154			style: co::WS::CAPTION
155				| co::WS::SYSMENU
156				| co::WS::CLIPCHILDREN
157				| co::WS::BORDER
158				| co::WS::VISIBLE,
159			ex_style: co::WS_EX::LEFT | co::WS_EX::TOOLWINDOW,
160		}
161	}
162}
