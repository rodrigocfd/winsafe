16/// An ordinary modeless window.

18pub(in crate::gui) struct RawModeless(Pin<Arc<Obj>>);

20impl RawModeless {
21	#[must_use]
22	pub(in crate::gui) fn new(
23		parent: &impl AsRef<Base>,
24		opts: WindowModelessOpts,
25	) -> Self
26	{
27		let new_self = Self(
28			Arc::pin(
29				Obj {
30					raw_base: RawBase::new(Some(parent)),
31					opts,
32					_pin: PhantomPinned,
33				},
34			),
35		);
36		new_self.default_message_handlers(parent.as_ref());
37		new_self
38	}
39
40	#[must_use]
41	pub(in crate::gui) fn base(&self) -> &Base {
42		self.0.raw_base.base()
43	}
44
45	fn default_message_handlers(&self, parent: &Base) {
46		let self2 = self.clone();
47		parent.before_user_on().wm_create_or_initdialog(move |_, _| {
48			let parent_base_ref = self2.base().parent().unwrap();
49			let opts = &self2.0.opts;
50
51			let parent_hinst = self2.base().parent_hinstance()?;
52			let mut wcx = WNDCLASSEX::default();
53			let mut class_name_buf = WString::new();
54			RawBase::fill_wndclassex(
55				&parent_hinst,
56				&opts.class_name, opts.class_style,
57				&opts.class_icon, &opts.class_icon,
58				&opts.class_bg_brush, &opts.class_cursor, &mut wcx,
59				&mut class_name_buf)?;
60			let atom = self2.0.raw_base.register_class(&mut wcx)?;
61
62			let wnd_pos = adjust_modeless_pos(
63				parent_base_ref, POINT::new(opts.position.0, opts.position.1))?;
64
65			let mut wnd_sz = SIZE::new(opts.size.0 as _, opts.size.1 as _);
66			multiply_dpi_or_dtu(parent_base_ref, None, Some(&mut wnd_sz))?;
67
68			self2.0.raw_base.create_window(
69				Some(parent_base_ref.hwnd()),
70				atom,
71				Some(&opts.title),
72				IdMenu::Menu(&HMENU::NULL),
73				wnd_pos, wnd_sz,
74				opts.ex_style, opts.style,
75			)?;
76			Ok(WmRet::NotHandled)
77		});
78	}
79}

81/// Options to create a [`WindowModeless`](crate::gui::WindowModeless)
82/// programmatically with
83/// [`WindowModeless::new`](crate::gui::WindowModeless::new).
84pub struct WindowModelessOpts {
85	/// Window class name to be
86	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
87	///
88	/// Defaults to an auto-generated string.
89	pub class_name: String,
90	/// Window class styles to be
91	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
92	///
93	/// Defaults to `co::CS::DBLCLKS`.
94	pub class_style: co::CS,
95	/// Window main icon to be
96	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
97	///
98	/// Defaults to `gui::Icon::None`.
99	pub class_icon: Icon,
100	/// Window cursor to be
101	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
102	///
103	/// Defaults to `gui::Cursor::Idc(co::IDC::ARROW)`.
104	pub class_cursor: Cursor,
105	/// Window background brush to be
106	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
107	///
108	/// Defaults to `gui::Brush::Color(co::COLOR::BTNFACE)`.
109	pub class_bg_brush: Brush,
110
111	/// Window title to be
112	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
113	///
114	/// Defaults to empty string.
115	pub title: String,
116	/// Left and top position coordinates of control within parent's client
117	/// area, to be
118	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
119	///
120	/// If the parent window is a dialog, the values are in Dialog Template
121	/// Units; otherwise in pixels, which will be multiplied to match current
122	/// system DPI.
123	///
124	/// Defaults to `(0, 0)`.
125	pub position: (i32, i32),
126	/// Width and height of window client area, in pixels, to be
127	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
128	/// Does not include title bar or borders.
129	///
130	/// Will be adjusted to match current system DPI.
131	///
132	/// Defaults to `(220, 150)`.
133	pub size: (u32, u32),
134	/// Window styles to be
135	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
136	///
137	/// Defaults to `WS::CAPTION | WS::SYSMENU | WS::CLIPCHILDREN | WS::BORDER | WS::VISIBLE`.
138	///
139	/// Suggestions:
140	/// * `WS::SIZEBOX` to make the window resizable.
141	pub style: co::WS,
142	/// Extended window styles to be
143	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
144	///
145	/// Defaults to `WS_EX::LEFT | WS_EX::TOOLWINDOW`.
146	pub ex_style: co::WS_EX,
147}
148

151		Self {

153			class_style: co::CS::DBLCLKS,
154			class_icon: Icon::None,

156			class_bg_brush: Brush::Color(co::COLOR::BTNFACE),
157			title: "".to_owned(),
158			position: (0, 0),
159			size: (220, 150),
160			style: co::WS::CAPTION | co::WS::SYSMENU | co::WS::CLIPCHILDREN | co::WS::BORDER | co::WS::VISIBLE,
161			ex_style: co::WS_EX::LEFT | co::WS_EX::TOOLWINDOW,
162		}
163	}
