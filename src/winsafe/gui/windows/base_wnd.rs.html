7/// Allocated on the heap and passed through `WM_UI_THREAD`.
8struct ThreadPack {
9	func: Box<dyn FnOnce() -> AnyResult<()>>,
10}

12/// Base to `RawBase` and `DlgBase`, which means all container windows.
13///
14/// Stores the message closures which are added internally and by the user.
15/// These closures are called by wndproc and dlgproc.

18	wnd_ty: WndTy,
19	layout: Layout,

90	pub(in crate::gui) fn spawn_thread<F>(&self, func: F)
91	where
92		F: FnOnce() -> AnyResult<()> + Send + 'static,
93	{
94		let hwnd = unsafe { self.hwnd.raw_copy() };
95		std::thread::spawn(move || {
96			func().unwrap_or_else(|err| {
97				// If the user func returned an error, create another function
98				// which just returns it, then forward it to WM_UI_THREAD.
99				let pack = Box::new(ThreadPack { func: Box::new(|| Err(err)) });
100				let ptr_pack = Box::into_raw(pack);
101				hwnd.GetAncestor(co::GA::ROOTOWNER).map(|hwnd| unsafe {
102					hwnd.SendMessage(WndMsg {
103						msg_id: Self::WM_UI_THREAD,
104						wparam: Self::WM_UI_THREAD.raw() as _,
105						lparam: ptr_pack as _, // send pointer
106					});
107				});
108			});
109		});
110	}

112	pub(in crate::gui) fn run_ui_thread<F>(&self, func: F)
113	where
114		F: FnOnce() -> AnyResult<()> + Send + 'static,
115	{
116		// This method is analog to SendMessage (synchronous), but intended to
117		// be called from another thread, so a callback function can, tunelled
118		// by wndproc, run in the original thread of the window, thus allowing
119		// GUI updates. With this, the user doesn't have to deal with a custom
120		// WM_ message.
121
122		// https://users.rust-lang.org/t/sending-a-boxed-trait-over-ffi/21708/2
123		let pack = Box::new(ThreadPack { func: Box::new(func) });
124		let ptr_pack = Box::into_raw(pack);
125
126		// Bypass any modals and send straight to main window. This avoids any
127		// blind spots of unhandled messages by a modal being created/destroyed.
128		self.hwnd.GetAncestor(co::GA::ROOTOWNER).map(|hwnd| unsafe {
129			hwnd.SendMessage(WndMsg {
130				msg_id: Self::WM_UI_THREAD,
131				wparam: Self::WM_UI_THREAD.raw() as _,
132				lparam: ptr_pack as _, // send pointer
133			});
134		});
135	}

137	pub(in crate::gui) fn default_message_handlers(&self) {
138		let layout = self.layout.clone();
139		self.before_events.wm_size(move |p| {
140			layout.rearrange(p);
141			Ok(())
142		});
143
144		self.before_events.wm(Self::WM_UI_THREAD, |p| {
145			// WPARAM is just an additional safety check.
146			if unsafe { co::WM::from_raw(p.wparam as _) } == Self::WM_UI_THREAD {
147				let ptr_pack = p.lparam as *mut ThreadPack; // retrieve pointer
148				let pack = unsafe { Box::from_raw(ptr_pack) };
149				let func = pack.func;
150				func().unwrap_or_else(|err| quit_error::post_quit_error(p, err));
151			}
152			Ok(0) // ignored
153		});
154	}

156	pub(in crate::gui) fn run_main_loop(
157		haccel: Option<&HACCEL>,
158		process_dlg_msgs: bool,
159	) -> AnyResult<i32> {
160		let mut msg = MSG::default();
161
162		loop {
163			// GetMessage only fails if hWnd is invalid, what should not happen.
164			if !GetMessage(&mut msg, None, 0, 0).expect(DONTFAIL) {
165				// WM_QUIT was sent, gracefully terminate the program.
166				// wParam has the program exit code.
167				// https://learn.microsoft.com/en-us/windows/win32/winmsg/using-messages-and-message-queues
168				// PostQuitMessage() may have been called internally, so check QUIT_ERROR.
169				return match {
170					let mut msg_error = quit_error::QUIT_ERROR.lock().unwrap();
171					msg_error.take()
172				} {
173					Some(msg_err) => Err(msg_err.into()), // MsgError wrapped into AnyResult
174					None => Ok(msg.wParam as _),          // successfull exit with ret code
175				};
176			}
177
178			// If a child window, will retrieve its top-level parent.
179			// If a top-level, use itself.
180			let hwnd_top_level = msg
181				.hwnd
182				.GetAncestor(co::GA::ROOT)
183				.unwrap_or(unsafe { msg.hwnd.raw_copy() });
184
185			// If we have an accelerator table, try to translate the message.
186			if let Some(haccel) = haccel {
187				if hwnd_top_level
188					.TranslateAccelerator(haccel, &mut msg)
189					.is_ok()
190				{
191					continue; // message translated
192				}
193			}
194
195			// Try to process keyboard actions for child controls.
196			if process_dlg_msgs && hwnd_top_level.IsDialogMessage(&mut msg) {
197				continue;
198			}
199
200			TranslateMessage(&msg);
201			unsafe {
202				DispatchMessage(&msg);
203			}
204		}
205	}

207	pub(in crate::gui) fn run_modal_loop(&self, process_dlg_msgs: bool) -> AnyResult<i32> {
208		let mut msg = MSG::default();
209
210		loop {
211			// GetMessage only fails if hWnd is invalid, what should not happen.
212			if !GetMessage(&mut msg, None, 0, 0).expect(DONTFAIL) {
213				// WM_QUIT was sent, exit modal loop now and signal parent.
214				// wParam has the program exit code.
215				// https://devblogs.microsoft.com/oldnewthing/20050222-00/?p=36393
216				// https://stackoverflow.com/a/29359913/6923555
217				PostQuitMessage(msg.wParam as _);
218				return Ok(0); // raw modals will always return 0
219			}
220
221			if self.hwnd == HWND::NULL || !self.hwnd.IsWindow() {
222				return Ok(0); // our modal was destroyed, terminate loop
223			}
224
225			// If a child window, will retrieve its top-level parent.
226			// If a top-level, use itself.
227			let hwnd_top_level = msg
228				.hwnd
229				.GetAncestor(co::GA::ROOT)
230				.unwrap_or(unsafe { msg.hwnd.raw_copy() });
231
232			// Try to process keyboard actions for child controls.
233			if process_dlg_msgs && hwnd_top_level.IsDialogMessage(&mut msg) {
234				// Processed all keyboard actions for child controls.
235				if self.hwnd == HWND::NULL {
236					return Ok(0); // our modal was destroyed, terminate loop
237				} else {
238					continue;
239				}
240			}
241
242			TranslateMessage(&msg);
243			unsafe {
244				DispatchMessage(&msg);
245			}
246
247			if self.hwnd == HWND::NULL || !self.hwnd.IsWindow() {
248				return Ok(0); // our modal was destroyed, terminate loop
249			}
250		}
251	}
