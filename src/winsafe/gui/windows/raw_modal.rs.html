18/// An ordinary modal window.
19///
20/// Hierarchy: `BaseWnd` -> `RawBase` -> `RawModal`.
21#[derive(Clone)]
22pub(in crate::gui) struct RawModal(Pin<Arc<RawModalObj>>);
23
24impl RawModal {
25	#[must_use]
26	pub(in crate::gui) fn new(opts: WindowModalOpts) -> Self {
27		let new_self = Self(Arc::pin(RawModalObj {
28			raw_base: RawBase::new(),
29			opts: opts.into(),
30			hchild_prev_focus_parent: UnsafeCell::new(HWND::NULL),
31			_pin: PhantomPinned,
32		}));
33		new_self.default_message_handlers();
34		new_self
35	}
36
37	fn default_message_handlers(&self) {
38		let self2 = self.clone();
39		self.0.raw_base.base().before_on().wm_set_focus(move |_| {
40			self2.0.raw_base.delegate_focus_to_first_child();
41			Ok(())
42		});
43
44		let self2 = self.clone();
45		self.0.raw_base.base().on().wm_close(move || {
46			if let Ok(hparent) = self2.0.raw_base.base().hwnd().GetWindow(co::GW::OWNER) {
47				hparent.EnableWindow(true); // re-enable parent
48				self2
49					.0
50					.raw_base
51					.base()
52					.hwnd()
53					.DestroyWindow()
54					.expect(DONTFAIL); // then destroy modal
55
56				let hchild_prev_focus_parent =
57					unsafe { &mut *self2.0.hchild_prev_focus_parent.get() };
58
59				if *hchild_prev_focus_parent != HWND::NULL {
60					hchild_prev_focus_parent.SetFocus(); // this focus could be set on WM_DESTROY as well
61				}
62			}
63			Ok(())
64		});
65	}
66
67	#[must_use]
68	pub(in crate::gui) fn raw_base(&self) -> &RawBase {
69		&self.0.raw_base
70	}
71
72	pub(in crate::gui) fn show_modal(&self, parent: &impl GuiParent) -> AnyResult<()> {
73		let hinst = parent.hwnd().hinstance();
74		let opts = &self.0.opts;
75		let atom = self.0.raw_base.register_class(
76			&hinst,
77			&opts.class_name,
78			opts.class_style,
79			&opts.class_icon,
80			&opts.class_bg_brush,
81			&opts.class_cursor,
82		);
83
84		*unsafe { &mut *self.0.hchild_prev_focus_parent.get() } =
85			HWND::GetFocus().unwrap_or(HWND::NULL);
86		parent.hwnd().EnableWindow(false); // https://devblogs.microsoft.com/oldnewthing/20040227-00/?p=40463
87
88		let mut rc_wnd = RECT {
89			left: 0, // client area, will be adjusted to size with title bar and borders
90			top: 0,
91			right: opts.size.0,
92			bottom: opts.size.1,
93		};
94		rc_wnd = AdjustWindowRectEx(rc_wnd, opts.style, false, opts.ex_style).expect(DONTFAIL);
95
96		let rc_parent = parent.hwnd().GetWindowRect().expect(DONTFAIL); // relative to screen
97		let wnd_pos = POINT::with(
98			rc_parent.left + (rc_parent.right - rc_parent.left) / 2
99				- (rc_wnd.right - rc_wnd.left) / 2, // center on parent
100			rc_parent.top + (rc_parent.bottom - rc_parent.top) / 2
101				- (rc_wnd.bottom - rc_wnd.top) / 2,
102		);
103
104		self.0.raw_base.create_window(
105			opts.ex_style,
106			atom,
107			Some(&opts.title),
108			opts.style,
109			POINT::with(wnd_pos.x, wnd_pos.y),
110			SIZE::with(rc_wnd.right - rc_wnd.left, rc_wnd.bottom - rc_wnd.top),
111			Some(parent.hwnd()),
112			IdMenu::None,
113			&hinst,
114		);
115
116		self.0
117			.raw_base
118			.base()
119			.run_modal_loop(opts.process_dlg_msgs) // blocks until window is closed
120			.map(|_| ())
121	}
122}

124/// Options to create a [`WindowModal`](crate::gui::WindowModal)
125/// programmatically with [`WindowModal::new`](crate::gui::WindowModal::new).
126pub struct WindowModalOpts<'a> {
127	/// Window class name to be
128	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
129	///
130	/// Defaults to an auto-generated string.
131	pub class_name: &'a str,
132	/// Window class styles to be
133	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
134	///
135	/// Defaults to `co::CS::DBLCLKS`.
136	pub class_style: co::CS,
137	/// Window main icon to be
138	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
139	///
140	/// Defaults to `gui::Icon::None`.
141	pub class_icon: Icon,
142	/// Window cursor to be
143	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
144	///
145	/// Defaults to `gui::Cursor::Idc(co::IDC::ARROW)`.
146	pub class_cursor: Cursor,
147	/// Window background brush to be
148	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
149	///
150	/// Defaults to `gui::Brush::Color(co::COLOR::BTNFACE)`.
151	pub class_bg_brush: Brush,
152
153	/// Window title to be
154	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
155	///
156	/// Defaults to empty string.
157	pub title: &'a str,
158	/// Width and height of window client area, in pixels, to be
159	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
160	/// Does not include title bar or borders.
161	///
162	/// Defaults to `gui::dpi(500, 400)`.
163	pub size: (i32, i32),
164	/// Window styles to be
165	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
166	///
167	/// Defaults to `WS::CAPTION | WS::SYSMENU | WS::CLIPCHILDREN | WS::BORDER | WS::VISIBLE`.
168	///
169	/// Suggestions:
170	/// * `WS::SIZEBOX` to make the window resizable;
171	/// * `WS::MAXIMIZEBOX` to have a maximize button.
172	pub style: co::WS,
173	/// Extended window styles to be
174	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
175	///
176	/// Defaults to `WS_EX::LEFT | WS_EX::DLGMODALFRAME`.
177	pub ex_style: co::WS_EX,
178	/// In most applications, the window loop calls
179	/// [`IsDialogMessage`](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-isdialogmessagew)
180	/// so child control messages will properly work. However, this has the side
181	/// effect of inhibiting
182	/// [`WM_CHAR`](https://learn.microsoft.com/en-us/windows/win32/inputdev/wm-char)
183	/// messages from being sent to the window procedure. So, applications which
184	/// do not have child controls and deal directly with character processing –
185	/// like text editors – will never be able to receive `WM_CHAR`.
186	///
187	/// This flag, when `true`, will enable the normal `IsDialogMessage` call in
188	/// the window loop. When `false`, the call will be suppressed.
189	///
190	/// Defaults to `true`.
191	pub process_dlg_msgs: bool,
192}

200			class_cursor: Cursor::Idc(co::IDC::ARROW),
201			class_bg_brush: Brush::Color(co::COLOR::BTNFACE),
202			title: "",
203			size: dpi(500, 400),
204			style: co::WS::CAPTION
205				| co::WS::SYSMENU
206				| co::WS::CLIPCHILDREN
207				| co::WS::BORDER
208				| co::WS::VISIBLE,
209			ex_style: co::WS_EX::LEFT | co::WS_EX::DLGMODALFRAME,
210			process_dlg_msgs: true,
211		}
212	}
213}
