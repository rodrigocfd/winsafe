7/// Base to all ordinary windows.

9/// Owns the window procedure for all ordinary windows.

14impl Drop for RawBase {
15	fn drop(&mut self) {
16		if *self.base.hwnd() != HWND::NULL {
17			unsafe { self.base.hwnd().SetWindowLongPtr(co::GWLP::USERDATA, 0); } // clear passed pointer
18		}
19	}
20}

33	pub(in crate::gui) fn delegate_focus_to_first_child(&self) {
34		if let Some(hwnd_cur_focus) = HWND::GetFocus() {
35			if *self.base.hwnd() == hwnd_cur_focus {
36				// https://stackoverflow.com/a/2835220/6923555
37				if let Ok(hchild_first) = self.base.hwnd().GetWindow(co::GW::CHILD) {
38					hchild_first.SetFocus(); // if window receives focus, delegate to first child
39				}
40			}
41		}
42	}

44	/// Fills `WNDCLASSEX` with the given values, and generates a class name as
45	/// a hash of all fields.
46	pub(in crate::gui) fn fill_wndclassex<'a>(
47		hinst: &'a HINSTANCE,
48		class_name: &'a str,
49		class_style: co::CS,
50		class_icon: &'a Icon,
51		class_icon_sm: &'a Icon,
52		class_bg_brush: &'a Brush,
53		class_cursor: &'a Cursor,
54		wcx: &mut WNDCLASSEX<'a>,
55		class_name_buf: &'a mut WString,
56	) -> SysResult<()>
57	{
58		wcx.lpfnWndProc = Some(Self::window_proc);
59		wcx.hInstance = unsafe { hinst.raw_copy() };
60		wcx.style = class_style;
61		wcx.hIcon = class_icon.as_hicon(hinst)?;
62		wcx.hIconSm = class_icon_sm.as_hicon(hinst)?;
63		wcx.hbrBackground = class_bg_brush.as_hbrush();
64		wcx.hCursor = class_cursor.as_hcursor(hinst)?;
65
66		if class_name.trim().is_empty() { // an actual class name was not provided?
67			*class_name_buf = WString::from_str(
68				&format!(
69					"WNDCLASS.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}",
70					wcx.style,
71					wcx.lpfnWndProc.map_or(0, |p| p as usize),
72					wcx.cbClsExtra, wcx.cbWndExtra,
73					wcx.hInstance, wcx.hIcon, wcx.hCursor, wcx.hbrBackground,
74					wcx.lpszMenuName(),
75					wcx.hIconSm,
76				),
77			);
78		} else {
79			*class_name_buf = WString::from_str(class_name);
80		}
81		wcx.set_lpszClassName(Some(class_name_buf));
82
83		Ok(())
84	}

86	pub(in crate::gui) fn register_class(&self,
87		wcx: &mut WNDCLASSEX,
88	) -> SysResult<ATOM>
89	{
90		SetLastError(co::ERROR::SUCCESS);
91		match unsafe { RegisterClassEx(&wcx) } {
92			Ok(atom) => Ok(atom),
93			Err(err) => match err {
94				co::ERROR::CLASS_ALREADY_EXISTS => {
95					// https://devblogs.microsoft.com/oldnewthing/20150429-00/?p=44984
96					// https://devblogs.microsoft.com/oldnewthing/20041011-00/?p=37603
97					// Retrieve ATOM of existing window class.
98					let hinst = unsafe { wcx.hInstance.raw_copy() };
99					let (atom, _) = hinst.GetClassInfoEx(&wcx.lpszClassName().unwrap())?;
100					Ok(atom)
101				},
102				err => Err(err),
103			},
104		}
105	}

107	pub(in crate::gui) fn create_window(
108		&self,
109		hparent: Option<&HWND>, // passed because message-only window is a special case
110		class_name: ATOM,
111		title: Option<&str>,
112		hmenu: IdMenu,
113		pos: POINT,
114		sz: SIZE,
115		ex_styles: co::WS_EX,
116		styles: co::WS,
117	) -> SysResult<()>
118	{
119		if *self.base.hwnd() != HWND::NULL {
120			panic!("Cannot create window twice.");
121		}
122
123		// Our hwnd member is set during WM_NCCREATE processing; already set when
124		// CreateWindowEx returns.
125		unsafe {
126			HWND::CreateWindowEx(
127				ex_styles,
128				AtomStr::Atom(class_name),
129				title, styles,
130				pos, sz,
131				hparent,
132				hmenu,
133				&self.base.parent_hinstance()?,
134				// Pass pointer to Self.
135				// At this moment, the parent struct is already created and pinned.
136				Some(self as *const _ as _),
137			)?;
138		}
139
140		Ok(())
141	}

143	extern "system" fn window_proc(
144		hwnd: HWND,
145		msg: co::WM,
146		wparam: usize,
147		lparam: isize,
148	) -> isize
149	{
150		let wm_any = WndMsg::new(msg, wparam, lparam);
151		Self::window_proc_proc(hwnd, wm_any)
152			.unwrap_or_else(|err| { post_quit_error(wm_any, err); 0 })
153	}
154
155	fn window_proc_proc(hwnd: HWND, wm_any: WndMsg) -> AnyResult<isize> {
156		let ptr_self = match wm_any.msg_id {
157			co::WM::NCCREATE => { // first message being handled
158				let wm_ncc = unsafe { wm::NcCreate::from_generic_wm(wm_any) };
159				let ptr_self = wm_ncc.createstruct.lpCreateParams as *mut Self;
160				unsafe { hwnd.SetWindowLongPtr(co::GWLP::USERDATA, ptr_self as _); } // store
161				let ref_self = unsafe { &mut *ptr_self };
162				ref_self.base.set_hwnd(unsafe { hwnd.raw_copy() }); // store HWND in struct field
163				ptr_self
164			},
165			_ => hwnd.GetWindowLongPtr(co::GWLP::USERDATA) as *mut Self, // retrieve
166		};
167
168		// If no pointer stored, then no processing is done.
169		// Prevents processing before WM_NCCREATE and after WM_NCDESTROY.
170		if ptr_self.is_null() {
171			return Ok(unsafe { hwnd.DefWindowProc(wm_any) });
172		}
173
174		// Execute before-user closures, keep track if at least one was executed.
175		let ref_self = unsafe { &mut *ptr_self };
176		let at_least_one_before_user = ref_self.base.process_before_user_messages(wm_any)?;
177
178		// Execute user closure, if any.
179		let process_result = ref_self.base.process_user_message(wm_any)?;
180
181		// Execute post-user closures, keep track if at least one was executed.
182		let at_least_one_after_user = ref_self.base.process_after_user_messages(wm_any)?;
183
184		if wm_any.msg_id == co::WM::NCDESTROY { // always check
185			unsafe { hwnd.SetWindowLongPtr(co::GWLP::USERDATA, 0); } // clear passed pointer
186			ref_self.base.set_hwnd(HWND::NULL); // clear stored HWND
187			ref_self.base.clear_events(); // prevents circular references
188		}
189
190		Ok(match process_result {
191			WmRet::HandledWithRet(res) => res,
192			WmRet::HandledOk => 0,
193			WmRet::NotHandled => if at_least_one_before_user || at_least_one_after_user {
194				0
195			} else {
196				unsafe { hwnd.DefWindowProc(wm_any) }.into()
197			},
198		})
199	}
