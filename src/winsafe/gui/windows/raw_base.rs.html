7/// Base to all ordinary windows.

14impl Drop for RawBase {
15	fn drop(&mut self) {
16		if *self.base.hwnd() != HWND::NULL {
17			unsafe {
18				self.base.hwnd().SetWindowLongPtr(co::GWLP::USERDATA, 0); // clear passed pointer
19			}
20		}
21	}

35	pub(in crate::gui) fn register_class(
36		&self,
37		hinst: &HINSTANCE,
38		class_name: &str,
39		class_style: co::CS,
40		class_icon: &Icon,
41		class_bg_brush: &Brush,
42		class_cursor: &Cursor,
43	) -> ATOM {
44		let mut wcx = WNDCLASSEX::default();
45		wcx.lpfnWndProc = Some(Self::wnd_proc);
46		wcx.hInstance = unsafe { hinst.raw_copy() };
47		wcx.style = class_style;
48		wcx.hIcon = class_icon.as_hicon(hinst).expect(DONTFAIL);
49		wcx.hIconSm = class_icon.as_hicon(hinst).expect(DONTFAIL);
50		wcx.hbrBackground = class_bg_brush.as_hbrush();
51		wcx.hCursor = class_cursor.as_hcursor(hinst).expect(DONTFAIL);
52
53		let mut wclass_name = if class_name.trim().is_empty() {
54			WString::from_str(&format!(
55				"WNDCLASS.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}",
56				wcx.style,
57				wcx.lpfnWndProc.map_or(0, |p| p as usize),
58				wcx.cbClsExtra,
59				wcx.cbWndExtra,
60				wcx.hInstance,
61				wcx.hIcon,
62				wcx.hCursor,
63				wcx.hbrBackground,
64				wcx.lpszMenuName(),
65				wcx.hIconSm,
66			))
67		} else {
68			WString::from_str(class_name)
69		};
70		wcx.set_lpszClassName(Some(&mut wclass_name));
71
72		SetLastError(co::ERROR::SUCCESS);
73		match unsafe { RegisterClassEx(&wcx) } {
74			Ok(atom) => atom,
75			Err(err) => match err {
76				co::ERROR::CLASS_ALREADY_EXISTS => {
77					// https://devblogs.microsoft.com/oldnewthing/20150429-00/?p=44984
78					// https://devblogs.microsoft.com/oldnewthing/20041011-00/?p=37603
79					// Retrieve ATOM of existing window class.
80					let hinst = unsafe { wcx.hInstance.raw_copy() };
81					let (atom, _) = hinst
82						.GetClassInfoEx(&wcx.lpszClassName().unwrap())
83						.expect(DONTFAIL);
84					atom
85				},
86				err => panic!("ERROR: RawBase::register_class: {}", err.to_string()),
87			},
88		}
89	}

91	pub(in crate::gui) fn create_window(
92		&self,
93		ex_style: co::WS_EX,
94		class_name: ATOM,
95		title: Option<&str>,
96		style: co::WS,
97		pos: POINT,
98		size: SIZE,
99		hparent: Option<&HWND>,
100		hmenu: IdMenu,
101		hinst: &HINSTANCE,
102	) {
103		if *self.base.hwnd() != HWND::NULL {
104			panic!("Cannot create window twice.");
105		}
106
107		unsafe {
108			// The hwnd member is saved in WM_INITDIALOG processing in wnd_proc.
109			HWND::CreateWindowEx(
110				ex_style,
111				AtomStr::Atom(class_name),
112				title,
113				style,
114				pos,
115				size,
116				hparent,
117				hmenu,
118				hinst,
119				Some(self as *const _ as _), // pass pointer to object itself
120			)
121		}
122		.expect(DONTFAIL);
123	}

125	pub(in crate::gui) fn delegate_focus_to_first_child(&self) {
126		if let Some(hwnd_cur_focus) = HWND::GetFocus() {
127			if *self.base.hwnd() == hwnd_cur_focus {
128				// https://stackoverflow.com/a/2835220/6923555
129				if let Ok(hchild_first) = self.base.hwnd().GetWindow(co::GW::CHILD) {
130					hchild_first.SetFocus(); // if window receives focus, delegate to first child
131				}
132			}
133		}
134	}

136	extern "system" fn wnd_proc(hwnd: HWND, msg: co::WM, wparam: usize, lparam: isize) -> isize {
137		let wm_any = WndMsg::new(msg, wparam, lparam);
138		Self::wnd_proc_proc(hwnd, wm_any).unwrap_or_else(|err| {
139			quit_error::post_quit_error(wm_any, err);
140			0
141		})
142	}

144	fn wnd_proc_proc(hwnd: HWND, p: WndMsg) -> AnyResult<isize> {
145		let ptr_self = match p.msg_id {
146			co::WM::NCCREATE => {
147				// First message being handled.
148				let msg = unsafe { wm::NcCreate::from_generic_wm(p) };
149				let ptr_self = msg.createstruct.lpCreateParams as *const Self;
150				unsafe {
151					hwnd.SetWindowLongPtr(co::GWLP::USERDATA, ptr_self as _); // store
152				}
153				let ref_self = unsafe { &*ptr_self };
154				ref_self.base.set_hwnd(unsafe { hwnd.raw_copy() }); // store HWND in struct field
155				ptr_self
156			},
157			_ => hwnd.GetWindowLongPtr(co::GWLP::USERDATA) as *const Self, // retrieve
158		};
159
160		// If no pointer stored, then no processing is done.
161		// Prevents processing before WM_NCCREATE and after WM_NCDESTROY.
162		if ptr_self.is_null() {
163			return Ok(unsafe { hwnd.DefWindowProc(p) });
164		}
165		let ref_self = unsafe { &*ptr_self };
166
167		// Execute before-user closures, keep track if at least one was executed.
168		// Execute user closure, if any.
169		// Execute post-user closures, keep track if at least one was executed.
170		let (at_least_one_before, user_ret, at_least_one_after) = ref_self.base.process_msgs(p)?;
171
172		// Always check.
173		if p.msg_id == co::WM::NCDESTROY {
174			unsafe {
175				hwnd.SetWindowLongPtr(co::GWLP::USERDATA, 0); // clear passed pointer
176			}
177			ref_self.base.set_hwnd(HWND::NULL); // clear stored HWND
178			ref_self.base.clear_messages(); // prevents circular references
179		}
180
181		if let Some(user_ret) = user_ret {
182			Ok(user_ret)
183		} else if at_least_one_before || at_least_one_after {
184			Ok(0)
185		} else {
186			Ok(unsafe { hwnd.DefWindowProc(p) })
187		}
188	}
