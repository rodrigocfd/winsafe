35	pub(in crate::gui) fn register_class(
36		&self,
37		hinst: &HINSTANCE,
38		class_name: &str,
39		class_style: co::CS,
40		class_icon: &Icon,
41		class_bg_brush: &Brush,
42		class_cursor: &Cursor,
43	) -> SysResult<ATOM> {
44		let mut wcx = WNDCLASSEX::default();
45		wcx.lpfnWndProc = Some(Self::wnd_proc);
46		wcx.hInstance = unsafe { hinst.raw_copy() };
47		wcx.style = class_style;
48		wcx.hIcon = class_icon.as_hicon(hinst)?;
49		wcx.hIconSm = class_icon.as_hicon(hinst)?;
50		wcx.hbrBackground = class_bg_brush.as_hbrush();
51		wcx.hCursor = class_cursor.as_hcursor(hinst)?;
52
53		let mut wclass_name = if class_name.trim().is_empty() {
54			WString::from_str(&format!(
55				"WNDCLASS.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}.{:#x}",
56				wcx.style,
57				wcx.lpfnWndProc.map_or(0, |p| p as usize),
58				wcx.cbClsExtra,
59				wcx.cbWndExtra,
60				wcx.hInstance,
61				wcx.hIcon,
62				wcx.hCursor,
63				wcx.hbrBackground,
64				wcx.lpszMenuName(),
65				wcx.hIconSm,
66			))
67		} else {
68			WString::from_str(class_name)
69		};
70		wcx.set_lpszClassName(Some(&mut wclass_name));
71
72		SetLastError(co::ERROR::SUCCESS);
73		match unsafe { RegisterClassEx(&wcx) } {
74			Ok(atom) => Ok(atom),
75			Err(err) => match err {
76				co::ERROR::CLASS_ALREADY_EXISTS => {
77					// https://devblogs.microsoft.com/oldnewthing/20150429-00/?p=44984
78					// https://devblogs.microsoft.com/oldnewthing/20041011-00/?p=37603
79					// Retrieve ATOM of existing window class.
80					let hinst = unsafe { wcx.hInstance.raw_copy() };
81					let (atom, _) = hinst.GetClassInfoEx(&wcx.lpszClassName().unwrap())?;
82					Ok(atom)
83				},
84				err => Err(err),
85			},
86		}
87	}

90		&self,

109				style,

112				hparent,
113				hmenu,
114				hinst,
115				Some(self as *const _ as _), // pass pointer to object itself
116			)?;
117		}
118		Ok(())
119	}

121	pub(in crate::gui) fn delegate_focus_to_first_child(&self) {
122		if let Some(hwnd_cur_focus) = HWND::GetFocus() {
123			if *self.base.hwnd() == hwnd_cur_focus {
124				// https://stackoverflow.com/a/2835220/6923555
125				if let Ok(hchild_first) = self.base.hwnd().GetWindow(co::GW::CHILD) {
126					hchild_first.SetFocus(); // if window receives focus, delegate to first child
127				}
128			}

132	extern "system" fn wnd_proc(hwnd: HWND, msg: co::WM, wparam: usize, lparam: isize) -> isize {
133		let wm_any = WndMsg::new(msg, wparam, lparam);
134		Self::wnd_proc_proc(hwnd, wm_any).unwrap_or_else(|err| {
135			quit_error::post_quit_error(wm_any, err);
136			0
137		})
138	}

140	fn wnd_proc_proc(hwnd: HWND, p: WndMsg) -> AnyResult<isize> {
141		let ptr_self = match p.msg_id {
142			co::WM::NCCREATE => {
143				// first message being handled
144				let msg = unsafe { wm::NcCreate::from_generic_wm(p) };
145				let ptr_self = msg.createstruct.lpCreateParams as *mut Self;
146				unsafe {
147					hwnd.SetWindowLongPtr(co::GWLP::USERDATA, ptr_self as _); // store
148				}
149				let ref_self = unsafe { &mut *ptr_self };
150				ref_self.base.set_hwnd(unsafe { hwnd.raw_copy() }); // store HWND in struct field
151				ptr_self
152			},
153			_ => hwnd.GetWindowLongPtr(co::GWLP::USERDATA) as *mut Self, // retrieve
154		};
155
156		// If no pointer stored, then no processing is done.
157		// Prevents processing before WM_NCCREATE and after WM_NCDESTROY.
158		if ptr_self.is_null() {
159			return Ok(unsafe { hwnd.DefWindowProc(p) });
160		}
161		let ref_self = unsafe { &mut *ptr_self };
162
163		// Execute before-user closures, keep track if at least one was executed.
164		let at_least_one_before = ref_self.base.process_before_messages(p)?;
165
166		// Execute user closure, if any.
167		let user_ret = ref_self.base.process_user_message(p).transpose()?;
168
169		// Execute post-user closures, keep track if at least one was executed.
170		let at_least_one_after = ref_self.base.process_after_messages(p)?;
171
172		// Always check.
173		if p.msg_id == co::WM::NCDESTROY {
174			unsafe {
175				hwnd.SetWindowLongPtr(co::GWLP::USERDATA, 0); // clear passed pointer
176			}
177			ref_self.base.set_hwnd(HWND::NULL); // clear stored HWND
178			ref_self.base.clear_messages(); // prevents circular references
179		}
180
181		if let Some(user_ret) = user_ret {
182			Ok(user_ret)
183		} else if at_least_one_before || at_least_one_after {
184			Ok(0)
185		} else {
186			Ok(unsafe { hwnd.DefWindowProc(p) })
187		}
188	}
