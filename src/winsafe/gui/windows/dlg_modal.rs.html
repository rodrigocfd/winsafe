15/// A dialog-based modal window.
16///
17/// Hierarchy: `BaseWnd` -> `RawBase` -> `DlgModal`.
18#[derive(Clone)]
19pub(in crate::gui) struct DlgModal(Pin<Arc<DlgModalObj>>);
20
21impl DlgModal {
22	#[must_use]
23	pub(in crate::gui) fn new(dlg_id: u16) -> Self {
24		let new_self = Self(Arc::pin(DlgModalObj {
25			dlg_base: DlgBase::new(dlg_id),
26			_pin: PhantomPinned,
27		}));
28		new_self.default_message_handlers();
29		new_self
30	}
31
32	fn default_message_handlers(&self) {
33		let self2 = self.clone();
34		self.0.dlg_base.base().before_on().wm_init_dialog(move |_| {
35			let hwnd = self2.0.dlg_base.base().hwnd();
36			let rc = hwnd.GetWindowRect().expect(DONTFAIL);
37			let rc_parent = hwnd
38				.GetParent()
39				.expect(DONTFAIL)
40				.GetWindowRect()
41				.expect(DONTFAIL);
42
43			hwnd.SetWindowPos(
44				HwndPlace::None,
45				POINT::with(
46					rc_parent.left + ((rc_parent.right - rc_parent.left) / 2) // center modal on parent
47						- (rc.right - rc.left) / 2,
48					rc_parent.top + ((rc_parent.bottom - rc_parent.top) / 2)
49						- (rc.bottom - rc.top) / 2,
50				),
51				SIZE::default(),
52				co::SWP::NOSIZE | co::SWP::NOZORDER,
53			)
54			.expect(DONTFAIL);
55
56			Ok(false) // return value is discarded
57		});
58
59		let self2 = self.clone();
60		self.0.dlg_base.base().on().wm_close(move || {
61			self2.0.dlg_base.base().hwnd().EndDialog(0).expect(DONTFAIL); // user clicked the X button
62			Ok(())
63		});
64	}
65
66	#[must_use]
67	pub(in crate::gui) fn dlg_base(&self) -> &DlgBase {
68		&self.0.dlg_base
69	}
70
71	pub(in crate::gui) fn show_modal(&self, parent: &impl GuiParent) {
72		let hinst = parent.hwnd().hinstance();
73		self.0
74			.dlg_base
75			.dialog_box_param(&hinst, Some(parent.hwnd()));
