19impl RawControl {
20	pub(in crate::gui) fn new(
21		parent: &impl AsRef<Base>,
22		opts: WindowControlOpts,
23	) -> Self
24	{
25		let resize_behavior = opts.resize_behavior;
26		let opts = auto_ctrl_id_if_zero(opts);
27		let new_self = Self(
28			Arc::pin(
29				Obj {
30					raw_base: RawBase::new(Some(parent)),
31					opts,
32					_pin: PhantomPinned,
33				},
34			),
35		);
36		new_self.default_message_handlers(parent.as_ref(), resize_behavior);
37		new_self
38	}
39
40	#[must_use]
41	pub(in crate::gui) fn base(&self) -> &Base {
42		self.0.raw_base.base()
43	}
44
45	#[must_use]
46	pub(in crate::gui) fn ctrl_id(&self) -> u16 {
47		self.0.opts.ctrl_id
48	}
49
50	fn default_message_handlers(&self,
51		parent: &Base,
52		resize_behavior: (Horz, Vert),
53	) {
54		let self2 = self.clone();
55		parent.before_user_on().wm_create_or_initdialog(move |_, _| {
56			let parent_base_ref = self2.base().parent().unwrap();
57			let opts = &self2.0.opts;
58
59			let parent_hinst = self2.base().parent_hinstance()?;
60			let mut wcx = WNDCLASSEX::default();
61			let mut class_name_buf = WString::new();
62			RawBase::fill_wndclassex(
63				&parent_hinst,
64				&opts.class_name, opts.class_style,
65				&opts.class_icon, &opts.class_icon,
66				&opts.class_bg_brush, &opts.class_cursor, &mut wcx,
67				&mut class_name_buf)?;
68			let atom = self2.0.raw_base.register_class(&mut wcx)?;
69
70			let mut wnd_pos = POINT::new(opts.position.0, opts.position.1);
71			let mut wnd_sz = SIZE::new(opts.size.0 as _, opts.size.1 as _);
72			multiply_dpi_or_dtu(parent_base_ref, Some(&mut wnd_pos), Some(&mut wnd_sz))?;
73
74			self2.0.raw_base.create_window(
75				Some(parent_base_ref.hwnd()),
76				atom,
77				None,
78				IdMenu::Id(opts.ctrl_id),
79				wnd_pos, wnd_sz,
80				opts.ex_style, opts.style,
81			)?;
82
83			parent_base_ref.add_to_layout_arranger(self2.base().hwnd(), resize_behavior)?;
84			Ok(WmRet::HandledOk)
85		});
86
87		let self2 = self.clone();
88		self.base().before_user_on().wm_nc_paint(move |p| {
89			paint_control_borders(self2.base().hwnd(), p)?;
90			Ok(())
91		});
92	}
93}

95/// Options to create a [`WindowControl`](crate::gui::WindowControl)
96/// programmatically with [`WindowControl::new`](crate::gui::WindowControl::new).
97pub struct WindowControlOpts {
98	/// Window class name to be
99	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
100	///
101	/// Defaults to an auto-generated string.
102	pub class_name: String,
103	/// Window class styles to be
104	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
105	///
106	/// Defaults to `co::CS::DBLCLKS`.
107	pub class_style: co::CS,
108	/// Window main icon to be
109	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
110	///
111	/// Defaults to `gui::Icon::None`.
112	pub class_icon: Icon,
113	/// Window cursor to be
114	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
115	///
116	/// Defaults to `gui::Cursor::Idc(co::IDC::ARROW)`.
117	pub class_cursor: Cursor,
118	/// Window background brush to be
119	/// [registered](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-registerclassexw).
120	///
121	/// Defaults to `gui::Brush::Color(co::COLOR::WINDOW)`.
122	pub class_bg_brush: Brush,
123
124	/// Left and top position coordinates of control within parent's client
125	/// area, to be
126	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
127	///
128	/// If the parent window is a dialog, the values are in Dialog Template
129	/// Units; otherwise in pixels, which will be multiplied to match current
130	/// system DPI.
131	///
132	/// Defaults to `(0, 0)`.
133	pub position: (i32, i32),
134	/// Width and height of window to be
135	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
136	///
137	/// If the parent window is a dialog, the values are in Dialog Template
138	/// Units; otherwise in pixels, which will be multiplied to match current
139	/// system DPI.
140	///
141	/// Defaults to `(100, 80)`.
142	pub size: (u32, u32),
143	/// Window styles to be
144	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
145	///
146	/// Defaults to `WS::CHILD | WS::TABSTOP | WS::GROUP | WS::VISIBLE | WS::CLIPCHILDREN | WS::CLIPSIBLINGS`.
147	pub style: co::WS,
148	/// Extended window styles to be
149	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
150	///
151	/// Defaults to `WS_EX::LEFT`.
152	///
153	/// Suggestion:
154	/// * `WS_EX::CLIENTEDGE` to have a border.
155	pub ex_style: co::WS_EX,
156
157	/// The control ID.
158	///
159	/// Defaults to an auto-generated ID.
160	pub ctrl_id: u16,
161	/// Horizontal and vertical behavior of the control when the parent window
162	/// is resized.
163	///
164	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
165	pub resize_behavior: (Horz, Vert),
166}
167
168impl Default for WindowControlOpts {
169	fn default() -> Self {
170		Self {
171			class_name: "".to_owned(),
172			class_style: co::CS::DBLCLKS,
173			class_icon: Icon::None,
174			class_cursor: Cursor::Idc(co::IDC::ARROW),
175			class_bg_brush: Brush::Color(co::COLOR::WINDOW),
176			position: (0, 0),
177			size: (100, 80),
178			style: co::WS::CHILD | co::WS::TABSTOP | co::WS::GROUP | co::WS::VISIBLE | co::WS::CLIPCHILDREN | co::WS::CLIPSIBLINGS,
179			ex_style: co::WS_EX::LEFT,
180			ctrl_id: 0,
181			resize_behavior: (Horz::None, Vert::None),
182		}
183	}
184}
