17/// A dialog-based main window.
18///
19/// Hierarchy: `BaseWnd` -> `RawBase` -> `DlgMain`.
20#[derive(Clone)]
21pub(in crate::gui) struct DlgMain(Pin<Arc<DlgMainObj>>);
22
23impl DlgMain {
24	#[must_use]
25	pub(in crate::gui) fn new(
26		dlg_id: u16,
27		icon_id: Option<u16>,
28		accel_tbl_id: Option<u16>,
29	) -> Self {
30		let new_self = Self(Arc::pin(DlgMainObj {
31			dlg_base: DlgBase::new(dlg_id),
32			icon_id,
33			accel_tbl_id,
34			_pin: PhantomPinned,
35		}));
36		new_self.default_message_handlers();
37		new_self
38	}
39
40	fn default_message_handlers(&self) {
41		let self2 = self.clone();
42		self.0.dlg_base.base().on().wm_close(move || {
43			self2.0.dlg_base.base().hwnd().DestroyWindow().ok(); // ignore errors
44			Ok(())
45		});
46
47		self.0.dlg_base.base().on().wm_nc_destroy(|| {
48			PostQuitMessage(0);
49			Ok(())
50		});
51	}
52
53	#[must_use]
54	pub(in crate::gui) fn dlg_base(&self) -> &DlgBase {
55		&self.0.dlg_base
56	}
57
58	pub(in crate::gui) fn run_main(
59		&self,
60		hinst: &HINSTANCE,
61		cmd_show: Option<co::SW>,
62	) -> AnyResult<i32> {
63		self.0.dlg_base.create_dialog_param(hinst, None);
64		if let Some(id) = self.0.icon_id {
65			self.0.dlg_base.set_icon(hinst, id)?;
66		}
67
68		let haccel = self
69			.0
70			.accel_tbl_id
71			.map(|id| hinst.LoadAccelerators(IdStr::Id(id))) // resources are automatically freed
72			.transpose()?;
73
74		self.0
75			.dlg_base
76			.base()
77			.hwnd()
78			.ShowWindow(cmd_show.unwrap_or(co::SW::SHOW));
79		BaseWnd::run_main_loop(haccel.as_deref(), true) // blocks until window is closed
80	}
