16/// A dialog-based main window.

20impl DlgMain {

23		dlg_id: u16,
24		icon_id: Option<u16>,
25		accel_tbl_id: Option<u16>,
26	) -> Self {
27		let new_self = Self(Arc::pin(DlgMainObj {
28			dlg_base: DlgBase::new(dlg_id),
29			icon_id,
30			accel_tbl_id,

32		}));
33		new_self.default_message_handlers();
34		new_self
35	}

37	fn default_message_handlers(&self) {
38		let self2 = self.clone();
39		self.0.dlg_base.base().on().wm_close(move || {
40			self2.0.dlg_base.base().hwnd().DestroyWindow().ok(); // ignore errors
41			Ok(())
42		});
43
44		self.0.dlg_base.base().on().wm_nc_destroy(|| {
45			PostQuitMessage(0);
46			Ok(())
47		});
48	}
49
50	#[must_use]
51	pub(in crate::gui) fn dlg_base(&self) -> &DlgBase {
52		&self.0.dlg_base
53	}

55	pub(in crate::gui) fn run_main(
56		&self,
57		hinst: &HINSTANCE,
58		cmd_show: Option<co::SW>,
59	) -> AnyResult<i32> {
60		self.0.dlg_base.create_dialog_param(hinst)?;
61		if let Some(id) = self.0.icon_id {
62			self.0.dlg_base.set_icon(hinst, id)?;
63		}
64
65		let haccel = self
66			.0
67			.accel_tbl_id
68			.map(|id| hinst.LoadAccelerators(IdStr::Id(id))) // resources are automatically freed
69			.transpose()?;
70
71		self.0
72			.dlg_base
73			.base()
74			.hwnd()
75			.ShowWindow(cmd_show.unwrap_or(co::SW::SHOW));
76		BaseWnd::run_main_loop(haccel.as_deref(), true) // blocks until window is closed
77	}
