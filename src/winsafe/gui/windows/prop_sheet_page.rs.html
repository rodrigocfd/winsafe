15/// Native
16/// [property sheet](https://learn.microsoft.com/en-us/windows/win32/controls/property-sheets)
17/// page. Behaves as a dialog window. Passed to
18/// [`PropSheetOpts`](crate::gui::PropSheetOpts).
19#[derive(Clone)]
20pub struct PropSheetPage(Pin<Arc<PropSheetPageObj>>);
21
22unsafe impl Send for PropSheetPage {}
23
24impl AsRef<BaseWnd> for PropSheetPage {
25	fn as_ref(&self) -> &BaseWnd {
26		self.0.dlg_base.base()
27	}
28}
29
30impl GuiWindow for PropSheetPage {
31	fn hwnd(&self) -> &HWND {
32		self.as_ref().hwnd()
33	}
34
35	fn as_any(&self) -> &dyn Any {
36		self
37	}
38}
39
40impl GuiParent for PropSheetPage {}
41
42impl PropSheetPage {
43	/// Instantiates a new `PropSheetPage` object, to be used with a
44	/// [`PropSheet`](crate::gui::PropSheet).
45	#[must_use]
46	pub fn new_dlg(dlg_id: u16, title: &str) -> Self {
47		let new_self = Self(Arc::pin(PropSheetPageObj {
48			dlg_base: DlgBase::new(dlg_id),
49			title: WString::from_str(title),
50		}));
51		new_self
52	}
53
54	/// # Safety
55	///
56	/// Demands `Self` to be safely allocated, and not to be moved.
57	pub(crate) unsafe fn gen_pspage(&self) -> PROPSHEETPAGE<'_, '_, '_, '_, '_, '_, '_> {
58		let mut ps_page = self.0.dlg_base.generate_propsheetpage();
59		ps_page.dwFlags |= co::PSP::USETITLE | co::PSP::PREMATURE;
60		ps_page.pszTitle = self.0.title.as_ptr() as _;
61		ps_page
62	}
63}
