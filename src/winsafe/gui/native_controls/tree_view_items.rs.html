1use crate::co;

7/// Exposes item methods of a [`TreeView`](crate::gui::TreeView) control.
8///
9/// You cannot directly instantiate this object, it is created internally by the
10/// control.
11pub struct TreeViewItems<'a, T: 'static> {
12	owner: &'a TreeView<T>,
13}
14
15impl<'a, T> TreeViewItems<'a, T> {
16	#[must_use]
17	pub(in crate::gui) const fn new(owner: &'a TreeView<T>) -> Self {
18		Self { owner }
19	}
20
21	/// Adds a new root item by sending a
22	/// [`tvm::InsertItem`](crate::msg::tvm::InsertItem) message, and returns
23	/// the newly added item.
24	pub fn add_root(
25		&self,
26		text: &str,
27		icon_index: Option<u32>,
28		data: T,
29	) -> SysResult<TreeViewItem<'a, T>> {
30		self.owner.raw_insert_item(None, text, icon_index, data)
31	}
32
33	/// Deletes all items by sending a
34	/// [`tvm::DeleteItem`](crate::msg::tvm::DeleteItem) message.
35	pub fn delete_all(&self) -> SysResult<()> {
36		unsafe {
37			self.owner
38				.hwnd()
39				.SendMessage(tvm::DeleteItem { hitem: &HTREEITEM::NULL })
40		}
41	}
42
43	/// Retrieves the total number of items by sending a
44	/// [`tvm::GetCount`](crate::msg::tvm::GetCount) message.
45	#[must_use]
46	pub fn count(&self) -> u32 {
47		unsafe { self.owner.hwnd().SendMessage(tvm::GetCount {}) }
48	}
49
50	/// Retrieves the number of visible items by sending a
51	/// [`tvm::GetVisibleCount`](crate::msg::tvm::GetVisibleCount) message.
52	#[must_use]
53	pub fn count_visible(&self) -> u32 {
54		unsafe { self.owner.hwnd().SendMessage(tvm::GetVisibleCount {}) }
55	}
56
57	/// Ends the editing of the item's text by sending a
58	/// [`tvm::EndEditLabelNow`](crate::msg::tvm::EndEditLabelNow) message.
59	pub fn end_edit_label_now(&self, save: bool) -> SysResult<()> {
60		unsafe { self.owner.hwnd().SendMessage(tvm::EndEditLabelNow { save }) }
61	}
62
63	/// Retrieves the item of the given handle.
64	///
65	/// **Note:** This method is cheap â€“ even if `htreeitem` is invalid, an
66	/// object will still be returned. However, operations upon this object will
67	/// produce no effect.
68	#[must_use]
69	pub fn get(&self, hitem: &HTREEITEM) -> TreeViewItem<'a, T> {
70		TreeViewItem::new(self.owner, unsafe { hitem.raw_copy() })
71	}
72
73	/// Returns an iterator over the selected items.
74	#[must_use]
75	pub fn iter_selected(&self) -> impl Iterator<Item = TreeViewItem<'a, T>> + 'a {
76		TreeViewItemIter::new(self.owner, None, co::TVGN::CARET)
77	}
78
79	/// Returns an iterator over the root items.
80	#[must_use]
81	pub fn iter_root(&self) -> impl Iterator<Item = TreeViewItem<'a, T>> + 'a {
82		TreeViewChildItemIter::new(self.owner, None)
83	}
84}
