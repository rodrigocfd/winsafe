7/// Exposes column methods of a [`ListView`](crate::gui::ListView) control.
8///
9/// You cannot directly instantiate this object, it is created internally by the
10/// control.
11pub struct ListViewColumns<'a, T: 'static> {
12	owner: &'a ListView<T>,
13}
14
15impl<'a, T> ListViewColumns<'a, T> {
16	#[must_use]
17	pub(in crate::gui) const fn new(owner: &'a ListView<T>) -> Self {
18		Self { owner }
19	}
20
21	/// Adds many columns at once by sending an
22	/// [`lvm::InsertColumn`](crate::msg::lvm::InsertColumn) message.
23	///
24	/// Widths will be adjusted to match current system DPI.
25	///
26	/// # Examples
27	///
28	/// ```no_run
29	/// use winsafe::{self as w, prelude::*, gui};
30	///
31	/// let my_list: gui::ListView; // initialized somewhere
32	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
33	/// # let my_list = gui::ListView::<()>::new(&wnd, gui::ListViewOpts::default());
34	///
35	/// my_list.columns().add(&[
36	///     ("Name", 300),
37	///     ("Address", 500),
38	/// ]);
39	/// ```
40	pub fn add(&self, texts_and_widths: &[(impl AsRef<str>, u32)]) {
41		for (text, width) in texts_and_widths.iter() {
42			let mut col_cx = SIZE::new(*width as _, 0);
43			multiply_dpi(None, Some(&mut col_cx)).unwrap();
44
45			let mut lvc = LVCOLUMN::default();
46			lvc.mask = co::LVCF::TEXT | co::LVCF::WIDTH;
47			lvc.cx = col_cx.cx;
48
49			let mut wtext = WString::from_str(text.as_ref());
50			lvc.set_pszText(Some(&mut wtext));
51
52			unsafe {
53				self.owner.hwnd()
54					.SendMessage(lvm::InsertColumn {
55						index: 0xffff, // insert as the last columns
56						column: &lvc,
57					})
58			}.unwrap();
59		}
60	}
61
62	/// Retrieves the number of columns by sending an
63	/// [`hdm::GetItemCount`](crate::msg::hdm::GetItemCount) message to the
64	/// handle returned by [`lvm::GetHeader`](crate::msg::lvm::GetHeader).
65	#[must_use]
66	pub fn count(&self) -> u32 {
67		unsafe {
68			self.owner.hwnd()
69				.SendMessage(lvm::GetHeader {})
70				.unwrap()
71				.SendMessage(hdm::GetItemCount {})
72				.unwrap()
73		}
74	}
75
76	/// Retrieves the column at the given zero-based position.
77	///
78	/// **Note:** This method is cheap â€“ even if `index` is beyond the range of
79	/// existing columns, an object will still be returned. However, operations
80	/// upon this object will produce no effect.
81	#[must_use]
82	pub const fn get(&self, index: u32) -> ListViewColumn<'a, T> {
83		ListViewColumn::new(self.owner, index)
84	}
85
86	/// Returns an iterator over all columns.
87	#[must_use]
88	pub fn iter(&self) -> impl Iterator<Item = ListViewColumn<'a, T>> + 'a {
89		ListViewColumnIter::new(self.owner)
90	}
