12/// A single item of a [`TreeView`](crate::gui::TreeView) control.
13///
14/// Each object keeps an unique [`HTREEITEM`](crate::HTREEITEM) handle.
15///
16/// You cannot directly instantiate this object, it is created internally by the
17/// control.
18pub struct TreeViewItem<'a, T: 'static = ()> {
19	owner: &'a TreeView<T>,
20	hitem: HTREEITEM,
21}
22
23impl<'a, T> Clone for TreeViewItem<'a, T> {
24	fn clone(&self) -> Self {
25		Self {
26			owner: self.owner,
27			hitem: unsafe { self.hitem.raw_copy() },
28		}
29	}
30}
31
32impl<'a, T> TreeViewItem<'a, T> {
33	#[must_use]
34	pub(in crate::gui) const fn new(owner: &'a TreeView<T>, hitem: HTREEITEM) -> Self {
35		Self { owner, hitem }
36	}
37
38	/// Adds a new child item by sending a
39	/// [`tvm::InsertItem`](crate::msg::tvm::InsertItem) message, and returns
40	/// the newly added item.
41	pub fn add_child(&self, text: &str, icon_index: Option<u32>, data: T) -> SysResult<Self> {
42		self.owner
43			.raw_insert_item(Some(&self.hitem), text, icon_index, data)
44	}
45
46	/// Returns a [`Rc`](std::rc::Rc)/[`RefCell`](std::cell::RefCell) with the
47	/// stored data by sending an [`lvm::GetItem`](crate::msg::lvm::GetItem)
48	/// message.
49	///
50	/// Returns `None` if the `ListView` holds a `()`, or if the item holds an
51	/// invalid index.
52	#[must_use]
53	pub fn data(&self) -> SysResult<Option<Rc<RefCell<T>>>> {
54		Ok(self.data_lparam()?.map(|pdata| {
55			let rc_data = ManuallyDrop::new(unsafe { Rc::from_raw(pdata) });
56			Rc::clone(&rc_data)
57		}))
58	}
59
60	#[must_use]
61	pub(in crate::gui) fn data_lparam(&self) -> SysResult<Option<*mut RefCell<T>>> {
62		let mut tvix = TVITEMEX::default();
63		tvix.hItem = unsafe { self.hitem.raw_copy() };
64		tvix.mask = co::TVIF::PARAM;
65
66		unsafe {
67			self.owner
68				.hwnd()
69				.SendMessage(tvm::GetItem { tvitem: &mut tvix })?;
70		}
71
72		Ok(match tvix.lParam {
73			0 => None,
74			lp => Some(lp as _),
75		})
76	}
77
78	/// Deletes the item by sending a
79	/// [`tvm::DeleteItem`](crate::msg::tvm::DeleteItem) message.
80	pub fn delete(&self) -> SysResult<()> {
81		unsafe {
82			self.owner
83				.hwnd()
84				.SendMessage(tvm::DeleteItem { hitem: &self.hitem })
85		}
86	}
87
88	/// Begins in-place editing of the item's text by sending a
89	/// [`tvm::EditLabel`](crate::msg::tvm::EditLabel) message.
90	///
91	/// Returns a handle to the edit control.
92	pub fn edit_label(&self) -> SysResult<HWND> {
93		unsafe {
94			self.owner
95				.hwnd()
96				.SendMessage(tvm::EditLabel { hitem: &self.hitem })
97		}
98	}
99
100	/// Ensures that a tree-view item is visible, expanding the parent item or
101	/// scrolling the tree-view control, if necessary, by sending a
102	/// [`tvm::EnsureVisible`](crate::msg::tvm::EnsureVisible) message.
103	///
104	/// Returns whether a scroll occurred and no items were expanded.
105	pub fn ensure_visible(&self) -> bool {
106		unsafe {
107			self.owner
108				.hwnd()
109				.SendMessage(tvm::EnsureVisible { hitem: &self.hitem })
110				!= 0
111		}
112	}
113
114	/// Expands or collapse the item by sending a
115	/// [`tvm::Expand`](crate::msg::tvm::Expand) message.
116	pub fn expand(&self, expand: bool) -> SysResult<()> {
117		unsafe {
118			self.owner.hwnd().SendMessage(tvm::Expand {
119				hitem: &self.hitem,
120				action: if expand { co::TVE::EXPAND } else { co::TVE::COLLAPSE },
121			})
122		}
123	}
124
125	/// Returns the underlying handle of the item.
126	#[must_use]
127	pub const fn htreeitem(&self) -> &HTREEITEM {
128		&self.hitem
129	}
130
131	/// Tells if the item is expanded by sending a
132	/// [`tvm::GetItemState`](crate::msg::tvm::GetItemState) message.
133	#[must_use]
134	pub fn is_expanded(&self) -> bool {
135		unsafe {
136			self.owner.hwnd().SendMessage(tvm::GetItemState {
137				hitem: &self.hitem,
138				mask: co::TVIS::EXPANDED,
139			})
140		}
141		.has(co::TVIS::EXPANDED)
142	}
143
144	/// Tells if the item is a root by sending a
145	/// [`tvm::GetNextItem`](crate::msg::tvm::GetNextItem) message.
146	#[must_use]
147	pub fn is_root(&self) -> bool {
148		self.parent().is_none()
149	}

168
169	/// Retrieves the parent of the item by sending a
170	/// [`tvm::GetNextItem`](crate::msg::tvm::GetNextItem) message.
171	#[must_use]
172	pub fn parent(&self) -> Option<Self> {
173		unsafe {
174			self.owner.hwnd().SendMessage(tvm::GetNextItem {
175				relationship: co::TVGN::PARENT,
176				hitem: Some(&self.hitem),
177			})
178		}
179		.map(|hitem| TreeViewItem::new(self.owner, hitem))
180	}
181
182	/// Sets the text of the item by sending a
183	/// [`tvm::SetItem`](crate::msg::tvm::SetItem) message.
184	pub fn set_text(&self, text: &str) -> SysResult<()> {
185		let mut buf = WString::from_str(text);
186
187		let mut tvix = TVITEMEX::default();
188		tvix.hItem = unsafe { self.hitem.raw_copy() };
189		tvix.mask = co::TVIF::TEXT;
190		tvix.set_pszText(Some(&mut buf));
191
192		unsafe {
193			self.owner
194				.hwnd()
195				.SendMessage(tvm::SetItem { tvitem: &tvix })
196		}
197	}
198
199	/// Retrieves the text of the item by sending a
200	/// [`tvm::GetItem`](crate::msg::tvm::GetItem) message.
201	#[must_use]
202	pub fn text(&self) -> SysResult<String> {
203		let mut tvix = TVITEMEX::default();
204		tvix.hItem = unsafe { self.hitem.raw_copy() };
205		tvix.mask = co::TVIF::TEXT;
206
207		let mut buf = WString::new_alloc_buf(MAX_PATH + 1); // arbitrary
208		tvix.set_pszText(Some(&mut buf));
209
210		unsafe {
211			self.owner
212				.hwnd()
213				.SendMessage(tvm::GetItem { tvitem: &mut tvix })?;
214		}
215
216		Ok(buf.to_string())
217	}
