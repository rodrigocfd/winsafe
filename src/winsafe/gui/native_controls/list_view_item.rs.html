12/// A single item of a [`ListView`](crate::gui::ListView) control.
13///
14/// **Note:** Each object keeps the zero-based index of an item. If new items
15/// are added/removed from the list view control, the object may then point to a
16/// different item.
17///
18/// You cannot directly instantiate this object, it is created internally by the
19/// control.
20pub struct ListViewItem<'a, T: 'static = ()> {
21	owner: &'a ListView<T>,
22	index: u32,
23}
24

27	fn clone(&self) -> Self {
28		Self { owner: self.owner, index: self.index }
29	}
30}

39	/// Returns a [`Rc`](std::rc::Rc)/[`RefCell`](std::cell::RefCell) with the
40	/// stored data by sending an [`lvm::GetItem`](crate::msg::lvm::GetItem)
41	/// message.
42	///
43	/// # Panics
44	///
45	/// Panics if the `ListView` doesn't have an actual type, that is, if it was
46	/// declared as `ListView<()>`.
47	///
48	/// Panics if the item index is invalid.
49	///
50	/// # Examples
51	///
52	/// ```no_run
53	/// use winsafe::{self as w, prelude::*, gui};
54	///
55	/// struct Person {
56	///     name: String,
57	///     age: u32,
58	/// }
59	///
60	/// let my_list: gui::ListView<Person>; // initialized somewhere
61	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
62	/// # let my_list = gui::ListView::<Person>::new(&wnd, gui::ListViewOpts::default());
63	///
64	/// // Create item with data.
65	/// let person = Person { name: "".to_owned(), age: 50 };
66	/// my_list.items().add(&["foo"], None, person)?;
67	///
68	/// // Retrieve and modify item data.
69	/// let rc_person = my_list.items().get(0).data()?;
70	/// rc_person.try_borrow_mut()?.age = 60;
71	/// # w::AnyResult::Ok(())
72	/// ```
73	#[must_use]
74	pub fn data(&self) -> SysResult<Rc<RefCell<T>>> {
75		if TypeId::of::<T>() == TypeId::of::<()>() {
76			panic!("ListView<()> will hold no data."); // user didn't define the generic type
77		}
78
79		let rc_ptr = self.data_lparam()?;
80		if rc_ptr.is_null() {
81			panic!("ListViewItem with invalid index, no data.");
82		}
83
84		let rc_obj = ManuallyDrop::new(unsafe { Rc::from_raw(rc_ptr) });
85		Ok(Rc::clone(&rc_obj))
86	}
87
88	#[must_use]
89	pub(in crate::gui) fn data_lparam(&self) -> SysResult<*mut RefCell<T>> {
90		let mut lvi = LVITEM::default();
91		lvi.iItem = self.index as _;
92		lvi.mask = co::LVIF::PARAM;
93
94		unsafe {
95			self.owner
96				.hwnd()
97				.SendMessage(lvm::GetItem { lvitem: &mut lvi })?;
98		}
99
100		Ok(match lvi.lParam {
101			0 => std::ptr::null_mut(),
102			lp => lp as _,
103		})
104	}

263	/// Sets the text of the item under a column by sending an
264	/// [`lvm::SetItemText`](crate::msg::lvm::SetItemText) message.
265	///
266	/// Returns the same item, so further operations can be chained.
267	pub fn set_text(&self, column_index: u32, text: &str) -> SysResult<Self> {
268		let mut lvi = LVITEM::default();
269		lvi.iSubItem = column_index as _;
270
271		let mut wtext = WString::from_str(text);
272		lvi.set_pszText(Some(&mut wtext));
273
274		unsafe {
275			self.owner
276				.hwnd()
277				.SendMessage(lvm::SetItemText { index: self.index, lvitem: &lvi })?;
278		}
279		Ok(*self)
280	}

282	/// Retrieves the text of an item under a column by sending an
283	/// [`lvm::GetItemText`](crate::msg::lvm::GetItemText) message.
284	#[must_use]
285	pub fn text(&self, column_index: u32) -> String {
286		// https://forums.codeguru.com/showthread.php?351972-Getting-listView-item-text-length
287		let mut buf_sz = WString::SSO_LEN; // start with no string heap allocation
288		loop {
289			let mut lvi = LVITEM::default();
290			lvi.iSubItem = column_index as _;
291
292			let mut buf = WString::new_alloc_buf(buf_sz);
293			lvi.set_pszText(Some(&mut buf));
294
295			let returned_chars = unsafe {
296				self.owner
297					.hwnd() // char count without terminating null
298					.SendMessage(lvm::GetItemText { index: self.index, lvitem: &mut lvi })
299			} + 1; // plus terminating null count
300
301			if (returned_chars as usize) < buf_sz {
302				return buf.to_string(); // to break, must have at least 1 char gap
303			}
304
305			buf_sz *= 2; // double the buffer size to try again
306		}
307	}
