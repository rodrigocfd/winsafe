54impl GuiNativeControlEvents<TabEvents> for Tab {
55	fn on(&self) -> &TabEvents {
56		if *self.hwnd() != HWND::NULL {
57			panic!("Cannot add events after the control creation.");
58		} else if *self.0.base.parent().hwnd() != HWND::NULL {
59			panic!("Cannot add events after the parent window creation.");
60		}
61		&self.0.events
62	}

65impl Tab {
66	/// Instantiates a new `Tab` object, to be created on the parent window with
67	/// [`HWND::CreateWindowEx`](crate::prelude::user_Hwnd::CreateWindowEx).
68	///
69	/// # Panics
70	///
71	/// Panics if the parent window was already created â€“ that is, you cannot
72	/// dynamically create a `TreeView` in an event closure.
73	#[must_use]
74	pub fn new(parent: &impl GuiParent, opts: TabOpts) -> Self {
75		let mut opts = auto_ctrl_id_if_zero(opts);
76		let ctrl_id = opts.ctrl_id;
77		let children = opts.items.drain(..).collect::<Vec<_>>();
78
79		let new_self = Self(
80			Arc::pin(
81				Obj {
82					base: BaseNativeControl::new(parent, ctrl_id),
83					events: TabEvents::new(parent, ctrl_id),
84					children,
85					_pin: PhantomPinned,
86				},
87			),
88		);
89
90		let self2 = new_self.clone();
91		parent.as_ref().before_user_on().wm_create_or_initdialog(move |_, _| {
92			self2.create(OptsResz::Wnd(&opts))?;
93			Ok(WmRet::NotHandled)
94		});
95
96		new_self.default_message_handlers(parent.as_ref(), ctrl_id);
97		new_self
98	}
99
100	/// Instantiates a new `Tab` object, to be loaded from a dialog resource
101	/// with [`HWND::GetDlgItem`](crate::prelude::user_Hwnd::GetDlgItem).
102	///
103	/// # Panics
104	///
105	/// Panics if the parent dialog was already created â€“ that is, you cannot
106	/// dynamically create a `TreeView` in an event closure.
107	#[must_use]
108	pub fn new_dlg(
109		parent: &impl GuiParent,
110		ctrl_id: u16,
111		resize_behavior: (Horz, Vert),
112		items: Vec<(String, Box<dyn AsRef<WindowControl>>)>,
113	) -> Self
114	{
115		let new_self = Self(
116			Arc::pin(
117				Obj {
118					base: BaseNativeControl::new(parent, ctrl_id),
119					events: TabEvents::new(parent, ctrl_id),
120					children: items,
121					_pin: PhantomPinned,
122				},
123			),
124		);
125
126		let self2 = new_self.clone();
127		parent.as_ref().before_user_on().wm_init_dialog(move |_| {
128			self2.create(OptsResz::Dlg(resize_behavior))?;
129			Ok(false) // this return value is discarded
130		});
131
132		new_self.default_message_handlers(parent.as_ref(), ctrl_id);
133		new_self
134	}
135
136	fn create(&self, opts_resz: OptsResz<&TabOpts>) -> SysResult<()> {
137		match opts_resz {
138			OptsResz::Wnd(opts) => {
139				let mut pos = POINT::new(opts.position.0, opts.position.1);
140				let mut sz = SIZE::new(opts.size.0 as _, opts.size.1 as _);
141				multiply_dpi_or_dtu(
142					self.0.base.parent(), Some(&mut pos), Some(&mut sz))?;
143
144				self.0.base.create_window( // may panic
145					"SysTabControl32", None, pos, sz,
146					opts.window_ex_style,
147					opts.window_style | opts.tab_style.into(),
148				)?;
149
150				unsafe {
151					self.hwnd().SendMessage(wm::SetFont {
152						hfont: ui_font(),
153						redraw: true,
154					});
155				}
156
157				if opts.tab_ex_style != co::TCS_EX::NoValue {
158					self.set_extended_style(true, opts.tab_ex_style);
159				}
160			},
161			OptsResz::Dlg(_) => self.0.base.create_dlg()?,
162		}
163
164		self.0.children.iter()
165			.for_each(|(text, _)| unsafe { self.items().add(text); }); // add the tabs
166		self.display_tab(0)?; // 1st tab selected by default
167
168		self.0.base.parent()
169			.add_to_layout_arranger(self.hwnd(), opts_resz.resize_behavior())
170	}
171
172	fn default_message_handlers(&self, parent: &Base, ctrl_id: u16) {
173		let self2 = self.clone();
174		parent.before_user_on().wm_notify(ctrl_id, co::TCN::SELCHANGE, move |_| {
175			if let Some(sel_item) = self2.items().selected() {
176				self2.display_tab(sel_item.index())?;
177			}
178			Ok(WmRet::HandledOk)
179		});
180
181		let self2 = self.clone();
182		parent.after_user_on().wm_destroy(move || {
183			self2.image_list().map(|hil| {
184				let _ = unsafe { ImageListDestroyGuard::new(hil.raw_copy()) };
185			});
186			Ok(())
187		});
188	}
189
190	fn display_tab(&self, index: u32) -> SysResult<()> {
191		self.0.children.iter()
192			.enumerate()
193			.filter(|(i, _)| *i != index as usize)
194			.for_each(|(_, (_, item))| {
195				item.as_ref().as_ref().hwnd().ShowWindow(co::SW::HIDE); // hide all others
196			});
197
198		if let Some((_, item)) = self.0.children.get(index as usize) {
199			let mut rc = self.hwnd()
200				.GetParent()?
201				.ScreenToClientRc(self.hwnd().GetWindowRect()?)?;
202			unsafe {
203				self.hwnd().SendMessage(tcm::AdjustRect {
204					display_rect: false,
205					rect: &mut rc,
206				});
207			}
208			item.as_ref().as_ref().hwnd().SetWindowPos(
209				HwndPlace::None,
210				POINT::new(rc.left, rc.top),
211				SIZE::new(rc.right - rc.left, rc.bottom - rc.top),
212				co::SWP::NOZORDER | co::SWP::SHOWWINDOW,
213			)?;
214		}
215
216		Ok(())
217	}

219	/// Retrieves a reference to the associated image list by sending a
220	/// [`tcm::GetImageList`](crate::msg::tcm::GetImageList) message.
221	///
222	/// The image list is owned by the control.
223	#[must_use]
224	pub fn image_list(&self) -> Option<&HIMAGELIST> {
225		unsafe { self.hwnd().SendMessage(tcm::GetImageList {}) }
226			.map(|hil| {
227				let hil_ptr = &hil as *const HIMAGELIST;
228				unsafe { &*hil_ptr }
229			})
230	}

238	/// Sets or unsets the given extended list view styles by sending a
239	/// [`tcm::SetExtendedStyle`](crate::msg::tcm::SetExtendedStyle) message.
240	pub fn set_extended_style(&self, set: bool, ex_style: co::TCS_EX) {
241		unsafe {
242			self.hwnd().SendMessage(tcm::SetExtendedStyle {
243				mask: ex_style,
244				style: if set { ex_style } else { co::TCS_EX::NoValue },
245			});
246		}
247	}
248
249	/// Sets the associated image list by sending a
250	/// [`tcm::SetImageList`](crate::msg::tcm::SetImageList) message.
251	///
252	/// The image list will be owned by the control. Returns the previous one,
253	/// if any.
254	pub fn set_image_list(&self,
255		himagelist: ImageListDestroyGuard,
256	) -> Option<ImageListDestroyGuard>
257	{
258		let mut himagelist = himagelist;
259		let hil = himagelist.leak();
260
261		unsafe {
262			self.hwnd()
263				.SendMessage(tcm::SetImageList { himagelist: Some(hil) })
264				.map(|prev_hil| ImageListDestroyGuard::new(prev_hil))
265		}
266	}

269/// Options to create a [`Tab`](crate::gui::Tab) programmatically with
270/// [`Tab::new`](crate::gui::Tab::new).
271pub struct TabOpts {
272	/// Left and top position coordinates of control within parent's client
273	/// area, to be
274	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
275	///
276	/// If the parent window is a dialog, the values are in Dialog Template
277	/// Units; otherwise in pixels, which will be multiplied to match current
278	/// system DPI.
279	///
280	/// Defaults to `(0, 0)`.
281	pub position: (i32, i32),
282	/// Width and height of control to be
283	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
284	///
285	/// If the parent window is a dialog, the values are in Dialog Template
286	/// Units; otherwise in pixels, which will be multiplied to match current
287	/// system DPI.
288	///
289	/// Defaults to `(80, 50)`.
290	pub size: (u32, u32),
291	/// Tab styles to be
292	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
293	///
294	/// Defaults to `TCS::NoValue`.
295	pub tab_style: co::TCS,
296	/// Extended tab styles to be
297	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
298	///
299	/// Defaults to `TCS_EX::NoValue`.
300	pub tab_ex_style: co::TCS_EX,
301	/// Window styles to be
302	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
303	///
304	/// Defaults to `WS::CHILD | WS::VISIBLE | WS::TABSTOP | WS::GROUP`.
305	pub window_style: co::WS,
306	/// Extended window styles to be
307	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
308	///
309	/// Defaults to `WS_EX::NoValue`.
310	pub window_ex_style: co::WS_EX,
311
312	/// The control ID.
313	///
314	/// Defaults to an auto-generated ID.
315	pub ctrl_id: u16,
316	/// Horizontal and vertical behavior of the control when the parent window
317	/// is resized.
318	///
319	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
320	pub resize_behavior: (Horz, Vert),
321
322	/// Items to be added as soon as the control is created. The tuple contains
323	/// the title of the tab and the window to be rendered inside of it.
324	///
325	/// Note that, in order to make the focus rotation work properly, the child
326	/// windows must be created with the
327	/// [`co::WS_EX::CONTROLPARENT`](crate::co::WS_EX::CONTROLPARENT) extended
328	/// style.
329	///
330	/// Defaults to none.
331	pub items: Vec<(String, Box<dyn AsRef<WindowControl>>)>,
332}
333
334impl Default for TabOpts {
335	fn default() -> Self {
336		Self {
337			position: (0, 0),
338			size: (80, 50),
339			tab_style: co::TCS::NoValue,
340			tab_ex_style: co::TCS_EX::NoValue,
341			window_style: co::WS::CHILD | co::WS::VISIBLE | co::WS::TABSTOP | co::WS::GROUP,
342			window_ex_style: co::WS_EX::NoValue,
343			ctrl_id: 0,
344			resize_behavior: (Horz::None, Vert::None),
345			items: Vec::default(),
346		}
347	}
348}
