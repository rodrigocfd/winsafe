29	pub(in crate::gui) fn new(
30		parent: &(impl GuiParent + 'static),
31		opts: RadioButtonOpts,
32		is_first: bool,
33	) -> Self {
34		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
35		let new_self = Self(Arc::pin(RadioButtonObj {
36			base: BaseCtrl::new(ctrl_id),
37			events: BaseCtrlEvents::new(parent, ctrl_id),
38			_pin: PhantomPinned,
39		}));
40
41		let self2 = new_self.clone();
42		let parent2 = parent.clone();
43		let text2 = opts.text.to_owned();
44		parent
45			.as_ref()
46			.before_on()
47			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
48				self2.0.base.create_window(
49					opts.window_ex_style,
50					"BUTTON",
51					Some(&text2),
52					if is_first {
53						opts.window_style | co::WS::GROUP | co::WS::TABSTOP
54					} else {
55						opts.window_style & !(co::WS::GROUP | co::WS::TABSTOP)
56					} | opts.control_style.into(),
57					opts.position.into(),
58					if opts.size == (0, 0) {
59						text_calc::bound_box_with_check(&text_calc::remove_accel_ampersands(&text2))
60					} else {
61						opts.size.into()
62					},
63					&parent2,
64				);
65				ui_font::set(self2.hwnd());
66				if opts.selected {
67					self2.select(true);
68				}
69				parent2
70					.as_ref()
71					.add_to_layout(self2.hwnd(), opts.resize_behavior);
72				Ok(0) // ignored
73			});
74
75		new_self
76	}

109	/// Tells if this radio button is the currently selected one by sending a
110	/// [`bm::GetCheck`](crate::msg::bm::GetCheck) message.
111	#[must_use]
112	pub fn is_selected(&self) -> bool {
113		unsafe { self.hwnd().SendMessage(bm::GetCheck {}) == co::BST::CHECKED }
114	}
115
116	/// Sets the this radio button as the currently selected one by sending a
117	/// [`bm::SetCheck`](crate::msg::bm::SetCheck) message.
118	pub fn select(&self, selected: bool) {
119		unsafe {
120			self.hwnd().SendMessage(bm::SetCheck {
121				state: if selected { co::BST::CHECKED } else { co::BST::UNCHECKED },
122			});
123		}
124	}
125
126	/// Sets the this radio button as the currently selected one by sending a
127	/// [`bm::SetCheck`](crate::msg::bm::SetCheck) message, then sends a
128	/// [`wm::Command`](crate::msg::wm::Command) message to the parent, so it
129	/// can handle the event.
130	pub fn select_and_trigger(&self, selected: bool) -> SysResult<()> {
131		self.select(selected);
132		self.hwnd().GetParent()?.SendCommand(AccelMenuCtrl::Ctrl {
133			notif_code: co::BN::CLICKED.into(),
134			ctrl_id: self.ctrl_id(),
135			ctrl_hwnd: unsafe { self.hwnd().raw_copy() },
136		});
137		Ok(())
138	}
139
140	/// Calls [`HWND::SetWindowText`](crate::HWND::SetWindowText) to set the
141	/// text and resizes the control to exactly fit it.
142	pub fn set_text_and_resize(&self, text: &str) -> SysResult<()> {
143		let bound_box = text_calc::bound_box_with_check(&text_calc::remove_accel_ampersands(text));
144		self.hwnd().SetWindowPos(
145			HwndPlace::None,
146			POINT::default(),
147			bound_box,
148			co::SWP::NOZORDER | co::SWP::NOMOVE,
149		)?;
150		self.hwnd().SetWindowText(text)?;
151		Ok(())
152	}
153
154	/// Fires the click event for the button by sending a
155	/// [`bm::Click`](crate::msg::bm::Click) message.
156	pub fn trigger_click(&self) {
157		unsafe {
158			self.hwnd().SendMessage(bm::Click {});
159		}
160	}

163/// Options to create a [`RadioButton`](crate::gui::RadioButton)
164/// programmatically with [`RadioGroup::new`](crate::gui::RadioGroup::new).
165#[derive(Clone)]
166pub struct RadioButtonOpts<'a> {
167	/// Text of the control to be
168	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
169	///
170	/// Defaults to empty string.
171	pub text: &'a str,
172	/// Left and top position coordinates of control within parent's client
173	/// area, to be
174	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
175	///
176	/// Defaults to `gui::dpi(0, 0)`.
177	pub position: (i32, i32),
178	/// Width and height of control to be
179	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
180	///
181	/// Defaults to the size needed to fit the text.
182	pub size: (i32, i32),
183	/// Radio button styles to be
184	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
185	///
186	/// Defaults to `BS::AUTORADIOBUTTON`.
187	pub control_style: co::BS,
188	/// Window styles to be
189	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
190	///
191	/// Defaults to `WS::CHILD | WS::VISIBLE`.
192	///
193	/// The first RadioButton of a group will also have `WS::TABSTOP |
194	/// WS::GROUP`. This will be automatically set by the owning
195	/// [`RadioGroup`](crate::gui::RadioGroup).
196	pub window_style: co::WS,
197	/// Extended window styles to be
198	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
199	///
200	/// Defaults to `WS_EX::LEFT`.
201	pub window_ex_style: co::WS_EX,
202
203	/// The control ID.
204	///
205	/// Defaults to an auto-generated ID.
206	pub ctrl_id: u16,
207	/// Horizontal and vertical behavior of the control when the parent window
208	/// is resized.
209	///
210	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
211	pub resize_behavior: (Horz, Vert),
212
213	/// Initial selection state.
214	///
215	/// Defaults to `false`.
216	pub selected: bool,
217}
218
219impl<'a> Default for RadioButtonOpts<'a> {
220	fn default() -> Self {
221		Self {
222			text: "",
223			position: (0, 0),
224			size: (0, 0), // will resize to fit the text
225			control_style: co::BS::AUTORADIOBUTTON,
226			window_style: co::WS::CHILD | co::WS::VISIBLE,
227			window_ex_style: co::WS_EX::LEFT,
228			ctrl_id: 0,
229			resize_behavior: (Horz::None, Vert::None),
230			selected: false,
231		}
232	}
