24impl Trackbar {
25	/// Instantiates a new `Trackbar` object, to be created on the parent window
26	/// with
27	/// [`HWND::CreateWindowEx`](crate::prelude::user_Hwnd::CreateWindowEx).
28	///
29	/// # Panics
30	///
31	/// Panics if the parent window was already created – that is, you cannot
32	/// dynamically create a `Trackbar` in an event closure.
33	#[must_use]
34	pub fn new(parent: &(impl GuiParent + 'static), opts: TrackbarOpts) -> Self {
35		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
36		let new_self = Self(Arc::pin(TrackbarObj {
37			base: BaseCtrl::new(ctrl_id),
38			events: TrackbarEvents::new(parent, ctrl_id),
39			_pin: PhantomPinned,
40		}));
41
42		let self2 = new_self.clone();
43		let parent2 = parent.clone();
44		parent
45			.as_ref()
46			.before_on()
47			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
48				self2.0.base.create_window(
49					opts.window_ex_style,
50					"msctls_trackbar32",
51					None,
52					opts.window_style | opts.control_style.into(),
53					opts.position.into(),
54					opts.size.into(),
55					&parent2,
56				)?;
57				if opts.range != (0, 100) {
58					self2.set_range(opts.range.0, opts.range.1);
59				}
60				if opts.value != 0 {
61					self2.set_pos(opts.value);
62				}
63				parent2
64					.as_ref()
65					.add_to_layout(self2.hwnd(), opts.resize_behavior)?;
66				Ok(0) // ignored
67			});
68
69		new_self
70	}
71
72	/// Instantiates a new `Trackbar` object, to be loaded from a dialog
73	/// resource with
74	/// [`HWND::GetDlgItem`](crate::prelude::user_Hwnd::GetDlgItem).
75	///
76	/// # Panics
77	///
78	/// Panics if the parent dialog was already created – that is, you cannot
79	/// dynamically create a `Trackbar` in an event closure.
80	#[must_use]
81	pub fn new_dlg(
82		parent: &(impl GuiParent + 'static),
83		ctrl_id: u16,
84		resize_behavior: (Horz, Vert),
85	) -> Self {
86		let new_self = Self(Arc::pin(TrackbarObj {
87			base: BaseCtrl::new(ctrl_id),
88			events: TrackbarEvents::new(parent, ctrl_id),
89			_pin: PhantomPinned,
90		}));
91
92		let self2 = new_self.clone();
93		let parent2 = parent.clone();
94		parent.as_ref().before_on().wm_init_dialog(move |_| {
95			self2.0.base.assign_dlg(&parent2)?;
96			parent2
97				.as_ref()
98				.add_to_layout(self2.hwnd(), resize_behavior)?;
99			Ok(true) // ignored
100		});
101
102		new_self
103	}

105	/// Retrieves the current position by sending a
106	/// [`trbm::GetPos`](crate::msg::trbm::GetPos) message.
107	#[must_use]
108	pub fn pos(&self) -> u32 {
109		unsafe { self.hwnd().SendMessage(trbm::GetPos {}) }
110	}
111
112	/// Retrieves the minimum and maximum position values by sending
113	/// [`trbm::GetRangeMin`](crate::msg::trbm::GetRangeMin) and
114	/// [`trbm::GetRangeMax`](crate::msg::trbm::GetRangeMax) messages.
115	#[must_use]
116	pub fn range(&self) -> (u32, u32) {
117		unsafe {
118			(
119				self.hwnd().SendMessage(trbm::GetRangeMin {}),
120				self.hwnd().SendMessage(trbm::GetRangeMax {}),
121			)
122		}
123	}
124
125	/// Sets the current position by sending a
126	/// [`trbm::SetPos`](crate::msg::trbm::SetPos) message.
127	pub fn set_pos(&self, pos: u32) {
128		unsafe {
129			self.hwnd().SendMessage(trbm::SetPos { redraw: true, pos });
130		}
131	}
132
133	/// Sets the minimum and maximum position values by sending
134	/// [`trbm::SetRangeMin`](crate::msg::trbm::SetRangeMin) and
135	/// [`trbm::SetRangeMax`](crate::msg::trbm::SetRangeMax) messages.
136	pub fn set_range(&self, min: u32, max: u32) {
137		unsafe {
138			self.hwnd()
139				.SendMessage(trbm::SetRangeMin { redraw: false, min });
140			self.hwnd()
141				.SendMessage(trbm::SetRangeMax { redraw: true, max });
142		}
143	}
144}

146/// Options to create a [`Trackbar`](crate::gui::Trackbar) programmatically with
147/// [`Trackbar::new`](crate::gui::Trackbar::new).
148pub struct TrackbarOpts {
149	/// Left and top position coordinates of control within parent's client
150	/// area, to be
151	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
152	///
153	/// Defaults to `gui::dpi(0, 0)`.
154	pub position: (i32, i32),
155	/// Width and height of control to be
156	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
157	///
158	/// Defaults to `gui::dpi(120, 23)`.
159	pub size: (i32, i32),
160	/// Trackbar styles to be
161	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
162	///
163	/// Defaults to `TBS::HORZ | TBS::AUTOTICKS`.
164	pub control_style: co::TBS,
165	/// Window styles to be
166	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
167	///
168	/// Defaults to `WS::CHILD | WS::GROUP | WS::TABSTOP | WS::VISIBLE`.
169	pub window_style: co::WS,
170	/// Extended window styles to be
171	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
172	///
173	/// Defaults to `WS_EX::LEFT`.
174	pub window_ex_style: co::WS_EX,
175
176	/// The control ID.
177	///
178	/// Defaults to an auto-generated ID.
179	pub ctrl_id: u16,
180	/// Horizontal and vertical behavior of the control when the parent window
181	/// is resized.
182	///
183	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
184	pub resize_behavior: (Horz, Vert),
185
186	/// The minimum and maximum position values.
187	///
188	/// Defaults to `(0, 100)`.
189	pub range: (u32, u32),
190	/// Initial position value.
191	///
192	/// Defaults to `0`.
193	pub value: u32,
194}
195
196impl Default for TrackbarOpts {
197	fn default() -> Self {
198		Self {
199			position: dpi(0, 0),
200			size: dpi(120, 23),
201			control_style: co::TBS::HORZ | co::TBS::AUTOTICKS,
202			window_style: co::WS::CHILD | co::WS::GROUP | co::WS::TABSTOP | co::WS::VISIBLE,
203			window_ex_style: co::WS_EX::LEFT,
204			ctrl_id: 0,
205			resize_behavior: (Horz::None, Vert::None),
206			range: (0, 100),
207			value: 0,
208		}
209	}
