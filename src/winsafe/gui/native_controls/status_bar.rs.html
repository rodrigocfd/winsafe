13/// Used when adding the parts in
14/// [`StatusBar::new`](crate::gui::StatusBar::new).
15#[derive(Clone, Copy, PartialEq, Eq)]
16pub enum SbPart {
17	/// A part that has a fixed size, in pixels.
18	Fixed(i32),
19	/// A part that will resize when the parent window resizes, filling the
20	/// space left by the fixed-size parts. Has the resizing proportion.
21	///
22	/// How proportion works:
23	///
24	/// 1. Suppose you have 3 parts, respectively with proportions of 1, 1 and 2.
25	/// 2. If available client area width is 400px, respective part widths will be 100, 100 and 200px.
26	/// 3. If parent is resized to have a client area of 600px, parts will then have 200, 200 and 400px.
27	///
28	/// If you're uncertain, just give all resizable parts the proportion 1.
29	Proportional(u8),
30}
31
32struct StatusBarObj {
33	base: BaseCtrl,
34	events: BaseCtrlEvents,
35	parts_info: UnsafeCell<Vec<SbPart>>,
36	right_edges: UnsafeCell<Vec<i32>>, // buffer to speed up resize calls
37	_pin: PhantomPinned,
38}

43	/// control, which has one or more parts.

46impl StatusBar {
47	/// Instantiates a new `StatusBar` object, to be created on the parent
48	/// window with [`HWND::CreateWindowEx`](crate::HWND::CreateWindowEx).
49	///
50	/// # Panics
51	///
52	/// Panics if the parent window was already created â€“ that is, you cannot
53	/// dynamically create a `StatusBar` in an event closure.
54	///
55	/// # Examples
56	///
57	/// ```no_run
58	/// use winsafe::{self as w, prelude::*, gui};
59	///
60	/// let wnd: gui::WindowMain; // initialized somewhere
61	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
62	///
63	/// let status_bar = gui::StatusBar::new(
64	///     &wnd,
65	///     &[
66	///         gui::SbPart::Fixed(200),      // 200 pixels, never resizes
67	///         gui::SbPart::Proportional(1), // these two will fill the remaning space
68	///         gui::SbPart::Proportional(1),
69	///     ],
70	/// );
71	/// ```
72	#[must_use]
73	pub fn new(parent: &(impl GuiParent + 'static), parts: &[SbPart]) -> Self {
74		let ctrl_id = auto_id::next();
75		let new_self = Self(Arc::pin(StatusBarObj {
76			base: BaseCtrl::new(ctrl_id),
77			events: BaseCtrlEvents::new(parent, ctrl_id),
78			parts_info: UnsafeCell::new(parts.to_vec()),
79			right_edges: UnsafeCell::new(vec![0; parts.len()]),
80			_pin: PhantomPinned,
81		}));
82
83		let self2 = new_self.clone();
84		let parent2 = parent.clone();
85		parent
86			.as_ref()
87			.before_on()
88			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
89				let parent_style = parent2.hwnd().style();
90				let is_parent_resizable =
91					parent_style.has(co::WS::MAXIMIZEBOX) || parent_style.has(co::WS::SIZEBOX);
92
93				self2.0.base.create_window(
94					co::WS_EX::LEFT,
95					"msctls_statusbar32",
96					None,
97					co::WS::CHILD
98						| co::WS::VISIBLE | co::SBARS::TOOLTIPS.into()
99						| if is_parent_resizable {
100							co::SBARS::SIZEGRIP
101						} else {
102							co::SBARS::NoValue
103						}
104						.into(),
105					POINT::default(),
106					SIZE::default(),
107					&parent2,
108				);
109
110				// Force first resizing, so the panels are created.
111				let parent_rc = parent2.hwnd().GetClientRect().expect(DONTFAIL);
112				self2.resize(&mut wm::Size {
113					client_area: SIZE::with(parent_rc.right, parent_rc.bottom),
114					request: co::SIZE_R::RESTORED,
115				});
116
117				Ok(0) // ignored
118			});
119
120		let self2 = new_self.clone();
121		parent.as_ref().before_on().wm_size(move |mut p| {
122			self2.resize(&mut p);
123			Ok(())
124		});
125
126		new_self
127	}
128
129	fn resize(&self, p: &mut wm::Size) {
130		if p.request == co::SIZE_R::MINIMIZED || *self.hwnd() == HWND::NULL {
131			return; // nothing to do
132		}
133
134		unsafe {
135			self.hwnd().SendMessage(p.as_generic_wm()); // send WM_SIZE to status bar, so it resizes itself to fit parent
136		}
137
138		let mut total_proportions = 0u8;
139		let mut cx_available = p.client_area.cx as i32;
140
141		let parts_info = unsafe { &mut *self.0.parts_info.get() };
142		for part_info in parts_info.iter() {
143			match part_info {
144				SbPart::Fixed(pixels) => {
145					cx_available -= if *pixels > cx_available { 0 } else { *pixels }; // prevent subtract overflow
146				},
147				SbPart::Proportional(prop) => total_proportions += prop,
148			}
149		}
150
151		let right_edges = unsafe { &mut *self.0.right_edges.get() };
152		let mut total_cx = p.client_area.cx;
153
154		for (idx, part_info) in parts_info.iter().rev().enumerate() {
155			right_edges[parts_info.len() - idx - 1] = total_cx as _;
156			let minus = match part_info {
157				SbPart::Fixed(pixels) => *pixels,
158				SbPart::Proportional(pp) => {
159					(cx_available / total_proportions as i32) * (*pp as i32)
160				},
161			};
162			total_cx -= if minus > total_cx { 0 } else { minus }; // prevent subtract overflow
163		}
164		*right_edges.last_mut().unwrap() = -1;
165
166		unsafe {
167			self.hwnd()
168				.SendMessage(sb::SetParts { right_edges: &right_edges })
169		}
170		.expect(DONTFAIL);
171	}

176		StatusBarParts::new(self)
177	}
