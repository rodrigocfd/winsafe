7/// A single column of a [`ListView`](crate::gui::ListView) control.
8///
9/// **Note:** Each object keeps the zero-based index of a column. If new columns
10/// are added/removed from the list view control, the object may then point to a
11/// different item.
12///
13/// You cannot directly instantiate this object, it is created internally by the
14/// control.
15#[derive(Clone, Copy)]
16pub struct ListViewColumn<'a, T: 'static = ()> {
17	owner: &'a ListView<T>,
18	index: u32,
19}
20
21impl<'a, T> ListViewColumn<'a, T> {
22	#[must_use]
23	pub(in crate::gui) const fn new(owner: &'a ListView<T>, index: u32) -> Self {
24		Self { owner, index }
25	}
26
27	/// Sets the title of the column by sending an
28	/// [`lvm::SetColumn`](crate::msg::lvm::SetColumn) message.
29	pub fn set_title(&self, text: &str) {
30		let mut lvc = LVCOLUMN::default();
31		lvc.iSubItem = self.index as _;
32		lvc.mask = co::LVCF::TEXT;
33
34		let mut buf = WString::from_str(text);
35		lvc.set_pszText(Some(&mut buf));
36
37		unsafe {
38			self.owner.hwnd()
39				.SendMessage(lvm::SetColumn {
40					index: self.index,
41					lvcolumn: &mut lvc,
42				})
43		}.unwrap();
44	}
45
46	/// Sets the width of the column by sending an
47	/// [`lvm::SetColumnWidth`](crate::msg::lvm::SetColumnWidth) message.
48	///
49	/// Width will be adjusted to match current system DPI.
50	pub fn set_width(&self, width: u32) {
51		let mut col_cx = SIZE::new(width as _, 0);
52		multiply_dpi(None, Some(&mut col_cx)).unwrap();
53
54		unsafe {
55			self.owner.hwnd()
56				.SendMessage(lvm::SetColumnWidth {
57					index: self.index,
58					width: col_cx.cx as _,
59				})
60		}.unwrap();
61	}
62
63	/// Sets the width of the column by sending an
64	/// [`lvm::SetColumnWidth`](crate::msg::lvm::SetColumnWidth) message. The
65	/// width will be calculated to fill the remaining space.
66	pub fn set_width_to_fill(&self) {
67		let num_cols = self.owner.columns().count();
68		if num_cols > 0 {
69			let mut cx_used = 0;
70
71			for i in 0..num_cols {
72				if i != self.index {
73					cx_used += self.owner.columns().get(i).width(); // retrieve cx of each column, but us
74				}
75			}
76
77			let rc = self.owner.hwnd().GetClientRect().unwrap(); // list view client area
78
79			unsafe {
80				self.owner.hwnd()
81					.SendMessage(lvm::SetColumnWidth {
82						index: self.index,
83						width: rc.right as u32 - cx_used,
84					})
85			}.unwrap();
86		}
87	}
88
89	/// Retrieves the title of the column by sending an
90	/// [`lvm::GetColumn`](crate::msg::lvm::GetColumn) message.
91	#[must_use]
92	pub fn title(&self) -> String {
93		let mut lvc = LVCOLUMN::default();
94		lvc.iSubItem = self.index as _;
95		lvc.mask = co::LVCF::TEXT;
96
97		let mut buf = WString::new_alloc_buf(128); // arbitrary
98		lvc.set_pszText(Some(&mut buf));
99
100		unsafe {
101			self.owner.hwnd()
102				.SendMessage(lvm::GetColumn {
103					index: self.index,
104					lvcolumn: &mut lvc,
105				})
106		}.unwrap();
107
108		buf.to_string()
109	}
110
111	/// Retrieves the width of the column by sending an
112	/// [`lvm::GetColumnWidth`](crate::msg::lvm::GetColumnWidth) message.
113	#[must_use]
114	pub fn width(&self) -> u32 {
115		unsafe {
116			self.owner.hwnd()
117				.SendMessage(lvm::GetColumnWidth { index: self.index })
118		}.unwrap()
119	}
120}
