33impl<T> TreeView<T> {
34	/// Instantiates a new `TreeView` object, to be created on the parent window
35	/// with [`HWND::CreateWindowEx`](crate::HWND::CreateWindowEx).
36	///
37	/// # Panics
38	///
39	/// Panics if the parent window was already created – that is, you cannot
40	/// dynamically create a `TreeView` in an event closure.
41	#[must_use]
42	pub fn new(parent: &(impl GuiParent + 'static), opts: TreeViewOpts) -> Self {
43		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
44		let new_self = Self(Arc::pin(TreeViewObj {
45			base: BaseCtrl::new(ctrl_id),
46			events: BaseCtrlEvents::new(parent, ctrl_id),
47			_pin: PhantomPinned,
48			_data: PhantomData,
49		}));
50
51		let self2 = new_self.clone();
52		let parent2 = parent.clone();
53		parent
54			.as_ref()
55			.before_on()
56			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
57				self2.0.base.create_window(
58					opts.window_ex_style,
59					"SysTreeView32",
60					None,
61					opts.window_style | opts.control_style.into(),
62					opts.position.into(),
63					opts.size.into(),
64					&parent2,
65				);
66				parent2
67					.as_ref()
68					.add_to_layout(self2.hwnd(), opts.resize_behavior);
69				Ok(0) // ignored
70			});
71
72		new_self.default_message_handlers(parent);
73		new_self
74	}

76	/// Instantiates a new `TreeView` object, to be loaded from a dialog
77	/// resource with [`HWND::GetDlgItem`](crate::HWND::GetDlgItem).
78	///
79	/// # Panics
80	///
81	/// Panics if the parent dialog was already created – that is, you cannot
82	/// dynamically create a `TreeView` in an event closure.
83	#[must_use]
84	pub fn new_dlg(
85		parent: &(impl GuiParent + 'static),
86		ctrl_id: u16,
87		resize_behavior: (Horz, Vert),
88	) -> Self {
89		let new_self = Self(Arc::pin(TreeViewObj {
90			base: BaseCtrl::new(ctrl_id),
91			events: BaseCtrlEvents::new(parent, ctrl_id),
92			_pin: PhantomPinned,
93			_data: PhantomData,
94		}));
95
96		let self2 = new_self.clone();
97		let parent2 = parent.clone();
98		parent.as_ref().before_on().wm_init_dialog(move |_| {
99			self2.0.base.assign_dlg(&parent2);
100			parent2
101				.as_ref()
102				.add_to_layout(self2.hwnd(), resize_behavior);
103			Ok(true) // ignored
104		});
105
106		new_self.default_message_handlers(parent);
107		new_self
108	}

110	fn default_message_handlers(&self, parent: &impl AsRef<BaseWnd>) {
111		let self2 = self.clone();
112		parent
113			.as_ref()
114			.after_on()
115			.wm_notify(self.ctrl_id(), co::TVN::DELETEITEM, move |p| {
116				let nmtv = unsafe { p.cast_nmhdr::<NMTREEVIEW>() };
117				let rc_ptr = self2
118					.items()
119					.get(&nmtv.itemOld.hItem)
120					.data_lparam()
121					.expect(DONTFAIL);
122
123				if !rc_ptr.is_null() {
124					let _ = unsafe { Rc::from_raw(rc_ptr) }; // free allocated LPARAM
125				}
126				Ok(0) // ignored
127			});
128
129		let self2 = self.clone();
130		parent.as_ref().after_on().wm_destroy(move || {
131			[co::TVSIL::NORMAL, co::TVSIL::STATE]
132				.into_iter()
133				.for_each(|kind| unsafe {
134					self2
135						.hwnd()
136						.SendMessage(tvm::GetImageList { kind })
137						.map(|h| {
138							self2
139								.hwnd()
140								.SendMessage(tvm::SetImageList { himagelist: None, kind }); // remove from control
141							let _ = ImageListDestroyGuard::new(h); // destroy
142						});
143				});
144			Ok(())
145		});
146	}

148	pub(in crate::gui) fn raw_insert_item(
149		&self,
150		hparent: Option<&HTREEITEM>,
151		text: &str,
152		icon_index: Option<u32>,
153		data: T,
154	) -> SysResult<TreeViewItem<'_, T>> {
155		let mut tvix = TVITEMEX::default();
156		tvix.mask = co::TVIF::TEXT;
157
158		let mut buf = WString::from_str(text);
159		tvix.set_pszText(Some(&mut buf));
160
161		if let Some(icon_index) = icon_index {
162			tvix.mask |= co::TVIF::IMAGE;
163			tvix.iImage = icon_index as _;
164		}
165
166		// User defined an actual type?
167		if TypeId::of::<T>() != TypeId::of::<()>() {
168			tvix.mask |= co::TVIF::PARAM;
169			let rc_data = Rc::new(RefCell::new(data));
170			tvix.lParam = Rc::into_raw(rc_data) as _;
171		}
172
173		let mut tvis = TVINSERTSTRUCT::default();
174		if let Some(hparent) = hparent {
175			tvis.hParent = unsafe { hparent.raw_copy() };
176		}
177
178		tvis.set_hInsertAfter(TreeitemTvi::Tvi(co::TVI::LAST));
179		tvis.itemex = tvix;
180
181		let new_hitem = unsafe {
182			self.hwnd()
183				.SendMessage(tvm::InsertItem { item: &mut tvis })?
184		};
185
186		Ok(TreeViewItem::new(self, new_hitem))
187	}

189	/// Retrieves one of the associated image lists by sending a
190	/// [`tvm::GetImageList`](crate::msg::tvm::GetImageList) message.
191	///
192	/// Image lists are lazy-initialized: the first time you call this method
193	/// for a given image list, it will be created and assigned with
194	/// [`tvm::SetImageList`](crate::msg::tvm::SetImageList).
195	///
196	/// The image list is owned by the control.
197	#[must_use]
198	pub fn image_list(&self, kind: co::TVSIL) -> HrResult<HIMAGELIST> {
199		match unsafe { self.hwnd().SendMessage(tvm::GetImageList { kind }) } {
200			Some(h) => Ok(h), // already created
201			None => {
202				// Not created yet. Create a new image list and assign it to the list view.
203				let h = HIMAGELIST::Create(SIZE::with(16, 16), co::ILC::COLOR32, 1, 1)?.leak();
204				unsafe {
205					self.hwnd()
206						.SendMessage(tvm::SetImageList { himagelist: Some(h.raw_copy()), kind });
207				}
208				Ok(h)
209			},
210		}
211	}

219	/// Sets or unsets the given extended list view styles by sending a
220	/// [`tvm::SetExtendedStyle`](crate::msg::tvm::SetExtendedStyle) message.
221	pub fn set_extended_style(&self, set: bool, ex_style: co::TVS_EX) -> HrResult<()> {
222		unsafe {
223			self.hwnd().SendMessage(tvm::SetExtendedStyle {
224				mask: ex_style,
225				style: if set { ex_style } else { co::TVS_EX::NoValue },

231/// Options to create a [`TreeView`](crate::gui::TreeView) programmatically with
232/// [`TreeView::new`](crate::gui::TreeView::new).
233pub struct TreeViewOpts {
234	/// Left and top position coordinates of control within parent's client
235	/// area, to be
236	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
237	///
238	/// Defaults to `gui::dpi(0, 0)`.
239	pub position: (i32, i32),
240	/// Width and height of control to be
241	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
242	///
243	/// Defaults to `gui::dpi(120, 120)`.
244	pub size: (i32, i32),
245	/// Tree view styles to be
246	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
247	///
248	/// Defaults to `TVS::HASLINES | TVS::LINESATROOT | TVS::SHOWSELALWAYS | TVS::HASBUTTONS`.
249	pub control_style: co::TVS,
250	/// Extended tree view styles to be
251	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
252	///
253	/// Defaults to `TVS_EX::NoValue`.
254	pub contorl_ex_style: co::TVS_EX,
255	/// Window styles to be
256	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
257	///
258	/// Defaults to `WS::CHILD | WS::GROUP | WS::TABSTOP | WS::VISIBLE`.
259	pub window_style: co::WS,
260	/// Extended window styles to be
261	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
262	///
263	/// Defaults to `WS_EX::LEFT | WS_EX::CLIENTEDGE`.
264	pub window_ex_style: co::WS_EX,
265
266	/// The control ID.
267	///
268	/// Defaults to an auto-generated ID.
269	pub ctrl_id: u16,
270	/// Horizontal and vertical behavior of the control when the parent window
271	/// is resized.
272	///
273	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
274	pub resize_behavior: (Horz, Vert),
275}
276
277impl Default for TreeViewOpts {
278	fn default() -> Self {
279		Self {
280			position: dpi(0, 0),
281			size: dpi(120, 120),
282			control_style: co::TVS::HASLINES
283				| co::TVS::LINESATROOT
284				| co::TVS::SHOWSELALWAYS
285				| co::TVS::HASBUTTONS,
286			contorl_ex_style: co::TVS_EX::NoValue,
287			window_style: co::WS::CHILD | co::WS::GROUP | co::WS::TABSTOP | co::WS::VISIBLE,
288			window_ex_style: co::WS_EX::LEFT | co::WS_EX::CLIENTEDGE,
289			ctrl_id: 0,
290			resize_behavior: (Horz::None, Vert::None),
291		}
292	}
