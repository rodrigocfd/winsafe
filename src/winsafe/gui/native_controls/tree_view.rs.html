24	/// [tree view](https://learn.microsoft.com/en-us/windows/win32/controls/tree-view-controls)

27	/// The generic parameter specifies the type of the object that will be

33impl<T> TreeView<T> {
34	/// Instantiates a new `TreeView` object, to be created on the parent window
35	/// with [`HWND::CreateWindowEx`](crate::HWND::CreateWindowEx).
36	///
37	/// # Panics
38	///
39	/// Panics if the parent window was already created – that is, you cannot
40	/// dynamically create a `TreeView` in an event closure.
41	#[must_use]
42	pub fn new(parent: &(impl GuiParent + 'static), opts: TreeViewOpts) -> Self {
43		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
44		let new_self = Self(Arc::pin(TreeViewObj {
45			base: BaseCtrl::new(ctrl_id),
46			events: BaseCtrlEvents::new(parent, ctrl_id),
47			_pin: PhantomPinned,
48			_data: PhantomData,
49		}));
50
51		let self2 = new_self.clone();
52		let parent2 = parent.clone();
53		parent
54			.as_ref()
55			.before_on()
56			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
57				self2.0.base.create_window(
58					opts.window_ex_style,
59					"SysTreeView32",
60					None,
61					opts.window_style | opts.control_style.into(),
62					opts.position.into(),
63					opts.size.into(),
64					&parent2,
65				);
66				parent2
67					.as_ref()
68					.add_to_layout(self2.hwnd(), opts.resize_behavior);
69				Ok(0) // ignored
70			});
71
72		new_self.default_message_handlers(parent);
73		new_self
74	}

76	/// Instantiates a new `TreeView` object, to be loaded from a dialog
77	/// resource with [`HWND::GetDlgItem`](crate::HWND::GetDlgItem).
78	///
79	/// # Panics
80	///
81	/// Panics if the parent dialog was already created – that is, you cannot
82	/// dynamically create a `TreeView` in an event closure.
83	#[must_use]
84	pub fn new_dlg(
85		parent: &(impl GuiParent + 'static),
86		ctrl_id: u16,
87		resize_behavior: (Horz, Vert),
88	) -> Self {
89		let new_self = Self(Arc::pin(TreeViewObj {
90			base: BaseCtrl::new(ctrl_id),
91			events: BaseCtrlEvents::new(parent, ctrl_id),
92			_pin: PhantomPinned,
93			_data: PhantomData,
94		}));
95
96		let self2 = new_self.clone();
97		let parent2 = parent.clone();
98		parent.as_ref().before_on().wm_init_dialog(move |_| {
99			self2.0.base.assign_dlg(&parent2);
100			parent2
101				.as_ref()
102				.add_to_layout(self2.hwnd(), resize_behavior);
103			Ok(true) // ignored
104		});
105
106		new_self.default_message_handlers(parent);
107		new_self
108	}

110	fn default_message_handlers(&self, parent: &impl AsRef<BaseWnd>) {
111		let self2 = self.clone();
112		parent
113			.as_ref()
114			.after_on()
115			.wm_notify(self.ctrl_id(), co::TVN::DELETEITEM, move |p| {
116				let nmtv = unsafe { p.cast_nmhdr::<NMTREEVIEW>() };
117				let item = self2.items().get(&nmtv.itemOld.hItem);
118
119				if let Some(rc_ptr) = item.data_lparam() {
120					let _ = unsafe { Rc::from_raw(rc_ptr) }; // drop the stored Rc
121				}
122
123				Ok(0) // ignored
124			});
125
126		let self2 = self.clone();
127		parent.as_ref().after_on().wm_destroy(move || {
128			[co::TVSIL::NORMAL, co::TVSIL::STATE]
129				.into_iter()
130				.for_each(|kind| unsafe {
131					self2
132						.hwnd()
133						.SendMessage(tvm::GetImageList { kind })
134						.map(|h| {
135							self2
136								.hwnd()
137								.SendMessage(tvm::SetImageList { himagelist: None, kind }); // remove from control
138							let _ = ImageListDestroyGuard::new(h); // destroy
139						});
140				});
141			Ok(())
142		});
143	}

145	pub(in crate::gui) fn raw_insert_item(
146		&self,
147		hparent: Option<&HTREEITEM>,
148		text: &str,
149		icon_index: Option<u32>,
150		data: T,
151	) -> SysResult<TreeViewItem<'_, T>> {
152		let mut tvix = TVITEMEX::default();
153		tvix.mask = co::TVIF::TEXT;
154
155		let mut buf = WString::from_str(text);
156		tvix.set_pszText(Some(&mut buf));
157
158		if let Some(icon_index) = icon_index {
159			tvix.mask |= co::TVIF::IMAGE;
160			tvix.iImage = icon_index as _;
161		}
162
163		if TypeId::of::<T>() != TypeId::of::<()>() {
164			// User has defined a generic type, so we store the object right away.
165			tvix.mask |= co::TVIF::PARAM;
166			let rc_data = Rc::new(RefCell::new(data));
167			tvix.lParam = Rc::into_raw(rc_data) as _; // store the Rc pointer
168		}
169
170		let mut tvis = TVINSERTSTRUCT::default();
171		if let Some(hparent) = hparent {
172			tvis.hParent = unsafe { hparent.raw_copy() };
173		}
174
175		tvis.set_hInsertAfter(TreeitemTvi::Tvi(co::TVI::LAST));
176		tvis.itemex = tvix;
177
178		let new_hitem = unsafe {
179			self.hwnd()
180				.SendMessage(tvm::InsertItem { item: &mut tvis })?
181		};
182
183		Ok(TreeViewItem::new(self, new_hitem))
184	}

186	/// Retrieves one of the associated image lists by sending a
187	/// [`tvm::GetImageList`](crate::msg::tvm::GetImageList) message.
188	///
189	/// Image lists are lazy-initialized: the first time you call this method
190	/// for a given image list, it will be created and assigned with
191	/// [`tvm::SetImageList`](crate::msg::tvm::SetImageList).
192	///
193	/// The image list is owned by the control.
194	#[must_use]
195	pub fn image_list(&self, kind: co::TVSIL) -> HrResult<HIMAGELIST> {
196		match unsafe { self.hwnd().SendMessage(tvm::GetImageList { kind }) } {
197			Some(h) => Ok(h), // already created
198			None => {
199				// Not created yet. Create a new image list and assign it to the list view.
200				let h = HIMAGELIST::Create(SIZE::with(16, 16), co::ILC::COLOR32, 1, 1)?.leak();
201				unsafe {
202					self.hwnd()
203						.SendMessage(tvm::SetImageList { himagelist: Some(h.raw_copy()), kind });
204				}
205				Ok(h)
206			},
207		}
208	}

216	/// Sets or unsets the given extended list view styles by sending a
217	/// [`tvm::SetExtendedStyle`](crate::msg::tvm::SetExtendedStyle) message.
218	pub fn set_extended_style(&self, set: bool, ex_style: co::TVS_EX) -> HrResult<()> {
219		unsafe {
220			self.hwnd().SendMessage(tvm::SetExtendedStyle {
221				mask: ex_style,
222				style: if set { ex_style } else { co::TVS_EX::NoValue },
223			})
224		}

228/// Options to create a [`TreeView`](crate::gui::TreeView) programmatically with
229/// [`TreeView::new`](crate::gui::TreeView::new).
230pub struct TreeViewOpts {
231	/// Left and top position coordinates of control within parent's client
232	/// area, to be
233	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
234	///
235	/// Defaults to `gui::dpi(0, 0)`.
236	pub position: (i32, i32),
237	/// Width and height of control to be
238	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
239	///
240	/// Defaults to `gui::dpi(120, 120)`.
241	pub size: (i32, i32),
242	/// Tree view styles to be
243	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
244	///
245	/// Defaults to `TVS::HASLINES | TVS::LINESATROOT | TVS::SHOWSELALWAYS | TVS::HASBUTTONS`.
246	pub control_style: co::TVS,
247	/// Extended tree view styles to be
248	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
249	///
250	/// Defaults to `TVS_EX::NoValue`.
251	pub contorl_ex_style: co::TVS_EX,
252	/// Window styles to be
253	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
254	///
255	/// Defaults to `WS::CHILD | WS::GROUP | WS::TABSTOP | WS::VISIBLE`.
256	pub window_style: co::WS,
257	/// Extended window styles to be
258	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
259	///
260	/// Defaults to `WS_EX::LEFT | WS_EX::CLIENTEDGE`.
261	pub window_ex_style: co::WS_EX,
262
263	/// The control ID.
264	///
265	/// Defaults to an auto-generated ID.
266	pub ctrl_id: u16,
267	/// Horizontal and vertical behavior of the control when the parent window
268	/// is resized.
269	///
270	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
271	pub resize_behavior: (Horz, Vert),
272}
273
274impl Default for TreeViewOpts {
275	fn default() -> Self {
276		Self {
277			position: dpi(0, 0),
278			size: dpi(120, 120),
279			control_style: co::TVS::HASLINES
280				| co::TVS::LINESATROOT
281				| co::TVS::SHOWSELALWAYS
282				| co::TVS::HASBUTTONS,
283			contorl_ex_style: co::TVS_EX::NoValue,
284			window_style: co::WS::CHILD | co::WS::GROUP | co::WS::TABSTOP | co::WS::VISIBLE,
285			window_ex_style: co::WS_EX::LEFT | co::WS_EX::CLIENTEDGE,
286			ctrl_id: 0,
287			resize_behavior: (Horz::None, Vert::None),
288		}
289	}
