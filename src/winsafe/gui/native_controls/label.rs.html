23impl Label {
24	/// Instantiates a new `Label` object, to be created on the parent window
25	/// with [`HWND::CreateWindowEx`](crate::HWND::CreateWindowEx).
26	///
27	/// # Panics
28	///
29	/// Panics if the parent window was already created – that is, you cannot
30	/// dynamically create a `Label` in an event closure.
31	#[must_use]
32	pub fn new(parent: &(impl GuiParent + 'static), opts: LabelOpts) -> Self {
33		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
34		let new_self = Self(Arc::pin(LabelObj {
35			base: BaseCtrl::new(ctrl_id),
36			events: BaseCtrlEvents::new(parent, ctrl_id),
37			_pin: PhantomPinned,
38		}));
39
40		let self2 = new_self.clone();
41		let parent2 = parent.clone();
42		let text2 = opts.text.to_owned();
43		parent
44			.as_ref()
45			.before_on()
46			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
47				self2.0.base.create_window(
48					opts.window_ex_style,
49					"STATIC",
50					Some(&text2),
51					opts.window_style | opts.control_style.into(),
52					opts.position.into(),
53					if opts.size == (0, 0) {
54						text_calc::bound_box(&text_calc::remove_accel_ampersands(&text2))
55					} else {
56						opts.size.into()
57					},
58					&parent2,
59				);
60				ui_font::set(self2.hwnd());
61				parent2
62					.as_ref()
63					.add_to_layout(self2.hwnd(), opts.resize_behavior);
64				Ok(0) // ignored
65			});
66
67		new_self
68	}
69
70	/// Instantiates a new `Label` object, to be loaded from a dialog resource
71	/// with [`HWND::GetDlgItem`](crate::HWND::GetDlgItem).
72	///
73	/// # Panics
74	///
75	/// Panics if the parent dialog was already created – that is, you cannot
76	/// dynamically create a `Label` in an event closure.
77	#[must_use]
78	pub fn new_dlg(
79		parent: &(impl GuiParent + 'static),
80		ctrl_id: u16,
81		resize_behavior: (Horz, Vert),
82	) -> Self {
83		let new_self = Self(Arc::pin(LabelObj {
84			base: BaseCtrl::new(ctrl_id),
85			events: BaseCtrlEvents::new(parent, ctrl_id),
86			_pin: PhantomPinned,
87		}));
88
89		let self2 = new_self.clone();
90		let parent2 = parent.clone();
91		parent.as_ref().before_on().wm_init_dialog(move |_| {
92			self2.0.base.assign_dlg(&parent2);
93			parent2
94				.as_ref()
95				.add_to_layout(self2.hwnd(), resize_behavior);
96			Ok(true) // ignored
97		});
98
99		new_self
100	}
101
102	/// Calls [`HWND::SetWindowText`](crate::HWND::SetWindowText) to set the
103	/// text and resizes the control to exactly fit it.
104	pub fn set_text_and_resize(&self, text: &str) -> SysResult<()> {
105		let bound_box = text_calc::bound_box(&text_calc::remove_accel_ampersands(text));
106		self.hwnd().SetWindowPos(
107			HwndPlace::None,
108			POINT::default(),
109			bound_box,
110			co::SWP::NOZORDER | co::SWP::NOMOVE,
111		)?;
112		self.hwnd().SetWindowText(text)?;
113		Ok(())
114	}
115}

117/// Options to create a [`Label`](crate::gui::Label) programmatically with
118/// [`Label::new`](crate::gui::Label::new).
119pub struct LabelOpts<'a> {
120	/// Text of the control to be
121	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
122	///
123	/// Defaults to "Label".
124	pub text: &'a str,
125	/// Left and top position coordinates of control within parent's client
126	/// area, to be
127	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
128	///
129	/// Defaults to `gui::dpi(0, 0)`.
130	pub position: (i32, i32),
131	/// Width and height of control to be
132	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
133	///
134	/// Defaults to the size needed to fit the text.
135	pub size: (i32, i32),
136	/// Label styles to be
137	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
138	///
139	/// Defaults to `SS::LEFT | SS:NOTIFY`.
140	pub control_style: co::SS,
141	/// Window styles to be
142	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
143	///
144	/// Defaults to `WS::CHILD | WS::VISIBLE`.
145	pub window_style: co::WS,
146	/// Extended window styles to be
147	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
148	///
149	/// Defaults to `WS_EX::LEFT`.
150	pub window_ex_style: co::WS_EX,
151
152	/// The control ID.
153	///
154	/// Defaults to an auto-generated ID.
155	pub ctrl_id: u16,
156	/// Horizontal and vertical behavior of the control when the parent window
157	/// is resized.
158	///
159	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
160	pub resize_behavior: (Horz, Vert),
161}
162
163impl<'a> Default for LabelOpts<'a> {
164	fn default() -> Self {
165		Self {
166			text: "Label",
167			position: dpi(0, 0),
168			size: (0, 0), // will resize to fit the text
169			control_style: co::SS::LEFT | co::SS::NOTIFY,
170			window_style: co::WS::CHILD | co::WS::VISIBLE,
171			window_ex_style: co::WS_EX::LEFT,
172			ctrl_id: 0,
173			resize_behavior: (Horz::None, Vert::None),
174		}
175	}
