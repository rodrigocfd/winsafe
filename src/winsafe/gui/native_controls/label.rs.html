52impl GuiNativeControlEvents<LabelEvents> for Label {
53	fn on(&self) -> &LabelEvents {
54		if *self.hwnd() != HWND::NULL {
55			panic!("Cannot add events after the control creation.");
56		} else if *self.0.base.parent().hwnd() != HWND::NULL {
57			panic!("Cannot add events after the parent window creation.");
58		}
59		&self.0.events
60	}

63impl Label {
64	/// Instantiates a new `Label` object, to be created on the parent window
65	/// with
66	/// [`HWND::CreateWindowEx`](crate::prelude::user_Hwnd::CreateWindowEx).
67	///
68	/// # Panics
69	///
70	/// Panics if the parent window was already created â€“ that is, you cannot
71	/// dynamically create a `Label` in an event closure.
72	#[must_use]
73	pub fn new(parent: &impl GuiParent, opts: LabelOpts) -> Self {
74		let opts = auto_ctrl_id_if_zero(opts);
75		let ctrl_id = opts.ctrl_id;
76
77		let new_self = Self(
78			Arc::pin(
79				Obj {
80					base: BaseNativeControl::new(parent, ctrl_id),
81					events: LabelEvents::new(parent, ctrl_id),
82					_pin: PhantomPinned,
83				},
84			),
85		);
86
87		let self2 = new_self.clone();
88		parent.as_ref().before_user_on().wm_create_or_initdialog(move |_, _| {
89			self2.create(OptsResz::Wnd(&opts))?;
90			Ok(WmRet::NotHandled)
91		});
92
93		new_self
94	}
95
96	/// Instantiates a new `CheckBox` object, to be loaded from a dialog
97	/// resource with
98	/// [`HWND::GetDlgItem`](crate::prelude::user_Hwnd::GetDlgItem).
99	///
100	/// # Panics
101	///
102	/// Panics if the parent dialog was already created â€“ that is, you cannot
103	/// dynamically create a `Label` in an event closure.
104	#[must_use]
105	pub fn new_dlg(
106		parent: &impl GuiParent,
107		ctrl_id: u16,
108		resize_behavior: (Horz, Vert)
109	) -> Self
110	{
111		let new_self = Self(
112			Arc::pin(
113				Obj {
114					base: BaseNativeControl::new(parent, ctrl_id),
115					events: LabelEvents::new(parent, ctrl_id),
116					_pin: PhantomPinned,
117				},
118			),
119		);
120
121		let self2 = new_self.clone();
122		parent.as_ref().before_user_on().wm_init_dialog(move |_| {
123			self2.create(OptsResz::Dlg(resize_behavior))?;
124			Ok(false) // return value is discarded
125		});
126
127		new_self
128	}
129
130	fn create(&self, opts_resz: OptsResz<&LabelOpts>) -> SysResult<()> {
131		match opts_resz {
132			OptsResz::Wnd(opts) => {
133				let mut pos = POINT::new(opts.position.0, opts.position.1);
134				multiply_dpi_or_dtu(
135					self.0.base.parent(), Some(&mut pos), None)?;
136
137				let mut sz = SIZE::new(opts.size.0 as _, opts.size.1 as _);
138				if sz.cx == -1 && sz.cy == -1 {
139					sz = calc_text_bound_box(&opts.text)?; // resize to fit text
140				} else {
141					multiply_dpi_or_dtu(
142						self.0.base.parent(), None, Some(&mut sz))?; // user-defined size
143				}
144
145				self.0.base.create_window(
146					"STATIC", Some(&opts.text), pos, sz,
147					opts.window_ex_style,
148					opts.window_style | opts.label_style.into(),
149				)?;
150
151				unsafe {
152					self.hwnd().SendMessage(wm::SetFont {
153						hfont: ui_font(),
154						redraw: true,
155					});
156				}
157			},
158			OptsResz::Dlg(_) => self.0.base.create_dlg()?,
159		}
160
161		self.0.base.parent()
162			.add_to_layout_arranger(self.hwnd(), opts_resz.resize_behavior())
163	}
164
165	/// Calls [`set_text`](crate::prelude::GuiWindowText::set_text) and resizes
166	/// the control to exactly fit the new text.
167	///
168	/// # Examples
169	///
170	/// ```no_run
171	/// use winsafe::{self as w, prelude::*, gui};
172	///
173	/// let my_label: gui::Label; // initialized somewhere
174	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
175	/// # let my_label = gui::Label::new(&wnd, gui::LabelOpts::default());
176	///
177	/// my_label.set_text_and_resize("This my text");
178	/// ```
179	pub fn set_text_and_resize(&self, text: &str) {
180		self.set_text(text);
181		let bound_box = calc_text_bound_box(text).unwrap();
182		self.hwnd().SetWindowPos(
183			HwndPlace::None, POINT::default(), bound_box,
184			co::SWP::NOZORDER | co::SWP::NOMOVE).unwrap();
185	}
186}

188/// Options to create a [`Label`](crate::gui::Label) programmatically with
189/// [`label::new`](crate::gui::Label::new).
190pub struct LabelOpts {
191	/// Text of the control to be
192	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
193	///
194	/// Defaults to "X".
195	pub text: String,
196	/// Left and top position coordinates of control within parent's client
197	/// area, to be
198	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
199	///
200	/// If the parent window is a dialog, the values are in Dialog Template
201	/// Units; otherwise in pixels, which will be multiplied to match current
202	/// system DPI.
203	///
204	/// Defaults to `(0, 0)`.
205	pub position: (i32, i32),
206	/// Width and height of control to be
207	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
208	///
209	/// If the parent window is a dialog, the values are in Dialog Template
210	/// Units; otherwise in pixels, which will be multiplied to match current
211	/// system DPI.
212	///
213	/// Defaults to the size needed to fit the text.
214	pub size: (u32, u32),
215	/// label styles to be
216	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
217	///
218	/// Defaults to `SS::LEFT | SS:NOTIFY`.
219	pub label_style: co::SS,
220	/// Window styles to be
221	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
222	///
223	/// Defaults to `WS::CHILD | WS::VISIBLE`.
224	pub window_style: co::WS,
225	/// Extended window styles to be
226	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
227	///
228	/// Defaults to `WS_EX::LEFT`.
229	pub window_ex_style: co::WS_EX,
230
231	/// The control ID.
232	///
233	/// Defaults to an auto-generated ID.
234	pub ctrl_id: u16,
235	/// Horizontal and vertical behavior of the control when the parent window
236	/// is resized.
237	///
238	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
239	pub resize_behavior: (Horz, Vert),
240}
241
242impl Default for LabelOpts {
243	fn default() -> Self {
244		Self {
245			text: "X".to_owned(),
246			position: (0, 0),
247			size: (-1i32 as _, -1i32 as _), // will resize to fit the text
248			label_style: co::SS::LEFT | co::SS::NOTIFY,
249			window_style: co::WS::CHILD | co::WS::VISIBLE,
250			window_ex_style: co::WS_EX::LEFT,
251			ctrl_id: 0,
252			resize_behavior: (Horz::None, Vert::None),
253		}
254	}
255}
