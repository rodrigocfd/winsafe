24impl Button {
25	/// Instantiates a new `Button` object, to be created on the parent window
26	/// with
27	/// [`HWND::CreateWindowEx`](crate::prelude::user_Hwnd::CreateWindowEx).
28	///
29	/// # Panics
30	///
31	/// Panics if the parent window was already created – that is, you cannot
32	/// dynamically create a `Button` in an event closure.
33	///
34	/// # Examples
35	///
36	/// ```no_run
37	/// use winsafe::{self as w, prelude::*, gui};
38	///
39	/// let wnd: gui::WindowMain; // initialized somewhere
40	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
41	///
42	/// let btn = gui::Button::new(
43	///     &wnd,
44	///     gui::ButtonOpts {
45	///         position: gui::dpi(10, 10),
46	///         text: "&Click me".to_owned(),
47	///         ..Default::default()
48	///     },
49	/// );
50	/// ```
51	#[must_use]
52	pub fn new(parent: &(impl GuiParent + 'static), opts: ButtonOpts) -> Self {
53		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
54		let new_self = Self(Arc::pin(ButtonObj {
55			base: BaseCtrl::new(ctrl_id),
56			events: ButtonEvents::new(parent, ctrl_id),
57			_pin: PhantomPinned,
58		}));
59
60		let self2 = new_self.clone();
61		let parent2 = parent.clone();
62		parent
63			.as_ref()
64			.before_on()
65			.wm(parent.as_ref().is_dlg().create_msg(), move |_| {
66				self2.0.base.create_window(
67					opts.window_ex_style,
68					"BUTTON",
69					Some(&opts.text),
70					opts.window_style | opts.control_style.into(),
71					opts.position.into(),
72					SIZE::new(opts.width, opts.height),
73					&parent2,
74				)?;
75				ui_font::set(self2.hwnd())?;
76				parent2
77					.as_ref()
78					.add_to_layout(self2.hwnd(), opts.resize_behavior)?;
79				Ok(0) // ignored
80			});
81
82		new_self
83	}
84
85	/// Instantiates a new `Button` object, to be loaded from a dialog resource
86	/// with [`HWND::GetDlgItem`](crate::prelude::user_Hwnd::GetDlgItem).
87	///
88	/// # Panics
89	///
90	/// Panics if the parent dialog was already created – that is, you cannot
91	/// dynamically create a `Button` in an event closure.
92	#[must_use]
93	pub fn new_dlg(
94		parent: &(impl GuiParent + 'static),
95		ctrl_id: u16,
96		resize_behavior: (Horz, Vert),
97	) -> Self {
98		let new_self = Self(Arc::pin(ButtonObj {
99			base: BaseCtrl::new(ctrl_id),
100			events: ButtonEvents::new(parent, ctrl_id),
101			_pin: PhantomPinned,
102		}));
103
104		let self2 = new_self.clone();
105		let parent2 = parent.clone();
106		parent.as_ref().before_on().wm_init_dialog(move |_| {
107			self2.0.base.assign_dlg(&parent2)?;
108			parent2
109				.as_ref()
110				.add_to_layout(self2.hwnd(), resize_behavior)?;
111			Ok(true) // ignored
112		});
113
114		new_self
115	}

117	/// Fires the click event for the button by sending a
118	/// [`bm::Click`](crate::msg::bm::Click) message.
119	pub fn trigger_click(&self) {
120		unsafe {
121			self.hwnd().SendMessage(bm::Click {});
122		}
123	}
124}

126/// Options to create a [`Button`](crate::gui::Button) programmatically with
127/// [`Button::new`](crate::gui::Button::new).
128pub struct ButtonOpts {
129	/// Text of the control to be
130	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
131	///
132	/// Defaults to empty string.
133	pub text: String,
134	/// Left and top position coordinates of control within parent's client
135	/// area, to be
136	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
137	///
138	/// Defaults to `gui::dpi(0, 0)`.
139	pub position: (i32, i32),
140	/// Control width to be
141	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
142	///
143	/// Defaults to `gui::dpi_x(88)`.
144	pub width: i32,
145	/// Control height to be
146	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
147	///
148	/// Defaults to `gui::dpi_y(26)`.
149	pub height: i32,
150	/// Button styles to be
151	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
152	///
153	/// Defaults to `BS::PUSHBUTTON`.
154	///
155	/// Suggestions:
156	/// * replace with `BS::DEFPUSHBUTTON` for the default button of the window;
157	/// * add `BS::NOTIFY` to receive notifications other than the simple click.
158	pub control_style: co::BS,
159	/// Window styles to be
160	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
161	///
162	/// Defaults to `WS::CHILD | WS::GROUP | WS::TABSTOP | WS::VISIBLE`.
163	pub window_style: co::WS,
164	/// Extended window styles to be
165	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
166	///
167	/// Defaults to `WS_EX::LEFT`.
168	pub window_ex_style: co::WS_EX,
169
170	/// The control ID.
171	///
172	/// Defaults to an auto-generated ID.
173	pub ctrl_id: u16,
174	/// Horizontal and vertical behavior of the control when the parent window
175	/// is resized.
176	///
177	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
178	pub resize_behavior: (Horz, Vert),
179}
180
181impl Default for ButtonOpts {
182	fn default() -> Self {
183		Self {
184			text: "".to_owned(),
185			position: dpi(0, 0),
186			width: dpi_x(88),
187			height: dpi_y(26),
188			control_style: co::BS::PUSHBUTTON,
189			window_style: co::WS::CHILD | co::WS::GROUP | co::WS::TABSTOP | co::WS::VISIBLE,
190			window_ex_style: co::WS_EX::LEFT,
191			ctrl_id: 0,
192			resize_behavior: (Horz::None, Vert::None),
193		}
194	}
