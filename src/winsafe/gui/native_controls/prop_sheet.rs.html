18/// Native
19/// [property sheet](https://learn.microsoft.com/en-us/windows/win32/controls/property-sheets)
20/// control.
21///
22/// # Examples
23///
24/// The example below creates a property sheet with one
25/// [`PropSheetPage`](crate::gui::PropSheetPage) page.
26///
27/// ```no_run
28/// use winsafe::{self as w, co, gui, prelude::*, seq_ids};
29///
30/// seq_ids! {
31///     DLG_PAGE_1 = 100; // must match the resource dialog
32/// }
33///
34/// fn main() {
35///     let ps = gui::PropSheet::new(gui::PropSheetOpts {
36///         title: "Main title",
37///         pages: &[
38///             ("First page", FirstPage::new().into()),
39///         ],
40///         ..Default::default()
41///     });
42///
43///     // In a real application, the parent window is passed to show().
44///     if let Err(e) = ps.show(None::<&gui::WindowMain>) {
45///         eprintln!("{}", e);
46///     }
47/// }
48///
49/// #[derive(Clone)]
50/// struct FirstPage {
51///     wnd: gui::PropSheetPage,
52/// }
53/// impl Into<gui::PropSheetPage> for FirstPage {
54///     fn into(self) -> gui::PropSheetPage {
55///         self.wnd.clone() // so we can pass our custom struct to PropSheet::new()
56///     }
57/// }
58/// impl FirstPage {
59///     fn new() -> Self {
60///         let wnd = gui::PropSheetPage::new_dlg(DLG_PAGE_1);
61///         let new_self = Self { wnd };
62///         new_self.events();
63///         new_self
64///     }
65///     fn events(&self) {
66///         let self2 = self.clone();
67///         self.wnd.on().psn_apply(move || {
68///             // When the user clicks OK, each page will receive this notification.
69///             println!("OK from the first page");
70///             Ok(co::PSNRET::NOERROR)
71///         });
72///     }
73/// }
74/// ```
75#[derive(Clone)]
76pub struct PropSheet(Pin<Arc<PropSheetObj>>);
77
78unsafe impl Send for PropSheet {}
79
80impl PropSheet {
81	/// Instantiates a new `PropSheet` object, to be created with
82	/// [`PropertySheet`](crate::PropertySheet).
83	///
84	/// # Panics
85	///
86	/// Panics if [`PropSheetOpts`](crate::gui::PropSheetOpts) has no pages.
87	#[must_use]
88	pub fn new(opts: PropSheetOpts) -> Self {
89		if opts.pages.is_empty() {
90			panic!("Property sheet has no pages.");
91		}
92
93		let (page_titles, pages): (Vec<WString>, Vec<PropSheetPage>) = opts
94			.pages
95			.into_iter()
96			.map(|(page_title, page)| (WString::from_str(*page_title), page.clone()))
97			.unzip();
98
99		Self(Arc::pin(PropSheetObj {
100			title: opts.title.to_owned(),
101			page_titles,
102			pages,
103			apply_button: opts.apply_button,
104			context_help: opts.context_help,
105			margin: opts.margin,
106		}))
107	}
108
109	/// Physically creates the property sheet window. This method will block
110	/// until the window is closed.
111	///
112	/// # Panics
113	///
114	/// Panics if the window is already created.
115	pub fn show(&self, parent: Option<&impl GuiParent>) -> AnyResult<()> {
116		let ps_pages = self
117			.0
118			.pages
119			.iter()
120			.zip(&self.0.page_titles)
121			.map(|(page, page_title)| {
122				let mut ps_page = page.generate_propsheetpage();
123				ps_page.pszTitle = page_title.as_ptr();
124				ps_page
125			})
126			.collect::<Vec<_>>();
127
128		let mut psh = PROPSHEETHEADER::default();
129		psh.dwFlags = co::PSH::PROPSHEETPAGE;
130		if !self.0.apply_button {
131			psh.dwFlags |= co::PSH::NOAPPLYNOW;
132		}
133		if !self.0.context_help {
134			psh.dwFlags |= co::PSH::NOCONTEXTHELP;
135		}
136		if !self.0.margin {
137			psh.dwFlags |= co::PSH::NOMARGIN;
138		}
139		// psh.dwFlags = co::PSH::USECALLBACK;
140		// psh.pfnCallback = Some(Self::ps_proc);
141
142		match parent {
143			Some(parent) => {
144				psh.hInstance = parent.hwnd().hinstance();
145				psh.hwndParent = unsafe { parent.hwnd().raw_copy() };
146			},
147			None => {
148				psh.hInstance = HINSTANCE::GetModuleHandle(None).expect(DONTFAIL);
149			},
150		}
151
152		let mut wcaption = WString::from_str(&self.0.title);
153		psh.set_pszCaption(Some(&mut wcaption));
154
155		psh.nPages = ps_pages.len() as _;
156		psh.set_ppsp(&ps_pages);
157
158		unsafe {
159			PropertySheet(&psh)?;
160		}
161		Ok(())
162	}
163}

165/// Options to create a [`PropSheet`](crate::gui::PropSheet) programmatically
166/// with [`PropSheet::new`](crate::gui::PropSheet::new).
167pub struct PropSheetOpts<'a> {
168	/// Title of the property sheet container window.
169	///
170	/// Defaults to an empty string.
171	pub title: &'a str,
172	/// Pages to be added to the property sheet container window. The tuple
173	/// contains the title of the page and the window to be rendered inside of
174	/// it.
175	///
176	/// Defaults to none.
177	pub pages: &'a [(&'a str, PropSheetPage)],
178	/// Display the "Apply" button?
179	///
180	/// Defaults to `true`.
181	pub apply_button: bool,
182	/// Display the context help button in the titlebar?
183	///
184	/// Defaults to `false`.
185	pub context_help: bool,
186	/// Insert a margin between the page and the frame?
187	///
188	/// Defaults to `true`.
189	pub margin: bool,
190}
191
192impl<'a> Default for PropSheetOpts<'a> {
193	fn default() -> Self {
194		Self {
195			title: "",
196			pages: &[],
197			apply_button: true,
198			context_help: false,
199			margin: true,
200		}
201	}
202}
