15struct ListViewObj<T> {
16	base: BaseCtrl,
17	events: ListViewEvents,
18	context_menu: Option<DestroyMenuGuard>, // the context menu itself is the 1st submenu
19	header: UnsafeCell<Option<Header>>, // if doesn't exist, will be set to None on WM_CREATE and WM_INITDIALOG
20	_pin: PhantomPinned,
21	_data: PhantomData<T>,
22}

25	/// Native
26	/// [list view](https://learn.microsoft.com/en-us/windows/win32/controls/list-view-controls-overview)
27	/// control. Not to be confused with the simpler
28	/// [list box](crate::gui::ListBox) control.
29	///
30	/// The generic parameter specifies the type of the object that will be
31	/// embedded on each item – if you don't want to store anything, just use
32	/// `()` as the type. Internally, this storage is implemented with pointers
33	/// in the item's `LPARAM` fields.
34	///
35	/// You can have access to the internal header of the list view by creating
36	/// a [`Header`](crate::gui::Header) object.
37}

40	/// Instantiates a new `ListView` object, to be created on the parent window
41	/// with
42	/// [`HWND::CreateWindowEx`](crate::prelude::user_Hwnd::CreateWindowEx).
43	///
44	/// # Panics
45	///
46	/// Panics if the parent window was already created – that is, you cannot
47	/// dynamically create a `ListView` in an event closure.
48	#[must_use]
49	pub fn new(parent: &(impl GuiParent + 'static), opts: ListViewOpts) -> Self {
50		let mut opts = opts;
51		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
52		let context_menu = opts.context_menu.take();
53
54		let new_self = Self(
55			Arc::pin(
56				ListViewObj {
57					base: BaseCtrl::new(ctrl_id),
58					events: ListViewEvents::new(parent, ctrl_id),
59					context_menu,
60					header: UnsafeCell::new(Some(Header::from_list_view(parent))), // initially does exist
61					_pin: PhantomPinned,
62					_data: PhantomData,
63				},
64			),
65		);
66
67		let self2 = new_self.clone();
68		let parent2 = parent.clone();
69		parent.as_ref().before_on().wm(parent.as_ref().is_dlg().create_msg(), move |_| {
70			self2.0.base.create_window(opts.window_ex_style, "SysListView32", None,
71				opts.window_style | opts.control_style.into() | co::LVS::SHAREIMAGELISTS.into(),
72				opts.position.into(), opts.size.into(), &parent2)?;
73			if opts.control_ex_style != co::LVS_EX::NoValue {
74				self2.set_extended_style(opts.control_ex_style, true);
75			}
76			if !unsafe { &*self2.0.header.get() }.as_ref().unwrap().init_nested(self2.hwnd()) {
77				*unsafe { &mut *self2.0.header.get() } = None; // no header, delete it
78			}
79			for (text, cx) in opts.columns.iter() {
80				self2.cols().add(text, *cx)?;
81			}
82			parent2.as_ref().add_to_layout(self2.hwnd(), opts.resize_behavior)?;
83			Ok(0) // ignored
84		});
85
86		new_self.default_message_handlers(parent);
87		new_self
88	}

90	/// Instantiates a new `ListView` object, to be loaded from a dialog
91	/// resource with
92	/// [`HWND::GetDlgItem`](crate::prelude::user_Hwnd::GetDlgItem).
93	///
94	/// The `context_menu_id` must point to the root menu whose first submenu
95	/// will be effectively displayed as the context menu.
96	///
97	/// Since the image list is always managed by the control itself,
98	/// [`LVS::SHAREIMAGELISTS`](crate::co::LVS::SHAREIMAGELISTS) style will
99	/// always be added.
100	///
101	/// # Panics
102	///
103	/// Panics if the parent dialog was already created – that is, you cannot
104	/// dynamically create a `ListView` in an event closure.
105	///
106	/// Panics if the context menu, when specified, does not exist.
107	#[must_use]
108	pub fn new_dlg(
109		parent: &(impl GuiParent + 'static),
110		ctrl_id: u16,
111		resize_behavior: (Horz, Vert),
112		context_menu_id: Option<u16>,
113	) -> Self
114	{
115		let new_self = Self(
116			Arc::pin(
117				ListViewObj {
118					base: BaseCtrl::new(ctrl_id),
119					events: ListViewEvents::new(parent, ctrl_id),
120					context_menu: context_menu_id.map(|id|
121						parent.hwnd()
122							.hinstance()
123							.LoadMenu(IdStr::Id(id))
124							.expect("Invalid ListView context menu ID"),
125					),
126					header: UnsafeCell::new(Some(Header::from_list_view(parent))), // initially does exist
127					_pin: PhantomPinned,
128					_data: PhantomData,
129				},
130			),
131		);
132
133		let self2 = new_self.clone();
134		let parent2 = parent.clone();
135		parent.as_ref().before_on().wm_init_dialog(move |_| {
136			self2.0.base.assign_dlg(&parent2)?;
137			self2.hwnd().set_style(co::LVS::from(self2.hwnd().style()) | co::LVS::SHAREIMAGELISTS);
138			if !unsafe { &*self2.0.header.get() }.as_ref().unwrap().init_nested(self2.hwnd()) {
139				*unsafe { &mut *self2.0.header.get() } = None; // no header, delete it
140			}
141			parent2.as_ref().add_to_layout(self2.hwnd(), resize_behavior)?;
142			Ok(true) // ignored
143		});
144
145		new_self.default_message_handlers(parent);
146		new_self
147	}

149	fn default_message_handlers(&self, parent: &impl AsRef<BaseWnd>) {
150		let self2 = self.clone();
151		self.on_subclass().wm_get_dlg_code(move |p| {
152			if !p.is_query && p.vkey_code == co::VK::RETURN { // Enter key
153				let mut nmlvkd = NMLVKEYDOWN::default();
154				nmlvkd.hdr.hwndFrom = unsafe { self2.hwnd().raw_copy() };
155				nmlvkd.hdr.set_idFrom(self2.ctrl_id());
156				nmlvkd.hdr.code = co::LVN::KEYDOWN.into();
157				nmlvkd.wVKey = co::VK::RETURN;
158
159				let hparent = self2.hwnd().GetAncestor(co::GA::PARENT).unwrap();
160				unsafe { hparent.SendMessage(wm::Notify { nmhdr: &mut nmlvkd.hdr }); } // send Enter key to parent
161			}
162			let dlgc_system = unsafe { self2.hwnd().DefSubclassProc::<wm::GetDlgCode>(p.into()) };
163			Ok(dlgc_system)
164		});
165
166		let self2 = self.clone();
167		parent.as_ref().before_on().wm_notify(self.ctrl_id(), co::LVN::KEYDOWN, move |p| {
168			let lvnk = unsafe { p.cast_nmhdr::<NMLVKEYDOWN>() };
169			let has_ctrl = GetAsyncKeyState(co::VK::CONTROL);
170			let has_shift = GetAsyncKeyState(co::VK::SHIFT);
171
172			if has_ctrl && lvnk.wVKey == co::VK::CHAR_A { // Ctrl+A
173				self2.items().select_all(true)?;
174			} else if lvnk.wVKey == co::VK::APPS { // context menu key
175				self2.show_context_menu(false, has_ctrl, has_shift)?;
176			}
177			Ok(0) // ignored
178		});
179
180		let self2 = self.clone();
181		parent.as_ref().before_on().wm_notify(self.ctrl_id(), co::NM::RCLICK, move |p| {
182			let nmia = unsafe { p.cast_nmhdr::<NMITEMACTIVATE>() };
183			let has_ctrl = nmia.uKeyFlags.has(co::LVKF::CONTROL);
184			let has_shift = nmia.uKeyFlags.has(co::LVKF::SHIFT);
185
186			self2.show_context_menu(true, has_ctrl, has_shift)?;
187			Ok(0) // ignored
188		});
189
190		let self2 = self.clone();
191		parent.as_ref().after_on().wm_notify(self.ctrl_id(), co::LVN::DELETEITEM, move |p| {
192			let nmlv = unsafe { p.cast_nmhdr::<NMLISTVIEW>() };
193			let rc_ptr = self2.items().get(nmlv.iItem as _).data_lparam()?;
194			if !rc_ptr.is_null() {
195				let _ = unsafe { Rc::from_raw(rc_ptr) }; // free allocated LPARAM
196			}
197			Ok(0) // ignored
198		});
199
200		let self2 = self.clone();
201		parent.as_ref().after_on().wm_destroy(move || {
202			[co::LVSIL::NORMAL, co::LVSIL::SMALL, co::LVSIL::STATE, co::LVSIL::GROUPHEADER]
203				.iter()
204				.for_each(|lvsil| {
205					self2.image_list(*lvsil).map(|hil| { // destroy each image list, if any
206						let _ = unsafe { ImageListDestroyGuard::new(hil.raw_copy()) };
207					});
208				});
209			Ok(())
210		});
211	}

213	fn show_context_menu(&self,
214		follow_cursor: bool,
215		has_ctrl: bool,
216		has_shift: bool,
217	) -> SysResult<()>
218	{
219		let hmenu = match self.context_menu() {
220			Some(h) => h,
221			None => return Ok(()), // no menu, nothing to do
222		};
223
224		let menu_pos = if follow_cursor { // usually when fired by a right-click
225			let menu_pos = self.hwnd().ScreenToClient(
226				GetCursorPos()?, // relative to screen
227			)?; // now relative to list view
228
229			match self.items().hit_test(menu_pos) {
230				Some(item_over) => {
231					if !has_ctrl && !has_shift {
232						item_over.select(true)?; // if not yet
233						item_over.focus()?;
234					}
235				},
236				None => self.items().select_all(false)?, // no item was right-clicked
237			}
238
239			self.focus()?; // because a right-click won't set the focus by itself
240			menu_pos
241
242		} else { // usually fired by the context menu key
243			let focused_opt = self.items().focused();
244
245			if focused_opt.is_some() && focused_opt.unwrap().is_visible() {
246				let focused = focused_opt.unwrap();
247				let rc_item = focused.rect(co::LVIR::BOUNDS)?;
248				POINT::new(rc_item.left + 16,
249					rc_item.top + (rc_item.bottom - rc_item.top) / 2)
250
251			} else { // no item is focused and visible
252				POINT::new(6, 10) // arbitrary coordinates
253			}
254		};
255
256		hmenu.track_popup_menu_at_point(
257			menu_pos, &self.hwnd().GetParent()?, self.hwnd())
258	}

284	/// Returns the embedded [`Header`](crate::gui::Header) of the list view, if
285	/// any.
286	///
287	/// The `Header` is tried to be initialized during the internal
288	/// [`wm::Create`](crate::msg::wm::Create) and
289	/// [`wm::InitDialog`](crate::msg::wm::InitDialog) processing. If the list
290	/// view doesn't happen to have a header – only report view lists have
291	/// headers –, the internal `Header` is removed, and this method will
292	/// forever return `None`.
293	///
294	/// This means you may call `header()` to add events to the internal
295	/// `Header`, but if the list view has no header, these events will simply be
296	/// discarded.
297	///
298	/// # Examples
299	///
300	/// ```no_run
301	/// use winsafe::{self as w, prelude::*, gui};
302	///
303	/// let my_list: gui::ListView; // initialized somewhere
304	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
305	/// # let my_list = gui::ListView::<()>::new(&wnd, gui::ListViewOpts::default());
306	///
307	/// let my_list2 = my_list.clone();
308	/// my_list.header().unwrap().on().hdn_item_click(move |p| {
309	///     let h = my_list2.header().unwrap().items().get(p.iItem as _);
310	///     println!("Col: {}", h.text());
311	///     Ok(())
312	/// });
313	/// ```
314	#[must_use]
315	pub fn header(&self) -> Option<&Header> {
316		unsafe { &*self.0.header.get() }.as_ref()
317	}

319	/// Retrieves a reference to one of the associated image lists by sending an
320	/// [`lvm::GetImageList`](crate::msg::lvm::GetImageList) message.
321	///
322	/// The image list is owned by the control.
323	#[must_use]
324	pub fn image_list(&self, kind: co::LVSIL) -> Option<&HIMAGELIST> {
325		unsafe {
326			self.hwnd()
327				.SendMessage(lvm::GetImageList { kind })
328		}.map(|hil| {
329			let hil_ptr = &hil as *const HIMAGELIST;
330			unsafe { &*hil_ptr }
331		})
332	}

361	/// Sets the one of the associated image lists by sending an
362	/// [`lvm::SetImageList`](crate::msg::lvm::SetImageList) message.
363	///
364	/// The image list will be owned by the control. Returns the previous one,
365	/// if any.
366	pub fn set_image_list(&self,
367		kind: co::LVSIL,
368		himagelist: ImageListDestroyGuard,
369	) -> Option<ImageListDestroyGuard>
370	{
371		let mut himagelist = himagelist;
372		let hil = himagelist.leak();
373
374		unsafe {
375			self.hwnd()
376				.SendMessage(lvm::SetImageList { kind, himagelist: Some(hil) })
377				.map(|prev_hil| ImageListDestroyGuard::new(prev_hil))
378		}
379	}

391/// Options to create a [`ListView`](crate::gui::ListView) programmatically with
392/// [`ListView::new`](crate::gui::ListView::new).
393pub struct ListViewOpts {
394	/// Left and top position coordinates of control within parent's client
395	/// area, to be
396	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
397	///
398	/// Defaults to `gui::dpi(0, 0)`.
399	pub position: (i32, i32),
400	/// Width and height of control to be
401	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
402	///
403	/// Defaults to `gui::dpi(120, 120)`.
404	pub size: (i32, i32),
405	/// List view styles to be
406	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
407	///
408	/// Since the image list is always managed by the control itself,
409	/// [`LVS::SHAREIMAGELISTS`](crate::co::LVS::SHAREIMAGELISTS) style will
410	/// always be added.
411	///
412	/// Defaults to `LVS::REPORT | LVS::NOSORTHEADER | LVS::SHOWSELALWAYS | LVS::SHAREIMAGELISTS`.
413	pub control_style: co::LVS,
414	/// Extended list view styles to be
415	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
416	///
417	/// Defaults to `LVS_EX::FULLROWSELECT`.
418	pub control_ex_style: co::LVS_EX,
419	/// Window styles to be
420	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
421	///
422	/// Defaults to `WS::CHILD | WS::GROUP | WS::TABSTOP | WS::VISIBLE`.
423	pub window_style: co::WS,
424	/// Extended window styles to be
425	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
426	///
427	/// Defaults to `WS_EX::LEFT | WS_EX::CLIENTEDGE`.
428	pub window_ex_style: co::WS_EX,
429
430	/// The control ID.
431	///
432	/// Defaults to an auto-generated ID.
433	pub ctrl_id: u16,
434	/// Horizontal and vertical behavior of the control when the parent window
435	/// is resized.
436	///
437	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
438	pub resize_behavior: (Horz, Vert),
439
440	/// Context popup menu.
441	///
442	/// This menu will be owned by the control, which will automatically destroy
443	/// it.
444	///
445	/// The first submenu is the one which will be effectively displayed as the
446	/// context menu.
447	///
448	/// Defaults to `None`.
449	pub context_menu: Option<DestroyMenuGuard>,
450	/// Text and width of columns to be added. The columns only show in report
451	/// mode.
452	///
453	/// Defaults to none.
454	pub columns: Vec<(String, i32)>,
455}
456
457impl Default for ListViewOpts {
458	fn default() -> Self {
459		Self {
460			position: dpi(0, 0),
461			size: dpi(120, 120),
462			control_style: co::LVS::REPORT | co::LVS::NOSORTHEADER | co::LVS::SHOWSELALWAYS | co::LVS::SHAREIMAGELISTS,
463			control_ex_style: co::LVS_EX::FULLROWSELECT,
464			window_style: co::WS::CHILD | co::WS::GROUP | co::WS::TABSTOP | co::WS::VISIBLE,
465			window_ex_style: co::WS_EX::LEFT | co::WS_EX::CLIENTEDGE,
466			ctrl_id: 0,
467			resize_behavior: (Horz::None, Vert::None),
468			context_menu: None,
469			columns: Vec::new(),
470		}
471	}
