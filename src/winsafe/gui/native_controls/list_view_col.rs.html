7/// A single column of a [`ListView`](crate::gui::ListView) control.
8///
9/// **Note:** Each object keeps the zero-based index of a column. If new columns
10/// are added/removed from the list view control, the object may then point to a
11/// different item.
12///
13/// You cannot directly instantiate this object, it is created internally by the
14/// control.
15pub struct ListViewCol<'a, T: 'static = ()> {
16	owner: &'a ListView<T>,
17	index: u32,
18}
19
20impl<'a, T> Clone for ListViewCol<'a, T> {
21	// https://stackoverflow.com/q/39415052/6923555
22	fn clone(&self) -> Self {
23		Self { owner: self.owner, index: self.index }
24	}
25}
26impl<'a, T> Copy for ListViewCol<'a, T> {}
27
28impl<'a, T> ListViewCol<'a, T> {
29	#[must_use]
30	pub(in crate::gui) const fn new(owner: &'a ListView<T>, index: u32) -> Self {
31		Self { owner, index }
32	}
33
34	/// Returns the zero-based index of the column.
35	#[must_use]
36	pub const fn index(&self) -> u32 {
37		self.index
38	}
39
40	/// Sets the title of the column by sending an
41	/// [`lvm::SetColumn`](crate::msg::lvm::SetColumn) message.
42	///
43	/// Returns the same column, so further operations can be chained.
44	pub fn set_title(&self, text: &str) -> SysResult<Self> {
45		let mut lvc = LVCOLUMN::default();
46		lvc.iSubItem = self.index as _;
47		lvc.mask = co::LVCF::TEXT;
48
49		let mut buf = WString::from_str(text);
50		lvc.set_pszText(Some(&mut buf));
51
52		unsafe {
53			self.owner
54				.hwnd()
55				.SendMessage(lvm::SetColumn { index: self.index, lvcolumn: &mut lvc })?;
56		}
57		Ok(*self)
58	}
59
60	/// Sets the width of the column by sending an
61	/// [`lvm::SetColumnWidth`](crate::msg::lvm::SetColumnWidth) message.
62	///
63	/// Returns the same column, so further operations can be chained.
64	pub fn set_width(&self, width: i32) -> SysResult<Self> {
65		unsafe {
66			self.owner
67				.hwnd()
68				.SendMessage(lvm::SetColumnWidth { index: self.index, width: width as _ })?;
69		}
70		Ok(*self)
71	}
72
73	/// Sets the width of the column by sending an
74	/// [`lvm::SetColumnWidth`](crate::msg::lvm::SetColumnWidth) message. The
75	/// width will be calculated to fill the remaining space.
76	///
77	/// Returns the same column, so further operations can be chained.
78	pub fn set_width_to_fill(&self) -> SysResult<Self> {
79		let num_cols = self.owner.cols().count()?;
80		if num_cols > 0 {
81			let mut cx_used = 0;
82
83			for i in 0..num_cols {
84				if i != self.index {
85					cx_used += self.owner.cols().get(i).width()?; // retrieve cx of each column, but us
86				}
87			}
88
89			let rc = self.owner.hwnd().GetClientRect()?; // list view client area
90
91			unsafe {
92				self.owner.hwnd().SendMessage(lvm::SetColumnWidth {
93					index: self.index,
94					width: std::cmp::max(0, rc.right - cx_used as i32) as _, // avoid negative values
95				})?;
96			}
97		}
98		Ok(*self)
99	}
100
101	/// Retrieves the title of the column by sending an
102	/// [`lvm::GetColumn`](crate::msg::lvm::GetColumn) message.
103	#[must_use]
104	pub fn title(&self) -> SysResult<String> {
105		let mut lvc = LVCOLUMN::default();
106		lvc.iSubItem = self.index as _;
107		lvc.mask = co::LVCF::TEXT;
108
109		let mut buf = WString::new_alloc_buf(128); // arbitrary
110		lvc.set_pszText(Some(&mut buf));
111
112		unsafe {
113			self.owner
114				.hwnd()
115				.SendMessage(lvm::GetColumn { index: self.index, lvcolumn: &mut lvc })?;
116		}
117		Ok(buf.to_string())
118	}
119
120	/// Retrieves the width of the column by sending an
121	/// [`lvm::GetColumnWidth`](crate::msg::lvm::GetColumnWidth) message.
122	#[must_use]
123	pub fn width(&self) -> SysResult<u32> {
124		unsafe {
125			self.owner
126				.hwnd()
127				.SendMessage(lvm::GetColumnWidth { index: self.index })
128		}
129	}
130}
