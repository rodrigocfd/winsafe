25impl Header {
26	/// Instantiates a new `Header` object, to be created on the parent window
27	/// with
28	/// [`HWND::CreateWindowEx`](crate::prelude::user_Hwnd::CreateWindowEx).
29	///
30	/// # Panics
31	///
32	/// Panics if the parent window was already created – that is, you cannot
33	/// dynamically create a `Header` in an event closure.
34	#[must_use]
35	pub fn new(parent: &(impl GuiParent + 'static), opts: HeaderOpts) -> Self {
36		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
37		let new_self = Self(Arc::pin(HeaderObj {
38			base: BaseCtrl::new(ctrl_id),
39			events: HeaderEvents::new(parent, ctrl_id),
40			_pin: PhantomPinned,
41		}));
42
43		let self2 = new_self.clone();
44		let parent2 = parent.clone();
45		parent
46			.as_ref()
47			.before_on()
48			.wm(parent.as_ref().is_dlg().create_msg(), move |_| {
49				self2.0.base.create_window(
50					opts.window_ex_style,
51					"SysHeader32",
52					None,
53					opts.window_style | opts.control_style.into(),
54					opts.position.into(),
55					SIZE::new(opts.width, opts.height),
56					&parent2,
57				)?;
58				ui_font::set(self2.hwnd())?;
59				for (text, width) in opts.items.iter() {
60					self2.items().add(text, *width)?;
61				}
62				parent2
63					.as_ref()
64					.add_to_layout(self2.hwnd(), opts.resize_behavior)?;
65				Ok(0) // ignored
66			});
67
68		new_self
69	}
70
71	/// Instantiates a new `Header` object, to be loaded from a dialog
72	/// resource with
73	/// [`HWND::GetDlgItem`](crate::prelude::user_Hwnd::GetDlgItem).
74	///
75	/// # Panics
76	///
77	/// Panics if the parent dialog was already created – that is, you cannot
78	/// dynamically create a `Header` in an event closure.
79	#[must_use]
80	pub fn new_dlg(
81		parent: &(impl GuiParent + 'static),
82		ctrl_id: u16,
83		resize_behavior: (Horz, Vert),
84	) -> Self {
85		let new_self = Self(Arc::pin(HeaderObj {
86			base: BaseCtrl::new(ctrl_id),
87			events: HeaderEvents::new(parent, ctrl_id),
88			_pin: PhantomPinned,
89		}));
90
91		let self2 = new_self.clone();
92		let parent2 = parent.clone();
93		parent.as_ref().before_on().wm_init_dialog(move |_| {
94			self2.0.base.assign_dlg(&parent2)?;
95			parent2
96				.as_ref()
97				.add_to_layout(self2.hwnd(), resize_behavior)?;
98			Ok(true) // ignored
99		});
100
101		new_self
102	}

115	/// For the nested Header inside ListView.
116	#[must_use]
117	pub(in crate::gui) fn init_nested(&self, hlist: &HWND) -> bool {
118		if let Ok(hheader) = unsafe { hlist.SendMessage(lvm::GetHeader {}) } {
119			unsafe {
120				hheader.SetWindowLongPtr(co::GWLP::ID, self.ctrl_id() as _); // give the header an ID; initially zero
121			}
122			self.0.base.set_hwnd(hheader);
123			true // header object initialized
124		} else {
125			false // not inialized
126		}
127	}

129	/// Retrieves a reference to one of the associated image lists by sending an
130	/// [`hdm::GetImageList`](crate::msg::hdm::GetImageList) message.
131	///
132	/// The image list is owned by the control.
133	#[must_use]
134	pub fn image_list(&self, kind: co::HDSIL) -> Option<&HIMAGELIST> {
135		unsafe { self.hwnd().SendMessage(hdm::GetImageList { kind }) }.map(|hil| {
136			let hil_ptr = &hil as *const HIMAGELIST;
137			unsafe { &*hil_ptr }
138		})
139	}
140
141	/// Item methods.
142	#[must_use]
143	pub const fn items(&self) -> HeaderItems<'_> {
144		HeaderItems::new(self)
145	}
146
147	/// Sets the one of the associated image lists by sending an
148	/// [`hdm::SetImageList`](crate::msg::hdm::SetImageList) message.
149	///
150	/// The image list will be owned by the control. Returns the previous one,
151	/// if any.
152	pub fn set_image_list(
153		&self,
154		kind: co::HDSIL,
155		himagelist: ImageListDestroyGuard,
156	) -> Option<ImageListDestroyGuard> {
157		let mut himagelist = himagelist;
158		let hil = himagelist.leak();
159
160		unsafe {
161			self.hwnd()
162				.SendMessage(hdm::SetImageList { kind, himagelist: Some(hil) })
163				.map(|prev_hil| ImageListDestroyGuard::new(prev_hil))
164		}
165	}

168/// Options to create a [`Header`](crate::gui::Header) programmatically with
169/// [`Header::new`](crate::gui::Header::new).
170pub struct HeaderOpts {
171	/// Left and top position coordinates of control within parent's client
172	/// area, to be
173	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
174	///
175	/// Defaults to `gui::dpi(0, 0)`.
176	pub position: (i32, i32),
177	/// Control width to be
178	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
179	///
180	/// Defaults to `gui::dpi_x(100)`.
181	pub width: i32,
182	/// Control height to be
183	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
184	///
185	/// Defaults to `gui::dpi_y(23)`.
186	pub height: i32,
187	/// Header styles to be
188	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
189	///
190	/// Defaults to `HDS::BUTTONS | HDS::HORZ`.
191	pub control_style: co::HDS,
192	/// Window styles to be
193	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
194	///
195	/// Defaults to `WS::BORDER | WS::CHILD | WS::GROUP | WS::TABSTOP | WS::VISIBLE`.
196	pub window_style: co::WS,
197	/// Extended window styles to be
198	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
199	///
200	/// Defaults to `WS_EX::LEFT`.
201	pub window_ex_style: co::WS_EX,
202
203	/// The control ID.
204	///
205	/// Defaults to an auto-generated ID.
206	pub ctrl_id: u16,
207	/// Horizontal and vertical behavior of the control when the parent window
208	/// is resized.
209	///
210	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
211	pub resize_behavior: (Horz, Vert),
212
213	/// Items to be added to the control. Each item is composed by the text and
214	/// the width.
215	///
216	/// Defaults to none.
217	pub items: Vec<(String, i32)>,
218}
219
220impl Default for HeaderOpts {
221	fn default() -> Self {
222		Self {
223			position: dpi(0, 0),
224			width: dpi_x(100),
225			height: dpi_y(23),
226			control_style: co::HDS::BUTTONS | co::HDS::HORZ,
227			window_style: co::WS::BORDER
228				| co::WS::CHILD
229				| co::WS::GROUP
230				| co::WS::TABSTOP
231				| co::WS::VISIBLE,
232			window_ex_style: co::WS_EX::LEFT,
233			ctrl_id: 0,
234			resize_behavior: (Horz::None, Vert::None),
235			items: Vec::default(),
236		}
237	}
