24impl ListBox {
25	/// Instantiates a new `ListBox` object, to be created on the parent window
26	/// with [`HWND::CreateWindowEx`](crate::HWND::CreateWindowEx).
27	///
28	/// # Panics
29	///
30	/// Panics if the parent window was already created – that is, you cannot
31	/// dynamically create a `ListBox` in an event closure.
32	#[must_use]
33	pub fn new(parent: &(impl GuiParent + 'static), opts: ListBoxOpts) -> Self {
34		let ctrl_id = auto_id::set_if_zero(opts.ctrl_id);
35		let new_self = Self(Arc::pin(ListBoxObj {
36			base: BaseCtrl::new(ctrl_id),
37			events: BaseCtrlEvents::new(parent, ctrl_id),
38			_pin: PhantomPinned,
39		}));
40
41		let self2 = new_self.clone();
42		let parent2 = parent.clone();
43		let items2 = opts.owned_items();
44		parent
45			.as_ref()
46			.before_on()
47			.wm(parent.as_ref().wnd_ty().creation_msg(), move |_| {
48				self2.0.base.create_window(
49					opts.window_ex_style,
50					"ListBox",
51					None,
52					opts.window_style | opts.control_style.into(),
53					opts.position.into(),
54					opts.size.into(),
55					&parent2,
56				);
57				ui_font::set(self2.hwnd());
58				self2.items().add(&items2)?;
59				parent2
60					.as_ref()
61					.add_to_layout(self2.hwnd(), opts.resize_behavior);
62				Ok(0) // ignored
63			});
64
65		new_self
66	}
67
68	/// Instantiates a new `ListBox` object, to be loaded from a dialog resource
69	/// with [`HWND::GetDlgItem`](crate::HWND::GetDlgItem).
70	///
71	/// # Panics
72	///
73	/// Panics if the parent dialog was already created – that is, you cannot
74	/// dynamically create a `ListBox` in an event closure.
75	#[must_use]
76	pub fn new_dlg(
77		parent: &(impl GuiParent + 'static),
78		ctrl_id: u16,
79		resize_behavior: (Horz, Vert),
80	) -> Self {
81		let new_self = Self(Arc::pin(ListBoxObj {
82			base: BaseCtrl::new(ctrl_id),
83			events: BaseCtrlEvents::new(parent, ctrl_id),
84			_pin: PhantomPinned,
85		}));
86
87		let self2 = new_self.clone();
88		let parent2 = parent.clone();
89		parent.as_ref().before_on().wm_init_dialog(move |_| {
90			self2.0.base.assign_dlg(&parent2);
91			parent2
92				.as_ref()
93				.add_to_layout(self2.hwnd(), resize_behavior);
94			Ok(true) // ignored
95		});
96
97		new_self
98	}

100	/// Item methods.

103		ListBoxItems::new(self)
104	}

107	/// [`lb::SetHorizontalExtent`](crate::msg::lb::SetHorizontalExtent) message.

113	}

116/// Options to create a [`ListBox`](crate::gui::ListBox) programmatically with
117/// [`ListBox::new`](crate::gui::ListBox::new).
118pub struct ListBoxOpts<'a> {
119	/// Left and top position coordinates of control within parent's client
120	/// area, to be
121	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
122	///
123	/// Defaults to `gui::dpi(0, 0)`.
124	pub position: (i32, i32),
125	/// Width and height of control to be
126	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
127	///
128	/// Defaults to `gui::dpi(120, 120)`.
129	pub size: (i32, i32),
130	/// List box styles to be
131	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
132	///
133	/// To allow multiple selection, add `LBS::MULTIPLESEL`.
134	///
135	/// Defaults to `LBS::NOTIFY`.
136	pub control_style: co::LBS,
137	/// Window styles to be
138	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
139	///
140	/// Defaults to `WS::CHILD | WS::GROUP | WS::TABSTOP | WS::VISIBLE`.
141	pub window_style: co::WS,
142	/// Extended window styles to be
143	/// [created](https://learn.microsoft.com/en-us/windows/win32/api/winuser/nf-winuser-createwindowexw).
144	///
145	/// Defaults to `WS_EX::LEFT | WS_EX::CLIENTEDGE`.
146	pub window_ex_style: co::WS_EX,
147
148	/// The control ID.
149	///
150	/// Defaults to an auto-generated ID.
151	pub ctrl_id: u16,
152	/// Horizontal and vertical behavior of the control when the parent window
153	/// is resized.
154	///
155	/// Defaults to `(gui::Horz::None, gui::Vert::None)`.
156	pub resize_behavior: (Horz, Vert),
157
158	/// Items to be added.
159	///
160	/// Defaults to none.
161	pub items: &'a [&'a str],
162}
163
164impl<'a> Default for ListBoxOpts<'a> {
165	fn default() -> Self {
166		Self {
167			position: dpi(0, 0),
168			size: dpi(120, 120),
169			control_style: co::LBS::NOTIFY,
170			window_style: co::WS::CHILD | co::WS::GROUP | co::WS::TABSTOP | co::WS::VISIBLE,
171			window_ex_style: co::WS_EX::LEFT | co::WS_EX::CLIENTEDGE,
172			ctrl_id: 0,
173			resize_behavior: (Horz::None, Vert::None),
174			items: &[],
175		}
176	}
177}
178
179impl<'a> ListBoxOpts<'a> {
180	#[must_use]
181	fn owned_items(&self) -> Vec<String> {
182		self.items.iter().map(|item| (*item).to_owned()).collect()
183	}
