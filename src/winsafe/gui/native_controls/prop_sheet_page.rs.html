17/// A page of a
18/// [property sheet](https://learn.microsoft.com/en-us/windows/win32/controls/property-sheets),
19/// passed to [`PropSheet`](crate::gui::PropSheet).
20#[derive(Clone)]
21pub struct PropSheetPage(Pin<Arc<PropSheetPageObj>>);
22
23unsafe impl Send for PropSheetPage {}

25impl AsRef<BaseWnd> for PropSheetPage {

43impl PropSheetPage {
44	/// Instantiates a new `PropSheetPage` object, to be used with a
45	/// [`PropSheet`](crate::gui::PropSheet). Pages are always built upon a
46	/// dialog template.
47	#[must_use]
48	pub fn new_dlg(dlg_id: u16) -> Self {
49		let new_self = Self(Arc::pin(PropSheetPageObj { base: BaseWnd::new(WndTy::Dlg), dlg_id }));
50		new_self
51	}
52
53	#[must_use]
54	pub(in crate::gui) fn generate_propsheetpage(&self) -> PROPSHEETPAGE {
55		if *self.0.base.hwnd() != HWND::NULL {
56			panic!("Cannot create property sheet page dialog twice.");
57		}
58
59		let mut ps_page = PROPSHEETPAGE::default(); // to be passed to PropertyPage()
60		ps_page.pfnDlgProc = Some(Self::dlg_proc);
61		ps_page.pszTemplate_pResource = MAKEINTRESOURCE(self.0.dlg_id as _) as _;
62		ps_page.lParam = self as *const _ as _; // pointer to object itself
63		ps_page.dwFlags |= co::PSP::USETITLE | co::PSP::PREMATURE;
64		ps_page
65	}
66
67	/// Exposes methods to handle window messages.
68	///
69	/// # Panics
70	///
71	/// Panics if the window is already created. Events must be set before
72	/// window creation.
73	#[must_use]
74	pub fn on(&self) -> &impl GuiEventsPropSheetPage {
75		self.as_ref().on()
76	}
77
78	extern "system" fn dlg_proc(hwnd: HWND, msg: co::WM, wparam: usize, lparam: isize) -> isize {
79		let wm_any = WndMsg::new(msg, wparam, lparam);
80		Self::dlg_proc_proc(hwnd, wm_any).unwrap_or_else(|err| {
81			quit_error::post_quit_error(wm_any, err);
82			true as _
83		})
84	}
85
86	fn dlg_proc_proc(hwnd: HWND, p: WndMsg) -> AnyResult<isize> {
87		let ptr_self = match p.msg_id {
88			co::WM::INITDIALOG => {
89				// First message being handled.
90				let msg = unsafe { wm::InitDialog::from_generic_wm(p) };
91
92				// For property sheet page dialogs, LPARAM contains PROPSHEETPAGE.
93				let ptr_psp = msg.additional_data as *const PROPSHEETPAGE;
94				let ref_psp = unsafe { &*ptr_psp };
95				let ptr_self = ref_psp.lParam as *const Self;
96
97				unsafe {
98					hwnd.SetWindowLongPtr(co::GWLP::DWLP_USER, ptr_self as _); // store
99				}
100				let ref_self = unsafe { &*ptr_self };
101				ref_self.0.base.set_hwnd(unsafe { hwnd.raw_copy() }); // store HWND in struct field
102				ptr_self
103			},
104			_ => hwnd.GetWindowLongPtr(co::GWLP::DWLP_USER) as *const Self, // retrieve
105		};
106
107		if ptr_self.is_null() {
108			// If no pointer stored, then no processing is done.
109			// Prevents processing before WM_INITDIALOG and after WM_NCDESTROY.
110			Ok(0) // FALSE
111		} else {
112			let ref_self = unsafe { &*ptr_self };
113			DlgBase::dlg_proc_proc2(&ref_self.0.base, hwnd, p)
114		}
115	}
