8/// Exposes the item methods of a [`Header`](crate::gui::Header) control.
9///
10/// You cannot directly instantiate this object, it is created internally by the
11/// control.
12pub struct HeaderItems<'a> {
13	owner: &'a Header,
14}
15
16impl<'a> HeaderItems<'a> {
17	#[must_use]
18	pub(in crate::gui) const fn new(owner: &'a Header) -> Self {
19		Self { owner }
20	}
21
22	/// Adds a new item by sending an
23	/// [`hdm::InsertItem`](crate::msg::hdm::InsertItem) message, returning the
24	/// new item.
25	pub fn add(&self, text: &str, width: i32) -> SysResult<HeaderItem<'a>> {
26		let mut hdi = HDITEM::default();
27		hdi.mask = co::HDI::TEXT | co::HDI::WIDTH;
28		hdi.cxy = width;
29
30		let mut wtext = WString::from_str(text);
31		hdi.set_pszText(Some(&mut wtext));
32
33		let idx = unsafe {
34			self.owner
35				.hwnd()
36				.SendMessage(hdm::InsertItem { index_after: 0xffff, item: &hdi })?
37		};
38		Ok(self.get(idx))
39	}
40
41	/// Retrieves the total number of items by sending a
42	/// [`hdm::GetItemCount`](crate::msg::hdm::GetItemCount) message.
43	#[must_use]
44	pub fn count(&self) -> SysResult<u32> {
45		unsafe { self.owner.hwnd().SendMessage(hdm::GetItemCount {}) }
46	}
47
48	/// Retrieves the item at the given zero-based position.
49	///
50	/// **Note:** This method is cheap â€“ even if `index` is beyond the range of
51	/// existing items, an object will still be returned. However, operations
52	/// upon this object will produce no effect.
53	#[must_use]
54	pub const fn get(&self, index: u32) -> HeaderItem<'a> {
55		HeaderItem::new(self.owner, index)
56	}
57
58	/// Returns an iterator over all items.
59	#[must_use]
60	pub fn iter(&self) -> SysResult<impl DoubleEndedIterator<Item = HeaderItem<'a>> + 'a> {
61		HeaderItemIter::new(self.owner)
62	}
63
64	/// Returns the last item, if any.
65	pub fn last(&self) -> SysResult<Option<HeaderItem<'a>>> {
66		let count = self.count()?;
67		Ok(if count > 0 { Some(self.get(count - 1)) } else { None })
68	}
69}
70
71struct HeaderItemIter<'a> {
72	owner: &'a Header,
73	double_idx: DoubleIterIndex,
74}
75
76impl<'a> Iterator for HeaderItemIter<'a> {
77	type Item = HeaderItem<'a>;
78
79	fn next(&mut self) -> Option<Self::Item> {
80		self.grab(true)
81	}
82}
83impl<'a> DoubleEndedIterator for HeaderItemIter<'a> {
84	fn next_back(&mut self) -> Option<Self::Item> {
85		self.grab(false)
86	}
87}
88
89impl<'a> HeaderItemIter<'a> {
90	#[must_use]
91	fn new(owner: &'a Header) -> SysResult<Self> {
92		Ok(Self {
93			owner,
94			double_idx: DoubleIterIndex::new(owner.items().count()?),
95		})
96	}
97
98	fn grab(&mut self, is_front: bool) -> Option<HeaderItem<'a>> {
99		self.double_idx.grab(is_front, |cur_idx| {
100			let item = self.owner.items().get(cur_idx);
101			DoubleIter::Yield(item)
102		})
103	}
104}
