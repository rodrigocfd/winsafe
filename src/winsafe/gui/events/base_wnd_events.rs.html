27/// Implements all window event traits.
28pub(in crate::gui) struct BaseWndEvents {
29	wnd_ty: WndTy,
30	msgs: UnsafeCell<Vec<StorageMsg>>, // ordinary WM messages
31	cmds: UnsafeCell<Vec<StorageCmd>>, // WM_COMMAND
32	nfys: UnsafeCell<Vec<StorageNfy>>, // WM_NOTIFY
33	tmrs: UnsafeCell<Vec<StorageTmr>>, // WM_TIMER

49		unsafe {

51			{ &mut *self.cmds.get() }.clear();

57	#[must_use]
58	pub(in crate::gui) fn has_message(&self) -> bool {
59		unsafe {
60			!{ &*self.msgs.get() }.is_empty()
61				|| !{ &*self.cmds.get() }.is_empty()
62				|| !{ &*self.nfys.get() }.is_empty()
63				|| !{ &*self.tmrs.get() }.is_empty()
64		}
65	}

67	/// Returns `true` if at least one message was processed. Result values are
68	/// ignored.
69	///
70	/// Stops on error. Since this is called only for internal messages, errors
71	/// shouldn't really happen.
72	#[must_use]
73	pub(in crate::gui) fn process_all_messages(&self, p: WndMsg) -> AnyResult<bool> {
74		let mut at_least_one = false;
75
76		if p.msg_id == co::WM::COMMAND {
77			let wm_cmd = unsafe { wm::Command::from_generic_wm(p) };
78			let key_cmd = (wm_cmd.event.ctrl_id(), wm_cmd.event.code());
79			let cmds = unsafe { &*self.cmds.get() };
80			for stored_cmd in cmds.iter().filter(|obj| obj.id == key_cmd) {
81				if let Err(e) = (stored_cmd.fun)() {
82					return Err(e); // stop on error
83				}
84				at_least_one = true;
85			}
86		} else if p.msg_id == co::WM::NOTIFY {
87			let wm_nfy = unsafe { wm::Notify::from_generic_wm(p) };
88			let key_nfy = (wm_nfy.nmhdr.idFrom(), wm_nfy.nmhdr.code);
89			let nfys = unsafe { &*self.nfys.get() };
90			for stored_nfy in nfys.iter().filter(|obj| obj.id == key_nfy) {
91				if let Err(e) = (stored_nfy.fun)(unsafe { wm::Notify::from_generic_wm(p) }) {
92					// wm::Notify cannot be Copy
93					return Err(e); // stop on error
94				}
95				at_least_one = true;
96			}
97		} else if p.msg_id == co::WM::TIMER {
98			let wm_tmr = unsafe { wm::Timer::from_generic_wm(p) };
99			let tmrs = unsafe { &*self.tmrs.get() };
100			for stored_tmr in tmrs.iter().filter(|obj| obj.id == wm_tmr.timer_id) {
101				if let Err(e) = (stored_tmr.fun)() {
102					return Err(e); // stop on error
103				}
104				at_least_one = true;
105			}
106		}
107
108		let msgs = unsafe { &*self.msgs.get() };
109		for stored_msg in msgs.iter().filter(|stored_msg| stored_msg.id == p.msg_id) {
110			if let Err(e) = (stored_msg.fun)(p) {
111				return Err(e); // stop on error
112			}
113			at_least_one = true;
114		}
115
116		Ok(at_least_one)
117	}

119	/// Returns the user return value, if processed.
120	#[must_use]
121	pub(in crate::gui) fn process_last_message(&self, p: WndMsg) -> Option<AnyResult<isize>> {
122		if p.msg_id == co::WM::COMMAND {
123			let wm_cmd = unsafe { wm::Command::from_generic_wm(p) };
124			let key_cmd = (wm_cmd.event.ctrl_id(), wm_cmd.event.code());
125			let cmds = unsafe { &*self.cmds.get() };
126			if let Some(stored_cmd) = cmds.iter().rev().find(|obj| obj.id == key_cmd) {
127				let ret = (stored_cmd.fun)().map(|_| self.wnd_ty.def_proc_val());
128				return Some(ret); // handled, stop here
129			}
130		} else if p.msg_id == co::WM::NOTIFY {
131			let wm_nfy = unsafe { wm::Notify::from_generic_wm(p) };
132			let key_nfy = (wm_nfy.nmhdr.idFrom(), wm_nfy.nmhdr.code);
133			let nfys = unsafe { &*self.nfys.get() };
134			if let Some(stored_nfy) = nfys.iter().rev().find(|obj| obj.id == key_nfy) {
135				let ret = (stored_nfy.fun)(unsafe { wm::Notify::from_generic_wm(p) }); // wm::Notify cannot be Copy
136				return Some(ret); // handled, stop here
137			}
138		} else if p.msg_id == co::WM::TIMER {
139			let wm_tmr = unsafe { wm::Timer::from_generic_wm(p) };
140			let tmrs = unsafe { &*self.tmrs.get() };
141			if let Some(stored_tmr) = tmrs.iter().rev().find(|obj| obj.id == wm_tmr.timer_id) {
142				let ret = (stored_tmr.fun)().map(|_| self.wnd_ty.def_proc_val());
143				return Some(ret); // handled, stop here
144			}
145		}
146
147		let msgs = unsafe { &*self.msgs.get() };
148		if let Some(stored_msg) = msgs
149			.iter()
150			.rev()
151			.find(|stored_msg| stored_msg.id == p.msg_id)
152		{
153			let ret = (stored_msg.fun)(p);
154			return Some(ret); // handled, stop here
155		}
156
157		None
158	}

187	/// Event to any [window message](crate::co::WM).
188	///
189	/// Instead of using this event, you should always prefer the specific
190	/// events, which will give you the correct message parameters. This generic
191	/// method should be used only when you have a custom, non-standard window
192	/// message – which should be pretty rare.
193	///
194	/// # Examples
195	///
196	/// Handling a custom, user-defined message:
197	///
198	/// ```no_run
199	/// use winsafe::{self as w, prelude::*, co, gui, msg};
200	///
201	/// let wnd: gui::WindowMain; // initialized somewhere
202	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
203	///
204	/// let CUSTOM_MSG = unsafe { co::WM::from_raw(0x1234) };
205	///
206	/// wnd.on().wm(
207	///     CUSTOM_MSG,
208	///     move |p: msg::WndMsg| -> w::AnyResult<isize> {
209	///         println!("Msg ID: {}", p.msg_id);
210	///         Ok(0)
211	///     },
212	/// );
213	/// ```

914/// This trait is enabled with the `gui` feature, and provides events for window
915/// [notifications](https://learn.microsoft.com/en-us/windows/win32/winmsg/about-messages-and-message-queues).
916///
917/// Prefer importing this trait through the prelude:
918///
919/// ```no_run
920/// use winsafe::prelude::*;
921/// ```
922pub trait GuiEventsParent: GuiEventsWindow {
923	/// [`WM_COMMAND`](https://learn.microsoft.com/en-us/windows/win32/menurc/wm-command)
924	/// message, for specific code and control ID.
925	///
926	/// A command notification must be narrowed by the
927	/// [command code](crate::co::CMD) and the control ID, so the closure will
928	/// be fired for that specific control at that specific event.
929	///
930	/// Instead of using this event, you should always prefer the specific
931	/// command notifications, which will give you the correct message
932	/// parameters. This generic method should be used only when you have a
933	/// custom, non-standard window notification.
934	///
935	/// ```no_run
936	/// use winsafe::{self as w, prelude::*, co, gui};
937	///
938	/// let wnd: gui::WindowMain; // initialized somewhere
939	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
940	///
941	/// const CTRL_ID: u16 = 1010;
942	///
943	/// wnd.on().wm_command(
944	///     CTRL_ID,
945	///     co::BN::CLICKED,
946	///     move || -> w::AnyResult<()> {
947	///         println!("Button clicked!");
948	///         Ok(())
949	///     },
950	/// );
951	/// ```
952	fn wm_command<F>(&self, ctrl_id: impl Into<u16>, code: impl Into<co::CMD>, func: F)
953	where
954		F: Fn() -> AnyResult<()> + 'static;
955
956	/// [`WM_COMMAND`](https://learn.microsoft.com/en-us/windows/win32/menurc/wm-command)
957	/// message, handling both `CMD::Accelerator` and `CMD::Menu`, for a
958	/// specific command ID.
959	///
960	/// Ideal to be used with menu commands whose IDs are shared with
961	/// accelerators, like menu items.
962	///
963	/// # Examples
964	///
965	/// ```no_run
966	/// use winsafe::{self as w, prelude::*, co, gui};
967	///
968	/// let wnd: gui::WindowMain; // initialized somewhere
969	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
970	///
971	/// const CTRL_ID: u16 = 1010;
972	///
973	/// wnd.on().wm_command_acc_menu(
