5/// Exposes list view control
6/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-list-view-control-reference-notifications).
7///
8/// These event methods are just proxies to the
9/// [`WindowEvents`](crate::gui::events::WindowEvents) of the parent window, who
10/// is the real responsible for the child event handling.
11///
12/// You cannot directly instantiate this object, it is created internally by the
13/// control.
14pub struct ListViewEvents(BaseCtrlEvents);
15
16impl ListViewEvents {
17	#[must_use]
18	pub(in crate::gui) fn new(parent: &impl AsRef<BaseWnd>, ctrl_id: u16) -> Self {
19		Self(BaseCtrlEvents::new(parent, ctrl_id))
20	}

119	/// notification.
120	pub fn lvn_get_empty_markup<F>(&self, func: F) -> &Self
121	where
122		F: Fn(&mut NMLVEMPTYMARKUP) -> AnyResult<bool> + 'static,
123	{
124		self.0.wm_notify(co::LVN::GETEMPTYMARKUP, move |p| {

223	/// [`LVN_ODFINDITEM`](https://learn.microsoft.com/en-us/windows/win32/controls/lvn-odfinditem)
224	/// notification.
225	pub fn lvn_od_find_item<F>(&self, func: F) -> &Self
226	where
227		F: Fn(&mut NMLVFINDITEM) -> AnyResult<Option<u32>> + 'static,
228	{
229		self.0.wm_notify(co::LVN::ODFINDITEM, move |p| {
230			Ok(match func(unsafe { p.cast_nmhdr_mut::<NMLVFINDITEM>() })? {
231				Some(idx) => idx as isize,
232				None => -1,
233			})
234		});
235		self
236	}
