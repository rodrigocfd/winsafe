1use std::ptr::NonNull;
2
3use crate::co;
4use crate::decl::*;
5use crate::gui::privs::*;
6use crate::msg::*;
7
8/// Base to all native control events. This is actually a proxy to the events of
9/// the parent window; events added to a native control are actually added as
10/// `WM_COMMAND` or `WM_NOTIFY` messages under the parent window.
11pub(in crate::gui) struct BaseCtrlEvents {
12	parent_ptr: NonNull<BaseWnd>, // used only to add the events to parent, before the first message is processed
13	wnd_ty: WndTy,
14	ctrl_id: u16,
15}
16
17impl BaseCtrlEvents {
18	#[must_use]
19	pub(in crate::gui) fn new(parent: &impl AsRef<BaseWnd>, ctrl_id: u16) -> Self {
20		Self {
21			parent_ptr: NonNull::from(parent.as_ref()),
22			ctrl_id,
23			wnd_ty: parent.as_ref().wnd_ty(),
24		}
25	}
26
27	#[must_use]
28	pub(in crate::gui) const fn wnd_ty(&self) -> WndTy {
29		self.wnd_ty
30	}
31
32	/// Adds a `WM` event to the parent window.
33	pub(in crate::gui) fn wm<F>(&self, msg: co::WM, func: F)
34	where
35		F: Fn(WndMsg) -> AnyResult<isize> + 'static,
36	{
37		let parent_base_ref = unsafe { self.parent_ptr.as_ref() };
38		parent_base_ref.on().wm(msg, func);
39	}
40
41	/// Adds a `WM_COMMAND` event to the parent window.
42	pub(in crate::gui) fn wm_command<F>(&self, code: impl Into<co::CMD>, func: F)
43	where
44		F: Fn() -> AnyResult<()> + 'static,
45	{
46		let parent_base_ref = unsafe { self.parent_ptr.as_ref() };
47		parent_base_ref.on().wm_command(self.ctrl_id, code, func);
48	}
49
50	/// Adds a `WM_NOTIFY` event to the parent window.
51	pub(in crate::gui) fn wm_notify<F>(&self, code: impl Into<NmhdrCode>, func: F)
52	where
53		F: Fn(wm::Notify) -> AnyResult<isize> + 'static,
54	{
55		let parent_base_ref = unsafe { self.parent_ptr.as_ref() };
56		parent_base_ref.on().wm_notify(self.ctrl_id, code, func);
57	}
58}
