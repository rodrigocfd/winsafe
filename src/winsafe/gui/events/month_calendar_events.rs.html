5/// This trait is enabled with the `gui` feature, and exposes month calendar control
6/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-month-calendar-control-reference-notifications).
7///
8/// These event methods are just proxies to the
9/// [`GuiEventsParent`](crate::prelude::GuiEventsParent) of the parent window,
10/// who is the real responsible for the child event handling.
11///
12/// Prefer importing this trait through the prelude:
13///
14/// ```no_run
15/// use winsafe::prelude::*;
16/// ```
17pub trait GuiEventsMonthCalendar: priv_ctrl_events::GuiEvents {
18	fn_nfy_withparm_noret! { mcn_get_day_state, co::MCN::GETDAYSTATE, NMDAYSTATE;
19		/// [`MCN_GETDAYSTATE`](https://learn.microsoft.com/en-us/windows/win32/controls/mcn-getdaystate)
20		/// notification.
21	}
22
23	fn_nfy_withparm_noret! { mcn_sel_change, co::MCN::SELCHANGE, NMSELCHANGE;
24		/// [`MCN_SELCHANGE`](https://learn.microsoft.com/en-us/windows/win32/controls/mcn-selchange)
25		/// notification.
26		///
27		/// # Examples
28		///
29		/// ```no_run
30		/// use winsafe::{self as w, prelude::*, gui};
31		///
32		/// let wnd: gui::WindowMain; // initialized somewhere
33		/// let mcal: gui::MonthCalendar;
34		/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
35		/// # let mcal = gui::MonthCalendar::new(&wnd, gui::MonthCalendarOpts::default());
36		///
37		/// mcal.on().mcn_sel_change(
38		///     move |p: &w::NMSELCHANGE| -> w::AnyResult<()> {
39		///         let d = &p.stSelStart;
40		///         println!("{}-{}-{}", d.wYear, d.wMonth, d.wDay);
41		///         Ok(())
42		///     },
43		/// );
44		/// ```
45	}
46
47	fn_nfy_withparm_noret! { mcn_select, co::MCN::SELECT, NMSELCHANGE;
48		/// [`MCN_SELECT`](https://learn.microsoft.com/en-us/windows/win32/controls/mcn-select)
49		/// notification.
50	}
51
52	fn_nfy_withparm_noret! { mcn_view_change, co::MCN::VIEWCHANGE, NMVIEWCHANGE;
53		/// [`MCN_VIEWCHANGE`](https://learn.microsoft.com/en-us/windows/win32/controls/mcn-viewchange)
54		/// notification.
55	}
56
57	fn_nfy_noparm_noret! { nm_released_capture, co::NM::RELEASEDCAPTURE;
58		/// [`NM_RELEASEDCAPTURE`](https://learn.microsoft.com/en-us/windows/win32/controls/nm-releasedcapture-monthcal-)
59		/// notification.
60	}
61}
