7/// Exposes trackbar control
8/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-trackbar-control-reference-notifications).
9///
10/// These event methods are just proxies to the
11/// [`WindowEvents`](crate::gui::events::WindowEvents) of the parent window, who
12/// is the real responsible for the child event handling.
13///
14/// You cannot directly instantiate this object, it is created internally by the
15/// control.
16pub struct TrackbarEvents(BaseCtrlEvents);
17
18impl TrackbarEvents {
19	#[must_use]
20	pub(in crate::gui) fn new(parent: &impl AsRef<BaseWnd>, ctrl_id: u16) -> Self {
21		Self(BaseCtrlEvents::new(parent, ctrl_id))
22	}
23
24	pub_fn_nfy_withparm_noret! { trbn_thumb_pos_changing, co::TRBN::THUMBPOSCHANGING, NMTRBTHUMBPOSCHANGING;
25		/// [`TRBN_THUMBPOSCHANGING`](https://learn.microsoft.com/en-us/windows/win32/controls/trbn-thumbposchanging)
26		/// notification.
27	}
28
29	/// [`WM_HSCROLL`](https://learn.microsoft.com/en-us/windows/win32/controls/wm-hscroll--trackbar-)
30	/// notification.
31	pub fn wm_h_scroll<F>(&self, func: F)
32	where
33		F: Fn(wm::HScroll) -> AnyResult<()> + 'static,
34	{
35		let def_proc_val = self.0.is_dlg().def_proc_val();
36		self.0.wm(co::WM::HSCROLL, move |p| {
37			func(unsafe { wm::HScroll::from_generic_wm(p) })?;
38			Ok(def_proc_val)
39		});
40	}
41
42	/// [`WM_VSCROLL`](https://learn.microsoft.com/en-us/windows/win32/controls/wm-vscroll--trackbar-)
43	/// notification.
44	pub fn wm_v_scroll<F>(&self, func: F)
45	where
46		F: Fn(wm::VScroll) -> AnyResult<()> + 'static,
47	{
48		let def_proc_val = self.0.is_dlg().def_proc_val();
49		self.0.wm(co::WM::VSCROLL, move |p| {
50			func(unsafe { wm::VScroll::from_generic_wm(p) })?;
51			Ok(def_proc_val)
52		});
53	}
54
55	/// [`NM_CUSTOMDRAW`](https://learn.microsoft.com/en-us/windows/win32/controls/nm-customdraw-trackbar)
56	/// notification.
57	pub fn nm_custom_draw<F>(&self, func: F)
58	where
59		F: Fn(&NMCUSTOMDRAW) -> AnyResult<co::CDRF> + 'static,
60	{
61		self.0.wm_notify(co::NM::CUSTOMDRAW, move |p| {
62			Ok(func(unsafe { p.cast_nmhdr::<NMCUSTOMDRAW>() })?.raw() as _)
63		});
64	}
65

67		/// [`NM_RELEASEDCAPTURE`](https://learn.microsoft.com/en-us/windows/win32/controls/nm-releasedcapture-trackbar-)
68		/// notification.
69	}
