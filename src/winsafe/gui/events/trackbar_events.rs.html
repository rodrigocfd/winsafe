7/// This trait is enabled with the `gui` feature, and exposes trackbar control
8/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-trackbar-control-reference-notifications).
9///
10/// These event methods are just proxies to the
11/// [`GuiEventsParent`](crate::prelude::GuiEventsParent) of the parent window,
12/// who is the real responsible for the child event handling.
13///
14/// Prefer importing this trait through the prelude:
15///
16/// ```no_run
17/// use winsafe::prelude::*;
18/// ```
19pub trait GuiEventsTrackbar: priv_ctrl_events::GuiEvents {
20	fn_nfy_withparm_noret! { trbn_thumb_pos_changing, co::TRBN::THUMBPOSCHANGING, NMTRBTHUMBPOSCHANGING;
21		/// [`TRBN_THUMBPOSCHANGING`](https://learn.microsoft.com/en-us/windows/win32/controls/trbn-thumbposchanging)
22		/// notification.
23	}
24
25	/// [`WM_HSCROLL`](https://learn.microsoft.com/en-us/windows/win32/controls/wm-hscroll--trackbar-)
26	/// notification.
27	fn wm_h_scroll<F>(&self, func: F)
28	where
29		F: Fn(wm::HScroll) -> AnyResult<()> + 'static,
30	{
31		let def_proc_val = self.wnd_ty().def_proc_val();
32		self.wm(co::WM::HSCROLL, move |p| {
33			func(unsafe { wm::HScroll::from_generic_wm(p) })?;
34			Ok(def_proc_val)
35		});
36	}
37
38	/// [`WM_VSCROLL`](https://learn.microsoft.com/en-us/windows/win32/controls/wm-vscroll--trackbar-)
39	/// notification.
40	fn wm_v_scroll<F>(&self, func: F)
41	where
42		F: Fn(wm::VScroll) -> AnyResult<()> + 'static,
43	{
44		let def_proc_val = self.wnd_ty().def_proc_val();
45		self.wm(co::WM::VSCROLL, move |p| {
46			func(unsafe { wm::VScroll::from_generic_wm(p) })?;
47			Ok(def_proc_val)
48		});
49	}
50
51	/// [`NM_CUSTOMDRAW`](https://learn.microsoft.com/en-us/windows/win32/controls/nm-customdraw-trackbar)
52	/// notification.
53	fn nm_custom_draw<F>(&self, func: F)
54	where
55		F: Fn(&NMCUSTOMDRAW) -> AnyResult<co::CDRF> + 'static,
56	{
57		self.wm_notify(co::NM::CUSTOMDRAW, move |p| {
58			Ok(func(unsafe { p.cast_nmhdr::<NMCUSTOMDRAW>() })?.raw() as _)
59		});
60	}
61
62	fn_nfy_noparm_noret! { nm_released_capture, co::NM::RELEASEDCAPTURE;
63		/// [`NM_RELEASEDCAPTURE`](https://learn.microsoft.com/en-us/windows/win32/controls/nm-releasedcapture-trackbar-)
64		/// notification.
65	}
66}
67
68impl GuiEventsTrackbar for BaseCtrlEvents {}
