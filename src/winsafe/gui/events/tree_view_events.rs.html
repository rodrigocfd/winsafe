5/// Exposes tree view control
6/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-tree-view-control-reference-notifications).
7///
8/// These event methods are just proxies to the
9/// [`WindowEvents`](crate::gui::events::WindowEvents) of the parent window, who
10/// is the real responsible for the child event handling.
11///
12/// You cannot directly instantiate this object, it is created internally by the
13/// control.
14pub struct TreeViewEvents(BaseCtrlEvents);
15
16impl TreeViewEvents {

19		Self(BaseCtrlEvents::new(parent, ctrl_id))
20	}

65		///             p.itemOld.pszText().unwrap(),
66		///             p.itemNew.pszText().unwrap(),
67		///             unsafe { co::TVC::from_raw(p.action) },
68		///         );

85	/// [`NM_CUSTOMDRAW`](https://learn.microsoft.com/en-us/windows/win32/controls/nm-customdraw-tree-view)
86	/// notification.
87	pub fn nm_custom_draw<F>(&self, func: F) -> &Self
88	where
89		F: Fn(&mut NMTVCUSTOMDRAW) -> AnyResult<co::CDRF> + 'static,
90	{
91		self.0.wm_notify(co::NM::CUSTOMDRAW, move |p| {
92			Ok(func(unsafe { p.cast_nmhdr_mut::<NMTVCUSTOMDRAW>() })?.raw() as _)
93		});
94		self
95	}
