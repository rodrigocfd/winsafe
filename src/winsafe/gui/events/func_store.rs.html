1/// Stores closures, associating them with an identifier.
2pub(in crate::gui) struct FuncStore<K: Copy + Eq, F> {
3	elems: Vec<(K, F)>,
4}
5
6impl<K: Copy + Eq, F> FuncStore<K, F> {
7	/// Creates a new, empty store.
8	#[must_use]
9	pub(in crate::gui) const fn new() -> Self {
10		Self { elems: Vec::new() }
11	}
12
13	/// Adds a new function into the store, associated to the given identifier.
14	pub(in crate::gui) fn push(&mut self, id: K, func: F) {
15		if self.elems.is_empty() {
16			self.elems.reserve(16); // arbitrary, prealloc for speed
17		}
18		self.elems.push((id, func));
19	}
20
21	/// Returns an iterator over all the functions associated to the given
22	/// identifier, if any.
23	#[must_use]
24	pub(in crate::gui) fn filter(&self, id: K) -> impl Iterator<Item = &F> {
25		// https://depth-first.com/articles/2020/06/22/returning-rust-iterators
26		self.elems.iter()
27			.filter(move |(elem_id, _)| *elem_id == id)
28			.map(|(_, func)| func)
29	}
30
31	/// Returns an iterator, in reverse order, over all the functions associated
32	/// to the given identifier, if any.
33	#[must_use]
34	pub(in crate::gui) fn filter_rev(&self, id: K) -> impl Iterator<Item = &F> {
35		self.elems.iter()
36			.rev()
37			.filter(move |(elem_id, _)| *elem_id == id)
38			.map(|(_, func)| func)
39	}
40
41	/// Tells whether no functions have been added.
42	#[must_use]
43	pub(in crate::gui) fn is_empty(&self) -> bool {
44		self.elems.is_empty()
45	}
46
47	/// Removes all identifiers and closures.
48	pub(in crate::gui) fn clear(&mut self) {
49		self.elems.clear();
50	}
51}
