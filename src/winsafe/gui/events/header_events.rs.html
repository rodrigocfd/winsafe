5/// Exposes header control
6/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-header-control-reference-notifications).
7///
8/// These event methods are just proxies to the
9/// [`WindowEvents`](crate::gui::events::WindowEvents) of the parent window, who
10/// is the real responsible for the child event handling.
11///
12/// You cannot directly instantiate this object, it is created internally by the
13/// control.
14pub struct HeaderEvents(BaseCtrlEvents);
15
16impl HeaderEvents {

18	pub(in crate::gui) fn new(parent: &impl AsRef<BaseWnd>, ctrl_id: u16) -> Self {
19		Self(BaseCtrlEvents::new(parent, ctrl_id))
20	}

72	/// [`HDN_GETDISPINFO`](https://learn.microsoft.com/en-us/windows/win32/controls/hdn-getdispinfo)
73	/// notification.
74	pub fn hdn_get_disp_info<F>(&self, func: F) -> &Self
75	where
76		F: Fn(&mut NMHDDISPINFO) -> AnyResult<isize> + 'static,
77	{
78		self.0.wm_notify(co::HDN::GETDISPINFO, move |p| {
79			Ok(func(unsafe { p.cast_nmhdr_mut::<NMHDDISPINFO>() })?)
80		});
81		self
82	}
