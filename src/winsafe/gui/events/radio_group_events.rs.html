8/// Exposes button control
9/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-button-control-reference-notifications)
10/// for a [`RadioGroup`](crate::gui::RadioGroup).
11///
12/// These event methods are just proxies to the
13/// [`WindowEvents`](crate::gui::events::WindowEvents) of the parent window, who
14/// is the real responsible for the child event handling.
15///
16/// You cannot directly instantiate this object, it is created internally by the
17/// control.
18pub struct RadioGroupEvents {
19	parent_ptr: NonNull<BaseWnd>, // used only to add the events to parent, before the first message is processed
20	ctrl_ids: Vec<u16>,
21}
22
23impl RadioGroupEvents {
24	#[must_use]
25	pub(in crate::gui) fn new(parent: &impl AsRef<BaseWnd>, ctrl_ids: Vec<u16>) -> Self {
26		Self {
27			parent_ptr: NonNull::from(parent.as_ref()),
28			ctrl_ids,
29		}
30	}
31
32	/// [`BN_CLICKED`](https://learn.microsoft.com/en-us/windows/win32/controls/bn-clicked)
33	/// command notification for all radio buttons in the group.
34	///
35	/// Sent when the user clicks a button.
36	///
37	/// # Examples
38	///
39	/// ```no_run
40	/// use winsafe::{self as w, prelude::*, gui};
41	///
42	/// let wnd: gui::WindowMain; // initialized somewhere
43	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
44	/// let radios: gui::RadioGroup;
45	/// # let radios = gui::RadioGroup::new(&wnd, &[]);
46	///
47	/// radios.on().bn_clicked({
48	///     let radios = radios.clone();
49	///     move || -> w::AnyResult<()> {
50	///         println!("Selected {}",
51	///             radios.selected().unwrap()
52	///                 .hwnd().GetWindowText()?,
53	///         );
54	///         Ok(())
55	///     }
56	/// });
57	/// # w::SysResult::Ok(())
58	/// ```
59	pub fn bn_clicked<F>(&self, func: F)
60		where F: Fn() -> AnyResult<()> + 'static,
61	{
62		let parent_base_ref = unsafe { self.parent_ptr.as_ref() };
63		let shared_func = Rc::new(func);
64
65		for ctrl_id in self.ctrl_ids.iter() {
66			parent_base_ref.on().wm_command(*ctrl_id, co::BN::CLICKED, {
67				let shared_func = shared_func.clone();
68				move || {
69					shared_func()
70				}
71			});
72		}
73	}
