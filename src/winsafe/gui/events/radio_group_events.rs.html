4/// This trait is enabled with the `gui` feature, and exposes button control
5/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-button-control-reference-notifications)
6/// for a [`RadioGroup`](crate::gui::RadioGroup).
7///
8/// These event methods are just proxies to the
9/// [`GuiEventsParent`](crate::prelude::GuiEventsParent) of the parent window,
10/// who is the real responsible for the child event handling.
11///
12/// Prefer importing this trait through the prelude:
13///
14/// ```no_run
15/// use winsafe::prelude::*;
16/// ```
17pub trait GuiEventsRadioGroup: priv_ctrl_events::GuiEvents {
18	fn_cmd_noparm_noret! { bn_clicked, co::BN::CLICKED;
19		/// [`BN_CLICKED`](https://learn.microsoft.com/en-us/windows/win32/controls/bn-clicked)
20		/// command notification for all radio buttons in the group.
21		///
22		/// Sent when the user clicks a button.
23		///
24		/// # Examples
25		///
26		/// ```no_run
27		/// use winsafe::{self as w, prelude::*, gui};
28		///
29		/// let wnd: gui::WindowMain; // initialized somewhere
30		/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
31		/// let radios: gui::RadioGroup;
32		/// # let radios = gui::RadioGroup::new(&wnd, &[]);
33		///
34		/// radios.on().bn_clicked({
35		///     let radios = radios.clone();
36		///     move || -> w::AnyResult<()> {
37		///         println!("Selected {}",
38		///             radios.selected().unwrap()
39		///                 .hwnd().GetWindowText()?,
40		///         );
41		///         Ok(())
42		///     }
43		/// });
44		/// # w::SysResult::Ok(())
45		/// ```
46	}
47
48	fn_cmd_noparm_noret! { bn_dbl_clk, co::BN::DBLCLK;
49		/// [`BN_DBLCLK`](https://learn.microsoft.com/en-us/windows/win32/controls/bn-dblclk)
50		/// command notification for all radio buttons in the group.
51		///
52		/// Sent when the user double-clicks a button. This notification code is
53		/// sent automatically for
54		/// [`BS::USERBUTTON`](crate::co::BS::USERBUTTON),
55		/// [`BS::RADIOBUTTON`](crate::co::BS::RADIOBUTTON), and
56		/// [`BS::OWNERDRAW`](crate::co::BS::OWNERDRAW) buttons. Other button
57		/// types send only if they have the
58		/// [`BS::NOTIFY`](crate::co::BS::NOTIFY) style.
59	}
60
61	fn_cmd_noparm_noret! { bn_kill_focus, co::BN::KILLFOCUS;
62		/// [`BN_KILLFOCUS`](https://learn.microsoft.com/en-us/windows/win32/controls/bn-killfocus)
63		/// command notification for all radio buttons in the group.
64		///
65		/// Sent when a button loses the keyboard focus. The button must have the
66		/// [`BS::NOTIFY`](crate::co::BS::NOTIFY) style to send this notification
67		/// code.
68	}
69
70	fn_cmd_noparm_noret! { bn_set_focus, co::BN::SETFOCUS;
71		/// [`BN_SETFOCUS`](https://learn.microsoft.com/en-us/windows/win32/controls/bn-setfocus)
72		/// command notification for all radio buttons in the group.
73		///
74		/// Sent when a button receives the keyboard focus. The button must have
75		/// the [`BS::NOTIFY`](crate::co::BS::NOTIFY) style to send this
76		/// notification code.
77	}
78}
