5/// This trait is enabled with the `gui` feature, and exposes date and time
6/// picker control
7/// [notifications](https://learn.microsoft.com/en-us/windows/win32/controls/bumper-date-and-time-picker-control-reference-notifications).
8///
9/// These event methods are just proxies to the
10/// [`GuiEventsParent`](crate::prelude::GuiEventsParent) of the parent window,
11/// who is the real responsible for the child event handling.
12///
13/// Prefer importing this trait through the prelude:
14///
15/// ```no_run
16/// use winsafe::prelude::*;
17/// ```
18pub trait GuiEventsDateTimePicker: priv_ctrl_events::GuiEvents {
19	fn_nfy_noparm_noret! { dtn_close_up, co::DTN::CLOSEUP;
20		/// [`DTN_CLOSEUP`](https://learn.microsoft.com/en-us/windows/win32/controls/dtn-closeup)
21		/// notification.
22	}
23
24	fn_nfy_withparm_noret! { dtn_date_time_change, co::DTN::DATETIMECHANGE, NMDATETIMECHANGE;
25		/// [`DTN_DATETIMECHANGE`](https://learn.microsoft.com/en-us/windows/win32/controls/dtn-datetimechange)
26		/// notification.
27		///
28		/// # Examples
29		///
30		/// ```no_run
31		/// use winsafe::{self as w, prelude::*, gui};
32		///
33		/// let wnd: gui::WindowMain; // initialized somewhere
34		/// let dtp: gui::DateTimePicker;
35		/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
36		/// # let dtp = gui::DateTimePicker::new(&wnd, gui::DateTimePickerOpts::default());
37		///
38		/// dtp.on().dtn_date_time_change(
39		///     move |p: &w::NMDATETIMECHANGE| -> w::AnyResult<()> {
40		///         println!("{}-{}-{}", p.st.wYear, p.st.wMonth, p.st.wDay);
41		///         Ok(())
42		///     },
43		/// );
44		/// ```
45	}
46
47	fn_nfy_noparm_noret! { dtn_drop_down, co::DTN::DROPDOWN;
48		/// [`DTN_DROPDOWN`](https://learn.microsoft.com/en-us/windows/win32/controls/dtn-dropdown)
49		/// notification.
50	}
51
52	fn_nfy_withmutparm_noret! { dtn_format, co::DTN::FORMAT, NMDATETIMEFORMAT;
53		/// [`DTN_FORMAT`](https://learn.microsoft.com/en-us/windows/win32/controls/dtn-format)
54		/// notification.
55	}
56
57	fn_nfy_withmutparm_noret! { dtn_format_query, co::DTN::FORMATQUERY, NMDATETIMEFORMATQUERY;
58		/// [`DTN_FORMATQUERY`](https://learn.microsoft.com/en-us/windows/win32/controls/dtn-formatquery)
59		/// notification.
60	}
61
62	fn_nfy_withmutparm_noret! { dtn_user_string, co::DTN::USERSTRING, NMDATETIMESTRING;
63		/// [`DTN_USERSTRING`](https://learn.microsoft.com/en-us/windows/win32/controls/dtn-userstring)
64		/// notification.
65	}
66
67	fn_nfy_withparm_noret! { dtn_wm_key_down, co::DTN::WMKEYDOWN, NMDATETIMEWMKEYDOWN;
68		/// [`DTN_WMKEYDOWN`](https://learn.microsoft.com/en-us/windows/win32/controls/dtn-wmkeydown)
69		/// notification.
70	}
71
72	fn_nfy_noparm_noret! { nm_kill_focus, co::NM::KILLFOCUS;
73		/// [`NM_KILLFOCUS`](https://learn.microsoft.com/en-us/windows/win32/controls/nm-killfocus-date-time)
74		/// notification.
75	}
76
77	fn_nfy_noparm_noret! { nm_set_focus, co::NM::SETFOCUS;
78		/// [`NM_SETFOCUS`](https://learn.microsoft.com/en-us/windows/win32/controls/nm-setfocus-date-time-)
79		/// notification.
80	}
81}
