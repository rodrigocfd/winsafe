7/// This trait is enabled with the `gui` feature, and is implemented by all
8/// windows. Exposes the underlying window handle.
9///
10/// Prefer importing this trait through the prelude:
11///
12/// ```no_run
13/// use winsafe::prelude::*;
14/// ```
15pub trait GuiWindow: Send {
16	/// Returns the underlying handle for this window.
17	///
18	/// Note that the handle is initially null, receiving an actual value only
19	/// after the window is physically created, what usually happens right
20	/// before
21	/// [`WM_CREATE`](https://learn.microsoft.com/en-us/windows/win32/winmsg/wm-create)
22	/// or
23	/// [`WM_INITDIALOG`](https://learn.microsoft.com/en-us/windows/win32/dlgbox/wm-initdialog)
24	/// events.
25	#[must_use]
26	fn hwnd(&self) -> &HWND;
27
28	/// Converts a reference to the [`Any`] trait. This is useful when storing a
29	/// collection of polymorphic controls, because `Any` allows downcasting.
30	///
31	/// # Examples
32	///
33	/// ```no_run
34	/// use std::sync::Arc;
35	/// use winsafe::{self as w, prelude::*, gui};
36	///
37	/// let parent: gui::WindowMain; // initialized somewhere
38	/// # let parent = gui::WindowMain::new(gui::WindowMainOpts::default());
39	///
40	/// let ctrls: Vec<Arc<dyn GuiControl>> = vec![
41	///     Arc::new( gui::Edit::new(&parent, gui::EditOpts::default()) ),
42	///     Arc::new( gui::Button::new(&parent, gui::ButtonOpts::default()) ),
43	/// ];
44	///
45	/// let edit = ctrls[0].as_any() // retrieve 1st element, which is an Edit
46	///     .downcast_ref::<gui::Edit>()
47	///     .expect("This Edit downcast should never fail.");
48	///
49	/// edit.set_text("Foo");
50	/// ```
51	#[must_use]
52	fn as_any(&self) -> &dyn Any;
53}

55/// This trait is enabled with the `gui` feature, and is implemented by all
56/// windows which can host child controls.
57///
58/// Prefer importing this trait through the prelude:
59///
60/// ```no_run
61/// use winsafe::prelude::*;
62/// ```
63#[allow(private_bounds)]
64pub trait GuiParent: GuiWindow + Clone + AsRef<BaseWnd> {
65	/// This method calls [`std::thread::spawn`], but it allows the returning of
66	/// an error value. This error value will be forwarded to the original UI
67	/// thread, allowing it to be caught at
68	/// [`WindowMain::run_main`](crate::gui::WindowMain::run_main).
69	///
70	/// It's a way to ensure that, upon an unexpected error, you application
71	/// will be terminated gracefully.
72	///
73	/// # Examples
74	///
75	/// The example below shows the event of a
76	/// [button click](crate::prelude::GuiEventsButton::bn_clicked) which spawns
77	/// a new thread.
78	///
79	/// ```no_run
80	/// use winsafe::{self as w, prelude::*, gui};
81	///
82	/// let wnd: gui::WindowMain; // initialized somewhere
83	/// let btn: gui::Button;
84	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
85	/// # let btn = gui::Button::new(&wnd, gui::ButtonOpts::default());
86	///
87	/// btn.on().bn_clicked({
88	///     let wnd = wnd.clone();
89	///     move || -> w::AnyResult<()> {
90	///         println!("Click event at {:#x}", w::GetCurrentThreadId());
91	///
92	///         wnd.spawn_thread({
93	///             let wnd = wnd.clone();
94	///             move || {
95	///                 println!("This is another thread: {:#x}", w::GetCurrentThreadId());
96	///                 if 1 != 2 {
97	///                     Err("Unexpected condition, goodbye.".into())
98	///                 } else {
99	///                     Ok(())
100	///                 }
101	///             }
102	///         });
103	///
104	///         Ok(())
105	///     }
106	/// });
107	/// ```
108	fn spawn_thread<F>(&self, func: F)
109	where
110		F: FnOnce() -> AnyResult<()> + Send + 'static,
111	{
112		self.as_ref().spawn_thread(func)
113	}
114
115	/// Runs a closure synchronously in the window's original UI thread,
116	/// allowing UI updates without the risk of a deadlock.
117	///
118	/// # Rationale
119	///
120	/// If you perform a very long task in the UI thread, the UI freezes until
121	/// the task is complete â€“ this may cause the impression that your
122	/// application crashed. That's why long tasks should be performed in
123	/// parallel threads. However, at some point you'll want to update the UI to
124	/// reflect the task progress, but if you update the UI from another thread
125	/// (different from the original UI thread), the UI may deadlock, and you
126	/// application crashes.
127	///
128	/// This is what this `run_ui_thread` does, step-by-step:
129	///
130	/// 1. blocks current thread;
131	/// 2. switches to the window's original UI thread;
132	/// 3. runs the given `FnOnce`;
133	/// 4. switches back to the first thread, which is then unblocked.
134	///
135	/// When working in a parallel thread, you **must** call `run_ui_thread` to
136	/// update the UI.
137	///
138	/// # Examples
139	///
140	/// The example below shows the event of a
141	/// [button click](crate::prelude::GuiEventsButton::bn_clicked) which starts
142	/// a long task in a parallel thread. As it progresses, the status is
143	/// printed at the windows's titlebar.
144	///
145	/// ```no_run
146	/// use winsafe::{self as w, prelude::*, gui};
147	///
148	/// let wnd: gui::WindowMain; // initialized somewhere
149	/// # let wnd = gui::WindowMain::new(gui::WindowMainOpts::default());
150	/// let btn: gui::Button;
151	/// # let btn = gui::Button::new(&wnd, gui::ButtonOpts::default());
152	///
153	/// btn.on().bn_clicked({
154	///     let wnd = wnd.clone();
155	///     move || -> w::AnyResult<()> {
156	///         println!("Click event at {:#x}", w::GetCurrentThreadId());
157	///
158	///         std::thread::spawn({
159	///             let wnd = wnd.clone();
160	///             move || {
161	///                 println!("Parallel task starts at {:#x}", w::GetCurrentThreadId());
162	///                 w::Sleep(2000);
163	///
164	///                 wnd.run_ui_thread({
165	///                     let wnd = wnd.clone();
166	///                     move || -> w::AnyResult<()> {
167	///                         println!("Updating UI at {:#x}", w::GetCurrentThreadId());
168	///                         wnd.hwnd().SetWindowText("Status... 50%")?;
169	///                         Ok(())
170	///                     }
171	///                 });
172	///
173	///                 println!("Parallel task keeps going at {:#x}", w::GetCurrentThreadId());
174	///                 w::Sleep(2000);
175	///
176	///                 wnd.run_ui_thread({
177	///                     let wnd = wnd.clone();
178	///                     move || -> w::AnyResult<()> {
179	///                         println!("Updating UI at {:#x}", w::GetCurrentThreadId());
180	///                         wnd.hwnd().SetWindowText("Status... 100%")?;
181	///                         Ok(())
182	///                     }
183	///                 });
184	///             }
185	///         });
186	///
187	///         Ok(())
188	///     }
189	/// });
190	/// ```
191	fn run_ui_thread<F>(&self, func: F)
192	where
193		F: FnOnce() -> AnyResult<()> + Send + 'static,
194	{
195		self.as_ref().run_ui_thread(func)
196	}
197}

199/// This trait is enabled with the `gui` feature, and is implemented by all
200/// child controls.
201///
202/// Prefer importing this trait through the prelude:
203///
204/// ```no_run
205/// use winsafe::prelude::*;
206/// ```
207pub trait GuiControl: GuiWindow {
208	/// Returns the control ID, which is defined at control creation.
209	///
210	/// The control ID should be unique within a parent.
211	#[must_use]
212	fn ctrl_id(&self) -> u16;
213
214	/// If the control belongs to an ordinary window, simply calls
215	/// [`HWND:SetFocus`](crate::HWND::SetFocus).
216	///
217	/// If the control belongs to a dialog window, sends a
218	/// [`wm::NextDlgCtl`] message. This is preferable to the `HWND::SetFocus`
219	/// because it takes care of border highlighting, like the native
220	/// [`Button`](crate::gui::Button) control needs.
221	fn focus(&self) -> SysResult<()> {
222		let hparent = self.hwnd().GetParent()?;
223		if hparent.is_dialog() {
224			unsafe {
225				hparent.SendMessage(wm::NextDlgCtl {
226					hwnd_focus: HwndFocus::Hwnd(self.hwnd().raw_copy()),
227				});
228			}
229		} else {
230			self.hwnd().SetFocus();
231		}
232		Ok(())
233	}
