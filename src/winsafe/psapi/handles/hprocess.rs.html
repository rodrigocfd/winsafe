15	/// [`EnumProcessModules`](https://learn.microsoft.com/en-us/windows/win32/api/psapi/nf-psapi-enumprocessmodules)
16	/// function.
17	#[must_use]
18	pub fn EnumProcessModules(&self) -> SysResult<Vec<HINSTANCE>> {
19		loop {
20			let mut bytes_needed = 0u32;
21			BoolRet(unsafe {
22				ffi::EnumProcessModules(self.ptr(), std::ptr::null_mut(), 0, &mut bytes_needed)
23			})
24			.to_sysresult()?;
25
26			let elems_needed = bytes_needed / (std::mem::size_of::<HINSTANCE>() as u32);
27			let mut buf = (0..elems_needed)
28				.map(|_| HINSTANCE::NULL)
29				.collect::<Vec<_>>();
30
31			let mut bytes_got = 0u32;
32			BoolRet(unsafe {
33				ffi::EnumProcessModules(
34					self.ptr(),
35					buf.as_mut_ptr() as _,
36					bytes_needed,
37					&mut bytes_got,
38				)
39			})
40			.to_sysresult()?;
41
42			if bytes_needed == bytes_got {
43				return Ok(buf);
44			}
45		}
46	}
