18	/// [`GetFileVersionInfo`](https://learn.microsoft.com/en-us/windows/win32/api/winver/nf-winver-getfileversioninfow)
19	/// function.
20	///
21	/// The returned buffer will be automatically allocated with
22	/// [`HVERSIONINFO::GetFileVersionInfoSize`](crate::HVERSIONINFO::GetFileVersionInfoSize).
23	#[must_use]
24	pub fn GetFileVersionInfo(file_name: &str) -> SysResult<VersionInfoGuard> {
25		let block_sz = Self::GetFileVersionInfoSize(file_name)?;
26		let mut hglobal =
27			HGLOBAL::GlobalAlloc(co::GMEM::FIXED | co::GMEM::ZEROINIT, block_sz as _)?;
28		let hglobal_ptr = hglobal.leak();
29
30		BoolRet(unsafe {
31			ffi::GetFileVersionInfoW(
32				WString::from_str(file_name).as_ptr(),
33				0,
34				block_sz,
35				hglobal_ptr.ptr(),
36			)
37		})
38		.to_sysresult()
39		.map(|_| unsafe {
40			VersionInfoGuard::new(
41				HVERSIONINFO::from_ptr(hglobal_ptr.ptr()), // simply use the HGLOBAL pointer
42			)
43		})
44	}

46	/// [`GetFileVersionInfoSize`](https://learn.microsoft.com/en-us/windows/win32/api/winver/nf-winver-getfileversioninfosizew)
47	/// function.
48	///
49	/// You don't need to call this function directly, because
50	/// [`HVERSIONINFO::GetFileVersionInfo`](crate::HVERSIONINFO::GetFileVersionInfo)
51	/// already calls it.
52	#[must_use]
53	pub fn GetFileVersionInfoSize(file_name: &str) -> SysResult<u32> {
54		let mut dw_handle = 0u32;
55		match unsafe {
56			ffi::GetFileVersionInfoSizeW(WString::from_str(file_name).as_ptr(), &mut dw_handle)
57		} {
58			0 => Err(GetLastError()),
59			sz => Ok(sz),
60		}
61	}

63	/// Calls
64	/// [`HVERSIONINFO::VerQueryValue`](crate::HVERSIONINFO::VerQueryValue) to
65	/// retrieve a reference to a slice with all languages and code pages.
66	///
67	/// # Examples
68	///
69	/// Listing all pairs of language and code page:
70	///
71	/// ```no_run
72	/// use winsafe::{self as w, prelude::*};
73	///
74	/// let exe_name = w::HINSTANCE::NULL.GetModuleFileName()?;
75	/// let hversion = w::HVERSIONINFO::GetFileVersionInfo(&exe_name)?;
76	///
77	/// for (lang, cp) in hversion.langs_and_cps()?.iter() {
78	///     println!("{} {}", lang, cp);
79	/// }
80	/// # w::SysResult::Ok(())
81	/// ```
82	#[must_use]
83	pub fn langs_and_cps(&self) -> SysResult<&[(LANGID, co::CP)]> {
84		unsafe {
85			self.VerQueryValue::<(LANGID, co::CP)>("\\VarFileInfo\\Translation")
86				.map(|(pblocks, sz)| {
87					std::slice::from_raw_parts(
88						pblocks,
89						sz as usize / std::mem::size_of::<(LANGID, co::CP)>(),
90					)
91				})
92		}
93	}

95	/// [`VerQueryValue`](https://learn.microsoft.com/en-us/windows/win32/api/winver/nf-winver-verqueryvaluew)
96	/// function.
97	///
98	/// # Safety
99	///
100	/// The returned pointer and size vary according to `lpSubBlock`. If you set
101	/// it wrong, you're likely to cause a buffer overrun.
102	///
103	/// This is a low-level method, prefer using:
104	/// * [`langs_and_cps`](crate::HVERSIONINFO::langs_and_cps);
105	/// * [`str_val`](crate::HVERSIONINFO::str_val);
106	/// * [`version_info`](crate::HVERSIONINFO::version_info).
107	///
108	/// # Examples
109	///
110	/// Reading version information from resource:
111	///
112	/// ```no_run
113	/// use winsafe::{self as w, prelude::*};
114	///
115	/// let exe_name = w::HINSTANCE::NULL.GetModuleFileName()?;
116	/// let hversion = w::HVERSIONINFO::GetFileVersionInfo(&exe_name)?;
117	///
118	/// let (pvsf, sz_data) = unsafe {
119	///     hversion.VerQueryValue::<w::VS_FIXEDFILEINFO>("\\")?
120	/// };
121	///
122	/// let ver = unsafe { &*pvsf }.dwFileVersion();
123	/// println!("Version {}.{}.{}.{}",
124	///     ver[0], ver[1], ver[2], ver[3]);
125	/// # w::SysResult::Ok(())
126	/// ```
127	#[must_use]
128	pub unsafe fn VerQueryValue<T>(&self, sub_block: &str) -> SysResult<(*const T, u32)> {
129		let mut lp_lp_buffer = std::ptr::null();
130		let mut pu_len = 0;
131
132		BoolRet(unsafe {
133			ffi::VerQueryValueW(
134				self.ptr(),
135				WString::from_str(sub_block).as_ptr(),
136				pvoid(&mut lp_lp_buffer),
137				&mut pu_len,
138			)
139		})
140		.to_sysresult()
141		.map(|_| (lp_lp_buffer as *const T, pu_len))
142	}

144	/// Calls
145	/// [`HVERSIONINFO::VerQueryValue`](crate::HVERSIONINFO::VerQueryValue)
146	/// to retrieve a string value.
147	///
148	/// Common value names are:
149	/// * Comments
150	/// * CompanyName
151	/// * FileDescription
152	/// * FileVersion
153	/// * InternalName
154	/// * LegalCopyright
155	/// * LegalTrademarks
156	/// * OriginalFilename
157	/// * ProductName
158	/// * ProductVersion
159	/// * PrivateBuild
160	/// * SpecialBuild
161	///
162	/// # Examples
163	///
164	/// Reading product name and legal copyright from resource:
165	///
166	/// ```no_run
167	/// use winsafe::{self as w, prelude::*};
168	///
169	/// let exe_name = w::HINSTANCE::NULL.GetModuleFileName()?;
170	/// let hversion = w::HVERSIONINFO::GetFileVersionInfo(&exe_name)?;
171	///
172	/// let lang_cp = hversion.langs_and_cps()?[0]; // first language and code page
173	///
174	/// println!(
175	///     "{}\n{}",
176	///     hversion.str_val(lang_cp, "ProductName")?,
177	///     hversion.str_val(lang_cp, "LegalCopyright")?,
178	/// );
179	/// # w::SysResult::Ok(())
180	/// ```
181	#[must_use]
182	pub fn str_val(&self, lang_and_cp: (LANGID, co::CP), name: &str) -> SysResult<String> {
183		unsafe {
184			self.VerQueryValue::<u16>(&format!(
185				"\\StringFileInfo\\{:04x}{:04x}\\{}",
186				u16::from(lang_and_cp.0),
187				u16::from(lang_and_cp.1),
188				name
189			))
190			.map(|(pstr, len)| {
191				WString::from_wchars_slice(std::slice::from_raw_parts(pstr, len as _)).to_string()
192			})
193		}
194	}
