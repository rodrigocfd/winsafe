18	/// [`GetFileVersionInfo`](https://learn.microsoft.com/en-us/windows/win32/api/winver/nf-winver-getfileversioninfow)
19	/// function.
20	///
21	/// The returned buffer will be automatically allocated with
22	/// [`HVERSIONINFO::GetFileVersionInfoSize`](crate::HVERSIONINFO::GetFileVersionInfoSize).
23	#[must_use]
24	pub fn GetFileVersionInfo(file_name: &str) -> SysResult<VersionInfoGuard> {
25		let block_sz = Self::GetFileVersionInfoSize(file_name)?;
26		let mut hglobal =
27			HGLOBAL::GlobalAlloc(co::GMEM::FIXED | co::GMEM::ZEROINIT, block_sz as _)?;
28		let hglobal_ptr = hglobal.leak();
29
30		bool_to_sysresult(unsafe {
31			ffi::GetFileVersionInfoW(
32				WString::from_str(file_name).as_ptr(),
33				0,
34				block_sz,
35				hglobal_ptr.ptr(),
36			)
37		})
38		.map(|_| unsafe {
39			VersionInfoGuard::new(
40				HVERSIONINFO::from_ptr(hglobal_ptr.ptr()), // simply use the HGLOBAL pointer
41			)
42		})
43	}

45	/// [`GetFileVersionInfoSize`](https://learn.microsoft.com/en-us/windows/win32/api/winver/nf-winver-getfileversioninfosizew)
46	/// function.
47	///
48	/// You don't need to call this function directly, because
49	/// [`HVERSIONINFO::GetFileVersionInfo`](crate::HVERSIONINFO::GetFileVersionInfo)
50	/// already calls it.
51	#[must_use]
52	pub fn GetFileVersionInfoSize(file_name: &str) -> SysResult<u32> {
53		let mut dw_handle = 0u32;
54		match unsafe {
55			ffi::GetFileVersionInfoSizeW(WString::from_str(file_name).as_ptr(), &mut dw_handle)
56		} {
57			0 => Err(GetLastError()),
58			sz => Ok(sz),
59		}
60	}

62	/// Calls
63	/// [`HVERSIONINFO::VerQueryValue`](crate::HVERSIONINFO::VerQueryValue) to
64	/// retrieve a reference to a slice with all languages and code pages.
65	///
66	/// # Examples
67	///
68	/// Listing all pairs of language and code page:
69	///
70	/// ```no_run
71	/// use winsafe::{self as w, prelude::*};
72	///
73	/// let exe_name = w::HINSTANCE::NULL.GetModuleFileName()?;
74	/// let hversion = w::HVERSIONINFO::GetFileVersionInfo(&exe_name)?;
75	///
76	/// for (lang, cp) in hversion.langs_and_cps()?.iter() {
77	///     println!("{} {}", lang, cp);
78	/// }
79	/// # w::SysResult::Ok(())
80	/// ```
81	#[must_use]
82	pub fn langs_and_cps(&self) -> SysResult<&[(LANGID, co::CP)]> {
83		unsafe {
84			self.VerQueryValue::<(LANGID, co::CP)>("\\VarFileInfo\\Translation")
85				.map(|(pblocks, sz)| {
86					std::slice::from_raw_parts(
87						pblocks,
88						sz as usize / std::mem::size_of::<(LANGID, co::CP)>(),
89					)
90				})
91		}
92	}

94	/// [`VerQueryValue`](https://learn.microsoft.com/en-us/windows/win32/api/winver/nf-winver-verqueryvaluew)
95	/// function.
96	///
97	/// # Safety
98	///
99	/// The returned pointer and size vary according to `lpSubBlock`. If you set
100	/// it wrong, you're likely to cause a buffer overrun.
101	///
102	/// This is a low-level method, prefer using:
103	/// * [`langs_and_cps`](crate::HVERSIONINFO::langs_and_cps);
104	/// * [`str_val`](crate::HVERSIONINFO::str_val);
105	/// * [`version_info`](crate::HVERSIONINFO::version_info).
106	///
107	/// # Examples
108	///
109	/// Reading version information from resource:
110	///
111	/// ```no_run
112	/// use winsafe::{self as w, prelude::*};
113	///
114	/// let exe_name = w::HINSTANCE::NULL.GetModuleFileName()?;
115	/// let hversion = w::HVERSIONINFO::GetFileVersionInfo(&exe_name)?;
116	///
117	/// let (pvsf, sz_data) = unsafe {
118	///     hversion.VerQueryValue::<w::VS_FIXEDFILEINFO>("\\")?
119	/// };
120	///
121	/// let ver = unsafe { &*pvsf }.dwFileVersion();
122	/// println!("Version {}.{}.{}.{}",
123	///     ver[0], ver[1], ver[2], ver[3]);
124	/// # w::SysResult::Ok(())
125	/// ```
126	#[must_use]
127	pub unsafe fn VerQueryValue<T>(&self, sub_block: &str) -> SysResult<(*const T, u32)> {
128		let mut lp_lp_buffer = std::ptr::null();
129		let mut pu_len = 0;
130
131		bool_to_sysresult(unsafe {
132			ffi::VerQueryValueW(
133				self.ptr(),
134				WString::from_str(sub_block).as_ptr(),
135				pvoid(&mut lp_lp_buffer),
136				&mut pu_len,
137			)
138		})
139		.map(|_| (lp_lp_buffer as *const T, pu_len))
140	}

142	/// Calls
143	/// [`HVERSIONINFO::VerQueryValue`](crate::HVERSIONINFO::VerQueryValue)
144	/// to retrieve a string value.
145	///
146	/// Common value names are:
147	/// * Comments
148	/// * CompanyName
149	/// * FileDescription
150	/// * FileVersion
151	/// * InternalName
152	/// * LegalCopyright
153	/// * LegalTrademarks
154	/// * OriginalFilename
155	/// * ProductName
156	/// * ProductVersion
157	/// * PrivateBuild
158	/// * SpecialBuild
159	///
160	/// # Examples
161	///
162	/// Reading product name and legal copyright from resource:
163	///
164	/// ```no_run
165	/// use winsafe::{self as w, prelude::*};
166	///
167	/// let exe_name = w::HINSTANCE::NULL.GetModuleFileName()?;
168	/// let hversion = w::HVERSIONINFO::GetFileVersionInfo(&exe_name)?;
169	///
170	/// let lang_cp = hversion.langs_and_cps()?[0]; // first language and code page
171	///
172	/// println!(
173	///     "{}\n{}",
174	///     hversion.str_val(lang_cp, "ProductName")?,
175	///     hversion.str_val(lang_cp, "LegalCopyright")?,
176	/// );
177	/// # w::SysResult::Ok(())
178	/// ```
179	#[must_use]
180	pub fn str_val(&self, lang_and_cp: (LANGID, co::CP), name: &str) -> SysResult<String> {
181		unsafe {
182			self.VerQueryValue::<u16>(&format!(
183				"\\StringFileInfo\\{:04x}{:04x}\\{}",
184				u16::from(lang_and_cp.0),
185				u16::from(lang_and_cp.1),
186				name
187			))
188			.map(|(pstr, len)| {
189				WString::from_wchars_slice(std::slice::from_raw_parts(pstr, len as _)).to_string()
190			})
191		}
192	}
