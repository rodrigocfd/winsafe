156#[must_use]
157fn enum_printers<T: Default + Clone>(
158	flags: co::PRINTER_ENUM, lvl: u32,
159) -> SysResult<Vec<T>>
160{
161	let (mut sz_buf, mut count) = (u32::default(), u32::default());
162	unsafe {
163		ffi::EnumPrintersW(
164			flags.raw(),
165			std::ptr::null_mut(),
166			lvl,
167			std::ptr::null_mut(),
168			0,
169			&mut sz_buf,
170			&mut count,
171		);
172	}
173
174	let sz_unit = std::mem::size_of::<T>() as u32;
175	let num_elems = ((sz_buf + sz_unit) - ((sz_buf + sz_unit) % sz_unit)) / sz_unit;
176	let mut buf = vec![T::default(); num_elems as _];
177
178	bool_to_sysresult(
179		unsafe {
180			ffi::EnumPrintersW(
181				flags.raw(),
182				std::ptr::null_mut(),
183				lvl,
184				buf.as_mut_ptr() as _,
185				(buf.len() * std::mem::size_of::<T>()) as _,
186				&mut sz_buf,
187				&mut count,
188			)
189		},
190	).map(|_| {
191		buf.truncate(count as _);
192		buf
193	})
194}

196/// [`GetDefaultPrinter`](https://learn.microsoft.com/en-us/windows/win32/printdocs/getdefaultprinter)
197/// function.
198///
199/// **Note:** This function doesn't seem to work with `raw-dylib` Cargo feature.
200///
201/// * [`EnumPrinters2`](crate::EnumPrinters2)
202/// * [`EnumPrinters4`](crate::EnumPrinters4)
203#[must_use]
204pub fn GetDefaultPrinter() -> SysResult<String> {
205	let mut sz = u32::default();
206	unsafe { ffi::GetDefaultPrinterW(std::ptr::null_mut(), &mut sz); }
207	let get_size_err = GetLastError();
208	if get_size_err != co::ERROR::INSUFFICIENT_BUFFER {
209		return Err(get_size_err);
210	}
211
212	let mut name_buf = WString::new_alloc_buf(sz as _);
213	bool_to_sysresult(
214		unsafe { ffi::GetDefaultPrinterW(name_buf.as_mut_ptr(), &mut sz) },
215	).map(|_| name_buf.to_string())
216}
