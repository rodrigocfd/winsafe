129#[must_use]
130fn enum_printers<T: Default + Clone>(flags: co::PRINTER_ENUM, lvl: u32) -> SysResult<Vec<T>> {
131	let (mut sz_buf, mut count) = (u32::default(), u32::default());
132	unsafe {
133		ffi::EnumPrintersW(
134			flags.raw(),
135			std::ptr::null_mut(),
136			lvl,
137			std::ptr::null_mut(),
138			0,
139			&mut sz_buf,
140			&mut count,
141		);
142	}
143
144	let sz_unit = std::mem::size_of::<T>() as u32;
145	let num_elems = ((sz_buf + sz_unit) - ((sz_buf + sz_unit) % sz_unit)) / sz_unit;
146	let mut buf = vec![T::default(); num_elems as _];
147
148	bool_to_sysresult(unsafe {
149		ffi::EnumPrintersW(
150			flags.raw(),
151			std::ptr::null_mut(),
152			lvl,
153			buf.as_mut_ptr() as _,
154			(buf.len() * std::mem::size_of::<T>()) as _,
155			&mut sz_buf,
156			&mut count,
157		)
158	})
159	.map(|_| {
160		buf.truncate(count as _);
161		buf
162	})
163}

165/// [`GetDefaultPrinter`](https://learn.microsoft.com/en-us/windows/win32/printdocs/getdefaultprinter)
166/// function.
167///
168/// **Note:** This function doesn't seem to work with `raw-dylib` Cargo feature.
169///
170/// * [`EnumPrinters2`](crate::EnumPrinters2)
171/// * [`EnumPrinters4`](crate::EnumPrinters4)
172#[must_use]
173pub fn GetDefaultPrinter() -> SysResult<String> {
174	let mut sz = u32::default();
175	unsafe {
176		ffi::GetDefaultPrinterW(std::ptr::null_mut(), &mut sz);
177	}
178	let get_size_err = GetLastError();
179	if get_size_err != co::ERROR::INSUFFICIENT_BUFFER {
180		return Err(get_size_err);
181	}
182
183	let mut name_buf = WString::new_alloc_buf(sz as _);
184	bool_to_sysresult(unsafe { ffi::GetDefaultPrinterW(name_buf.as_mut_ptr(), &mut sz) })
185		.map(|_| name_buf.to_string())
186}
