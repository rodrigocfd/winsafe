14	/// [`IFileDialogEvents`](https://learn.microsoft.com/en-us/windows/win32/api/shobjidl_core/nn-shobjidl_core-ifiledialogevents)
15	/// COM interface.
16	///
17	/// Automatically calls
18	/// [`IUnknown::Release`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-release)
19	/// when the object goes out of scope.
20	///
21	/// # Examples
22	///
23	/// ```no_run
24	/// use winsafe::{self as w, prelude::*, co};
25	///
26	/// let hparent: w::HWND; // initialized somewhere
27	/// # let hparent = w::HWND::NULL;
28	///
29	/// let file_open = w::CoCreateInstance::<w::IFileOpenDialog>(
30	///     &co::CLSID::FileOpenDialog,
31	///     None::<&w::IUnknown>,
32	///     co::CLSCTX::INPROC_SERVER,
33	/// )?;
34	///
35	/// let file_dialog_events = w::IFileDialogEvents::new_impl();
36	///
37	/// file_dialog_events.OnFolderChanging(
38	///     move |fd: &w::IFileDialog, si: &w::IShellItem| -> w::AnyResult<()> {
39	///         println!("New folder: {}",
40	///             si.GetDisplayName(co::SIGDN::FILESYSPATH)?);
41	///         Ok(())
42	///     },
43	/// );
44	///
45	/// file_open.Advise(&file_dialog_events)?;
46	/// # w::HrResult::Ok(())
47	/// ```
48}
