8	/// Handle to an
9	/// [internal drop structure](https://learn.microsoft.com/en-us/windows/win32/winprog/windows-data-types#hdrop).

12impl HDROP {
13	/// [`DragQueryFile`](https://learn.microsoft.com/en-us/windows/win32/api/shellapi/nf-shellapi-dragqueryfilew)
14	/// function.
15	///
16	/// Returns an iterator over the dropped files.
17	///
18	/// # Examples
19	///
20	/// Iterating over the strings:
21	///
22	/// ```no_run
23	/// use winsafe::{self as w, prelude::*};
24	///
25	/// let mut hdrop: w::HDROP; // initialized somewhere
26	/// # let mut hdrop = w::HDROP::NULL;
27	///
28	/// for file_path in hdrop.DragQueryFile()? {
29	///     let file_path = file_path?;
30	///     println!("File: {}", file_path);
31	/// }
32	/// # w::SysResult::Ok(())
33	/// ```
34	///
35	/// Collecting the strings into a [`Vec`](std::vec::Vec):
36	///
37	/// ```no_run
38	/// use winsafe::{self as w, prelude::*};
39	///
40	/// let mut hdrop: w::HDROP; // initialized somewhere
41	/// # let mut hdrop = w::HDROP::NULL;
42	///
43	/// let file_paths = hdrop.DragQueryFile()?
44	///     .collect::<w::SysResult<Vec<_>>>()?;
45	/// # w::SysResult::Ok(())
46	/// ```
47	#[must_use]
48	pub fn DragQueryFile(&self) -> SysResult<impl Iterator<Item = SysResult<String>> + '_> {
49		Ok(HdropIter::new(self)?)
50	}
51
52	/// [`DragQueryPoint`](https://learn.microsoft.com/en-us/windows/win32/api/shellapi/nf-shellapi-dragquerypoint)
53	/// function.
54	///
55	/// Returns the coordinates and whether the drop occurred in the client area
56	/// of the window.
57	///
58	/// Note that you must call this method before
59	/// [`DragQueryFile`](crate::HDROP::DragQueryFile), because it invalidates
60	/// the `HDROP` handle after the iterator is consumed.
61	#[must_use]
62	pub fn DragQueryPoint(&self) -> (POINT, bool) {
63		let mut pt = POINT::default();
64		let client_area = unsafe { ffi::DragQueryPoint(self.ptr(), pvoid(&mut pt)) };
65		(pt, client_area != 0)
66	}
