18	/// function.
19	pub fn ShellAbout(
20		&self,
21		title_bar: &str,
22		first_line: Option<&str>,
23		other_stuff: Option<&str>,
24		hicon: Option<&HICON>,
25	) -> SysResult<()> {
26		BoolRet(unsafe {
27			ffi::ShellAboutW(
28				self.ptr(),
29				WString::from_str(&match first_line {
30					Some(line) => format!("{}#{}", title_bar, line),
31					None => title_bar.to_owned(),
32				})
33				.as_ptr(),
34				WString::from_opt_str(other_stuff).as_ptr(),
35				hicon.map_or(std::ptr::null_mut(), |h| h.ptr()),
36			)
37		})
38		.to_sysresult()
39	}

47		parameters: Option<&str>,
48		directory: Option<&str>,
49		show_cmd: co::SW,

54				WString::from_str(operation).as_ptr(),
55				WString::from_str(file).as_ptr(),
56				parameters.map_or(std::ptr::null(), |lp| WString::from_str(lp).as_ptr()),
57				directory.map_or(std::ptr::null(), |lp| WString::from_str(lp).as_ptr()),
58				show_cmd.raw(),
59			)
60		};
61
62		if ret as usize > 32 { Ok(()) } else { Err(GetLastError()) }
63	}
