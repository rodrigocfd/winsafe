46	/// [`IShellItem::BindToHandler`](https://learn.microsoft.com/en-us/windows/win32/api/shobjidl_core/nf-shobjidl_core-ishellitem-bindtohandler)
47	/// method.
48	///
49	/// # Examples
50	///
51	/// Retrieving the items inside a directory:
52	///
53	/// ```no_run
54	/// use winsafe::{self as w, prelude::*, co};
55	///
56	/// let sh_folder: w::IShellItem; // initialized somewhere
57	/// # let sh_folder = unsafe { w::IShellItem::null() };

60	///     None::<&w::IBindCtx>,
61	///     &co::BHID::EnumItems,
62	/// )?;
63	/// # w::HrResult::Ok(())
64	/// ```
65	#[must_use]
66	fn BindToHandler<T>(&self, bind_ctx: Option<&impl ole_IBindCtx>, bhid: &co::BHID) -> HrResult<T>
67	where
68		T: ole_IUnknown,
69	{
70		let mut queried = unsafe { T::null() };
71		HrRet(unsafe {
72			(vt::<IShellItemVT>(self).BindToHandler)(
73				self.ptr(),
74				bind_ctx.map_or(std::ptr::null_mut(), |p| p.ptr()),
75				pcvoid(bhid),
76				pcvoid(&T::IID),
77				queried.as_mut(),
78			)
79		})
80		.to_hrresult()
81		.map(|_| queried)
82	}
