12com_interface_userdef! { IDropTarget: IDropTargetImpl, "00000122-0000-0000-c000-000000000046";
13	/// [`IDropTarget`](https://learn.microsoft.com/en-us/windows/win32/api/oleidl/nn-oleidl-idroptarget)
14	/// COM interface.
15	///
16	/// Automatically calls
17	/// [`Release`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-release)
18	/// when the object goes out of scope.
19	///
20	/// # Examples
21	///
22	/// Retrieving dropped files:
23	///
24	/// ```no_run
25	/// use winsafe::{self as w, prelude::*, co};
26	///
27	/// let drop_target = w::IDropTarget::new_impl();
28	///
29	/// drop_target.Drop(
30	///     |d: &w::IDataObject, key_st: co::MK, pt: w::POINT, fx: &mut co::DROPEFFECT|
31	///         -> w::AnyResult<()>
32	///     {
33	///         let mut fmt = w::FORMATETC::default();
34	///         fmt.cfFormat = co::CF::HDROP;
35	///         fmt.dwAspect = co::DVASPECT::CONTENT;
36	///         fmt.tymed = co::TYMED::HGLOBAL;
37	///
38	///         let medium = unsafe { d.GetData(&fmt)? };
39	///         let hglobal = unsafe { medium.ptr_hglobal().unwrap() };
40	///         let ptr_lock = hglobal.GlobalLock()?;
41	///         let hdrop = unsafe { w::HDROP::from_ptr(ptr_lock.as_ptr() as _) };
42	///         let dropped_paths = hdrop.DragQueryFile()?
43	///             .collect::<w::SysResult<Vec<_>>>()?;
44	///
45	///         for f in dropped_paths {
46	///             println!("> {f}");
47	///         }
48	///
49	///         Ok(())
50	///     },
51	/// );
52	/// ```
53}

112struct IDropTargetImpl {

114	counter: AtomicU32,
115	DragEnter:
116		Option<Box<dyn Fn(&IDataObject, co::MK, POINT, &mut co::DROPEFFECT) -> AnyResult<()>>>,
117	DragOver: Option<Box<dyn Fn(co::MK, POINT, &mut co::DROPEFFECT) -> AnyResult<()>>>,
118	DragLeave: Option<Box<dyn Fn() -> AnyResult<()>>>,
119	Drop: Option<Box<dyn Fn(&IDataObject, co::MK, POINT, &mut co::DROPEFFECT) -> AnyResult<()>>>,
120}

123	#[must_use]
124	const fn new() -> Self {
125		Self {
126			vt: IDropTargetVT {
127				IUnknownVT: IUnknownVT {
128					QueryInterface: Self::QueryInterface,
129					AddRef: Self::AddRef,
130					Release: Self::Release,
131				},
132				DragEnter: Self::DragEnter,
133				DragOver: Self::DragOver,
134				DragLeave: Self::DragLeave,
135				Drop: Self::Drop,
136			},
137			counter: AtomicU32::new(1),
138			DragEnter: None,
139			DragOver: None,
140			DragLeave: None,

143	}

145	fn_com_userdef_iunknown_impls!(Self);

147	fn DragEnter(
148		p: COMPTR,
149		pDataObj: COMPTR,
150		grfKeyState: u32,
151		pt: u64,
152		pdwEffect: *mut u32,
153	) -> HRES {
154		let box_impl = box_impl_of::<Self>(p);
155		hrresult_to_hres(match &box_impl.DragEnter {
156			Some(func) => unsafe {
157				anyresult_to_hresult(func(
158					&ManuallyDrop::new(IDataObject::from_ptr(pDataObj)),
159					co::MK::from_raw(grfKeyState as _),
160					POINT::with(LODWORD(pt) as _, HIDWORD(pt) as _),
161					&mut *(pdwEffect as *mut co::DROPEFFECT),
162				))
163			},
164			None => Ok(()),
165		})
166	}
