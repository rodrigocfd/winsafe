53/// [`CoCreateInstance`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-cocreateinstance)
54/// function.
55///
56/// Returns an [`IUnknown`](crate::IUnknown)-derived COM object.
57///
58/// # Examples
59///
60/// Instantiating an [`ITaskbarList`](crate::ITaskbarList) object:
61///
62/// ```rust,ignore
63/// use winsafe::{self as w, prelude::*, co};
64///
65/// let obj = w::CoCreateInstance::<w::ITaskbarList>(
66///     &co::CLSID::TaskbarList,
67///     None,
68///     co::CLSCTX::INPROC_SERVER,
69/// )?;
70/// # w::HrResult::Ok(())
71/// ```
72#[must_use]
73pub fn CoCreateInstance<T>(
74	clsid: &co::CLSID,
75	iunk_outer: Option<&mut IUnknown>,
76	cls_context: co::CLSCTX,
77) -> HrResult<T>
78where
79	T: ole_IUnknown,
80{
81	let (mut queried, mut queried_outer) = unsafe { (T::null(), IUnknown::null()) };
82
83	ok_to_hrresult(unsafe {
84		ffi::CoCreateInstance(
85			clsid as *const _ as _,
86			iunk_outer
87				.as_ref()
88				.map_or(std::ptr::null_mut(), |_| queried_outer.as_mut()),
89			cls_context.raw(),
90			&T::IID as *const _ as _,
91			queried.as_mut(),
92		)
93	})
94	.map(|_| {
95		if let Some(iunk_outer) = iunk_outer {
96			*iunk_outer = queried_outer; // create outer IUnknown if due
97		}
98		queried // return new IUnknown-derived object
99	})
100}

131/// [`CoInitializeEx`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-coinitializeex)
132/// function, which
133/// [initializes](https://learn.microsoft.com/en-us/windows/win32/learnwin32/initializing-the-com-library)
134/// the COM library. When succeeding, returns an informational error code.
135///
136/// In the original C implementation, you must call
137/// [`CoUninitialize`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-couninitialize)
138/// as a cleanup operation.
139///
140/// Here, the cleanup is performed automatically, because `CoInitializeEx`
141/// returns a [`CoUninitializeGuard`](crate::guard::CoUninitializeGuard), which
142/// automatically calls `CoUninitialize` when the guard goes out of scope. You
143/// must, however, keep the guard alive, otherwise the cleanup will be performed
144/// right away.
145///
146/// # Examples
147///
148/// ```no_run
149/// use winsafe::{self as w, prelude::*, co};
150///
151/// let _com_guard = w::CoInitializeEx( // keep guard alive
152///     co::COINIT::APARTMENTTHREADED
153///     | co::COINIT::DISABLE_OLE1DDE,
154/// )?;
155///
156/// // program runs...
157///
158/// // CoUninitialize() automatically called
159/// # w::HrResult::Ok(())
160/// ```
161#[must_use]
162pub fn CoInitializeEx(coinit: co::COINIT) -> HrResult<CoUninitializeGuard> {
163	unsafe {
164		let hr = co::HRESULT::from_raw(ffi::CoInitializeEx(std::ptr::null_mut(), coinit.raw()));
165		match hr {
166			co::HRESULT::S_OK | co::HRESULT::S_FALSE | co::HRESULT::RPC_E_CHANGED_MODE => {
167				Ok(CoUninitializeGuard::new(hr))
168			},
169			hr => Err(hr),
170		}
171	}
172}

190/// [`CoTaskMemAlloc`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-cotaskmemalloc)
191/// function.
192///
193/// # Examples
194///
195/// ```no_run
196/// use winsafe::{self as w, prelude::*};
197///
198/// let pmem = w::CoTaskMemAlloc(100)?;
199///
200/// // use memory block...
201///
202/// // CoTaskMemFree() automatically called
203/// # w::HrResult::Ok(())
204/// ```
205///
206/// # Related functions
207///
208/// * [`CoTaskMemRealloc`](crate::CoTaskMemRealloc)
209#[must_use]
210pub fn CoTaskMemAlloc(cb: usize) -> HrResult<CoTaskMemFreeGuard> {
211	let p = unsafe { ffi::CoTaskMemAlloc(cb) };
212	if p.is_null() {
213		Err(co::HRESULT::E_OUTOFMEMORY)
214	} else {
215		Ok(unsafe { CoTaskMemFreeGuard::new(p, cb) })
216	}
217}

219/// [`CoTaskMemRealloc`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-cotaskmemfree)
220/// function.
221///
222/// Originally this method returns a pointer to the reallocated memory block;
223/// here the original pointer is automatically updated.
224///
225/// # Examples
226///
227/// ```no_run
228/// use winsafe::{self as w, prelude::*};
229///
230/// let mut pmem = w::CoTaskMemAlloc(100)?;
231/// w::CoTaskMemRealloc(&mut pmem, 120)?;
232///
233/// // use memory block...
234///
235/// // CoTaskMemFree() automatically called
236/// # w::HrResult::Ok(())
237/// ```
238///
239/// # Related functions
240///
241/// * [`CoTaskMemAlloc`](crate::CoTaskMemAlloc)
242#[must_use]
243pub fn CoTaskMemRealloc(pv: &mut CoTaskMemFreeGuard, cb: usize) -> HrResult<()> {
244	let (old_pmem, _) = pv.leak();
245	let p = unsafe { ffi::CoTaskMemRealloc(old_pmem, cb) };
246	if p.is_null() {
247		Err(co::HRESULT::E_OUTOFMEMORY)
248	} else {
249		*pv = unsafe { CoTaskMemFreeGuard::new(p, cb) };
250		Ok(())
251	}
252}

307/// [`OleInitialize`](https://learn.microsoft.com/en-us/windows/win32/api/ole2/nf-ole2-oleinitialize)
308/// function, which calls [`CoInitializeEx`](crate::CoInitializeEx) and enables
309/// OLE operations.
310///

312/// [`OleUninitialize`](https://learn.microsoft.com/en-us/windows/win32/api/ole2/nf-ole2-oleuninitialize)
313/// as a cleanup operation.
314///
315/// Here, the cleanup is performed automatically, because `OleInitialize`
316/// returns an [`OleUninitializeGuard`](crate::guard::OleUninitializeGuard),

320///
321/// # Examples
322///
323/// ```no_run
324/// use winsafe::{self as w, prelude::*, co};
325///
326/// let _ole_guard = w::OleInitialize()?;
327///
328/// // program runs...
329///
330/// // OleUninitialize() automatically called
331/// # w::HrResult::Ok(())
332/// ```
333#[must_use]
334pub fn OleInitialize() -> HrResult<OleUninitializeGuard> {
335	unsafe {
336		ok_to_hrresult(ffi::OleInitialize(std::ptr::null_mut()))
337			.map(|_| OleUninitializeGuard::new())
338	}
339}

341/// [`StringFromCLSID`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-stringfromclsid)
342/// function.
343#[must_use]
344pub fn StringFromCLSID(clsid: &co::CLSID) -> HrResult<String> {
345	let mut pstr = std::ptr::null_mut::<u16>();
346	ok_to_hrresult(unsafe { ffi::StringFromCLSID(clsid as *const _ as _, &mut pstr) }).map(|_| {
347		let name = unsafe { WString::from_wchars_nullt(pstr) };
348		let _ = unsafe { CoTaskMemFreeGuard::new(pstr as _, 0) };
349		name.to_string()
350	})
