125/// [`SNB`](https://learn.microsoft.com/en-us/windows/win32/stg/snb)
126/// struct.
127#[repr(transparent)]
128pub struct SNB(*mut *mut u16);
129
130impl_default!(SNB);
131
132impl Drop for SNB {
133	fn drop(&mut self) {
134		let _ = unsafe { CoTaskMemFreeGuard::new(self.0 as _, 0) }; // size is irrelevant
135	}
136}
137
138impl SNB {
139	/// Creates a new `SNB` by allocating a contiguous memory block for pointers
140	/// and strings.
141	#[must_use]
142	pub fn from_strs(strs: &[impl AsRef<str>]) -> HrResult<Self> {
143		if strs.is_empty() {
144			return Ok(Self::default());
145		}
146
147		let tot_bytes_ptrs = (strs.len() + 1) * std::mem::size_of::<*mut u16>(); // plus null
148
149		let wstrs = WString::from_str_vec(strs);
150		let num_valid_chars = wstrs
151			.as_slice() // count just the chars; important because stack alloc has zero fills
152			.iter()
153			.filter(|ch| **ch != 0x0000)
154			.count();
155		let tot_bytes_wstrs = ( num_valid_chars + strs.len() ) // plus one null for each string
156			* std::mem::size_of::<u16>();
157
158		let mut block = CoTaskMemAlloc(tot_bytes_ptrs + tot_bytes_wstrs)?; // alloc block for pointers and strings
159		let ptr_block = block.as_mut_ptr() as *mut u8;
160		let ptr_block_wstrs = unsafe { ptr_block.add(tot_bytes_ptrs) } as *mut u16; // start of strings block
161
162		let mut idx_cur_wstr = 0; // current string to be copied to block
163		*unsafe { block.as_mut_slice_aligned::<*mut u16>() }
164			.get_mut(idx_cur_wstr)
165			.unwrap() = ptr_block_wstrs; // copy pointer to 1st string
166		idx_cur_wstr += 1;
167
168		wstrs.as_slice().iter().enumerate().for_each(|(idx, ch)| {
169			if *ch == 0x0000 && idx_cur_wstr < strs.len() {
170				unsafe {
171					*block
172						.as_mut_slice_aligned::<*mut u16>() // copy pointer to subsequent strings in the block
173						.get_mut(idx_cur_wstr)
174						.unwrap() = ptr_block_wstrs.add(idx + 1);
175				}
176				idx_cur_wstr += 1;
177			}
178		});
179
180		*unsafe { block.as_mut_slice_aligned::<*mut u16>() }
181			.get_mut(idx_cur_wstr)
182			.unwrap() = std::ptr::null_mut(); // null pointer after string pointers
183
184		wstrs.copy_to_slice(
185			// Beyond pointers, copy each null-terminated string sequentially.
186			unsafe {
187				std::slice::from_raw_parts_mut(
188					ptr_block_wstrs,
189					tot_bytes_wstrs / std::mem::size_of::<u16>(),
190				)
191			},
192		);
193
194		let (ptr, _) = block.leak();
195		Ok(Self(ptr as _))
196	}
197
198	/// Returns the underlying pointer.
199	#[must_use]
200	pub const fn as_ptr(&self) -> *mut *mut u16 {
201		self.0
202	}
203
204	/// Converts the internal UTF-16 blocks into strings.
205	#[must_use]
206	pub fn to_strings(&self) -> Vec<String> {
207		let mut vec = Vec::<String>::new();
208		if !self.0.is_null() {
209			let mut idx_ptr = 0;
210			loop {
211				let ptr_ws = unsafe {
212					let sli_ptrs = std::slice::from_raw_parts(self.0, idx_ptr + 1);
213					*sli_ptrs.get_unchecked(idx_ptr) // get nth pointer to string
214				};
215				if ptr_ws.is_null() {
216					// a null pointer means the end of pointers block
217					break;
218				}
219				let ws = unsafe { WString::from_wchars_nullt(ptr_ws) };
220				vec.push(ws.to_string());
221				idx_ptr += 1;
222			}
223		}
224		vec
225	}
226}
