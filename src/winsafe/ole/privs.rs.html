9/// Returns a reference to the virtual table of the COM object.
10///
11/// # Safety
12///
13/// Make sure the inheritance is correct.
14#[must_use]
15pub(crate) unsafe fn vt<T>(obj: &impl ole_IUnknown) -> &T {
16	let ppvt = obj.ptr() as *mut *mut T;
17	unsafe { &**ppvt }
18}

20/// Wraps a `co::HRESULT` value returned by a FFI call, providing treatment
21/// options.
22pub(crate) struct HrRet(pub(crate) HRES);
23
24impl HrRet {
25	/// If value is `co::HRESULT::S_OK` yields `Ok()`, othersize `Err(hr)`.
26	#[must_use]
27	pub(crate) const fn to_hrresult(self) -> HrResult<()> {
28		match unsafe { co::HRESULT::from_raw(self.0) } {
29			co::HRESULT::S_OK => Ok(()),
30			hr => Err(hr),
31		}
32	}
33
34	/// If value is `co::HRESULT::S_OK` yields `Ok(true)`,
35	/// if `co::HRESULT::S_FALSE` yields `Ok(false)` othersize `Err(hr)`.
36	#[must_use]
37	pub(crate) const fn to_bool_hrresult(self) -> HrResult<bool> {
38		match unsafe { co::HRESULT::from_raw(self.0) } {
39			co::HRESULT::S_OK => Ok(true),
40			co::HRESULT::S_FALSE => Ok(false),
41			hr => Err(hr),
42		}
43	}
44}
45
46/// If value is `Ok` yields 0, otherwise the error code.
47#[must_use]
48pub(crate) fn hrresult_to_hres<T>(hrr: HrResult<T>) -> HRES {
49	match hrr {
50		Ok(_) => co::HRESULT::S_OK.raw(),
51		Err(e) => e.raw(),
52	}
53}

55/// Converts the `*mut u16` to `String`, and releases the pointer with
56/// `CoTaskMemFree`.
57#[must_use]
58pub(crate) fn htaskmem_ptr_to_str(p: *mut u16) -> String {
59	let wstr = unsafe { WString::from_wchars_nullt(p) };
60	let _ = unsafe { CoTaskMemFreeGuard::new(p as _, 0) };
61	wstr.to_string()
62}

64/// Given the pointer to the memory block, converts it to the `Box` of the
65/// allocated VT struct.
66///
67/// Used with COM interface impls.
68#[must_use]
69pub(crate) fn box_impl_of<T>(p: COMPTR) -> ManuallyDrop<Box<T>> {
70	let pp = p as *mut *mut T;
71	let box_impl = ManuallyDrop::new(unsafe { Box::from_raw(*pp) });
72	box_impl
73}

75/// If the error is `HrResult`, returns it, otherwise displays a message box
76/// with the error and returns `ERROR::E_UNEXPECTED`.
77///
78/// Used with COM interface impls.
79#[must_use]
80pub(crate) fn anyresult_to_hresult<T>(res: AnyResult<T>) -> HrResult<T> {
81	res.map_err(|err| {
82		if let Some(hr) = err.downcast_ref::<co::HRESULT>() {
83			*hr
84		} else if let Some(err) = err.downcast_ref::<co::ERROR>() {
85			err.to_hresult()
86		} else {
87			HWND::NULL
88				.MessageBox(
89					&format!("Unhandled error: {}", err.to_string()),
90					"Unhandled error in COM impl",
91					co::MB::ICONERROR,
92				)
93				.unwrap();
94			co::HRESULT::E_UNEXPECTED
95		}
96	})
