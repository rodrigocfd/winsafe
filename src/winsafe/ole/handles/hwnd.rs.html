8impl HWND {
9	/// [`RegisterDragDrop`](https://learn.microsoft.com/en-us/windows/win32/api/ole2/nf-ole2-registerdragdrop)
10	/// function.
11	///
12	/// Note that if you don't call [`OleInitialize`](crate::OleInitialize)
13	/// before this function, you'll receive an
14	/// [`ERROR::OUTOFMEMORY`](crate::co::ERROR::OUTOFMEMORY) error.
15	pub fn RegisterDragDrop(&self, drop_target: &IDropTarget) -> HrResult<()> {
16		HrRet(unsafe { ffi::RegisterDragDrop(self.ptr(), drop_target.ptr() as _) }).to_hrresult()
17	}
18
19	/// [`RevokeDragDrop`](https://learn.microsoft.com/en-us/windows/win32/api/ole2/nf-ole2-revokedragdrop)
20	/// function.
21	pub fn RevokeDragDrop(&self) -> HrResult<()> {
22		HrRet(unsafe { ffi::RevokeDragDrop(self.ptr()) }).to_hrresult()
23	}
24}
