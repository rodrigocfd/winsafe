9/// RAII implementation which automatically calls
10/// [`CoLockObjectExternal`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-colockobjectexternal)
11/// to unlock the COM pointer.
12pub struct CoLockObjectExternalGuard<'a, T>
13where
14	T: ole_IUnknown,
15{
16	com_obj: &'a T,
17}
18
19impl<'a, T> Drop for CoLockObjectExternalGuard<'a, T>
20where
21	T: ole_IUnknown,
22{
23	fn drop(&mut self) {
24		unsafe {
25			ffi::CoLockObjectExternal(self.com_obj.ptr(), 0, 1); // ignore errors
26		}
27	}
28}
29
30impl<'a, T> CoLockObjectExternalGuard<'a, T>
31where
32	T: ole_IUnknown,
33{
34	/// Constructs the guard by keeping the reference to the COM pointer.
35	///
36	/// # Safety
37	///
38	/// Be sure the COM pointer has been locked with a previous call to

40	#[must_use]
41	pub const unsafe fn new(com_obj: &'a T) -> Self {
42		Self { com_obj }
43	}
44}

46/// RAII implementation which automatically calls
47/// [`CoTaskMemFree`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-cotaskmemfree)
48/// when the object goes out of scope.
49pub struct CoTaskMemFreeGuard {
50	pmem: *mut std::ffi::c_void,
51	sz: usize,
52}
53
54impl Drop for CoTaskMemFreeGuard {
55	fn drop(&mut self) {
56		if !self.pmem.is_null() {
57			unsafe {
58				ffi::CoTaskMemFree(self.pmem);
59			}
60		}
61	}
62}
63
64impl CoTaskMemFreeGuard {
65	/// Constructs the guard.
66	///
67	/// # Safety
68	///
69	/// Be sure the pointer must be freed with
70	/// [`CoTaskMemFree`](https://learn.microsoft.com/en-us/windows/win32/api/combaseapi/nf-combaseapi-cotaskmemfree)
71	/// at the end of the scope, and the size is correct.
72	#[must_use]
73	pub const unsafe fn new(pmem: *mut std::ffi::c_void, sz: usize) -> Self {
74		Self { pmem, sz }
75	}
76
77	/// Ejects the underlying memory pointer and size, leaving null and zero in
78	/// their places.
79	///
80	/// Since the internal memory pointer will be invalidated, the destructor
81	/// will not run. It's your responsibility to run it, otherwise you'll cause
82	/// a memory leak.
83	#[must_use]
84	pub fn leak(&mut self) -> (*mut std::ffi::c_void, usize) {
85		(
86			std::mem::replace(&mut self.pmem, std::ptr::null_mut()),
87			std::mem::replace(&mut self.sz, 0),
88		)
89	}
90
91	pub_fn_mem_block!();
92}
