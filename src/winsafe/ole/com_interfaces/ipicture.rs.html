131	/// [`IPicture::Render`](https://learn.microsoft.com/en-us/windows/win32/api/ocidl/nf-ocidl-ipicture-render)
132	/// method.
133	///
134	/// This method will automatically perform the inverse height calculations –
135	/// convert `src_extent_himetric.cy` to a negative value and compensate
136	/// `src_offset_himetric.y`. This is necessary because HIMETRIC height is
137	/// rendered in reverse order (bottom to top) when compared to HDC height.
138	///
139	/// Default values:
140	///
141	/// | Parameter | Default value |
142	/// | -- | :--: |
143	/// | `dest_pt` | `0`, `0` |
144	/// | `dest_sz` | [`HDC`](crate::HDC) client rect |
145	/// | `src_offset_himetric` | `0`, `0` |
146	/// | `src_extent_himetric` | image size from [`get_Width`](crate::prelude::ole_IPicture::get_Width), [`get_Height`](crate::prelude::ole_IPicture::get_Height) |
147	fn Render(
148		&self,
149		hdc: &HDC,
150		dest_pt: Option<POINT>,
151		dest_sz: Option<SIZE>,
152		src_offset_himetric: Option<POINT>,
153		src_extent_himetric: Option<SIZE>,
154		metafile_bounds: Option<&RECT>,
155	) -> AnyResult<()> {
156		let dest_sz_hdc = match dest_sz {
157			Some(sz) => sz,
158			None => {
159				let rc = hdc.WindowFromDC().unwrap().GetClientRect()?;
160				SIZE::new(rc.right - rc.left, rc.bottom - rc.top)
161			},
162		};
163		let src_extent_compensated = match src_extent_himetric {
164			Some(sz) => SIZE::new(sz.cx, -sz.cy),
165			None => SIZE::new(self.get_Width()?, -self.get_Height()?),
166		};
167
168		ok_to_hrresult(unsafe {
169			(vt::<IPictureVT>(self).Render)(
170				self.ptr(),
171				hdc.ptr(),
172				dest_pt.map_or(0, |pt| pt.x),
173				dest_pt.map_or(0, |pt| pt.y),
174				dest_sz_hdc.cx,
175				dest_sz_hdc.cy,
176				src_offset_himetric.map_or(0, |off| off.x),
177				src_offset_himetric.map_or(0, |off| off.y) - src_extent_compensated.cy,
178				src_extent_compensated.cx,
179				src_extent_compensated.cy,
180				metafile_bounds.map_or(std::ptr::null_mut(), |rc| rc as *const _ as _),
181			)
182		})
183		.map_err(|e| e.into())
184	}
