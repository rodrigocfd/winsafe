7com_interface! { IUnknown: "00000000-0000-0000-c000-000000000046";
8	/// [`IUnknown`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nn-unknwn-iunknown)
9	/// COM interface. It's the base to all COM interfaces.
10	///
11	/// The `clone` method calls
12	/// [`AddRef`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-addref)
13	/// internally.
14	///
15	/// Automatically calls
16	/// [`Release`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-release)
17	/// when the object goes out of scope.
18}
19
20/// This trait is enabled with the `ole` feature, and provides methods for
21/// [`IUnknown`](crate::IUnknown). It is the base trait for all COM traits.
22///
23/// Prefer importing this trait through the prelude:
24///
25/// ```no_run
26/// use winsafe::prelude::*;
27/// ```
28///
29/// Note that the `IUnknown` virtual table has two other methods: `AddRef` and
30/// `Release`. While these methods are relevant in C++, here they are abstracted
31/// away as it follows:
32///
33/// * `AddRef` – called along the `clone` method from the
34/// [`Clone`](std::clone::Clone) trait;
35///
36/// * `Release` – called automatically by the [`Drop`](std::ops::Drop) trait, so
37/// you don't need to worry about it.
38pub trait ole_IUnknown: Clone {
39	/// The COM interface ID.
40	const IID: co::IID;
41
42	/// Creates an object from a COM virtual table pointer.
43	///
44	/// This method can be used as an escape hatch to interoperate with other
45	/// libraries.
46	///
47	/// # Panics
48	///
49	/// Panics if trying to create a custom COM implementation object.
50	///
51	/// # Safety
52	///
53	/// Be sure the pointer points to a properly allocated COM virtual table.
54	#[must_use]
55	unsafe fn from_ptr(p: *mut std::ffi::c_void) -> Self;
56
57	/// Returns the pointer to the underlying COM virtual table.
58	///
59	/// This method can be used as an escape hatch to interoperate with other
60	/// libraries.
61	#[must_use]
62	fn ptr(&self) -> *mut std::ffi::c_void;
63
64	/// Returns a mutable reference do the underlying COM virtual table pointer.
65	///
66	/// This method can be used as an escape hatch to interoperate with other
67	/// libraries.
68	///
69	/// # Panics
70	///
71	/// Panics if trying to modify a custom COM implementation object.
72	///
73	/// # Safety
74	///
75	/// Be sure the pointer being set points to a properly allocated COM virtual
76	/// table.
77	#[must_use]
78	unsafe fn as_mut(&mut self) -> &mut *mut std::ffi::c_void;
79
80	/// Creates an object from a null COM virtual table pointer.
81	///
82	/// # Safety
83	///
84	/// The pointer must be initialized before any call. Do not call methods on
85	/// a null COM pointer.
86	#[must_use]
87	unsafe fn null() -> Self {
88		Self::from_ptr(std::ptr::null_mut())
89	}
90
91	/// Returns the pointer to the underlying COM virtual table and sets the
92	/// internal pointer to null, so that
93	/// [`Release`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-release)
94	/// won't be called.
95	///
96	/// Be sure to release the pointer, otherwise, as the name of this method
97	/// implies, you will cause a resource leak.
98	#[must_use]
99	fn leak(&mut self) -> *mut std::ffi::c_void {
100		let p = self.ptr();
101		unsafe {
102			*self.as_mut() = std::ptr::null_mut();
103		}
104		p
105	}
106
107	/// [`IUnknown::QueryInterface`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-queryinterface(refiid_void))
108	/// method.
109	#[must_use]
110	fn QueryInterface<T>(&self) -> HrResult<T>
111	where
112		T: ole_IUnknown,
113	{
114		let mut queried = unsafe { T::null() };
115		ok_to_hrresult(unsafe {
116			(vt::<IUnknownVT>(self).QueryInterface)(
117				self.ptr(),
118				&T::IID as *const _ as _,
119				queried.as_mut(),
120			)
121		})
122		.map(|_| queried)
123	}
124}
