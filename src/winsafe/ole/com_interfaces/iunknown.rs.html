8com_interface! { IUnknown: "00000000-0000-0000-c000-000000000046";
9	/// [`IUnknown`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nn-unknwn-iunknown)
10	/// COM interface. It's the base to all COM interfaces.
11	///
12	/// The `clone` method calls
13	/// [`AddRef`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-addref)
14	/// internally.
15	///
16	/// Automatically calls
17	/// [`Release`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-release)
18	/// when the object goes out of scope.
19}
20
21/// This trait is enabled with the `ole` feature, and provides methods for
22/// [`IUnknown`](crate::IUnknown). It is the base trait for all COM traits.
23///
24/// Prefer importing this trait through the prelude:
25///
26/// ```no_run
27/// use winsafe::prelude::*;
28/// ```
29///
30/// Note that the `IUnknown` virtual table has two other methods: `AddRef` and
31/// `Release`. While these methods are relevant in C++, here they are abstracted
32/// away as it follows:
33///
34/// * `AddRef` – called along the `clone` method from the
35/// [`Clone`](std::clone::Clone) trait;
36///
37/// * `Release` – called automatically by the [`Drop`](std::ops::Drop) trait, so
38/// you don't need to worry about it.
39pub trait ole_IUnknown: Clone {
40	/// The COM interface ID.
41	const IID: co::IID;
42
43	/// Creates an object from a COM virtual table pointer.
44	///
45	/// This method can be used as an escape hatch to interoperate with other
46	/// libraries.
47	///
48	/// # Panics
49	///
50	/// Panics if trying to create a custom COM implementation object.
51	///
52	/// # Safety
53	///
54	/// Be sure the pointer points to a properly allocated COM virtual table.
55	#[must_use]
56	unsafe fn from_ptr(p: *mut std::ffi::c_void) -> Self;
57
58	/// Returns the pointer to the underlying COM virtual table.
59	///
60	/// This method can be used as an escape hatch to interoperate with other
61	/// libraries.
62	#[must_use]
63	fn ptr(&self) -> *mut std::ffi::c_void;
64
65	/// Returns a mutable reference do the underlying COM virtual table pointer.
66	///
67	/// This method can be used as an escape hatch to interoperate with other
68	/// libraries.
69	///
70	/// # Panics
71	///
72	/// Panics if trying to modify a custom COM implementation object.
73	///
74	/// # Safety
75	///
76	/// Be sure the pointer being set points to a properly allocated COM virtual
77	/// table.
78	#[must_use]
79	unsafe fn as_mut(&mut self) -> &mut *mut std::ffi::c_void;
80
81	/// Creates an object from a null COM virtual table pointer.
82	///
83	/// # Safety
84	///
85	/// The pointer must be initialized before any call. Do not call methods on
86	/// a null COM pointer.
87	#[must_use]
88	unsafe fn null() -> Self {
89		unsafe { Self::from_ptr(std::ptr::null_mut()) }
90	}
91
92	/// Returns the pointer to the underlying COM virtual table and sets the
93	/// internal pointer to null, so that
94	/// [`Release`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-release)
95	/// won't be called.
96	///
97	/// Be sure to release the pointer, otherwise, as the name of this method
98	/// implies, you will cause a resource leak.
99	#[must_use]
100	fn leak(&mut self) -> *mut std::ffi::c_void {
101		let p = self.ptr();
102		unsafe {
103			*self.as_mut() = std::ptr::null_mut();
104		}
105		p
106	}
107
108	/// [`IUnknown::QueryInterface`](https://learn.microsoft.com/en-us/windows/win32/api/unknwn/nf-unknwn-iunknown-queryinterface(refiid_void))
109	/// method.
110	#[must_use]
111	fn QueryInterface<T>(&self) -> HrResult<T>
112	where
113		T: ole_IUnknown,
114	{
115		let mut queried = unsafe { T::null() };
116		HrRet(unsafe {
117			(vt::<IUnknownVT>(self).QueryInterface)(self.ptr(), pcvoid(&T::IID), queried.as_mut())
118		})
119		.to_hrresult()
120		.map(|_| queried)
121	}
122}
