6const_basic_decl! { HRESULT: u32;
7	/// A native
8	/// [COM error code](https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-erref/0642cb2f-2075-4469-918c-4441e69c548a)
9	/// (`u32`), which can be considered a superset of
10	/// [`ERROR`](crate::co::ERROR).
11	///
12	/// The [`Result` alias](crate#errors-and-result-aliases) for this type is
13	/// [`HrResult`](crate::HrResult).
14	///
15	/// Implements the standard [`Error`](std::error::Error) trait.
16	///
17	/// Implements the [`Debug`](std::fmt::Debug) and
18	/// [`Display`](std::fmt::Display) traits to show the error code along with
19	/// the error description, taken from
20	/// [`SystemError`](crate::prelude::SystemError) trait. For example, the
21	/// code below:
22	///
23	/// ```no_run
24	/// use winsafe::{self as w, prelude::*, co};
25	///
26	/// println!("{}", co::HRESULT::E_INVALIDARG);
27	/// ```
28	///
29	/// Will print:
30	///
31	/// ```text
32	/// [0x80070057 2147942487] The parameter is incorrect.
33	/// ```
34	///
35	/// The numbers inside the brackets are the system error code, in
36	/// hexadecimal and decimal formats for convenience.
37	///
38	/// An [`ERROR`](crate::co::ERROR) can be seamlessly converted into an
39	/// `HRESULT`. This operation is equivalent to the
40	/// [`HRESULT_FROM_WIN32`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_from_win32)
41	/// macro:
42	///
43	/// ```no_run
44	/// use winsafe::{self as w, prelude::*, co};
45	///
46	/// let err = co::ERROR::FILE_NOT_FOUND;
47	///
48	/// let hr = err.to_hresult();
49	/// ```
50}
51
52impl std::error::Error for HRESULT {
53	fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
54		None
55	}
56}

58impl std::fmt::Display for HRESULT {
59	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
60		if self.0 > 0xffff {
61			write!(f, "[{:#010x}] {}", self.0, self.FormatMessage().trim_end())
62		} else {
63			write!(f, "[{:#06x}] {}", self.0, self.FormatMessage().trim_end())
64		}
65	}
66}
67impl std::fmt::Debug for HRESULT {
68	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
69		if self.0 > 0xffff {
70			write!(f, "HRESULT({:#010x} {}) {}", self.0, self.0, self.FormatMessage().trim_end())
71		} else {
72			write!(f, "HRESULT({:#06x} {}) {}", self.0, self.0, self.FormatMessage().trim_end())
73		}
74	}
75}

79impl co::ERROR {
80	/// [`HRESULT_FROM_WIN32`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_from_win32)
81	/// macro.
82	#[must_use]
83	pub const fn to_hresult(self) -> co::HRESULT {
84		unsafe {
85			if self.raw() as i32 <= 0 {
86				co::HRESULT::from_raw(self.raw())
87			} else {
88				co::HRESULT::from_raw(
89					(self.raw() & 0x0000_ffff) | co::FACILITY::WIN32.raw() << 16 | 0x8000_0000,
90				)
91			}
92		}
93	}
94}

96impl HRESULT {
97	/// [`HRESULT_CODE`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_code)
98	/// macro.
99	#[must_use]
100	pub const fn code(self) -> u16 {
101		(self.0 & 0xffff) as u16
102	}
103
104	/// [`HRESULT_FACILITY`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_facility)
105	/// macro.
106	#[must_use]
107	pub const fn facility(self) -> co::FACILITY {
108		unsafe { co::FACILITY::from_raw((self.0 >> 16) & 0x1fff) }
109	}
110
111	/// [`HRESULT_SEVERITY`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_severity)
112	/// macro.
113	#[must_use]
114	pub const fn severity(self) -> co::SEVERITY {
115		unsafe { co::SEVERITY::from_raw(((self.0 >> 31) & 0x1) as _) }
116	}
117
118	/// [`FAILED`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-failed)
119	/// macro.
120	#[must_use]
121	pub const fn FAILED(self) -> bool {
122		(self.0 as i32) < 0
123	}
124
125	/// [`SUCCEEDED`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-succeeded)
126	/// macro.
127	#[must_use]
128	pub const fn SUCCEEDED(self) -> bool {
129		(self.0 as i32) >= 0
130	}
