4const_basic_decl! { HRESULT: u32;
5	/// A native
6	/// [COM error code](https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-erref/0642cb2f-2075-4469-918c-4441e69c548a)
7	/// (`u32`), which can be considered a superset of
8	/// [`ERROR`](crate::co::ERROR).
9	///
10	/// The [`Result` alias](crate#errors-and-result-aliases) for this type is
11	/// [`HrResult`](crate::HrResult).
12	///
13	/// Implements the standard [`Error`](std::error::Error) trait.
14	///
15	/// Implements the [`Debug`](std::fmt::Debug) and
16	/// [`Display`](std::fmt::Display) traits to show the error code along with
17	/// the error description, taken from
18	/// [`SystemError`](crate::prelude::SystemError) trait. For example, the
19	/// code below:
20	///
21	/// ```no_run
22	/// use winsafe::{self as w, prelude::*, co};
23	///
24	/// println!("{}", co::HRESULT::E_INVALIDARG);
25	/// ```
26	///
27	/// Will print:
28	///
29	/// ```text
30	/// [0x80070057 2147942487] The parameter is incorrect.
31	/// ```
32	///
33	/// The numbers inside the brackets are the system error code, in
34	/// hexadecimal and decimal formats for convenience.
35	///
36	/// An [`ERROR`](crate::co::ERROR) can be seamlessly converted into an
37	/// `HRESULT`. This operation is equivalent to the
38	/// [`HRESULT_FROM_WIN32`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_from_win32)
39	/// macro:
40	///
41	/// ```no_run
42	/// use winsafe::{self as w, prelude::*, co};
43	///
44	/// let err = co::ERROR::FILE_NOT_FOUND;
45	///
46	/// let hr = err.to_hresult();
47	/// ```
48}

50impl std::error::Error for HRESULT {
51	fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
52		None
53	}
54}

56impl std::fmt::Display for HRESULT {
57	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
58		if self.0 > 0xffff {
59			write!(f, "[{:#010x}] {}", self.0, self.FormatMessage().trim_end())
60		} else {
61			write!(f, "[{:#06x}] {}", self.0, self.FormatMessage().trim_end())
62		}
63	}
64}
65impl std::fmt::Debug for HRESULT {
66	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
67		if self.0 > 0xffff {
68			write!(f, "HRESULT({:#010x} {}) {}", self.0, self.0, self.FormatMessage().trim_end())
69		} else {
70			write!(f, "HRESULT({:#06x} {}) {}", self.0, self.0, self.FormatMessage().trim_end())
71		}
72	}
73}

77impl co::ERROR {
78	/// [`HRESULT_FROM_WIN32`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_from_win32)
79	/// macro.
80	pub const fn to_hresult(self) -> co::HRESULT {
81		unsafe {
82			if self.raw() as i32 <= 0 {
83				co::HRESULT::from_raw(self.raw())
84			} else {
85				co::HRESULT::from_raw(
86					(self.raw() & 0x0000_ffff) | co::FACILITY::WIN32.raw() << 16 | 0x8000_0000,
87				)
88			}
89		}
90	}
91}

93impl HRESULT {
94	/// [`HRESULT_CODE`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_code)
95	/// macro.
96	pub const fn code(self) -> u16 {
97		(self.0 & 0xffff) as u16
98	}

100	/// [`HRESULT_FACILITY`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_facility)
101	/// macro.
102	pub const fn facility(self) -> co::FACILITY {
103		unsafe { co::FACILITY::from_raw((self.0 >> 16) & 0x1fff) }
104	}
105
106	/// [`HRESULT_SEVERITY`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_severity)
107	/// macro.
108	pub const fn severity(self) -> co::SEVERITY {
109		unsafe { co::SEVERITY::from_raw(((self.0 >> 31) & 0x1) as _) }
110	}
