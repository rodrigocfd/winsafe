7const_basic_decl! { HRESULT: u32;
8	/// A native
9	/// [COM error code](https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-erref/0642cb2f-2075-4469-918c-4441e69c548a)
10	/// (`u32`), which can be considered a superset of
11	/// [`ERROR`](crate::co::ERROR).
12	///
13	/// The [`Result` alias](crate#errors-and-result-aliases) for this type is
14	/// [`HrResult`](crate::HrResult).
15	///
16	/// Implements the standard [`Error`](std::error::Error) trait.
17	///
18	/// Implements the [`Debug`](std::fmt::Debug) and
19	/// [`Display`](std::fmt::Display) traits to show the error code along with
20	/// the error description, taken from
21	/// [`SystemError`](crate::prelude::SystemError) trait. For example, the
22	/// code below:
23	///
24	/// ```no_run
25	/// use winsafe::{self as w, prelude::*, co};
26	///
27	/// println!("{}", co::HRESULT::E_INVALIDARG);
28	/// ```
29	///
30	/// Will print:
31	///
32	/// ```text
33	/// [0x80070057 2147942487] The parameter is incorrect.
34	/// ```
35	///
36	/// The numbers inside the brackets are the system error code, in
37	/// hexadecimal and decimal formats for convenience.
38	///
39	/// An [`ERROR`](crate::co::ERROR) can be seamlessly converted into an
40	/// `HRESULT`. This operation is equivalent to the
41	/// [`HRESULT_FROM_WIN32`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_from_win32)
42	/// macro:
43	///
44	/// ```no_run
45	/// use winsafe::{self as w, prelude::*, co};
46	///
47	/// let err = co::ERROR::FILE_NOT_FOUND;
48	///
49	/// let hr = err.to_hresult();
50	/// ```
51}
52
53impl std::error::Error for HRESULT {
54	fn source(&self) -> Option<&(dyn std::error::Error + 'static)> {
55		None
56	}
57}

59impl std::fmt::Display for HRESULT {
60	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
61		if self.0 > 0xffff {
62			write!(f, "[{:#010x}] {}", self.0, self.FormatMessage().trim_end())
63		} else {
64			write!(f, "[{:#06x}] {}", self.0, self.FormatMessage().trim_end())
65		}
66	}
67}
68impl std::fmt::Debug for HRESULT {
69	fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
70		if self.0 > 0xffff {
71			write!(f, "HRESULT({:#010x} {}) {}", self.0, self.0, self.FormatMessage().trim_end())
72		} else {
73			write!(f, "HRESULT({:#06x} {}) {}", self.0, self.0, self.FormatMessage().trim_end())
74		}
75	}
76}

80impl co::ERROR {
81	/// [`HRESULT_FROM_WIN32`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_from_win32)
82	/// macro.
83	#[must_use]
84	pub const fn to_hresult(self) -> co::HRESULT {
85		unsafe {
86			if self.raw() as i32 <= 0 {
87				co::HRESULT::from_raw(self.raw())
88			} else {
89				co::HRESULT::from_raw(
90					(self.raw() & 0x0000_ffff) | co::FACILITY::WIN32.raw() << 16 | 0x8000_0000,
91				)
92			}
93		}
94	}
95}

97impl HRESULT {
98	/// [`HRESULT_CODE`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_code)
99	/// macro.
100	#[must_use]
101	pub const fn code(self) -> u16 {
102		(self.0 & 0xffff) as u16
103	}
104
105	/// [`HRESULT_FACILITY`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_facility)
106	/// macro.
107	#[must_use]
108	pub const fn facility(self) -> co::FACILITY {
109		unsafe { co::FACILITY::from_raw((self.0 >> 16) & 0x1fff) }
110	}
111
112	/// [`HRESULT_SEVERITY`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-hresult_severity)
113	/// macro.
114	#[must_use]
115	pub const fn severity(self) -> co::SEVERITY {
116		unsafe { co::SEVERITY::from_raw(((self.0 >> 31) & 0x1) as _) }
117	}
118
119	/// [`FAILED`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-failed)
120	/// macro.
121	#[must_use]
122	pub const fn FAILED(self) -> bool {
123		(self.0 as i32) < 0
124	}
125
126	/// [`SUCCEEDED`](https://learn.microsoft.com/en-us/windows/win32/api/winerror/nf-winerror-succeeded)
127	/// macro.
128	#[must_use]
129	pub const fn SUCCEEDED(self) -> bool {
130		(self.0 as i32) >= 0
131	}
