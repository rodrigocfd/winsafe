123/// Variant parameter for:
124///
125/// * [`HWND::TaskDialog`](crate::HWND::TaskDialog)
126#[derive(Clone)]
127pub enum IconRes<'a> {
128	/// No icon.
129	None,
130	/// Handle to the instance to load the icon from, and the icon resource
131	/// identifier.
132	Res(&'a HINSTANCE, IdStr),
133	/// The [`co::TD_ICON::WARNING`](crate::co::TD_ICON::WARNING) constant.
134	Warn,
135	/// The [`co::TD_ICON::ERROR`](crate::co::TD_ICON::ERROR) constant.
136	Error,
137	/// The [`co::TD_ICON::INFORMATION`](crate::co::TD_ICON::INFORMATION)
138	/// constant.
139	Info,
140	/// The [`co::TD_ICON::SHIELD`](crate::co::TD_ICON::SHIELD) constant.
141	Shield,
142}
143
144impl<'a> IconRes<'a> {
145	/// Returns the `HINSTANCE`, if any, and a pointer to the raw data content.
146	#[must_use]
147	pub fn as_ptr(&self) -> (HINSTANCE, *const u16) {
148		use IconRes::*;
149		match self {
150			None => (HINSTANCE::NULL, std::ptr::null()),
151			Res(hinst, id_str) => (unsafe { hinst.raw_copy() }, id_str.as_ptr()),
152			Warn => (HINSTANCE::NULL, MAKEINTRESOURCE(co::TD_ICON::WARNING.raw() as _)),
153			Error => (HINSTANCE::NULL, MAKEINTRESOURCE(co::TD_ICON::ERROR.raw() as _)),
154			Info => (HINSTANCE::NULL, MAKEINTRESOURCE(co::TD_ICON::INFORMATION.raw() as _)),
155			Shield => (HINSTANCE::NULL, MAKEINTRESOURCE(co::TD_ICON::SHIELD.raw() as _)),
156		}
157	}
158}

225/// Variant parameter for:
226///
227/// * [`TASKDIALOGCONFIG`](crate::TASKDIALOGCONFIG)
228pub enum Tdn {
229	ButtonClicked(u16),
230	Created,
231	Destroyed,
232	DialogConstructed,
233	ExpandoButtonClicked(bool),
234	Help,
235	HyperlinkClicked(String),
236	Navigated,
237	RadioButtonClicked(u16),
238	Timer(u32),
239	VerificationClicked(bool),
240}
241
242impl Tdn {
243	/// Constructs the enum from its message data.
244	///
245	/// # Panics
246	///
247	/// Panics if `tdn` value is invalid.
248	#[must_use]
249	pub unsafe fn from_msg(tdn: co::TDN, wp: usize, lp: isize) -> Self {
250		match tdn {
251			co::TDN::BUTTON_CLICKED => Self::ButtonClicked(wp as _),
252			co::TDN::CREATED => Self::Created,
253			co::TDN::DESTROYED => Self::Destroyed,
254			co::TDN::DIALOG_CONSTRUCTED => Self::DialogConstructed,
255			co::TDN::EXPANDO_BUTTON_CLICKED => Self::ExpandoButtonClicked(wp != 0),
256			co::TDN::HELP => Self::Help,
257			co::TDN::HYPERLINK_CLICKED => {
258				Self::HyperlinkClicked(unsafe { WString::from_wchars_nullt(lp as _) }.to_string())
259			},
260			co::TDN::NAVIGATED => Self::Navigated,
261			co::TDN::RADIO_BUTTON_CLICKED => Self::RadioButtonClicked(wp as _),
262			co::TDN::TIMER => Self::Timer(wp as _),
263			co::TDN::VERIFICATION_CLICKED => Self::VerificationClicked(wp != 0),
264			_ => panic!("Invalid TDN value."),
265		}
266	}
267}
