125pub enum IconRes<'a> {
126	/// No icon.
127	None,
128	/// Handle to the instance to load the icon from, and the icon resource
129	/// identifier.
130	Res(&'a HINSTANCE, IdStr),
131	/// The [`co::TD_ICON::WARNING`](crate::co::TD_ICON::WARNING) constant.
132	Warn,
133	/// The [`co::TD_ICON::ERROR`](crate::co::TD_ICON::ERROR) constant.
134	Error,
135	/// The [`co::TD_ICON::INFORMATION`](crate::co::TD_ICON::INFORMATION)
136	/// constant.
137	Info,
138	/// The [`co::TD_ICON::SHIELD`](crate::co::TD_ICON::SHIELD) constant.
139	Shield,
140}
141
142impl<'a> IconRes<'a> {
143	/// Returns the `HINSTANCE`, if any, and a pointer to the raw data content.
144	#[must_use]
145	pub fn as_ptr(&self) -> (HINSTANCE, *const u16) {
146		match self {
147			Self::None => (HINSTANCE::NULL, std::ptr::null()),
148			Self::Res(hinst, id_str) => (unsafe { hinst.raw_copy() }, id_str.as_ptr()),
149			Self::Warn => (HINSTANCE::NULL, MAKEINTRESOURCE(co::TD_ICON::WARNING.raw() as _)),
150			Self::Error => (HINSTANCE::NULL, MAKEINTRESOURCE(co::TD_ICON::ERROR.raw() as _)),
151			Self::Info => (HINSTANCE::NULL, MAKEINTRESOURCE(co::TD_ICON::INFORMATION.raw() as _)),
152			Self::Shield => (HINSTANCE::NULL, MAKEINTRESOURCE(co::TD_ICON::SHIELD.raw() as _)),
153		}
154	}
155}

221/// Variant parameter for:
222///
223/// * [`TASKDIALOGCONFIG`](crate::TASKDIALOGCONFIG)
224pub enum Tdn {
225	ButtonClicked(u16),
226	Created,
227	Destroyed,
228	DialogConstructed,
229	ExpandoButtonClicked(bool),
230	Help,
231	HyperlinkClicked(String),
232	Navigated,
233	RadioButtonClicked(u16),
234	Timer(u32),
235	VerificationClicked(bool),
236}
237
238impl Tdn {
239	/// Constructs the enum from its message data.
240	///
241	/// # Panics
242	///
243	/// Panics if `tdn` value is invalid.
244	#[must_use]
245	pub unsafe fn from_msg(tdn: co::TDN, wp: usize, lp: isize) -> Self {
246		match tdn {
247			co::TDN::BUTTON_CLICKED => Self::ButtonClicked(wp as _),
248			co::TDN::CREATED => Self::Created,
249			co::TDN::DESTROYED => Self::Destroyed,
250			co::TDN::DIALOG_CONSTRUCTED => Self::DialogConstructed,
251			co::TDN::EXPANDO_BUTTON_CLICKED => Self::ExpandoButtonClicked(wp != 0),
252			co::TDN::HELP => Self::Help,
253			co::TDN::HYPERLINK_CLICKED => {
254				Self::HyperlinkClicked(unsafe { WString::from_wchars_nullt(lp as _) }.to_string())
255			},
256			co::TDN::NAVIGATED => Self::Navigated,
257			co::TDN::RADIO_BUTTON_CLICKED => Self::RadioButtonClicked(wp as _),
258			co::TDN::TIMER => Self::Timer(wp as _),
259			co::TDN::VERIFICATION_CLICKED => Self::VerificationClicked(wp != 0),
260			_ => panic!("Invalid TDN value."),
261		}
262	}
263}
