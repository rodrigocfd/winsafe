9/// [`TASKDIALOGCONFIG`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/ns-commctrl-taskdialog_button)
10/// struct.
11///
12/// Used with [`TaskDialogIndirect`](crate::TaskDialogIndirect) function.
13///
14/// Not all `flags` constants are available, some of them are automatically set
15/// as you fill other parameters.
16#[derive(Default)]
17pub struct TASKDIALOGCONFIG<'a, 'b, 'c, 'd, 'e, 'f, 'g, 'h, 'i, 'j, 'k, 'l, 'm, 'n, 'o, 'p> {
18	pub hwnd_parent: Option<&'a HWND>,
19	pub hinstance: Option<&'b HINSTANCE>,
20	pub flags: co::TDF,
21	/// Predefined buttons. Will be placed after the custom `buttons`.
22	pub common_buttons: co::TDCBF,
23	/// Window caption. If not specified, the .exe name is used.
24	pub window_title: Option<&'c str>,
25	pub main_icon: IconIdTd<'d>,
26	/// Text shown before the main content, in a larger font.
27	pub main_instruction: Option<&'e str>,
28	/// The main text of the dialog.
29	pub content: Option<&'f str>,
30	/// Command ID and button text. Will be placed before the predefined
31	/// `common_buttons`.
32	pub buttons: &'g [(u16, &'h str)],
33	/// Any ID from `common_buttons` or `buttons`.
34	pub default_button_id: u16,
35	/// Command ID and radio button text.
36	pub radio_buttons: &'i [(u16, &'j str)],
37	/// Any ID from `radio_buttons`.
38	pub default_radio_button_id: u16,
39	/// Text of the label of the verification check box.
40	pub verification_text: Option<&'k str>,
41	/// Text of the collapsible section.
42	pub more_info: Option<&'l str>,
43	/// Text of the button that expands/collapses `more_info`, when the section
44	/// is expanded.
45	pub more_info_btn_expanded: Option<&'m str>,
46	/// Text of the button that expands/collapses `more_info`, when the section
47	/// is collapsed.
48	pub more_info_btn_collapsed: Option<&'n str>,
49	pub footer_icon: IconId<'o>,
50	pub footer_text: Option<&'p str>,
51	pub callback: Option<Box<dyn Fn(&HWND, Tdn) -> co::HRESULT>>,
52	pub width: u32,
53}

58	pub(in crate::comctl) fn to_raw(&self) -> TASKDIALOGCONFIG_buf {
59		let mut raw = TASKDIALOGCONFIG_raw::default();
60		raw.hwndParent = unsafe { self.hwnd_parent.unwrap_or(&HWND::NULL).raw_copy() };
61		raw.hInstance = unsafe { self.hinstance.unwrap_or(&HINSTANCE::NULL).raw_copy() };
62		raw.dwFlags = self.flags;
63		raw.dwCommonButtons = self.common_buttons;
64
65		let w_title = self
66			.window_title // must force heap because variable will be moved
67			.map_or(WString::new(), |s| WString::from_str_force_heap(s));
68		raw.pszWindowTitle = w_title.as_ptr();
69
70		match &self.main_icon {
71			IconIdTd::None => {
72				let new_flags = raw.dwFlags | TDF_USE_HICON_MAIN;
73				raw.dwFlags = new_flags;
74			},
75			IconIdTd::Icon(h) => {
76				raw.hMainIcon = h.ptr();
77				let new_flags = raw.dwFlags | TDF_USE_HICON_MAIN;
78				raw.dwFlags = new_flags;
79			},
80			IconIdTd::Id(id) => {
81				raw.hMainIcon = MAKEINTRESOURCE(*id as _) as _;
82			},
83			IconIdTd::Td(td) => {
84				raw.hMainIcon = td.raw() as _;
85			},
86		}
87
88		let w_instruc = self
89			.main_instruction
90			.map_or(WString::new(), |s| WString::from_str_force_heap(s));
91		raw.pszMainInstruction = w_instruc.as_ptr();
92
93		let w_content = self
94			.content
95			.map_or(WString::new(), |s| WString::from_str_force_heap(s));
96		raw.pszContent = w_content.as_ptr();
97
98		let btns_buf: (Vec<_>, Vec<_>) = self
99			.buttons
100			.iter()
101			.map(|(id, txt)| {
102				let txt_buf = WString::from_str_force_heap(*txt);
103				let btn_buf = TASKDIALOG_BUTTON {
104					nButtonID: *id as _,
105					pszButtonText: txt_buf.as_ptr(),
106				};
107				(txt_buf, btn_buf)
108			})
109			.unzip();
110		raw.cButtons = btns_buf.1.len() as _;
111		raw.pButtons = btns_buf.1.as_ptr() as _;
112		raw.nDefaultButton = self.default_button_id as _;
113
114		let radios_buf: (Vec<_>, Vec<_>) = self
115			.radio_buttons
116			.iter()
117			.map(|(id, txt)| {
118				let txt_buf = WString::from_str_force_heap(*txt);
119				let btn_buf = TASKDIALOG_BUTTON {
120					nButtonID: *id as _,
121					pszButtonText: txt_buf.as_ptr(),
122				};
123				(txt_buf, btn_buf)
124			})
125			.unzip();
126		raw.cRadioButtons = radios_buf.1.len() as _;
127		raw.pRadioButtons = radios_buf.1.as_ptr() as _;
128		raw.nDefaultRadioButton = self.default_radio_button_id as _;
129
130		let w_verif = self
131			.verification_text
132			.map_or(WString::new(), |s| WString::from_str_force_heap(s));
133		raw.pszVerificationText = w_verif.as_ptr();
134
135		let w_more_info = self
136			.more_info
137			.map_or(WString::new(), |s| WString::from_str_force_heap(s));
138		raw.pszExpandedInformation = w_more_info.as_ptr();
139
140		let w_expanded_info = self
141			.more_info_btn_expanded
142			.map_or(WString::new(), |s| WString::from_str_force_heap(s));
143		raw.pszExpandedControlText = w_expanded_info.as_ptr();
144
145		let w_collapsed_info = self
146			.more_info_btn_collapsed
147			.map_or(WString::new(), |s| WString::from_str_force_heap(s));
148		raw.pszCollapsedControlText = w_collapsed_info.as_ptr();
149
150		match &self.footer_icon {
151			IconId::None => {
152				let new_flags = raw.dwFlags | TDF_USE_HICON_FOOTER;
153				raw.dwFlags = new_flags;
154			},
155			IconId::Icon(h) => {
156				raw.hFooterIcon = h.ptr();
157				let new_flags = raw.dwFlags | TDF_USE_HICON_FOOTER;
158				raw.dwFlags = new_flags;
159			},
160			IconId::Id(id) => {
161				raw.hFooterIcon = MAKEINTRESOURCE(*id as _) as _;
162			},
163		}
164
165		let w_footer = self
166			.footer_text
167			.map_or(WString::new(), |s| WString::from_str_force_heap(s));
168		raw.pszFooter = w_footer.as_ptr();
169
170		raw.pfCallback = Some(callbacks::func_task_dialog_callback);
171		raw.lpCallbackData = self as *const _ as _; // object will exist until TaskDialogIndirect() returns
172		raw.cxWidth = self.width;
173
174		TASKDIALOGCONFIG_buf {
175			raw,
176			w_title,
177			w_instruc,
178			w_content,
179			btns_buf,
180			radios_buf,
181			w_verif,
182			w_more_info,
183			w_expanded_info,
184			w_collapsed_info,
185			w_footer,
186		}
187	}
188}
