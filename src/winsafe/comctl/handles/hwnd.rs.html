11	/// [`DefSubclassProc`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-defsubclassproc)
12	/// function.
13	///
14	/// The return type is variable, being defined by the `RetType` associated
15	/// type of the [`MsgSend`](crate::prelude::MsgSend) trait. That means each
16	/// message can define its own return type.
17	///
18	/// # Safety
19	///
20	/// Messages manipulate pointers, copies and window states. Improper use may
21	/// lead to undefined behavior.
22	pub unsafe fn DefSubclassProc<M>(&self, msg: M) -> M::RetType
23	where
24		M: MsgSend,
25	{
26		let mut msg = msg;
27		let wm_any = msg.as_generic_wm();
28		unsafe {
29			msg.isize_to_ret(ffi::DefSubclassProc(
30				self.ptr(),
31				wm_any.msg_id.raw(),
32				wm_any.wparam,
33				wm_any.lparam,
34			))
35		}
36	}

73	/// [`TaskDialog`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-taskdialog)
74	/// function.
75	///
76	/// If you need more customization, see the
77	/// [`TaskDialogIndirect`](crate::TaskDialogIndirect) function.
78	///
79	/// # Examples
80	///
81	/// An information message with just an OK button:
82	///
83	/// ```no_run
84	/// use winsafe::{self as w, prelude::*, co};
85	///
86	/// let hwnd: w::HWND; // initialized somewhere
87	/// # let hwnd = w::HWND::NULL;
88	///
89	/// hwnd.TaskDialog(
90	///     Some("Operation successful"),
91	///     None,
92	///     Some("The operation completed successfully."),
93	///     co::TDCBF::OK,
94	///     w::IconRes::Info,
95	/// )?;
96	/// # w::HrResult::Ok(())
97	/// ```
98	///
99	/// Prompt the user to click OK or Cancel upon a question:
100	///
101	/// ```no_run
102	/// use winsafe::{self as w, prelude::*, co};
103	///
104	/// let hwnd: w::HWND; // initialized somewhere
105	/// # let hwnd = w::HWND::NULL;
106	///
107	/// let answer = hwnd.TaskDialog(
108	///     Some("My app name"),
109	///     Some("File modified"),
110	///     Some("The file has been modified.\nProceed closing the application?"),
111	///     co::TDCBF::OK | co::TDCBF::CANCEL,
112	///     w::IconRes::Warn,
113	/// )?;
114	///
115	/// if answer == co::DLGID::OK {
116	///     println!("User clicked OK.");
117	/// }
118	/// # w::HrResult::Ok(())
119	/// ```
120	pub fn TaskDialog(
121		&self,
122		window_title: Option<&str>,
123		main_instruction: Option<&str>,
124		content: Option<&str>,
125		common_buttons: co::TDCBF,
126		icon: IconRes,
127	) -> HrResult<co::DLGID> {
128		// https://weblogs.asp.net/kennykerr/Windows-Vista-for-Developers-_1320_-Part-2-_1320_-Task-Dialogs-in-Depth
129		let mut pn_button = i32::default();
130		let (hinst, raw_ico) = icon.as_ptr();
131
132		ok_to_hrresult(unsafe {
133			ffi::TaskDialog(
134				self.ptr(),
135				hinst.ptr(),
136				WString::from_opt_str(window_title).as_ptr(),
137				WString::from_opt_str(main_instruction).as_ptr(),
138				WString::from_opt_str(content).as_ptr(),
139				common_buttons.raw(),
140				raw_ico,
141				&mut pn_button,
142			)
143		})
144		.map(|_| unsafe { co::DLGID::from_raw(pn_button as _) })
145	}
