17	/// Returns an iterator over all icons in the image list, by calling
18	/// [`HIMAGELIST::ExtractIcon`](crate::HIMAGELIST::ExtractIcon) for each
19	/// one.
20	///
21	/// # Examples
22	///
23	/// Collecting the icons into a [`Vec`](std::vec::Vec):
24	///
25	/// ```no_run
26	/// use winsafe::{self as w, prelude::*};
27	///
28	/// let himgl: w::HIMAGELIST; // initialized somewhere
29	/// # let himgl = w::HIMAGELIST::NULL;
30	///
31	/// let icons = himgl.iter()
32	///     .collect::<w::HrResult<Vec<_>>>()?;
33	/// # w::HrResult::Ok(())
34	/// ```
35	#[must_use]
36	pub fn iter(&self) -> impl DoubleEndedIterator<Item = HrResult<DestroyIconGuard>> + '_ {
37		HimagelistIter::new(self)
38	}

40	/// [`ImageList_Add`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_add)
41	/// function.
42	///
43	/// A copy of the bitmap is made and stored in the image list, so you're
44	/// free to release the original bitmap.
45	pub fn Add(&self, hbmp_image: &HBITMAP, hbmp_mask: Option<&HBITMAP>) -> HrResult<u32> {
46		match unsafe {
47			ffi::ImageList_Add(
48				self.ptr(),
49				hbmp_image.ptr(),
50				hbmp_mask.map_or(std::ptr::null_mut(), |h| h.ptr()),
51			)
52		} {
53			-1 => Err(co::HRESULT::E_FAIL),
54			idx => Ok(idx as _),
55		}
56	}

82	/// [`ImageList_BeginDrag`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_begindrag)
83	/// function.
84	///
85	/// In the original C implementation, you must call
86	/// [`ImageList_EndDrag`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_enddrag)
87	/// as a cleanup operation.
88	///
89	/// Here, the cleanup is performed automatically, because `BeginDrag`
90	/// returns an
91	/// [`ImageListEndDragGuard`](crate::guard::ImageListEndDragGuard), which
92	/// automatically calls `ImageList_EndDrag` when the guard goes out of
93	/// scope. You must, however, keep the guard alive, otherwise the cleanup
94	/// will be performed right away.
95	///
96	/// # Examples
97	///
98	/// ```no_run
99	/// use winsafe::{self as w, prelude::*};
100	///
101	/// let himgl: w::HIMAGELIST; // initialized somewhere
102	/// # let himgl = w::HIMAGELIST::NULL;
103	///
104	/// let _drag = himgl.BeginDrag(0, w::POINT::new())?; // keep guard alive
105	/// # w::HrResult::Ok(())
106	/// ```
107	pub fn BeginDrag(&self, itrack: u32, hotspot: POINT) -> HrResult<ImageListEndDragGuard<'_>> {
108		unsafe {
109			match ffi::ImageList_BeginDrag(self.ptr(), itrack as _, hotspot.x, hotspot.y) {
110				0 => Err(co::HRESULT::E_FAIL),
111				_ => Ok(ImageListEndDragGuard::new()),
112			}
113		}
114	}

116	/// [`ImageList_Create`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_create)
117	/// function.
118	///
119	/// # Examples
120	///
121	/// ```no_run
122	/// use winsafe::{self as w, prelude::*, co};
123	///
124	/// let himgl = w::HIMAGELIST::Create(
125	///     w::SIZE::with(16, 16),
126	///     co::ILC::COLOR32,
127	///     1,
128	///     1,
129	/// )?;
130	///
131	/// // ImageList_Destroy() automatically called
132	/// # w::HrResult::Ok(())
133	/// ```
134	#[must_use]
135	pub fn Create(
136		image_sz: SIZE,
137		flags: co::ILC,
138		initial_size: i32,
139		grow_size: i32,
140	) -> HrResult<ImageListDestroyGuard> {
141		unsafe {
142			match PtrRet(ffi::ImageList_Create(
143				image_sz.cx,
144				image_sz.cy,
145				flags.raw(),
146				initial_size,
147				grow_size,
148			))
149			.to_opt_handle()
150			{
151				None => Err(co::HRESULT::E_FAIL),
152				Some(h) => Ok(ImageListDestroyGuard::new(h)),
153			}
154		}
155	}
