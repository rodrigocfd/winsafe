17/// This trait is enabled with the `comctl` feature, and provides methods for
18/// [`HIMAGELIST`](crate::HIMAGELIST).
19///
20/// Prefer importing this trait through the prelude:
21///
22/// ```no_run
23/// use winsafe::prelude::*;
24/// ```
25pub trait comctl_Himagelist: Handle {
26	/// Returns an iterator over all icons in the image list, by calling
27	/// [`HIMAGELIST::ExtractIcon`](crate::prelude::comctl_Himagelist::ExtractIcon)
28	/// for each one.
29	///
30	/// # Examples
31	///
32	/// Collecting the icons into a [`Vec`](std::vec::Vec):
33	///
34	/// ```no_run
35	/// use winsafe::{self as w, prelude::*};
36	///
37	/// let himgl: w::HIMAGELIST; // initialized somewhere
38	/// # let himgl = w::HIMAGELIST::NULL;
39	///
40	/// let icons = himgl.iter()
41	///     .collect::<w::HrResult<Vec<_>>>()?;
42	/// # w::HrResult::Ok(())
43	/// ```
44	#[must_use]
45	fn iter(&self) -> impl Iterator<Item = HrResult<DestroyIconGuard>> + '_ {
46		HimagelistIter::new(self)
47	}

49	/// [`ImageList_Add`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_add)
50	/// function.
51	///
52	/// A copy of the bitmap is made and stored in the image list, so you're
53	/// free to release the original bitmap.
54	fn Add(&self, hbmp_image: &HBITMAP, hbmp_mask: Option<&HBITMAP>) -> HrResult<u32> {
55		match unsafe {
56			ffi::ImageList_Add(
57				self.ptr(),
58				hbmp_image.ptr(),
59				hbmp_mask.map_or(std::ptr::null_mut(), |h| h.ptr()),
60			)
61		} {
62			-1 => Err(co::HRESULT::E_FAIL),
63			idx => Ok(idx as _),
64		}
65	}

91	/// [`ImageList_BeginDrag`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_begindrag)
92	/// function.
93	///
94	/// In the original C implementation, you must call
95	/// [`ImageList_EndDrag`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_enddrag)
96	/// as a cleanup operation.
97	///
98	/// Here, the cleanup is performed automatically, because `BeginDrag`
99	/// returns an
100	/// [`ImageListEndDragGuard`](crate::guard::ImageListEndDragGuard), which
101	/// automatically calls `ImageList_EndDrag` when the guard goes out of
102	/// scope. You must, however, keep the guard alive, otherwise the cleanup
103	/// will be performed right away.
104	///
105	/// # Examples
106	///
107	/// ```no_run
108	/// use winsafe::{self as w, prelude::*};
109	///
110	/// let himgl: w::HIMAGELIST; // initialized somewhere
111	/// # let himgl = w::HIMAGELIST::NULL;
112	///
113	/// let _drag = himgl.BeginDrag(0, w::POINT::new(0, 0))?; // keep guard alive
114	/// # w::HrResult::Ok(())
115	/// ```
116	fn BeginDrag(&self, itrack: u32, hotspot: POINT) -> HrResult<ImageListEndDragGuard<'_>> {
117		unsafe {
118			match ffi::ImageList_BeginDrag(self.ptr(), itrack as _, hotspot.x, hotspot.y) {
119				0 => Err(co::HRESULT::E_FAIL),
120				_ => Ok(ImageListEndDragGuard::new()),
121			}
122		}
123	}

125	/// [`ImageList_Create`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_create)
126	/// function.
127	///
128	/// # Examples
129	///
130	/// ```no_run
131	/// use winsafe::{self as w, prelude::*, co};
132	///
133	/// let himgl = w::HIMAGELIST::Create(
134	///     w::SIZE::new(16, 16),
135	///     co::ILC::COLOR32,

137	///     1,

140	/// // ImageList_Destroy() automatically called
141	/// # w::HrResult::Ok(())

144	fn Create(
145		image_sz: SIZE,
146		flags: co::ILC,
147		initial_size: i32,
148		grow_size: i32,
149	) -> HrResult<ImageListDestroyGuard> {
150		unsafe {
151			match ptr_to_option_handle(ffi::ImageList_Create(
152				image_sz.cx,
153				image_sz.cy,
154				flags.raw(),
155				initial_size,
156				grow_size,
157			)) {
158				None => Err(co::HRESULT::E_FAIL),
159				Some(h) => Ok(ImageListDestroyGuard::new(h)),
160			}
161		}
162	}
