17/// This trait is enabled with the `comctl` feature, and provides methods for
18/// [`HIMAGELIST`](crate::HIMAGELIST).
19///
20/// Prefer importing this trait through the prelude:
21///
22/// ```no_run
23/// use winsafe::prelude::*;
24/// ```
25pub trait comctl_Himagelist: Handle {
26	/// Returns an iterator over all icons in the image list, by calling
27	/// [`HIMAGELIST::ExtractIcon`](crate::prelude::comctl_Himagelist::ExtractIcon)
28	/// for each one.
29	///
30	/// # Examples
31	///
32	/// Collecting the icons into a [`Vec`](std::vec::Vec):
33	///
34	/// ```no_run
35	/// use winsafe::{self as w, prelude::*};
36	///
37	/// let himgl: w::HIMAGELIST; // initialized somewhere
38	/// # let himgl = w::HIMAGELIST::NULL;
39	///
40	/// let icons = himgl.iter()
41	///     .collect::<w::HrResult<Vec<_>>>()?;
42	/// # w::HrResult::Ok(())
43	/// ```
44	#[must_use]
45	fn iter(&self) -> impl Iterator<Item = HrResult<DestroyIconGuard>> + '_ {
46		HimagelistIter::new(self)
47	}

49	/// [`ImageList_Add`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_add)
50	/// function.
51	///
52	/// A copy of the bitmap is made and stored in the image list, so you're
53	/// free to release the original bitmap.
54	fn Add(&self,
55		hbmp_image: &HBITMAP,
56		hbmp_mask: Option<&HBITMAP>,
57	) -> HrResult<u32>
58	{
59		match unsafe {
60			ffi::ImageList_Add(
61				self.ptr(),
62				hbmp_image.ptr(),
63				hbmp_mask.map_or(std::ptr::null_mut(), |h| h.ptr()),
64			)
65		} {
66			-1 => Err(co::HRESULT::E_FAIL),
67			idx => Ok(idx as _),
68		}
69	}

103	/// [`ImageList_BeginDrag`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_begindrag)
104	/// function.
105	///
106	/// In the original C implementation, you must call
107	/// [`ImageList_EndDrag`](https://learn.microsoft.com/en-us/windows/win32/api/commctrl/nf-commctrl-imagelist_enddrag)
108	/// as a cleanup operation.
109	///
110	/// Here, the cleanup is performed automatically, because `BeginDrag`
111	/// returns an
112	/// [`ImageListEndDragGuard`](crate::guard::ImageListEndDragGuard), which
113	/// automatically calls `ImageList_EndDrag` when the guard goes out of
114	/// scope. You must, however, keep the guard alive, otherwise the cleanup
115	/// will be performed right away.
116	///
117	/// # Examples
118	///
119	/// ```no_run
120	/// use winsafe::{self as w, prelude::*};
121	///
122	/// let himgl: w::HIMAGELIST; // initialized somewhere
123	/// # let himgl = w::HIMAGELIST::NULL;
124	///
125	/// let _drag = himgl.BeginDrag(0, w::POINT::new(0, 0))?; // keep guard alive
126	/// # w::HrResult::Ok(())
127	/// ```
128	fn BeginDrag(&self,
129		itrack: u32,
130		hotspot: POINT,
131	) -> HrResult<ImageListEndDragGuard<'_>>
132	{
133		unsafe {
134			match ffi::ImageList_BeginDrag(
135				self.ptr(),
136				itrack as _,
137				hotspot.x, hotspot.y,
138			) {
139				0 => Err(co::HRESULT::E_FAIL),
140				_ => Ok(ImageListEndDragGuard::new()),
141			}
142		}
143	}

184	}
