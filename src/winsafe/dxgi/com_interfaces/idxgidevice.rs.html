22/// This trait is enabled with the `dxgi` feature, and provides methods for
23/// [`IDXGIDevice`](crate::IDXGIDevice).

25/// Prefer importing this trait through the prelude:

28/// use winsafe::prelude::*;
29/// ```
30pub trait dxgi_IDXGIDevice: dxgi_IDXGIObject {
31	/// [`IDXGIDevice::CreateSurface`](https://learn.microsoft.com/en-us/windows/win32/api/dxgi/nf-dxgi-idxgidevice-createsurface)
32	/// method.
33	#[must_use]
34	fn CreateSurface(
35		&self,
36		desc: &DXGI_SURFACE_DESC,
37		num_surfaces: u32,
38		usage: co::DXGI_USAGE,
39		shared_resource: Option<&DXGI_SHARED_RESOURCE>,
40	) -> HrResult<IDXGISurface> {
41		let mut queried = unsafe { IDXGISurface::null() };
42		ok_to_hrresult(unsafe {
43			(vt::<IDXGIDeviceVT>(self).CreateSurface)(
44				self.ptr(),
45				pcvoid(desc),
46				num_surfaces,
47				usage.raw(),
48				pcvoid_or_null(shared_resource),
49				queried.as_mut(),
50			)
51		})
52		.map(|_| queried)
53	}
54
55	fn_com_interface_get! { GetAdapter: IDXGIDeviceVT => IDXGIAdapter;
56		/// [`IDXGIDevice::GetAdapter`](https://learn.microsoft.com/en-us/windows/win32/api/dxgi/nf-dxgi-idxgidevice-getadapter)
57		/// method.
58	}

60	/// [`IDXGIDevice::GetGPUThreadPriority`](https://learn.microsoft.com/en-us/windows/win32/api/dxgi/nf-dxgi-idxgidevice-getgputhreadpriority)
61	/// method.
62	#[must_use]
63	fn GetGPUThreadPriority(&self) -> HrResult<i8> {
64		let mut priority = 0i32;
65		ok_to_hrresult(unsafe {
66			(vt::<IDXGIDeviceVT>(self).GetGPUThreadPriority)(self.ptr(), &mut priority)
67		})
68		.map(|_| priority as _)
69	}

71	/// [`IDXGIDevice::QueryResourceResidency`](https://learn.microsoft.com/en-us/windows/win32/api/dxgi/nf-dxgi-idxgidevice-queryresourceresidency)
72	/// method.
73	#[must_use]
74	fn QueryResourceResidency(
75		&self,
76		resources: &[&impl dxgi_IDXGIResource],
77	) -> HrResult<Vec<co::DXGI_RESIDENCY>> {
78		let mut status = vec![co::DXGI_RESIDENCY::default(); resources.len()];
79		ok_to_hrresult(unsafe {
80			(vt::<IDXGIDeviceVT>(self).QueryResourceResidency)(
81				self.ptr(),
82				resources.as_ptr() as _,
83				status.as_mut_ptr() as _,
84				resources.len() as _,
85			)
86		})
87		.map(|_| status)
88	}
