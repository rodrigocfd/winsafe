fn iter(&self) -> HrResult<impl Iterator<Item = HrResult<IUnknown>> + '_>

Returns an iterator over the IUnknown elements by calling IMFCollection::GetElementCount and IMFCollection::GetElement consecutively.

ยงExamples

Iterating over the IUnknown objects:

use winsafe::{self as w, prelude::*, co};

let collection: w::IMFCollection; // initialized somewhere

for element in collection.iter()? {
    let element = element?;
    println!("{:?}", element.ptr());
}

Collecting the elements into a Vec:

use winsafe::{self as w, prelude::*, co};

let collection: w::IMFCollection; // initialized somewhere

let elements = collection.iter()?
    .collect::<w::HrResult<Vec<_>>>()?;
