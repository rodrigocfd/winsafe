Expand description

ITaskSettings COM interface.

Automatically calls Release when the object goes out of scope.

§Examples

use winsafe::{self as w, prelude::*};

let task: w::ITaskDefinition; // initialized somewhere

let settings = task
    .QueryInterface::<w::ITaskSettings>()?;
