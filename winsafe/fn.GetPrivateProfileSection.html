ยงExamples

Reading all key/value pairs of a section from an INI file:

use winsafe::{self as w, prelude::*};

let pairs = w::GetPrivateProfileSection(
    "MySection",
    "C:\\Temp\\foo.ini",
)?;

for (key, val) in pairs.iter() {
    println!("{} = {}", key, val);
}
