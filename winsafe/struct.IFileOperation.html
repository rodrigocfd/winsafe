Struct IFileOperation

pub struct IFileOperation(/* private fields */);
Available on crate feature shell only.
Expand description

IFileOperation COM interface.

Automatically calls IUnknown::Release when the object goes out of scope.

§Examples

Copying a file:

use winsafe::{self as w, prelude::*, co};

let _com_guard =
    w::CoInitializeEx(co::COINIT::APARTMENTTHREADED | co::COINIT::DISABLE_OLE1DDE)?;

let op = w::CoCreateInstance::<w::IFileOperation>(
    &co::CLSID::FileOperation,
    None::<&w::IUnknown>,
    co::CLSCTX::ALL,
)?;

let file = w::SHCreateItemFromParsingName::<w::IShellItem>(
    "C:\\Temp\\foo.txt",
    None::<&w::IBindCtx>,
)?;
let dest = w::SHCreateItemFromParsingName::<w::IShellItem>(
    "C:\\Temp",
    None::<&w::IBindCtx>,
)?;

op.CopyItem(&file, &dest, Some("new file.txt"), None)?;
op.PerformOperations()?;
