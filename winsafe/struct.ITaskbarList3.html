ITaskbarList3 COM interface over ITaskbarList3VT.

Automatically calls IUnknown::Release when the object goes out of scope.

Examples

use winsafe::{self as w, prelude::*, co};

let obj = w::CoCreateInstance::<w::ITaskbarList3>(
    &co::CLSID::TaskbarList,
    None,
    co::CLSCTX::INPROC_SERVER,
)?;
