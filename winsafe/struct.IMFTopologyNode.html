pub struct IMFTopologyNode(/* private fields */);

Expand description

IMFTopologyNode COM interface.

Automatically calls Release when the object goes out of scope.

Usually created with MFCreateTopologyNode function.

§Examples

use winsafe::{self as w, prelude::*, co};

let topology_node = w::MFCreateTopologyNode(co::MF_TOPOLOGY::OUTPUT_NODE)?;
